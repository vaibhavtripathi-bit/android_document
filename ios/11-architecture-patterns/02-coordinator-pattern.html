1. The Navigation Coordination Problem

In Apple's default MVC, view controllers navigate directly to other view controllers. This creates tight coupling — every VC knows about the VCs it can navigate to, making reuse impossible and deep linking a nightmare.

1.1 The Problem in Code

// Without coordinators — VC knows about its destination
class ProductListVC: UIViewController {
    func didSelectProduct(_ product: Product) {
        // ProductListVC knows about ProductDetailVC
        let detailVC = ProductDetailVC(product: product)
        navigationController?.pushViewController(detailVC, animated: true)
    }

    func didTapCart() {
        // ProductListVC knows about CartVC AND its dependencies
        let cartVC = CartVC(cartService: CartService(), paymentService: PaymentService())
        let nav = UINavigationController(rootViewController: cartVC)
        present(nav, animated: true)
    }

    func didTapLogin() {
        // ProductListVC knows about the entire auth flow
        let loginVC = LoginVC()
        loginVC.onSuccess = { [weak self] in
            self?.dismiss(animated: true)
            self?.reloadData()
        }
        present(UINavigationController(rootViewController: loginVC), animated: true)
    }
}

2. Basic Coordinator Protocol

The Coordinator pattern was popularized by Soroush Khanlou (2015). The core idea: a coordinator object owns the navigation controller and decides which screen to show next. View controllers delegate navigation decisions upward.

2.1 Protocol Definition

protocol Coordinator: AnyObject {
    var childCoordinators: [Coordinator] { get set }
    var navigationController: UINavigationController { get set }
    func start()
}

extension Coordinator {
    func addChild(_ coordinator: Coordinator) {
        childCoordinators.append(coordinator)
    }

    func removeChild(_ coordinator: Coordinator) {
        childCoordinators.removeAll { $0 === coordinator }
    }

    func removeAllChildren() {
        childCoordinators.removeAll()
    }
}

2.2 View Controller Communication

View controllers communicate navigation intent to coordinators via delegation, not direct navigation:

// VC declares what it CAN do, not WHERE to go
protocol ProductListVCDelegate: AnyObject {
    func productListDidSelect(_ product: Product)
    func productListDidTapCart()
    func productListDidTapSearch()
}

class ProductListVC: UIViewController {
    weak var delegate: ProductListVCDelegate?

    func didSelectProduct(_ product: Product) {
        delegate?.productListDidSelect(product)
    }

    func didTapCart() {
        delegate?.productListDidTapCart()
    }
}

3. UIKit Coordinator Implementation

3.1 Concrete Coordinator

final class ProductCoordinator: Coordinator {
    var childCoordinators: [Coordinator] = []
    var navigationController: UINavigationController

    private let dependencies: AppDependencies

    init(navigationController: UINavigationController,
         dependencies: AppDependencies) {
        self.navigationController = navigationController
        self.dependencies = dependencies
    }

    func start() {
        let viewModel = ProductListViewModel(
            productService: dependencies.productService
        )
        let vc = ProductListVC(viewModel: viewModel)
        vc.delegate = self
        navigationController.pushViewController(vc, animated: false)
    }

    private func showProductDetail(_ product: Product) {
        let viewModel = ProductDetailViewModel(
            product: product,
            cartService: dependencies.cartService
        )
        let vc = ProductDetailVC(viewModel: viewModel)
        vc.delegate = self
        navigationController.pushViewController(vc, animated: true)
    }

    private func showCart() {
        let cartCoordinator = CartCoordinator(
            navigationController: navigationController,
            dependencies: dependencies
        )
        addChild(cartCoordinator)
        cartCoordinator.onFinish = { [weak self] in
            self?.removeChild(cartCoordinator)
        }
        cartCoordinator.start()
    }
}

extension ProductCoordinator: ProductListVCDelegate {
    func productListDidSelect(_ product: Product) {
        showProductDetail(product)
    }

    func productListDidTapCart() {
        showCart()
    }

    func productListDidTapSearch() {
        let searchCoordinator = SearchCoordinator(
            navigationController: navigationController,
            dependencies: dependencies
        )
        addChild(searchCoordinator)
        searchCoordinator.start()
    }
}

extension ProductCoordinator: ProductDetailVCDelegate {
    func productDetailDidTapAddToCart(_ product: Product) {
        showCart()
    }

    func productDetailDidTapSeller(_ sellerId: String) {
        let sellerCoordinator = SellerCoordinator(
            sellerId: sellerId,
            navigationController: navigationController,
            dependencies: dependencies
        )
        addChild(sellerCoordinator)
        sellerCoordinator.start()
    }
}

3.2 UINavigationController Delegate Approach

To detect when a user pops a view controller (back button or swipe), the coordinator must be the navigation controller's delegate:

final class ProductCoordinator: NSObject, Coordinator {
    var childCoordinators: [Coordinator] = []
    var navigationController: UINavigationController
    private var completionHandlers: [UIViewController: () -> Void] = [:]

    func start() {
        navigationController.delegate = self
        // ... show first VC
    }

    func pushViewController(_ vc: UIViewController,
                           animated: Bool,
                           onPop: @escaping () -> Void) {
        completionHandlers[vc] = onPop
        navigationController.pushViewController(vc, animated: animated)
    }
}

extension ProductCoordinator: UINavigationControllerDelegate {
    func navigationController(
        _ navigationController: UINavigationController,
        didShow viewController: UIViewController,
        animated: Bool
    ) {
        guard let fromVC = navigationController.transitionCoordinator?.viewController(forKey: .from),
              !navigationController.viewControllers.contains(fromVC) else {
            return
        }
        // VC was popped
        if let handler = completionHandlers[fromVC] {
            handler()
            completionHandlers[fromVC] = nil
        }
    }
}

4.2 App Coordinator (Root)

final class AppCoordinator: Coordinator {
    var childCoordinators: [Coordinator] = []
    var navigationController: UINavigationController

    private let window: UIWindow
    private let dependencies: AppDependencies

    init(window: UIWindow, dependencies: AppDependencies) {
        self.window = window
        self.dependencies = dependencies
        self.navigationController = UINavigationController()
    }

    func start() {
        if dependencies.authService.isLoggedIn {
            showMainApp()
        } else {
            showAuth()
        }
        window.rootViewController = navigationController
        window.makeKeyAndVisible()
    }

    private func showAuth() {
        let authCoordinator = AuthCoordinator(
            navigationController: navigationController,
            dependencies: dependencies
        )
        authCoordinator.onLoginSuccess = { [weak self] in
            self?.removeChild(authCoordinator)
            self?.showMainApp()
        }
        addChild(authCoordinator)
        authCoordinator.start()
    }

    private func showMainApp() {
        let tabCoordinator = TabCoordinator(
            window: window,
            dependencies: dependencies
        )
        tabCoordinator.onLogout = { [weak self] in
            self?.removeChild(tabCoordinator)
            self?.showAuth()
        }
        addChild(tabCoordinator)
        tabCoordinator.start()
    }
}

// Tab coordinator manages UITabBarController
final class TabCoordinator: Coordinator {
    var childCoordinators: [Coordinator] = []
    var navigationController: UINavigationController
    var onLogout: (() -> Void)?

    private let tabBarController = UITabBarController()
    private let window: UIWindow
    private let dependencies: AppDependencies

    init(window: UIWindow, dependencies: AppDependencies) {
        self.window = window
        self.dependencies = dependencies
        self.navigationController = UINavigationController()
    }

    func start() {
        let homeNav = UINavigationController()
        let homeCoord = HomeCoordinator(navigationController: homeNav, dependencies: dependencies)

        let searchNav = UINavigationController()
        let searchCoord = SearchCoordinator(navigationController: searchNav, dependencies: dependencies)

        let profileNav = UINavigationController()
        let profileCoord = ProfileCoordinator(navigationController: profileNav, dependencies: dependencies)

        [homeCoord, searchCoord, profileCoord].forEach {
            addChild($0)
            $0.start()
        }

        tabBarController.viewControllers = [homeNav, searchNav, profileNav]
        window.rootViewController = tabBarController
    }
}

5. Deep Link Routing Through Coordinator Tree

Deep Linking is the Coordinator's Killer Feature

Without coordinators, deep linking requires manually constructing the entire navigation stack. With coordinators, you route a URL through the coordinator tree, and each coordinator handles its segment of the path.

5.1 Route Definition

enum DeepLink {
    case product(id: String)
    case productReviews(productId: String)
    case cart
    case checkout
    case userProfile(id: String)
    case settings
    case settingsNotifications

    static func parse(url: URL) -> DeepLink? {
        guard let components = URLComponents(url: url, resolvingAgainstBaseURL: false) else {
            return nil
        }
        let pathComponents = components.path.split(separator: "/").map(String.init)

        switch pathComponents {
        case ["product", let id]:
            return .product(id: id)
        case ["product", let id, "reviews"]:
            return .productReviews(productId: id)
        case ["cart"]:
            return .cart
        case ["user", let id]:
            return .userProfile(id: id)
        default:
            return nil
        }
    }
}

5.2 Routable Coordinator Protocol

protocol DeepLinkable {
    func handle(deepLink: DeepLink) -> Bool
}

extension AppCoordinator: DeepLinkable {
    func handle(deepLink: DeepLink) -> Bool {
        // First, try to route to existing child coordinators
        for child in childCoordinators {
            if let routable = child as? DeepLinkable,
               routable.handle(deepLink: deepLink) {
                return true
            }
        }

        // If no child can handle it, create the appropriate flow
        switch deepLink {
        case .product(let id):
            ensureMainApp()
            tabCoordinator?.selectTab(.home)
            return tabCoordinator?.homeCoordinator.handle(deepLink: deepLink) ?? false

        case .cart:
            ensureMainApp()
            tabCoordinator?.selectTab(.home)
            tabCoordinator?.homeCoordinator.showCart()
            return true

        case .userProfile(let id):
            ensureMainApp()
            tabCoordinator?.selectTab(.profile)
            return tabCoordinator?.profileCoordinator.handle(deepLink: deepLink) ?? false

        default:
            return false
        }
    }

    private func ensureMainApp() {
        guard dependencies.authService.isLoggedIn else {
            pendingDeepLink = deepLink
            showAuth()
            return
        }
        if tabCoordinator == nil {
            showMainApp()
        }
    }
}

6. SwiftUI Navigation with Coordinators

SwiftUI's NavigationStack (iOS 16+) with NavigationPath provides a value-based navigation model that works well with coordinators.

6.1 NavigationStack + Coordinator

@Observable
final class NavigationCoordinator {
    var path = NavigationPath()
    var sheet: SheetDestination?
    var fullScreenCover: CoverDestination?

    enum Route: Hashable {
        case productDetail(id: String)
        case reviews(productId: String)
        case sellerProfile(id: String)
        case cart
        case checkout
    }

    enum SheetDestination: Identifiable {
        case filter(FilterState)
        case share(Product)

        var id: String {
            switch self {
            case .filter: return "filter"
            case .share: return "share"
            }
        }
    }

    enum CoverDestination: Identifiable {
        case login
        case onboarding

        var id: String {
            switch self {
            case .login: return "login"
            case .onboarding: return "onboarding"
            }
        }
    }

    func push(_ route: Route) {
        path.append(route)
    }

    func pop() {
        guard !path.isEmpty else { return }
        path.removeLast()
    }

    func popToRoot() {
        path = NavigationPath()
    }

    func present(_ sheet: SheetDestination) {
        self.sheet = sheet
    }

    func handle(deepLink: DeepLink) {
        popToRoot()
        switch deepLink {
        case .product(let id):
            push(.productDetail(id: id))
        case .productReviews(let productId):
            push(.productDetail(id: productId))
            push(.reviews(productId: productId))
        default:
            break
        }
    }
}

// Root view using the coordinator
struct ProductFlowView: View {
    @State private var coordinator = NavigationCoordinator()

    var body: some View {
        NavigationStack(path: $coordinator.path) {
            ProductListView(coordinator: coordinator)
                .navigationDestination(for: NavigationCoordinator.Route.self) { route in
                    switch route {
                    case .productDetail(let id):
                        ProductDetailView(productId: id, coordinator: coordinator)
                    case .reviews(let productId):
                        ReviewsView(productId: productId)
                    case .sellerProfile(let id):
                        SellerProfileView(sellerId: id, coordinator: coordinator)
                    case .cart:
                        CartView(coordinator: coordinator)
                    case .checkout:
                        CheckoutView(coordinator: coordinator)
                    }
                }
        }
        .sheet(item: $coordinator.sheet) { destination in
            switch destination {
            case .filter(let state):
                FilterView(state: state)
            case .share(let product):
                ShareSheet(product: product)
            }
        }
        .fullScreenCover(item: $coordinator.fullScreenCover) { destination in
            switch destination {
            case .login:
                LoginView()
            case .onboarding:
                OnboardingView()
            }
        }
    }
}

6.2 Type-Safe Route Composition

protocol RouteProvider {
    associatedtype Route: Hashable
    @ViewBuilder func destination(for route: Route) -> some View
}

struct HomeRouteProvider: RouteProvider {
    enum Route: Hashable {
        case featured
        case category(Category)
        case product(Product.ID)
    }

    @ViewBuilder func destination(for route: Route) -> some View {
        switch route {
        case .featured:
            FeaturedView()
        case .category(let category):
            CategoryView(category: category)
        case .product(let id):
            ProductDetailView(productId: id)
        }
    }
}

7. Coordinator Lifecycle Management

Memory Leak Pitfall

The most common coordinator bug is a memory leak: child coordinators are added but never removed. If a user navigates back (pop) and the child coordinator isn't removed from the parent's array, it leaks along with all its view controllers and view models.

7.1 Lifecycle Tracking

protocol Coordinator: AnyObject {
    var id: UUID { get }
    var childCoordinators: [UUID: Coordinator] { get set }
    var onFinish: (() -> Void)? { get set }
    func start()
    func finish()
}

extension Coordinator {
    func addChild(_ coordinator: Coordinator) {
        childCoordinators[coordinator.id] = coordinator
        coordinator.onFinish = { [weak self, weak coordinator] in
            guard let coordinator else { return }
            self?.childCoordinators.removeValue(forKey: coordinator.id)
        }
    }

    func finish() {
        childCoordinators.forEach { $0.value.finish() }
        childCoordinators.removeAll()
        onFinish?()
    }
}

// Usage — child coordinator signals completion
final class CheckoutCoordinator: Coordinator {
    let id = UUID()
    var childCoordinators: [UUID: Coordinator] = [:]
    var onFinish: (() -> Void)?

    func orderCompleted() {
        navigationController.popToRootViewController(animated: true)
        finish()
    }
}

8. Architect: Coordinators for 100+ Screen Apps

Architect Must-Know: Coordinator Architecture at Scale

At 100+ screens, the coordinator tree becomes a critical piece of infrastructure. It must handle deep linking, A/B tested flows, authentication gates, and analytics — all while remaining debuggable and testable.

8.2 Coordinator Factory

protocol CoordinatorFactory {
    func makeProductCoordinator(
        productId: String,
        navigationController: UINavigationController
    ) -> Coordinator

    func makeCheckoutCoordinator(
        cart: Cart,
        navigationController: UINavigationController
    ) -> Coordinator
}

final class DefaultCoordinatorFactory: CoordinatorFactory {
    private let dependencies: AppDependencies
    private let featureFlags: FeatureFlagService
    private let abTesting: ABTestingService

    func makeCheckoutCoordinator(
        cart: Cart,
        navigationController: UINavigationController
    ) -> Coordinator {
        if abTesting.variant(for: .checkoutFlow) == .newCheckout {
            return NewCheckoutCoordinator(
                cart: cart,
                nav: navigationController,
                deps: dependencies
            )
        } else {
            return LegacyCheckoutCoordinator(
                cart: cart,
                nav: navigationController,
                deps: dependencies
            )
        }
    }
}

Best Practice: Coordinator Guidelines for Large Teams
  • One coordinator per user flow — Login flow, checkout flow, onboarding flow each get their own coordinator
  • Coordinators don't know about sibling coordinators — They communicate up to the parent, never sideways
  • Factory pattern for creation — Coordinators use factories, not direct instantiation, enabling A/B testing
  • Analytics middleware — Wrap coordinator transitions to automatically track screen views
  • Debug overlay — In debug builds, show the coordinator tree to diagnose navigation issues
