1. UIView.animate (Block-Based)

// Basic animation
UIView.animate(withDuration: 0.3) {
    view.alpha = 0
    view.transform = CGAffineTransform(scaleX: 0.8, y: 0.8)
}

// With completion
UIView.animate(withDuration: 0.3, animations: {
    view.alpha = 0
}, completion: { finished in
    if finished {
        view.removeFromSuperview()
    }
})

// Spring animation
UIView.animate(
    withDuration: 0.6,
    delay: 0,
    usingSpringWithDamping: 0.7,       // 0 = very bouncy, 1 = no bounce
    initialSpringVelocity: 0.5,       // Initial velocity relative to total distance
    options: [.curveEaseInOut],
    animations: {
        view.center = newCenter
    }
)

// iOS 17+ spring animation with Spring struct
UIView.animate(springDuration: 0.5, bounce: 0.3) {
    view.center = newCenter
}

// Keyframe animations
UIView.animateKeyframes(withDuration: 1.0, delay: 0) {
    UIView.addKeyframe(withRelativeStartTime: 0, relativeDuration: 0.25) {
        view.transform = CGAffineTransform(scaleX: 1.2, y: 1.2)
    }
    UIView.addKeyframe(withRelativeStartTime: 0.25, relativeDuration: 0.25) {
        view.transform = CGAffineTransform(rotationAngle: .pi)
    }
    UIView.addKeyframe(withRelativeStartTime: 0.5, relativeDuration: 0.25) {
        view.transform = CGAffineTransform(rotationAngle: .pi * 2)
    }
    UIView.addKeyframe(withRelativeStartTime: 0.75, relativeDuration: 0.25) {
        view.transform = .identity
    }
}

// Animatable UIView properties:
// frame, bounds, center, transform, alpha, backgroundColor,
// contentStretch (deprecated), layoutMargins (iOS 11+)
//
// NOT animatable via UIView.animate:
// cornerRadius, borderWidth, shadowOpacity (use Core Animation)

How UIView.animate Works Internally

// UIView.animate does the following:
// 1. Creates an explicit CATransaction
// 2. Temporarily modifies UIView's action(for:forKey:) delegate method
//    to return CABasicAnimation objects instead of NSNull()
// 3. Executes the animation block — property changes create CAAnimations
// 4. The model layer values are set IMMEDIATELY to the final values
// 5. The animations play on the presentation layer
// 6. Commits the transaction

// This means: during animation, the MODEL layer has the FINAL values
// The PRESENTATION layer shows the animated intermediate values
// This is why hit testing during animation can be surprising

2. UIViewPropertyAnimator (Interruptible)

UIViewPropertyAnimator (iOS 10+) is the modern animation API that supports interruption, reversal, and scrubbing. It's essential for gesture-driven animations.

// Basic usage
let animator = UIViewPropertyAnimator(duration: 0.3, curve: .easeInOut) {
    view.transform = CGAffineTransform(scaleX: 1.5, y: 1.5)
    view.alpha = 0.5
}
animator.startAnimation()

// Spring timing
let springAnimator = UIViewPropertyAnimator(
    duration: 0.5,
    dampingRatio: 0.7
) {
    view.center = newCenter
}

// Custom cubic bezier timing
let timing = UICubicTimingParameters(
    controlPoint1: CGPoint(x: 0.2, y: 0),
    controlPoint2: CGPoint(x: 0, y: 1)
)
let customAnimator = UIViewPropertyAnimator(duration: 0.4, timingParameters: timing)
customAnimator.addAnimations {
    view.frame = newFrame
}

// Interruptible animation — the key advantage
let interruptible = UIViewPropertyAnimator(duration: 0.5, curve: .easeOut) {
    view.center = targetCenter
}
interruptible.startAnimation()

// Later, while animation is running:
interruptible.pauseAnimation()           // Pause at current position
interruptible.isReversed = true           // Reverse direction
interruptible.continueAnimation(         // Resume with new timing
    withTimingParameters: UISpringTimingParameters(dampingRatio: 0.8),
    durationFactor: 0                     // 0 = remaining duration
)

// Scrubbing (gesture-driven)
interruptible.pauseAnimation()
interruptible.fractionComplete = 0.5     // Jump to 50% progress

// Adding animations and completions
interruptible.addAnimations {
    view.backgroundColor = .systemRed
}
interruptible.addCompletion { position in
    switch position {
    case .end:
        print("Completed forward")
    case .start:
        print("Reversed to start")
    case .current:
        print("Stopped at current position")
    @unknown default:
        break
    }
}

3. Core Animation Classes

// CABasicAnimation — animates a single property from A to B
let rotation = CABasicAnimation(keyPath: "transform.rotation.z")
rotation.fromValue = 0
rotation.toValue = CGFloat.pi * 2
rotation.duration = 1.0
rotation.repeatCount = .infinity
layer.add(rotation, forKey: "spin")

// CAKeyframeAnimation — animates through multiple values
let bounce = CAKeyframeAnimation(keyPath: "position.y")
bounce.values = [0, -30, 0, -15, 0, -5, 0]
bounce.keyTimes = [0, 0.2, 0.4, 0.5, 0.6, 0.7, 1.0]
bounce.duration = 0.8
bounce.isAdditive = true  // Values are added to current value
layer.add(bounce, forKey: "bounce")

// Path-based keyframe animation
let pathAnimation = CAKeyframeAnimation(keyPath: "position")
pathAnimation.path = UIBezierPath(ovalIn: orbitRect).cgPath
pathAnimation.duration = 3.0
pathAnimation.repeatCount = .infinity
pathAnimation.calculationMode = .paced  // Constant speed along path
layer.add(pathAnimation, forKey: "orbit")

// CASpringAnimation — physics-based spring
let spring = CASpringAnimation(keyPath: "position.x")
spring.fromValue = layer.position.x
spring.toValue = targetX
spring.mass = 1.0           // Mass of the object
spring.stiffness = 100       // Spring stiffness
spring.damping = 10          // Damping coefficient
spring.initialVelocity = 5   // Initial velocity
spring.duration = spring.settlingDuration  // Use calculated duration
layer.add(spring, forKey: "spring")

// CAAnimationGroup — multiple animations in parallel
let group = CAAnimationGroup()
group.animations = [rotation, bounce]
group.duration = 1.0
layer.add(group, forKey: "group")

// CATransition — built-in transition effects
let transition = CATransition()
transition.type = .push
transition.subtype = .fromRight
transition.duration = 0.3
layer.add(transition, forKey: kCATransition)

4. Presentation Layer vs Model Layer

This is one of the most important concepts in Core Animation and a frequent interview topic.

Core Animation Layer Trees: MODEL TREE PRESENTATION TREE RENDER TREE (your code sets) (current visual state) (render server) ┌─────────────┐ ┌─────────────┐ ┌─────────────┐ │ layer │ │ presentation│ │ render │ │ │ │ Layer │ │ (private) │ │ opacity: 0.0 │ ──────▶│ opacity: 0.7│ ──────▶│ opacity: 0.7 │ │ (final value)│ │ (animating) │ │ (GPU state) │ └─────────────┘ └─────────────┘ └─────────────┘ When you animate opacity from 1.0 to 0.0: • Model layer immediately becomes 0.0 • Presentation layer animates from 1.0 → 0.0 over duration • Render tree follows presentation tree (sent to GPU each frame) After animation completes: • All three trees show 0.0 
// Accessing the presentation layer
let currentOpacity = layer.presentation()?.opacity ?? layer.opacity
let currentPosition = layer.presentation()?.position ?? layer.position

// Why this matters for hit testing during animation:
// hitTest uses the MODEL layer (final values), not the presentation layer
// So a view animating from A to B is "touchable" at B from the start

// To hit test at the visual position during animation:
class AnimatingView: UIView {
    override func hitTest(_ point: CGPoint, with event: UIEvent?) -> UIView? {
        // Check the presentation layer's frame instead
        if let presentationFrame = layer.presentation()?.frame {
            if presentationFrame.contains(point) {
                return self
            }
        }
        return super.hitTest(point, with: event)
    }
}

// fillMode and isRemovedOnCompletion
let anim = CABasicAnimation(keyPath: "opacity")
anim.fromValue = 1
anim.toValue = 0
anim.duration = 0.5

// By default, animation is removed on completion
// The layer snaps back to the model value
anim.isRemovedOnCompletion = true  // default

// To keep the animation's final value visible:
// Option 1 (PREFERRED): Set the model value before adding animation
layer.opacity = 0  // Set model to final value
layer.add(anim, forKey: "fade")  // Animation plays from 1 → 0

// Option 2 (AVOID): Use fillMode — leaves animation in memory
anim.fillMode = .forwards
anim.isRemovedOnCompletion = false
// This keeps the animation object alive — memory leak risk
// The model layer still has the old value — confusing state
Common Pitfall

Never use fillMode = .forwards with isRemovedOnCompletion = false as a lazy solution. This leaves the animation in memory, creates a disconnect between model and presentation layers, and makes hit testing unreliable. Always set the model layer to the final value before (or at the same time as) adding the animation.

5. Interactive Transitions

// UIPercentDrivenInteractiveTransition — drives a transition with a gesture

class SwipeDismissInteraction: UIPercentDrivenInteractiveTransition {
    var isInteracting = false
    private weak var viewController: UIViewController?

    func attach(to vc: UIViewController) {
        viewController = vc
        let pan = UIPanGestureRecognizer(target: self, action: #selector(handlePan))
        vc.view.addGestureRecognizer(pan)
    }

    @objc private func handlePan(_ gesture: UIPanGestureRecognizer) {
        guard let view = gesture.view else { return }
        let translation = gesture.translation(in: view)
        let progress = min(1, max(0, translation.y / view.bounds.height))

        switch gesture.state {
        case .began:
            isInteracting = true
            viewController?.dismiss(animated: true)

        case .changed:
            update(progress)

        case .ended, .cancelled:
            isInteracting = false
            let velocity = gesture.velocity(in: view).y
            if progress > 0.5 || velocity > 1000 {
                finish()
            } else {
                cancel()
            }

        default:
            break
        }
    }
}

6. View Controller Transitions

// Custom view controller transition
class FadeTransition: NSObject, UIViewControllerAnimatedTransitioning {

    let isPresenting: Bool

    init(isPresenting: Bool) {
        self.isPresenting = isPresenting
    }

    func transitionDuration(
        using context: UIViewControllerContextTransitioning?
    ) -> TimeInterval {
        return 0.35
    }

    func animateTransition(
        using context: UIViewControllerContextTransitioning
    ) {
        let containerView = context.containerView

        if isPresenting {
            guard let toView = context.view(forKey: .to) else { return }
            containerView.addSubview(toView)
            toView.alpha = 0

            UIView.animate(
                withDuration: transitionDuration(using: context),
                animations: { toView.alpha = 1 },
                completion: { _ in
                    context.completeTransition(!context.transitionWasCancelled)
                }
            )
        } else {
            guard let fromView = context.view(forKey: .from) else { return }

            UIView.animate(
                withDuration: transitionDuration(using: context),
                animations: { fromView.alpha = 0 },
                completion: { _ in
                    fromView.removeFromSuperview()
                    context.completeTransition(!context.transitionWasCancelled)
                }
            )
        }
    }
}

// Transitioning delegate
class ModalPresenter: NSObject, UIViewControllerTransitioningDelegate {

    let interaction = SwipeDismissInteraction()

    func animationController(
        forPresented presented: UIViewController,
        presenting: UIViewController,
        source: UIViewController
    ) -> UIViewControllerAnimatedTransitioning? {
        return FadeTransition(isPresenting: true)
    }

    func animationController(
        forDismissed dismissed: UIViewController
    ) -> UIViewControllerAnimatedTransitioning? {
        return FadeTransition(isPresenting: false)
    }

    func interactionControllerForDismissal(
        using animator: UIViewControllerAnimatedTransitioning
    ) -> UIViewControllerInteractiveTransitioning? {
        return interaction.isInteracting ? interaction : nil
    }
}

7. Hero / Shared Element Transitions

// Shared element transition using UIViewControllerAnimatedTransitioning
class SharedElementTransition: NSObject, UIViewControllerAnimatedTransitioning {

    let originFrame: CGRect
    let isPresenting: Bool

    init(originFrame: CGRect, isPresenting: Bool) {
        self.originFrame = originFrame
        self.isPresenting = isPresenting
    }

    func transitionDuration(
        using context: UIViewControllerContextTransitioning?
    ) -> TimeInterval { 0.5 }

    func animateTransition(
        using context: UIViewControllerContextTransitioning
    ) {
        guard let toVC = context.viewController(forKey: .to),
              let toView = context.view(forKey: .to),
              let snapshot = toView.snapshotView(afterScreenUpdates: true)
        else { return }

        let containerView = context.containerView
        let finalFrame = context.finalFrame(for: toVC)

        snapshot.frame = originFrame
        snapshot.layer.cornerRadius = 12
        snapshot.clipsToBounds = true

        containerView.addSubview(toView)
        containerView.addSubview(snapshot)
        toView.isHidden = true

        let animator = UIViewPropertyAnimator(duration: 0.5, dampingRatio: 0.8) {
            snapshot.frame = finalFrame
            snapshot.layer.cornerRadius = 0
        }

        animator.addCompletion { _ in
            toView.isHidden = false
            snapshot.removeFromSuperview()
            context.completeTransition(!context.transitionWasCancelled)
        }

        animator.startAnimation()
    }
}

9. Architect: Interruptible Gesture-Driven Transitions

Architect Must-Know

Building interruptible, gesture-driven transitions at 120fps:

Modern iOS apps (Apple Maps, Photos, App Store) use gesture-driven transitions that can be interrupted, reversed, and completed with physics-based momentum. This requires combining UIViewPropertyAnimator with gesture recognizers.

Key principles:

  • Use UIViewPropertyAnimator — it supports pause, reverse, and scrubbing
  • Separate animation from gesture — The gesture controls fractionComplete, not view properties directly
  • Use spring timing for completion — When the gesture ends, complete with a spring for natural feel
  • Handle interruption — If the user grabs the view mid-animation, pause and transition to gesture control
// Complete interruptible transition system
class InterruptibleCardTransition {

    enum State {
        case collapsed, expanded
    }

    private var state: State = .collapsed
    private var runningAnimators: [UIViewPropertyAnimator] = []
    private var animationProgress: [CGFloat] = []

    func handlePan(_ gesture: UIPanGestureRecognizer) {
        switch gesture.state {
        case .began:
            startInteractiveTransition(to: state == .collapsed ? .expanded : .collapsed)

        case .changed:
            let translation = gesture.translation(in: gesture.view)
            var fraction = translation.y / totalDistance
            if state == .expanded { fraction *= -1 }
            updateInteractiveTransition(fraction: fraction)

        case .ended:
            let velocity = gesture.velocity(in: gesture.view)
            let shouldComplete = velocity.y > 500 ||
                (velocity.y >= 0 && runningAnimators.first?.fractionComplete ?? 0 > 0.5)
            continueInteractiveTransition(shouldComplete: shouldComplete)

        default:
            break
        }
    }

    private func startInteractiveTransition(to targetState: State) {
        if runningAnimators.isEmpty {
            createAnimators(to: targetState)
        }
        for animator in runningAnimators {
            animator.pauseAnimation()
        }
        animationProgress = runningAnimators.map { $0.fractionComplete }
    }

    private func updateInteractiveTransition(fraction: CGFloat) {
        for (index, animator) in runningAnimators.enumerated() {
            animator.fractionComplete = animationProgress[index] + fraction
        }
    }

    private func continueInteractiveTransition(shouldComplete: Bool) {
        for animator in runningAnimators {
            if !shouldComplete { animator.isReversed = true }
            animator.continueAnimation(
                withTimingParameters: UISpringTimingParameters(dampingRatio: 0.8),
                durationFactor: 0
            )
        }
    }
}
