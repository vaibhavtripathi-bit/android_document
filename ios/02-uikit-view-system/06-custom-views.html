1. Building Custom UIView Subclasses

A well-built custom view follows specific patterns for initialization, layout, and state management. Here's the complete template:

class AvatarBadgeView: UIView {

    // MARK: - Public API

    var image: UIImage? {
        didSet { imageView.image = image }
    }

    var badgeCount: Int = 0 {
        didSet {
            badgeLabel.text = badgeCount > 99 ? "99+" : "\(badgeCount)"
            badgeContainer.isHidden = badgeCount == 0
            invalidateIntrinsicContentSize()
        }
    }

    var avatarSize: CGFloat = 48 {
        didSet {
            invalidateIntrinsicContentSize()
            setNeedsLayout()
        }
    }

    // MARK: - Private Views

    private let imageView = UIImageView()
    private let badgeContainer = UIView()
    private let badgeLabel = UILabel()

    // MARK: - Initialization

    override init(frame: CGRect) {
        super.init(frame: frame)
        commonInit()
    }

    required init?(coder: NSCoder) {
        super.init(coder: coder)
        commonInit()
    }

    private func commonInit() {
        // Setup subviews
        imageView.contentMode = .scaleAspectFill
        imageView.clipsToBounds = true
        addSubview(imageView)

        badgeContainer.backgroundColor = .systemRed
        addSubview(badgeContainer)

        badgeLabel.font = .monospacedDigitSystemFont(ofSize: 11, weight: .bold)
        badgeLabel.textColor = .white
        badgeLabel.textAlignment = .center
        badgeContainer.addSubview(badgeLabel)

        badgeContainer.isHidden = true

        setupAccessibility()
    }

    // MARK: - Layout

    override var intrinsicContentSize: CGSize {
        return CGSize(width: avatarSize, height: avatarSize)
    }

    override func layoutSubviews() {
        super.layoutSubviews()

        imageView.frame = bounds
        imageView.layer.cornerRadius = bounds.height / 2

        let badgeSize = max(20, badgeLabel.sizeThatFits(
            CGSize(width: .greatestFiniteMagnitude, height: 20)
        ).width + 8)
        badgeContainer.frame = CGRect(
            x: bounds.width - badgeSize + 4,
            y: -4,
            width: badgeSize,
            height: 20
        )
        badgeContainer.layer.cornerRadius = 10
        badgeLabel.frame = badgeContainer.bounds
    }

    // MARK: - Accessibility

    private func setupAccessibility() {
        isAccessibilityElement = true
        accessibilityTraits = .image
    }

    override var accessibilityLabel: String? {
        get {
            var label = "Avatar"
            if badgeCount > 0 {
                label += ", \(badgeCount) notification\(badgeCount == 1 ? "" : "s")"
            }
            return label
        }
        set { }
    }
}
Best Practice

The commonInit() pattern: Always use a shared initialization method called from both init(frame:) and init?(coder:). This ensures your view works whether created programmatically or from a storyboard/xib. Never put setup code in only one initializer.

2. intrinsicContentSize Contract

// The intrinsicContentSize contract:
// 1. Return the natural size based on current content
// 2. Return UIView.noIntrinsicMetric for dimensions that have no natural size
// 3. Call invalidateIntrinsicContentSize() when content changes
// 4. The system creates implicit constraints from this value

class TagView: UIView {
    var title: String = "" {
        didSet {
            label.text = title
            invalidateIntrinsicContentSize()  // MUST call this
        }
    }

    private let label = UILabel()
    private let padding = UIEdgeInsets(top: 4, left: 12, bottom: 4, right: 12)

    override var intrinsicContentSize: CGSize {
        let labelSize = label.intrinsicContentSize
        return CGSize(
            width: labelSize.width + padding.left + padding.right,
            height: labelSize.height + padding.top + padding.bottom
        )
    }

    // For views that only have intrinsic height (like a separator):
    // override var intrinsicContentSize: CGSize {
    //     return CGSize(width: UIView.noIntrinsicMetric, height: 1)
    // }
}

3. sizeThatFits vs Auto Layout

// sizeThatFits(_:) — manual sizing API (pre-Auto Layout)
// Returns the best size for the view given a constraining size

class FlowLabel: UIView {
    var text: String = ""
    var font: UIFont = .systemFont(ofSize: 14)

    override func sizeThatFits(_ size: CGSize) -> CGSize {
        let textSize = (text as NSString).boundingRect(
            with: CGSize(width: size.width, height: .greatestFiniteMagnitude),
            options: [.usesLineFragmentOrigin],
            attributes: [.font: font],
            context: nil
        ).size
        return CGSize(
            width: ceil(textSize.width),
            height: ceil(textSize.height)
        )
    }

    // sizeToFit() calls sizeThatFits(bounds.size) and sets the frame
    // It's a convenience method — don't override it, override sizeThatFits
}

// When to use sizeThatFits vs intrinsicContentSize:
//
// sizeThatFits: Takes a constraining size parameter
//   → Use for manual layout (layoutSubviews)
//   → Use when the size depends on available space
//
// intrinsicContentSize: No parameters — returns the "natural" size
//   → Use with Auto Layout
//   → The system uses this + hugging/compression priorities
//
// systemLayoutSizeFitting: Uses Auto Layout to compute size
//   → Most expensive but most accurate for constraint-based views
//   → Used internally by self-sizing cells

4. State Invalidation Patterns

class ProgressRing: UIView {

    // Pattern: Property → invalidate → redraw on next cycle

    var progress: CGFloat = 0 {
        didSet {
            progress = min(1, max(0, progress))
            if progress != oldValue {
                setNeedsDisplay()          // Triggers draw(_:)
                updateAccessibility()
            }
        }
    }

    var ringWidth: CGFloat = 8 {
        didSet {
            setNeedsDisplay()              // Visual change → redraw
            invalidateIntrinsicContentSize() // Size might change
        }
    }

    var isExpanded: Bool = false {
        didSet {
            invalidateIntrinsicContentSize() // Size changes
            setNeedsLayout()                  // Subview positions change
        }
    }

    // Invalidation methods summary:
    // setNeedsDisplay()              → triggers draw(_:) on next render cycle
    // setNeedsLayout()               → triggers layoutSubviews() on next layout cycle
    // invalidateIntrinsicContentSize → recalculates intrinsic size, may trigger layout
    // setNeedsUpdateConstraints()    → triggers updateConstraints() on next cycle

    // IMPORTANT: These are all coalesced — calling them multiple times
    // in the same RunLoop iteration only triggers one actual pass.

    override func draw(_ rect: CGRect) {
        guard let context = UIGraphicsGetCurrentContext() else { return }

        let center = CGPoint(x: bounds.midX, y: bounds.midY)
        let radius = min(bounds.width, bounds.height) / 2 - ringWidth / 2

        // Background ring
        context.setStrokeColor(UIColor.systemGray5.cgColor)
        context.setLineWidth(ringWidth)
        context.setLineCap(.round)
        context.addArc(center: center, radius: radius,
                      startAngle: 0, endAngle: .pi * 2, clockwise: false)
        context.strokePath()

        // Progress ring
        context.setStrokeColor(UIColor.systemBlue.cgColor)
        let startAngle = -CGFloat.pi / 2
        let endAngle = startAngle + (.pi * 2 * progress)
        context.addArc(center: center, radius: radius,
                      startAngle: startAngle, endAngle: endAngle, clockwise: false)
        context.strokePath()
    }

    override var intrinsicContentSize: CGSize {
        let size = isExpanded ? 80 : 40
        return CGSize(width: size, height: size)
    }
}

5. @IBDesignable / @IBInspectable

// @IBDesignable renders the view in Interface Builder
// @IBInspectable exposes properties in the IB inspector

@IBDesignable
class RoundedButton: UIButton {

    @IBInspectable var cornerRadius: CGFloat = 8 {
        didSet { layer.cornerRadius = cornerRadius }
    }

    @IBInspectable var borderWidth: CGFloat = 0 {
        didSet { layer.borderWidth = borderWidth }
    }

    @IBInspectable var borderColor: UIColor? {
        didSet { layer.borderColor = borderColor?.cgColor }
    }

    override func prepareForInterfaceBuilder() {
        super.prepareForInterfaceBuilder()
        // Called only in IB — set sample data for preview
        setTitle("Button", for: .normal)
    }
}

// WHY @IBDesignable IS DYING:
// 1. Xcode builds a separate target to render designables — slow and fragile
// 2. Storyboards themselves are falling out of favor (programmatic UI, SwiftUI)
// 3. SwiftUI previews are a better solution for live preview
// 4. @IBDesignable often breaks with complex dependencies
// 5. Build times increase significantly with many designable views

// Modern alternative: SwiftUI preview for UIKit views
#if DEBUG
import SwiftUI

struct AvatarBadgeView_Preview: PreviewProvider {
    static var previews: some View {
        UIViewPreviewWrapper {
            let view = AvatarBadgeView()
            view.badgeCount = 5
            view.image = UIImage(systemName: "person.fill")
            return view
        }
        .frame(width: 60, height: 60)
        .previewLayout(.sizeThatFits)
    }
}

struct UIViewPreviewWrapper<V: UIView>: UIViewRepresentable {
    let builder: () -> V
    func makeUIView(context: Context) -> V { builder() }
    func updateUIView(_ uiView: V, context: Context) {}
}
#endif

6. Accessibility

Accessibility is not optional — it's required for App Store compliance and is a key differentiator in Staff-level interviews. Apple takes accessibility seriously, and so should you.

UIAccessibility Protocol

class StarRatingView: UIView {
    var rating: Int = 0 {
        didSet {
            setNeedsDisplay()
            updateAccessibility()
        }
    }
    let maxRating = 5

    override init(frame: CGRect) {
        super.init(frame: frame)

        // Basic accessibility setup
        isAccessibilityElement = true
        accessibilityTraits = .adjustable  // Supports increment/decrement
        updateAccessibility()
    }

    private func updateAccessibility() {
        accessibilityValue = "\(rating) out of \(maxRating) stars"
        accessibilityLabel = "Rating"
        accessibilityHint = "Swipe up or down to adjust"
    }

    // .adjustable trait requires these methods
    override func accessibilityIncrement() {
        rating = min(maxRating, rating + 1)
    }

    override func accessibilityDecrement() {
        rating = max(0, rating - 1)
    }

    required init?(coder: NSCoder) { fatalError() }
}

// Accessibility traits:
// .button         — activatable element
// .link           — opens a URL
// .header         — section header (VoiceOver navigation)
// .image          — image content
// .staticText     — non-interactive text
// .adjustable     — supports increment/decrement (slider-like)
// .notEnabled     — disabled element
// .selected       — currently selected
// .updatesFrequently — live-updating content

Custom Accessibility Actions

class MessageCell: UITableViewCell {
    override var accessibilityCustomActions: [UIAccessibilityCustomAction]? {
        get {
            return [
                UIAccessibilityCustomAction(
                    name: "Reply",
                    target: self,
                    selector: #selector(replyAction)
                ),
                UIAccessibilityCustomAction(
                    name: "Delete",
                    target: self,
                    selector: #selector(deleteAction)
                ),
                UIAccessibilityCustomAction(
                    name: "Forward",
                    target: self,
                    selector: #selector(forwardAction)
                )
            ]
        }
        set { }
    }

    @objc private func replyAction() -> Bool {
        // Perform reply — return true if handled
        return true
    }

    @objc private func deleteAction() -> Bool { return true }
    @objc private func forwardAction() -> Bool { return true }
}

Container Accessibility

// When a view contains multiple accessible elements
// but isn't an accessibility element itself

class DashboardCard: UIView {
    let titleLabel = UILabel()
    let valueLabel = UILabel()
    let actionButton = UIButton()

    func setupAccessibility() {
        // This view is NOT an accessibility element
        isAccessibilityElement = false

        // But it contains accessible elements
        accessibilityElements = [titleLabel, valueLabel, actionButton]

        // VoiceOver will navigate through these in order
    }

    // For grouping: make the card a single element
    func setupGroupedAccessibility() {
        isAccessibilityElement = true
        accessibilityLabel = "\(titleLabel.text ?? ""), \(valueLabel.text ?? "")"
        accessibilityTraits = .button
        accessibilityHint = "Double tap to view details"
    }
}

7. Dynamic Type Support

// Dynamic Type allows users to choose their preferred text size
// Your views MUST adapt to these changes

// Using preferred fonts (automatically scales)
label.font = UIFont.preferredFont(forTextStyle: .body)
label.adjustsFontForContentSizeCategory = true  // Auto-update on change

// Custom fonts with Dynamic Type scaling
let customFont = UIFont(name: "Avenir-Medium", size: 16)!
label.font = UIFontMetrics(forTextStyle: .body).scaledFont(for: customFont)
label.adjustsFontForContentSizeCategory = true

// Scaling other values (spacing, image sizes)
let metrics = UIFontMetrics(forTextStyle: .body)
let scaledSpacing = metrics.scaledValue(for: 16)  // Scales with text size
let scaledImageSize = metrics.scaledValue(for: 24)

// Responding to Dynamic Type changes
override func traitCollectionDidChange(_ previous: UITraitCollection?) {
    super.traitCollectionDidChange(previous)
    if previous?.preferredContentSizeCategory != traitCollection.preferredContentSizeCategory {
        // Text size changed — update layout
        invalidateIntrinsicContentSize()
        setNeedsLayout()
    }
}

// Accessibility text sizes (larger than standard range)
if traitCollection.preferredContentSizeCategory.isAccessibilityCategory {
    // Switch to a stacked layout for very large text
    stackView.axis = .vertical
} else {
    stackView.axis = .horizontal
}

8. Architect: Reusable Component Libraries

Architect Must-Know

Designing a reusable component library with accessibility built in:

  • Protocol-driven design — Define protocols for component configuration, not concrete types. This allows different data models to drive the same component.
  • Accessibility by default — Every component should be accessible out of the box. Make accessibility a requirement in code review, not an afterthought.
  • Dynamic Type as a first-class citizen — Design layouts that work from .extraSmall to .accessibilityExtraExtraExtraLarge. Use stack views that switch axis at accessibility sizes.
  • Theme support — Use semantic colors (.label, .systemBackground) and trait-based configuration for dark mode.
  • Testing — Snapshot test every component at multiple Dynamic Type sizes and in both light/dark mode.
// Protocol-driven component configuration
protocol CardContent {
    var title: String { get }
    var subtitle: String? { get }
    var accessibilityDescription: String { get }
}

class GenericCard<Content: CardContent>: UIView {
    var content: Content? {
        didSet { configure() }
    }

    private func configure() {
        guard let content = content else { return }
        titleLabel.text = content.title
        subtitleLabel.text = content.subtitle
        accessibilityLabel = content.accessibilityDescription
        invalidateIntrinsicContentSize()
    }

    private let titleLabel: UILabel = {
        let l = UILabel()
        l.font = .preferredFont(forTextStyle: .headline)
        l.adjustsFontForContentSizeCategory = true
        l.numberOfLines = 0
        return l
    }()

    private let subtitleLabel: UILabel = {
        let l = UILabel()
        l.font = .preferredFont(forTextStyle: .subheadline)
        l.adjustsFontForContentSizeCategory = true
        l.textColor = .secondaryLabel
        l.numberOfLines = 0
        return l
    }()
}
