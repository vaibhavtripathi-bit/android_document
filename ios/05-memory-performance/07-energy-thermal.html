ProcessInfo.ThermalState

iOS exposes the device's thermal state through ProcessInfo.processInfo.thermalState. As the device heats up, the system throttles CPU and GPU performance to prevent hardware damage.

State Meaning CPU Impact Your Response
.nominal Normal operating temperature Full performance No action needed
.fair Slightly elevated temperature Minor throttling possible Reduce non-essential work
.serious Significant thermal pressure Noticeable throttling (50-70%) Reduce quality, defer work
.critical Near thermal shutdown Severe throttling (30-50%) Minimum viable operation 
import Foundation

class ThermalStateManager {
    private var observer: NSObjectProtocol?

    init() {
        observer = NotificationCenter.default.addObserver(
            forName: ProcessInfo.thermalStateDidChangeNotification,
            object: nil,
            queue: .main
        ) { [weak self] _ in
            self?.handleThermalStateChange()
        }
    }

    func handleThermalStateChange() {
        let state = ProcessInfo.processInfo.thermalState

        switch state {
        case .nominal:
            restoreFullQuality()
        case .fair:
            reduceBackgroundWork()
        case .serious:
            reduceRenderQuality()
            pauseNonEssentialDownloads()
            reduceFrameRate(to: 30)
        case .critical:
            minimumViableMode()
            reduceFrameRate(to: 20)
            stopAllBackgroundWork()
        @unknown default:
            break
        }
    }

    deinit {
        if let observer = observer {
            NotificationCenter.default.removeObserver(observer)
        }
    }
}
Deep Dive: Thermal Throttling Mechanics

When the SoC temperature rises, iOS reduces clock frequencies in stages. The A-series chips have performance (P) and efficiency (E) cores. Throttling first migrates work from P-cores to E-cores, then reduces P-core frequency, then reduces E-core frequency. GPU throttling follows a similar pattern. At .critical, the device may also dim the display and disable the camera flash. The key insight: your app's performance characteristics change dynamically — code that runs in 16ms at .nominal might take 40ms at .serious, causing frame drops.

Adaptive Quality Pattern

class AdaptiveRenderer {
    enum QualityLevel {
        case high, medium, low, minimal
    }

    var currentQuality: QualityLevel {
        switch ProcessInfo.processInfo.thermalState {
        case .nominal:  return .high
        case .fair:     return .medium
        case .serious:  return .low
        case .critical: return .minimal
        @unknown default: return .medium
        }
    }

    func renderFrame() {
        switch currentQuality {
        case .high:
            renderAtFullResolution()
            applyPostProcessing()
            renderShadows(quality: .high)
        case .medium:
            renderAtFullResolution()
            renderShadows(quality: .medium)
        case .low:
            renderAtHalfResolution()
            disableShadows()
        case .minimal:
            renderAtQuarterResolution()
            disableAllEffects()
        }
    }
}

Background Energy Budget

iOS strictly limits background energy usage. Apps that consume too much energy in the background are throttled or terminated.

System delays your background launches │ │ 2. System reduces your background refresh frequency │ │ 3. Ultimately: app is terminated │ │ │ └──────────────────────────────────────────────────────────┘ 
// Efficient background work pattern
func applicationDidEnterBackground(_ application: UIApplication) {
    var taskID: UIBackgroundTaskIdentifier = .invalid

    taskID = application.beginBackgroundTask(withName: "SaveState") {
        // Expiration handler: clean up and end task
        application.endBackgroundTask(taskID)
        taskID = .invalid
    }

    DispatchQueue.global(qos: .utility).async {
        // Do minimal work — save state, flush caches
        self.saveApplicationState()
        self.flushPendingAnalytics()

        DispatchQueue.main.async {
            application.endBackgroundTask(taskID)
            taskID = .invalid
        }
    }
}

BGTaskScheduler Energy Awareness

import BackgroundTasks

class BackgroundTaskManager {

    func registerTasks() {
        BGTaskScheduler.shared.register(
            forTaskWithIdentifier: "com.app.refresh",
            using: nil
        ) { task in
            self.handleAppRefresh(task: task as! BGAppRefreshTask)
        }

        BGTaskScheduler.shared.register(
            forTaskWithIdentifier: "com.app.processing",
            using: nil
        ) { task in
            self.handleProcessing(task: task as! BGProcessingTask)
        }
    }

    func scheduleRefresh() {
        let request = BGAppRefreshTaskRequest(identifier: "com.app.refresh")
        request.earliestBeginDate = Date(timeIntervalSinceNow: 15 * 60)
        try? BGTaskScheduler.shared.submit(request)
    }

    func scheduleProcessing() {
        let request = BGProcessingTaskRequest(identifier: "com.app.processing")
        request.requiresNetworkConnectivity = true
        request.requiresExternalPower = true  // Only when charging
        try? BGTaskScheduler.shared.submit(request)
    }

    private func handleAppRefresh(task: BGAppRefreshTask) {
        scheduleRefresh()  // Schedule next refresh

        let operation = RefreshOperation()

        task.expirationHandler = {
            operation.cancel()
        }

        operation.completionBlock = {
            task.setTaskCompleted(success: !operation.isCancelled)
        }

        OperationQueue().addOperation(operation)
    }

    private func handleProcessing(task: BGProcessingTask) {
        let operation = HeavyProcessingOperation()

        task.expirationHandler = {
            operation.cancel()
        }

        operation.completionBlock = {
            task.setTaskCompleted(success: !operation.isCancelled)
        }

        OperationQueue().addOperation(operation)
    }
}
Best Practice: BGProcessingTask for Heavy Work

Use BGProcessingTask with requiresExternalPower = true for heavy operations like ML model training, database maintenance, or large sync operations. The system will only run these when the device is charging (usually overnight), giving you minutes of CPU time instead of seconds. This is the energy-optimal way to do heavy background work.

Network Coalescing

Every network request wakes the cellular/WiFi radio. The radio stays powered for ~10-30 seconds after each request (the "tail energy" problem). Coalescing requests reduces the number of radio wake-ups.

Network Energy: Individual vs Coalesced Requests Individual (bad): Radio: ████░░░░████░░░░████░░░░████░░░░████░░░░ req1 tail req2 tail req3 tail req4 tail req5 tail Total radio-on time: ~150 seconds Coalesced (good): Radio: ░░░░░░░░░░░░░░░░░░░░████████████░░░░░░░░ all 5 reqs tail Total radio-on time: ~40 seconds Energy savings: ~73% reduction in radio energy 
class NetworkCoalescer {
    private var pendingRequests: [URLRequest] = []
    private var flushTimer: Timer?
    private let maxBatchSize = 10
    private let maxDelay: TimeInterval = 5.0

    func enqueue(_ request: URLRequest) {
        pendingRequests.append(request)

        if pendingRequests.count >= maxBatchSize {
            flush()
        } else if flushTimer == nil {
            flushTimer = Timer.scheduledTimer(
                withTimeInterval: maxDelay, repeats: false
            ) { [weak self] _ in
                self?.flush()
            }
        }
    }

    private func flush() {
        flushTimer?.invalidate()
        flushTimer = nil

        let batch = pendingRequests
        pendingRequests.removeAll()

        // Send all requests in rapid succession
        // HTTP/2 multiplexing sends them on one connection
        for request in batch {
            URLSession.shared.dataTask(with: request) { _, _, _ in }.resume()
        }
    }
}

Push vs Poll

Approach Energy Cost Latency Use When
Push (APNs) Very low (system manages connection) Seconds Real-time updates, messages, alerts
Background Fetch Low (system-scheduled) Minutes to hours Content prefetching, feed updates
Polling (timer-based) High (radio wake per poll) Depends on interval Avoid — use push or background fetch
WebSocket Medium-high (keeps connection alive) Milliseconds Chat, live data, gaming
Server-Sent Events Medium (one-way connection) Milliseconds Live feeds, dashboards
Pitfall: Polling in Background

Never use timer-based polling in the background. iOS will throttle your timers, and each poll wakes the radio for 10-30 seconds. A 30-second poll interval means the radio is essentially always on. Use APNs silent push to trigger background fetches — the push infrastructure shares a single persistent connection across all apps, making it far more energy-efficient than per-app connections.

import MetricKit

class EnergyMetricsCollector: NSObject, MXMetricManagerSubscriber {

    func didReceive(_ payloads: [MXMetricPayload]) {
        for payload in payloads {
            // CPU metrics (energy proxy)
            if let cpu = payload.cpuMetrics {
                let cpuTime = cpu.cumulativeCPUTime
                report("cpu_time_seconds", value: cpuTime)
            }

            // GPU metrics
            if let gpu = payload.gpuMetrics {
                let gpuTime = gpu.cumulativeGPUTime
                report("gpu_time_seconds", value: gpuTime)
            }

            // Location metrics
            if let location = payload.locationActivityMetrics {
                report("location_best_accuracy_time",
                    value: location.cumulativeBestAccuracyTime)
                report("location_10m_accuracy_time",
                    value: location.cumulativeBestAccuracyForNavigationTime)
            }

            // Network metrics
            if let network = payload.networkTransferMetrics {
                report("cellular_upload",
                    value: network.cumulativeCellularUpload)
                report("cellular_download",
                    value: network.cumulativeCellularDownload)
                report("wifi_upload",
                    value: network.cumulativeWifiUpload)
                report("wifi_download",
                    value: network.cumulativeWifiDownload)
            }

            // Display metrics
            if let display = payload.displayMetrics {
                if let apl = display.averagePixelLuminance {
                    report("avg_pixel_luminance", value: apl)
                }
            }
        }
    }
}

Designing for Thermal Throttling

Architect Must-Know: Designing for Sustained Workloads

For apps with sustained heavy workloads (video processing, AR, gaming, ML inference), thermal throttling is not an edge case — it's the steady state. Here's how to design for it:

Sustained Performance Architecture

Adaptive Performance System: ┌──────────────────────────────────────────────────────────┐ │ │ │ ┌──────────────┐ ┌──────────────┐ │ │ │ Thermal State│ │ Frame Time │ │ │ │ Monitor │ │ Monitor │ │ │ └──────┬───────┘ └──────┬───────┘ │ │ │ │ │ │ ▼ ▼ │ │ ┌──────────────────────────────────┐ │ │ │ Performance Controller │ │ │ │ │ │ │ │ Target: Consistent frame time │ │ │ │ Input: thermal state + FPS │ │ │ │ Output: quality level │ │ │ └──────────────┬───────────────────┘ │ │ │ │ │ ┌───────┼───────┐ │ │ ▼ ▼ ▼ │ │ ┌──────────┐ ┌─────┐ ┌──────────┐ │ │ │ Render │ │ AI │ │ Physics │ │ │ │ Quality │ │ LOD │ │ Fidelity │ │ │ └──────────┘ └─────┘ └──────────┘ │ │ │ └──────────────────────────────────────────────────────────┘ 
class PerformanceController {
    private var frameTimeHistory: [CFTimeInterval] = []
    private let targetFrameTime: CFTimeInterval = 1.0 / 60.0
    private var displayLink: CADisplayLink?

    var qualityLevel: Float = 1.0  // 0.0 (minimum) to 1.0 (maximum)

    func startMonitoring() {
        displayLink = CADisplayLink(target: self, selector: #selector(tick))
        displayLink?.add(to: .main, forMode: .common)
    }

    @objc private func tick(_ link: CADisplayLink) {
        let frameTime = link.targetTimestamp - link.timestamp
        frameTimeHistory.append(frameTime)

        if frameTimeHistory.count > 60 {
            frameTimeHistory.removeFirst()
        }

        adjustQuality()
    }

    private func adjustQuality() {
        let avgFrameTime = frameTimeHistory.reduce(0, +) / Double(frameTimeHistory.count)
        let thermalState = ProcessInfo.processInfo.thermalState

        let thermalFactor: Float
        switch thermalState {
        case .nominal:  thermalFactor = 1.0
        case .fair:     thermalFactor = 0.8
        case .serious:  thermalFactor = 0.5
        case .critical: thermalFactor = 0.3
        @unknown default: thermalFactor = 0.5
        }

        let frameFactor = Float(targetFrameTime / avgFrameTime)

        // Blend thermal and frame-time signals
        let targetQuality = min(thermalFactor, frameFactor)
        qualityLevel = qualityLevel * 0.9 + targetQuality * 0.1  // Smooth
    }
}
