Part 1 — GoRouter Foundations
Navigator 2.0 Made Practical
GoRouter is the Flutter team's recommended router package. It wraps Navigator 2.0's RouterDelegate and RouteInformationParser into a declarative, URL-based API that works on all platforms including web.

1. GoRouter as a Navigator 2.0 Wrapper

Raw Navigator 2.0 requires implementing RouterDelegate, RouteInformationParser, and RouteInformationProvider — significant boilerplate for every app. GoRouter abstracts all three behind a simple configuration object.

// GoRouter configuration
final _router = GoRouter(
  initialLocation: '/home',
  debugLogDiagnostics: true,   // prints route changes to console
  routes: [
    GoRoute(
      path: '/home',
      name: 'home',
      builder: (context, state) => const HomeScreen(),
      routes: [
        // Nested route: /home/settings
        GoRoute(
          path: 'settings',
          name: 'settings',
          builder: (context, state) => const SettingsScreen(),
        ),
      ],
    ),
    GoRoute(
      path: '/product/:productId',
      name: 'product',
      builder: (context, state) {
        final id = state.pathParameters['productId']!;
        return ProductDetailScreen(productId: id);
      },
    ),
  ],
  errorBuilder: (context, state) => ErrorScreen(error: state.error),
);

// Wire into MaterialApp
MaterialApp.router(
  routerConfig: _router,
  title: 'My App',
  theme: ThemeData(useMaterial3: true),
)

2. GoRoute Parameters

Path Parameters

GoRoute(
  path: '/users/:userId/posts/:postId',
  builder: (context, state) {
    final userId = state.pathParameters['userId']!;
    final postId = state.pathParameters['postId']!;
    return PostDetailScreen(userId: userId, postId: postId);
  },

// Navigate:
context.go('/users/alice/posts/42');

Query Parameters

GoRoute(
  path: '/search',
  builder: (context, state) {
    final query  = state.uri.queryParameters['q'] ?? '';
    final filter = state.uri.queryParameters['filter'] ?? 'all';
    return SearchScreen(query: query, filter: filter);
  },
),

// Navigate:
context.go('/search?q=flutter&filter=packages');

Extra — Passing Objects (No URL Encoding)

GoRoute(
  path: '/checkout',
  builder: (context, state) {
    // extra is typed as Object? — cast carefully
    final cart = state.extra as Cart;
    return CheckoutScreen(cart: cart);
  },
),

// Navigate with extra:
context.go('/checkout', extra: cart);

// WARNING: extra is NOT preserved across app restarts or deep links.
// Use path/query parameters or reconstruct from state for deep linkable routes.
Warning
state.extra is an in-memory object and is lost on deep link cold start or browser refresh. For routes that must be deep-linkable, pass identifiers (IDs) as path parameters and fetch the object in the screen itself.

3. ShellRoute — Persistent Navigation Shell

ShellRoute provides a persistent shell widget (e.g., a BottomNavigationBar or NavigationRail) that wraps a set of child routes. The shell widget stays mounted while child routes are swapped — critical for tab-based navigation.

final _router = GoRouter(
  routes: [
    ShellRoute(
      // builder wraps all child routes with the shell widget
      builder: (context, state, child) {
        return AppShell(child: child); // shell with BottomNavigationBar
      },
      routes: [
        GoRoute(
          path: '/home',
          builder: (context, state) => const HomeTab(),
        ),
        GoRoute(
          path: '/search',
          builder: (context, state) => const SearchTab(),
        ),
        GoRoute(
          path: '/profile',
          builder: (context, state) => const ProfileTab(),
        ),
      ],
    ),
    // Routes outside shell (no bottom nav):
    GoRoute(
      path: '/onboarding',
      builder: (context, state) => const OnboardingScreen(),
    ),
  ],
);

AppShell with BottomNavigationBar

class AppShell extends StatelessWidget {
  final Widget child;
  const AppShell({required this.child, super.key});

  static const _tabs = ['/home', '/search', '/profile'];

  int _locationToTabIndex(String location) {
    final idx = _tabs.indexWhere((t) => location.startsWith(t));
    return idx < 0 ? 0 : idx;
  }

  @override
  Widget build(BuildContext context) {
    final location = GoRouterState.of(context).uri.toString();

    return Scaffold(
      body: child,
      bottomNavigationBar: NavigationBar(
        selectedIndex: _locationToTabIndex(location),
        onDestinationSelected: (index) => context.go(_tabs[index]),
        destinations: const [
          NavigationDestination(icon: Icon(Icons.home), label: 'Home'),
          NavigationDestination(icon: Icon(Icons.search), label: 'Search'),
          NavigationDestination(icon: Icon(Icons.person), label: 'Profile'),
        ],
      ),
    );
  }
}

Nested ShellRoutes for Separate Stacks Per Tab

// StatefulShellRoute.indexedStack — each tab maintains its own navigator stack
StatefulShellRoute.indexedStack(
  builder: (context, state, navigationShell) {
    return ScaffoldWithNavBar(navigationShell: navigationShell);
  },
  branches: [
    StatefulShellBranch(
      routes: [
        GoRoute(
          path: '/home',
          builder: (_, __) => const HomeScreen(),
          routes: [
            GoRoute(
              path: 'article/:id',
              builder: (_, state) =>
                ArticleScreen(id: state.pathParameters['id']!),
            ),
          ],
        ),
      ],
    ),
    StatefulShellBranch(
      routes: [
        GoRoute(
          path: '/search',
          builder: (_, __) => const SearchScreen(),
        ),
      ],
    ),
  ],
)
Staff-Level Insight
StatefulShellRoute.indexedStack maintains a separate Navigator stack per tab, so navigating to an article in the Home tab and then switching to Search and back preserves the Home stack's back history. This is the correct UX for apps like Instagram or YouTube. Without it, each tab switch resets the stack — a common UX bug in naive implementations.

4. Redirect Guards

Global Redirect (Auth Guard)

final _router = GoRouter(
  redirect: (context, state) {
    final isAuthenticated = context.read<AuthBloc>().state is AuthAuthenticated;
    final isLoggingIn = state.matchedLocation == '/login';

    if (!isAuthenticated && !isLoggingIn) {
      // Store intended destination for post-login redirect
      final from = state.uri.toString();
      return '/login?from=${Uri.encodeComponent(from)}';
    }

    if (isAuthenticated && isLoggingIn) {
      // Already logged in, redirect away from login
      final from = state.uri.queryParameters['from'];
      return from != null ? Uri.decodeComponent(from) : '/home';
    }

    return null; // null = no redirect, proceed normally
  },
  refreshListenable: GoRouterRefreshStream(authBloc.stream),
  routes: [...],
);

GoRouterRefreshStream — Reacting to Auth State Changes

// Triggers GoRouter to re-evaluate redirect when auth state changes
class GoRouterRefreshStream extends ChangeNotifier {
  late final StreamSubscription<dynamic> _subscription;

  GoRouterRefreshStream(Stream<dynamic> stream) {
    _subscription = stream.asBroadcastStream().listen((_) => notifyListeners());
  }

  @override
  void dispose() {
    _subscription.cancel();
    super.dispose();
  }
}

// Usage in GoRouter:
final _router = GoRouter(
  refreshListenable: GoRouterRefreshStream(authBloc.stream),
  redirect: (context, state) {
    // This is re-called every time authBloc emits a new state
    final isAuth = context.read<AuthBloc>().state is Authenticated;
    if (!isAuth && state.matchedLocation != '/login') return '/login';
    return null;
  },
  routes: [...],
);

Per-Route Redirect

GoRoute(
  path: '/admin',
  redirect: (context, state) {
    final user = context.read<AuthBloc>().state.user;
    if (user == null) return '/login';
    if (!user.isAdmin) return '/home'; // not authorized
    return null; // allow

5. GoRouterState

// Access in builder:
GoRoute(
  path: '/product/:id',
  builder: (context, state) {
    // Path parameters
    final productId = state.pathParameters['id']!;

    // Query parameters
    final tab = state.uri.queryParameters['tab'] ?? 'details';

    // Full URI
    final uri = state.uri; // Uri object

    // Matched location (without query params)
    final matched = state.matchedLocation; // e.g., '/product/123'

    // Full path (with query params)
    final fullPath = state.uri.toString(); // e.g., '/product/123?tab=reviews'

    // Extra (in-memory object)
    final extra = state.extra as Product?;

    // Route name
    final name = state.name; // 'product'

    return ProductScreen(id: productId, initialTab: tab);
  },
),

// Access GoRouterState anywhere in the subtree:
final state = GoRouterState.of(context);

6. Navigation Methods: go vs push vs replace vs pop

MethodBack StackURLUse Case
context.go(path) Resets stack to new route tree Updated Tab switches, post-auth redirect, home button
context.push(path) Pushes on top of current stack Updated Drill into detail screen (back returns to list)
context.replace(path) Replaces current route (no back) Updated Post-login (replace login with home), onboarding steps
context.pop() Pops top route Restored to previous Back button, dialog dismiss
context.pop(result) Pops and returns value Restored Picker/dialog returning selection
context.pushReplacement(path) Replaces top of stack Updated Replace current detail with another detail 
// go — resets stack (no back to previous)
context.go('/home');

// push — preserves back stack (can go back)
context.push('/product/123');

// Push and await result
final result = await context.push<bool>('/confirm-dialog');
if (result == true) { /* user confirmed */ }

// Named navigation (compile-safe)
context.goNamed('product', pathParameters: {'id': '123'});
context.pushNamed('settings');

// replace — no back stack entry added
context.replace('/home'); // after login
go vs push — The Key Difference
context.go('/home') is like pressing the Home button — it resets the navigator stack. context.push('/home') pushes home on top of the current route, so the user can press back and return. Use go for top-level navigation (tabs, auth flow transitions), push for drill-down navigation (list → detail).

8. Custom Page Transitions

GoRoute(
  path: '/modal',
  // pageBuilder gives full control over the Page object
  pageBuilder: (context, state) => CustomTransitionPage(
    key: state.pageKey,
    child: const ModalScreen(),
    transitionDuration: const Duration(milliseconds: 300),
    transitionsBuilder: (context, animation, secondaryAnimation, child) {
      // Slide up from bottom
      return SlideTransition(
        position: Tween(
          begin: const Offset(0, 1),
          end: Offset.zero,
        ).animate(CurvedAnimation(
          parent: animation,
          curve: Curves.easeOutCubic,
        )),
        child: child,
      );
    },
  ),
),

// For modal bottom sheets:
GoRoute(
  path: '/filter',
  pageBuilder: (context, state) => const NoTransitionPage(
    child: FilterSheet(),
  ),
),

11. Error Handling

final _router = GoRouter(
  errorBuilder: (context, state) {
    // state.error contains the routing error
    return Scaffold(
      body: Center(
        child: Column(
          mainAxisSize: MainAxisSize.min,
          children: [
            Text('Page not found: ${state.uri}'),
            ElevatedButton(
              onPressed: () => context.go('/home'),
              child: const Text('Go Home'),
            ),
          ],
        ),
      ),
    );
  },
  // Or use errorPageBuilder for custom page transitions:
  errorPageBuilder: (context, state) => MaterialPage(
    key: state.pageKey,
    child: ErrorScreen(error: state.error),

  routes: [...],
);
