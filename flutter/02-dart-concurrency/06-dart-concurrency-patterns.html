Logical Race Conditions in Async Dart

Dart's single-threaded concurrency model prevents data races at the memory level — you cannot have two threads simultaneously writing to the same variable. But logical race conditions are very much possible and common in async Dart code. A logical race is when the sequencing of async operations produces unexpected results even without shared memory mutation.

The Check-Then-Act Race

The classic pattern: check a condition, then act on it — but the condition may change between the check and the act because an await released the event loop.

// BUGGY — logical race condition
class AuthService {
  bool _isRefreshing = false;
  String? _cachedToken;

  Future<String> getValidToken() async {
    // Check if already refreshing
    if (_isRefreshing) {
      // Wait a bit and retry — naive, broken approach
      await Future.delayed(Duration(milliseconds: 100));
      return _cachedToken!;
    }

    if (_isTokenExpired()) {
      _isRefreshing = true;          // Set flag

      // RACE: another call to getValidToken() that arrives here
      // BETWEEN the _isRefreshing = true and the await below
      // will see _isRefreshing == true and branch to the wait above
      // but _cachedToken is still null — NPE!
      _cachedToken = await _refreshToken(); // await releases event loop
      _isRefreshing = false;
    }
    return _cachedToken!;
  }
}

The Double-Tap Race

// BUGGY — double-tap causes duplicate operations
class PaymentButton extends StatefulWidget { ... }

class _PaymentButtonState extends State<PaymentButton> {
  bool _isProcessing = false;

  Future<void> _onTap() async {
    if (_isProcessing) return; // Check
    _isProcessing = true;      // Act — but this is NOT atomic with the check!

    // If user taps twice before this line, both taps pass the check
    // because the event loop hasn't processed the state update yet
    await _processPayment(); // Duplicate payment!
    _isProcessing = false;
  }
}

// FIX: Use setState to mark processing AND prevent double-triggers
Future<void> _onTapFixed() async {
  if (_isProcessing) return;
  setState(() => _isProcessing = true); // Synchronous — immediate effect

  try {
    await _processPayment();
  } finally {
    if (mounted) setState(() => _isProcessing = false);
  }
}

Stale Closure Race

// BUGGY — stale closure captures old value
class SearchWidget extends StatefulWidget { ... }

class _SearchWidgetState extends State<SearchWidget> {
  String _query = '';
  List<Result> _results = [];

  void _onSearchChanged(String query) async {
    _query = query;
    final results = await _searchApi.search(query); // User types fast — multiple calls

    // By the time this completes, _query may have changed
    // We're setting results for an OLD query!
    setState(() => _results = results); // BUG: stale results
  }
}

// FIX: Capture and compare query at completion time
void _onSearchChangedFixed(String query) async {
  _query = query;
  final results = await _searchApi.search(query);

  // Only update if query hasn't changed since we started this search
  if (_query == query && mounted) {
    setState(() => _results = results);
  }
}

Async Mutex with package:synchronized

When You Need Mutual Exclusion

Sometimes you genuinely need to ensure only one async operation runs at a time for a given resource (e.g., database writes, token refresh, file writes). The synchronized package provides a Dart-native lock:

import 'package:synchronized/synchronized.dart';

class TokenManager {
  final _lock = Lock();
  String? _token;
  DateTime? _expiry;

  /// Thread-safe (in the async sense) token refresh
  Future<String> getValidToken() async {
    return _lock.synchronized(() async {
      // Only ONE caller executes this block at a time
      // Other callers wait for the lock to release
      if (_token != null && _expiry!.isAfter(DateTime.now())) {
        return _token!; // Still valid — return cached
      }

      // Refresh needed — only one refresh happens even if many callers arrived
      final newToken = await _refreshFromServer();
      _token = newToken.accessToken;
      _expiry = newToken.expiry;
      return _token!;
    });
  }

  // All callers that arrived during the refresh get the same new token
  // No duplicate refresh requests — this is the correct pattern
  Future<TokenResponse> _refreshFromServer() async { /* ... */ }
}

Reentrant Locks

// Lock() is NOT reentrant — calling synchronized() from within synchronized() deadlocks
// Use Lock(reentrant: true) for reentrant scenarios

final _reentrantLock = Lock(reentrant: true);

Future<void> outer() async {
  await _reentrantLock.synchronized(() async {
    await inner(); // Safe — reentrant lock allows this
  });
}

Future<void> inner() async {
  await _reentrantLock.synchronized(() async {
    // This would deadlock with Lock() — fine with reentrant: true
    print('inner critical section');
  });
}

Debounce Pattern

Timer-Based Debounce

Debouncing delays execution until a burst of calls has stopped. Classic use: search-as-you-type — don't fire a network request on every keystroke, only after typing pauses.

import 'dart:async';

class Debouncer {
  final Duration delay;
  Timer? _timer;

  Debouncer({required this.delay});

  void call(void Function() action) {
    _timer?.cancel(); // Cancel any pending timer
    _timer = Timer(delay, action); // Start fresh timer
  }

  void dispose() {
    _timer?.cancel();
  }
}

// Usage in StatefulWidget
class SearchBar extends StatefulWidget { ... }

class _SearchBarState extends State<SearchBar> {
  final _debouncer = Debouncer(delay: const Duration(milliseconds: 400));

  @override
  void dispose() {
    _debouncer.dispose();
    super.dispose();
  }

  void _onTextChanged(String query) {
    _debouncer.call(() {
      // Only fires 400ms after the user stops typing
      context.read<SearchCubit>().search(query);
    });
  }
}

Future-Based Debounce

/// Returns a Future that completes with the debounced result
class FutureDebouncer<T> {
  final Duration delay;
  Timer? _timer;
  Completer<T>? _completer;

  FutureDebouncer({required this.delay});

  Future<T> call(Future<T> Function() action) {
    _timer?.cancel();
    _completer ??= Completer<T>();
    final completer = _completer!;

    _timer = Timer(delay, () async {
      try {
        final result = await action();
        if (!completer.isCompleted) completer.complete(result);
      } catch (e, st) {
        if (!completer.isCompleted) completer.completeError(e, st);
      }
      _completer = null;
    });

    return completer.future;
  }
}

Throttle Pattern

Flag-Based Throttle

Throttling limits how frequently an action can execute. Unlike debounce (fires after a pause), throttle fires immediately and then ignores calls for a cooldown period.

class Throttler {
  final Duration cooldown;
  bool _isThrottled = false;

  Throttler({required this.cooldown});

  void call(void Function() action) {
    if (_isThrottled) return; // Ignore calls during cooldown

    action(); // Execute immediately
    _isThrottled = true;
    Future.delayed(cooldown, () => _isThrottled = false);
  }
}

// Scroll listener that fires at most once per 200ms
class ExpensiveScrollWidget extends StatefulWidget { ... }

class _ExpensiveScrollWidgetState extends State<ExpensiveScrollWidget> {
  final _throttler = Throttler(cooldown: const Duration(milliseconds: 200));
  final _controller = ScrollController();

  @override
  void initState() {
    super.initState();
    _controller.addListener(() {
      _throttler.call(_onScroll);
    });
  }

  void _onScroll() {
    // Expensive analytics call — throttled to max 5 calls/second
    Analytics.logScrollPosition(_controller.offset);
  }
}

RxDart Operators: debounceTime and throttleTime

import 'package:rxdart/rxdart.dart';

class SearchBloc {
  final _queryController = BehaviorSubject<String>();

  late final Stream<List<Result>> results = _queryController.stream
      .debounceTime(const Duration(milliseconds: 400)) // Debounce typing
      .distinct() // Skip duplicate queries
      .switchMap((query) => _searchApi.searchStream(query)) // Cancel previous
      .asBroadcastStream();

  void search(String query) => _queryController.add(query);

  void dispose() => _queryController.close();
}

// Throttle example with RxDart
final buttonPresses = _buttonController.stream
    .throttleTime(const Duration(seconds: 2)) // At most one per 2 seconds
    .listen((_) => _processPayment());

StreamQueue: Pull-Based Stream Consumption

import 'package:async/async.dart';

// Traditional stream listener — push-based (events arrive when they want)
// Problem: hard to coordinate multiple async steps with stream events

// StreamQueue — pull-based (you ask for the next event when you're ready)
Future<void> processStreamSequentially(Stream<int> stream) async {
  final queue = StreamQueue(stream);

  // Pull exactly the events you need, in sequence
  final first = await queue.next;    // Wait for first event
  final second = await queue.next;   // Wait for second event

  // Or pull a batch
  final batch = await queue.take(5); // Wait for next 5 events as a List

  // Or peek without consuming
  final peeked = await queue.peek;

  await queue.cancel();
}

// Real use case: parsing a binary protocol stream
Future<void> parseBinaryProtocol(Stream<List<int>> bytes) async {
  final queue = StreamQueue(bytes);

  while (await queue.hasNext) {
    // Read 4-byte length prefix
    final lengthBytes = await queue.take(4);
    final length = ByteData.view(Uint8List.fromList(lengthBytes.expand((b) => b).toList()).buffer)
        .getUint32(0, Endian.big);

    // Read exactly 'length' bytes of payload
    final payloadChunks = <List<int>>[];
    var remaining = length;
    while (remaining > 0) {
      final chunk = await queue.next;
      payloadChunks.add(chunk);
      remaining -= chunk.length;
    }
    _processPayload(payloadChunks.expand((b) => b).toList());
  }
}

CompositeSubscription: Managing Multiple Stream Subscriptions

/// Manually implemented CompositeSubscription for managing lifecycle
class CompositeSubscription {
  final List<StreamSubscription> _subscriptions = [];
  bool _disposed = false;

  void add(StreamSubscription subscription) {
    if (_disposed) {
      subscription.cancel();
      return;
    }
    _subscriptions.add(subscription);
  }

  Future<void> dispose() async {
    _disposed = true;
    final futures = _subscriptions.map((s) => s.cancel());
    _subscriptions.clear();
    await Future.wait(futures);
  }
}

// Usage in a StatefulWidget
class _DashboardState extends State<Dashboard> {
  final _subscriptions = CompositeSubscription();

  @override
  void initState() {
    super.initState();

    _subscriptions.add(
      context.read<UserBloc>().stream.listen(_onUserChanged),
    );
    _subscriptions.add(
      context.read<NetworkBloc>().stream.listen(_onNetworkChanged),
    );
    _subscriptions.add(
      _locationService.positionStream.listen(_onLocationChanged),
    );
  }

  @override
  void dispose() {
    _subscriptions.dispose(); // Cancels ALL subscriptions at once
    super.dispose();
  }
}

RxDart also provides a CompositeSubscription class with the same pattern. If you're already using RxDart, use theirs to avoid the boilerplate above.

Common Pitfalls

1. Unawaited Futures

// BUGGY — Future returned by _saveData is not awaited
void _onSave() {
  _saveData(); // Fire and forget — exceptions are swallowed!
  Navigator.pop(context); // May execute before save completes
}

// FIX A: Await the future
Future<void> _onSave() async {
  await _saveData(); // Wait for save to complete
  if (mounted) Navigator.pop(context);
}

// FIX B: If truly fire-and-forget, handle errors explicitly
void _onSave() {
  unawaited(_saveData().catchError((e) => logger.error('Save failed', e)));
  Navigator.pop(context);
}

// Enable linting: add to analysis_options.yaml
// linter:
//   rules:
//     - unawaited_futures

2. Async in initState Anti-Pattern

// BUGGY — initState is synchronous; async work here is fragile
@override
void initState() {
  super.initState();
  _loadData(); // This is fine — starts async work
  // But DON'T await here — initState cannot be async
}

// WRONG — this won't work:
@override
Future<void> initState() async { // Can't override as async!
  super.initState();
  await _loadData(); // Dart won't allow this
}

// CORRECT — start async work in initState, handle results safely
@override
void initState() {
  super.initState();
  WidgetsBinding.instance.addPostFrameCallback((_) {
    // Runs after first frame — safe to call setState
    _loadData();
  });
}

Future<void> _loadData() async {
  final data = await _repository.fetchData();
  if (!mounted) return; // Widget may have been disposed during fetch
  setState(() => _data = data);
}

3. BuildContext After Async Gap

// BUGGY — context used after async gap without mounted check
Future<void> _onDelete() async {
  await _repository.deleteItem(widget.itemId);
  // User may have navigated away during the delete!
  Navigator.of(context).pop(); // Throws if widget is unmounted
  ScaffoldMessenger.of(context).showSnackBar(...); // Same issue
}

// FIX — always check mounted after any await
Future<void> _onDeleteFixed() async {
  await _repository.deleteItem(widget.itemId);

  if (!mounted) return; // Widget was disposed — bail out

  Navigator.of(context).pop();
  ScaffoldMessenger.of(context).showSnackBar(
    const SnackBar(content: Text('Item deleted')),
  );
}

// ALTERNATIVE — capture before async gap
Future<void> _onDeleteAlternative() async {
  final navigator = Navigator.of(context);    // Capture before await
  final messenger = ScaffoldMessenger.of(context); // Capture before await

  await _repository.deleteItem(widget.itemId);

  // Now safe to use captured references (they don't require mounted check)
  navigator.pop();
  messenger.showSnackBar(const SnackBar(content: Text('Item deleted')));
}

4. setState After Dispose

// BUGGY — setState called after widget disposal
class _CountdownState extends State<CountdownWidget> {
  int _count = 10;
  Timer? _timer;

  @override
  void initState() {
    super.initState();
    _timer = Timer.periodic(Duration(seconds: 1), (_) {
      setState(() => _count--); // CRASH if widget is disposed before countdown ends!
    });
  }

  @override
  void dispose() {
    // Forgot to cancel the timer!
    super.dispose();
  }
}

// FIX — always cancel timers/subscriptions in dispose
@override
void dispose() {
  _timer?.cancel(); // Cancel BEFORE super.dispose()
  super.dispose();
}

// OR — check mounted in the callback
_timer = Timer.periodic(Duration(seconds: 1), (_) {
  if (!mounted) {
    _timer?.cancel();
    return;
  }
  setState(() => _count--);
});

5. Async Gaps in Loops

// PROBLEMATIC — state can change between iterations
Future<void> processAll(List<Item> items) async {
  for (final item in items) {
    await _processItem(item); // Each await yields control
    // Other event loop callbacks may run between iterations
    // 'items' might be modified by another callback!
  }
}

// FIX — snapshot the collection before the loop
Future<void> processAllFixed(List<Item> items) async {
  final snapshot = List<Item>.from(items); // Defensive copy
  for (final item in snapshot) {
    await _processItem(item);
  }
}

6. Zone Error Handling

import 'dart:async';

// Catching errors in async code that "escapes" try-catch
// Errors in unawaited futures are NOT caught by surrounding try-catch

void main() {
  // This does NOT catch errors from unawaited futures
  try {
    Future.error('uncaught!'); // Escapes try-catch — goes to Zone handler
  } catch (e) {
    print('NOT reached'); // Never called for unawaited Future errors
  }
}

// Correct approach: use runZonedGuarded for global async error handling
void main() {
  runZonedGuarded(
    () {
      runApp(MyApp());
    },
    (error, stackTrace) {
      // Catches ALL unhandled async errors in this zone
      FirebaseCrashlytics.instance.recordError(error, stackTrace);
    },
  );
}

// Flutter's FlutterError.onError for framework errors
FlutterError.onError = (details) {
  FirebaseCrashlytics.instance.recordFlutterFatalError(details);
};
Staff Insight

The most common source of production bugs in Flutter apps is not the architecture pattern (BLoC vs Riverpod vs Provider) — it's async mistakes: unawaited futures swallowing errors, BuildContext after async gap, and setState after dispose. A Staff engineer ensures these patterns are enforced at the team level via analysis_options.yaml rules, code review checklists, and lint rules like unawaited_futures, use_build_context_synchronously, and cancel_subscriptions.

Structured Concurrency Patterns

Cancellation Tokens

/// Simple cancellation token for cooperative cancellation
class CancellationToken {
  bool _cancelled = false;

  bool get isCancelled => _cancelled;

  void cancel() => _cancelled = true;

  void throwIfCancelled() {
    if (_cancelled) throw const CancelledException();
  }
}

class CancelledException implements Exception {
  const CancelledException();
}

// Usage: long-running operation that checks for cancellation
Future<List<ProcessedItem>> processLargeDataset(
  List<RawItem> items,
  CancellationToken token,
) async {
  final results = <ProcessedItem>[];
  for (final item in items) {
    token.throwIfCancelled(); // Check before each expensive operation
    final processed = await _processItem(item);
    results.add(processed);
  }
  return results;
}

// In widget:
CancellationToken? _token;

void _startProcessing() {
  _token = CancellationToken();
  processLargeDataset(items, _token!).then(/* ... */).catchError((e) {
    if (e is CancelledException) return; // Expected — ignore
    // Handle real errors
  });
}

void _cancelProcessing() {
  _token?.cancel();
}

Completer-Based One-Shot Events

/// Signal a one-time event across async code
class ReadySignal {
  final _completer = Completer<void>();

  Future<void> get ready => _completer.future;

  void markReady() {
    if (!_completer.isCompleted) _completer.complete();
  }
}

// Usage: wait for initialization before proceeding
class DatabaseService {
  final _ready = ReadySignal();

  DatabaseService() {
    _initialize();
  }

  Future<void> _initialize() async {
    await _openDatabase();
    await _runMigrations();
    _ready.markReady(); // Signal that DB is ready
  }

  Future<T> query<T>(String sql) async {
    await _ready.ready; // Wait for initialization if needed
    return _executeQuery(sql);
  }
}
