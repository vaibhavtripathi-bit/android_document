1. Wire Format

Every MethodChannel message is encoded by StandardMethodCodec, which wraps StandardMessageCodec. A method call is serialized as two consecutive StandardMessageCodec values: the method name string, then the arguments.

StandardMethodCodec Wire Format — MethodCall ═════════════════════════════════════════════ Byte offset Content ──────────── ────────────────────────────────────────────── 0x00 Type tag: 0x0c (String) 0x01 Method name length (variable-length integer) 0x02..N Method name bytes (UTF-8) N+1 Type tag for arguments (0x00=null, 0x0d=List, 0x0e=Map …) N+2..end Encoded argument value Example: invokeMethod("getBattery", null) ───────────────────────────────────────── 0c 0a 67 65 74 42 61 74 74 65 72 79 00 │ │ └──────────────────────────┘ └─ args = null (0x00) │ └─ length = 10 (varint) └─ String tag

1.1 Variable-Length Integer Encoding

// StandardMessageCodec uses a custom varint for sizes
// Values 0–253 : 1 byte  (the value itself)
// Values 254–65535 : 3 bytes (0xFE + 2-byte LE)
// Values ≥65536 : 5 bytes (0xFF + 4-byte LE)

// This means most field names/method names are 1-byte overhead for length.
// A String "getBattery" (10 chars) encodes as: 0x0c 0x0a [10 bytes]
//                                                tag  len  data

2. Calling Convention

2.1 invokeMethod — Basic Use

const MethodChannel _channel = MethodChannel('com.example/battery');

// Returns null if method not implemented on this platform.
// Throws PlatformException on native error.
// Throws MissingPluginException if no handler registered.
Future<int> getBatteryLevel() async {
  try {
    final level = await _channel.invokeMethod<int>('getBatteryLevel');
    return level ?? -1;
  } on PlatformException catch (e) {
    throw BatteryException(e.code, e.message);
  } on MissingPluginException {
    return -1; // platform doesn't support this API
  }
}

2.2 invokeListMethod / invokeMapMethod

// Convenience wrappers that cast the return type for you.
// invokeListMethod → Future<List<T>?>
// invokeMapMethod  → Future<Map<K,V>?>

final devices = await _channel
    .invokeListMethod<Map<String, dynamic>>('getBluetoothDevices');

// Without the helper you'd need:
// final raw = await _channel.invokeMethod('getBluetoothDevices');
// final devices = (raw as List).cast<Map<String, dynamic>>();

3. Native Handler — Android (Kotlin)

class BatteryPlugin : FlutterPlugin, MethodCallHandler {
    private lateinit var channel: MethodChannel
    private lateinit var context: Context

    // Called when the plugin is attached to a FlutterEngine.
    override fun onAttachedToEngine(binding: FlutterPlugin.FlutterPluginBinding) {
        context = binding.applicationContext
        channel = MethodChannel(binding.binaryMessenger, "com.example/battery")
        channel.setMethodCallHandler(this)
    }

    override fun onMethodCall(call: MethodCall, result: Result) {
        when (call.method) {
            "getBatteryLevel" -> {
                val level = getBatteryLevel()
                if (level != -1) result.success(level)
                else result.error(
                    "UNAVAILABLE",
                    "Battery level not available",
                    null  // optional details — any JSON-serializable object
                )
            }
            // Accessing typed arguments safely
            "setThreshold" -> {
                val threshold = call.argument<Int>("threshold")
                    ?: return result.error("MISSING_ARG", "threshold required", null)
                applyThreshold(threshold)
                result.success(null)
            }
            else -> result.notImplemented()
        }
    }

    override fun onDetachedFromEngine(binding: FlutterPlugin.FlutterPluginBinding) {
        // CRITICAL: tear down the channel to prevent memory leaks
        channel.setMethodCallHandler(null)
    }

    private fun getBatteryLevel(): Int {
        val bm = context.getSystemService(BatteryManager::class.java)
        return bm?.getIntProperty(BatteryManager.BATTERY_PROPERTY_CAPACITY) ?: -1
    }
}
Plugin Lifecycle

onAttachedToEngine replaces the deprecated registerWith(Registrar) pattern. Always tear down channel handlers in onDetachedFromEngine — failing to do so leaks the handler closure and can cause crashes when a second engine attaches to the same plugin class.

4. Native Handler — iOS (Swift)

public class BatteryPlugin: NSObject, FlutterPlugin {
    public static func register(with registrar: FlutterPluginRegistrar) {
        let channel = FlutterMethodChannel(
            name: "com.example/battery",
            binaryMessenger: registrar.messenger()
        )
        let instance = BatteryPlugin()
        registrar.addMethodCallDelegate(instance, channel: channel)
        // registrar.addMethodCallDelegate retains 'instance' weakly.
        // The channel itself holds a strong reference.
    }

    public func handle(_ call: FlutterMethodCall, result: @escaping FlutterResult) {
        switch call.method {
        case "getBatteryLevel":
            let device = UIDevice.current
            device.isBatteryMonitoringEnabled = true
            let level = device.batteryLevel
            if level < 0 {
                result(FlutterError(
                    code: "UNAVAILABLE",
                    message: "Battery level not available",
                    details: nil
                ))
            } else {
                result(Int(level * 100))
            }

        case "setThreshold":
            guard let args = call.arguments as? [String: Any],
                  let threshold = args["threshold"] as? Int
            else {
                result(FlutterError(code: "MISSING_ARG", message: "threshold required", details: nil))
                return
            }
            applyThreshold(threshold)
            result(nil)  // nil = success with null value

        default:
            result(FlutterMethodNotImplemented)
        }
    }
}

5. Error Handling

5.1 Error Flow

Error Propagation Path ══════════════════════ Native side: result.error("BATTERY_UNAVAILABLE", "Sensor offline", {"sensor": "bms"}) │ ▼ StandardMethodCodec encodes: [0x01][errorCode String][errorMessage String][details Any] │ ▼ BinaryMessenger transports bytes │ Dart side: StandardMethodCodec.decodeEnvelope() detects 0x01 tag → throws PlatformException { code: "BATTERY_UNAVAILABLE" message: "Sensor offline" details: {"sensor": "bms"} }

5.2 Comprehensive Dart Error Handling

sealed class BatteryResult {}
class BatteryLevel extends BatteryResult {
  final int level;
  BatteryLevel(this.level);
}
class BatteryError extends BatteryResult {
  final String code;
  final String? message;
  BatteryError(this.code, this.message);
}

Future<BatteryResult> fetchBattery() async {
  try {
    final level = await _channel
        .invokeMethod<int>('getBatteryLevel')
        .timeout(
          const Duration(seconds: 5),
          onTimeout: () => throw TimeoutException('Battery call timed out'),
        );
    return BatteryLevel(level ?? -1);
  } on PlatformException catch (e) {
    return BatteryError(e.code, e.message);
  } on MissingPluginException {
    return BatteryError('NOT_SUPPORTED', 'Not available on this platform');
  } on TimeoutException catch (e) {
    return BatteryError('TIMEOUT', e.message);
  }
}
Common Mistake

Platform channels have no built-in timeout. If the native side never calls result.success() / result.error(), the Dart Future hangs indefinitely. Always wrap invokeMethod calls with .timeout() for production code, especially for hardware-dependent APIs.

6. Calling Dart from Native

Plugins can invoke methods on the Dart side using the same MethodChannel from native. This is useful for push notifications, lifecycle events, or any native-initiated action.

6.1 Android → Dart

// In the plugin (Kotlin) — invoked from any thread
class PushPlugin : FlutterPlugin {
    private lateinit var channel: MethodChannel
    private lateinit var uiThreadHandler: Handler

    override fun onAttachedToEngine(binding: FlutterPlugin.FlutterPluginBinding) {
        channel = MethodChannel(binding.binaryMessenger, "com.example/push")
        uiThreadHandler = Handler(Looper.getMainLooper())
        // Register to receive Dart calls if needed
        channel.setMethodCallHandler { call, result ->
            if (call.method == "initialize") result.success(null)
            else result.notImplemented()
        }
    }

    // Call from FCM service (background thread) → Dart
    fun sendPushToFlutter(payload: Map<String, Any>) {
        // invokeMethod MUST be called on the main thread
        uiThreadHandler.post {
            channel.invokeMethod("onPushReceived", payload)
        }
    }
}

// Dart side — listen for incoming native calls
const _channel = MethodChannel('com.example/push');

void setupPushHandler() {
  _channel.setMethodCallHandler((call) async {
    if (call.method == 'onPushReceived') {
      final payload = call.arguments as Map<String, dynamic>;
      handlePush(payload);
    }
  });
}

7. Calling Channels from Background Isolates

// Pattern: root isolate passes token + SendPort to background isolate
void main() async {
  WidgetsFlutterBinding.ensureInitialized();

  final port = ReceivePort();
  final token = RootIsolateToken.instance!;

  await Isolate.spawn(
    _backgroundEntry,
    _IsolateInit(token, port.sendPort),
  );

  await for (final result in port) {
    // Process result on UI thread
  }
}

class _IsolateInit {
  final RootIsolateToken token;
  final SendPort sendPort;
  const _IsolateInit(this.token, this.sendPort);
}

void _backgroundEntry(_IsolateInit init) async {
  BackgroundIsolateBinaryMessenger.ensureInitialized(init.token);

  const channel = MethodChannel('com.example/file');
  final data = await channel.invokeMethod<Uint8List>('readFile', {'path': '/some/file'});
  init.sendPort.send(data);
}

8. Performance Characteristics

Metric Typical Value Notes
Round-trip latency 0.5 – 2 ms Varies by device, payload size, thread contention
Codec overhead (small Map) ~20–50 µs Dominated by Map iteration, not byte copying
Max safe call frequency ~100–200 calls/sec Beyond this, frame budget is at risk on 60 fps displays
Thread hop cost ~0.1–0.3 ms UI → main thread post and return post
Design Rule

Do not use MethodChannel for high-frequency calls — sensor polling, animation ticks, audio frames. At 100+ calls/second the combined serialization + thread-hop overhead will consume a significant fraction of a 16ms frame budget. For high-throughput data, use EventChannel (native-push), Dart FFI (synchronous, zero thread hop), or shared memory via dart:ffi + Uint8List.

9. Testing MethodChannels

9.1 Mocking in Unit Tests

void main() {
  TestWidgetsFlutterBinding.ensureInitialized();

  setUp(() {
    // Install a fake handler on the channel
    TestDefaultBinaryMessengerBinding.instance.defaultBinaryMessenger
        .setMockMethodCallHandler(
      const MethodChannel('com.example/battery'),
      (call) async {
        switch (call.method) {
          case 'getBatteryLevel': return 72;
          case 'setThreshold': return null;
          default: throw PlatformException(code: 'NOT_IMPL');
        }
      },
    );
  });

  tearDown(() {
    // Clear mock to avoid leaking state between tests
    TestDefaultBinaryMessengerBinding.instance.defaultBinaryMessenger
        .setMockMethodCallHandler(
      const MethodChannel('com.example/battery'),
      null,
    );
  });

  test('getBattery returns level from platform', () async {
    expect(await getBatteryLevel(), 72);
  });

  test('PlatformException propagates as BatteryError', () async {
    // Override mock to return error for this test
    TestDefaultBinaryMessengerBinding.instance.defaultBinaryMessenger
        .setMockMethodCallHandler(
      const MethodChannel('com.example/battery'),
      (_) async => throw PlatformException(code: 'UNAVAILABLE'),
    );
    final result = await fetchBattery();
    expect(result, isA<BatteryError>());
  });
}

9.2 Integration Testing with Real Channels

// integration_test/battery_test.dart
void main() {
  IntegrationTestWidgetsFlutterBinding.ensureInitialized();

  testWidgets('battery level is between 0 and 100', (tester) async {
    final level = await getBatteryLevel();
    // Real device: real channel call, no mock
    expect(level, inInclusiveRange(0, 100));
  });
}
Testing Strategy

Use setMockMethodCallHandler for unit tests — fast, no device needed, tests Dart-side logic in isolation. Use integration tests running on a real device or emulator to validate the actual platform implementation. Never use mocks in integration tests — the whole point is to exercise the native code path.
