03 — Production Plugin: Background State Streaming

Problem: Stream long-running native upload progress from Android/iOS to Flutter UI

Problem Statement

You are integrating a closed-source native file upload SDK. Build a Flutter plugin that:

  1. Streams real-time progress (bytes sent, transfer speed, ETA) from the native SDK to the Dart UI.
  2. Survives Android Activity recreation (screen rotation) without interrupting the upload.
  3. Allows pause, resume, and cancel from Dart.
  4. Cleans up all native resources correctly when the Flutter engine is detached (app goes to background and OS kills the engine).
Staff Signal
The hard part is not the channel setup — it is the lifecycle correctness. L6 candidates are expected to identify the Activity vs. Service distinction on Android, thread safety for the event sink, and the difference between engine detach and app backgrounding.

Plugin Structure: Federated Architecture

Federated plugins separate interface from implementation, allowing platform teams to ship their own implementations independently.

PackageResponsibility
file_uploaderApp-facing API. Delegates to platform interface.
file_uploader_platform_interfaceAbstract class defining the contract. No platform code.
file_uploader_androidAndroid implementation using Kotlin + native SDK.
file_uploader_iosiOS implementation using Swift + native SDK.

Channel Design

ChannelTypeMethods / Events
file_uploader/controlMethodChannelstart(params), pause(), resume(), cancel()
file_uploader/progressEventChannelEmits UploadProgress maps continuously
Design Choice
Use EventChannel (not polling via MethodChannel) for progress. EventChannel uses a push model — native calls into Dart — which is far more efficient than Dart polling every 100ms and avoids the complexity of managing timers on both sides.

  Future<void> cancelUpload();

class UploadParams {
  final String filePath;
  final String destinationUrl;
  final Map<String, String> headers;

  const UploadParams({
    required this.filePath,
    required this.destinationUrl,
    this.headers = const {},
  });

  Map<String, dynamic> toMap() => {
    'filePath': filePath,
    'destinationUrl': destinationUrl,
    'headers': headers,
  };
}

class UploadProgress {
  final int bytesSent;
  final int totalBytes;
  final double speedBytesPerSecond;
  final Duration eta;

  const UploadProgress({
    required this.bytesSent,
    required this.totalBytes,
    required this.speedBytesPerSecond,
    required this.eta,
  });

  double get fraction => totalBytes == 0 ? 0 : bytesSent / totalBytes;

  factory UploadProgress.fromMap(Map<dynamic, dynamic> map) {
    return UploadProgress(
      bytesSent: map['bytesSent'] as int,
      totalBytes: map['totalBytes'] as int,
      speedBytesPerSecond: (map['speedBytesPerSecond'] as num).toDouble(),
      eta: Duration(seconds: map['etaSeconds'] as int),
    );
  }
}

Dart: Method Channel Implementation

// file_uploader/lib/src/method_channel_file_uploader.dart

class MethodChannelFileUploader extends FileUploaderPlatform {
  static const _control = MethodChannel('file_uploader/control');
  static const _progressChannel = EventChannel('file_uploader/progress');

  // Cache the broadcast stream so multiple listeners work correctly.
  Stream<UploadProgress>? _progressStream;

  @override
  Stream<UploadProgress> get progressStream {
    _progressStream ??= _progressChannel
        .receiveBroadcastStream()
        .map((event) => UploadProgress.fromMap(event as Map));
    return _progressStream!;
  }

  @override
  Future<void> startUpload(UploadParams params) async {
    await _control.invokeMethod('start', params.toMap());
  }

  @override
  Future<void> pauseUpload() async {
    await _control.invokeMethod('pause');
  }

  @override
  Future<void> resumeUpload() async {
    await _control.invokeMethod('resume');
  }

  @override
  Future<void> cancelUpload() async {
    await _control.invokeMethod('cancel');
  }
}

Kotlin: Android Plugin Implementation

Plugin Registration and Lifecycle

// FileUploaderPlugin.kt

class FileUploaderPlugin : FlutterPlugin, MethodCallHandler {

  private lateinit var controlChannel: MethodChannel
  private lateinit var progressChannel: EventChannel
  private var eventSink: EventChannel.EventSink? = null

  // Upload runs in a Service, decoupled from the Activity lifecycle.
  private var uploadManager: UploadManager? = null

  // Handler ensures event sink is called on the main thread.
  private val mainHandler = Handler(Looper.getMainLooper())

  override fun onAttachedToEngine(binding: FlutterPlugin.FlutterPluginBinding) {
    controlChannel = MethodChannel(binding.binaryMessenger, "file_uploader/control")
    controlChannel.setMethodCallHandler(this)

    progressChannel = EventChannel(binding.binaryMessenger, "file_uploader/progress")
    progressChannel.setStreamHandler(object : EventChannel.StreamHandler {

      override fun onListen(arguments: Any?, sink: EventChannel.EventSink) {
        eventSink = sink
        // If an upload is already running (e.g., after rotation),
        // immediately emit the latest known progress.
        uploadManager?.latestProgress?.let { sink.success(it.toMap()) }
      }

      override fun onCancel(arguments: Any?) {
        // Dart side unsubscribed; stop emitting but don't cancel the upload.
        eventSink = null
      }
    })
  }

  override fun onDetachedFromEngine(binding: FlutterPlugin.FlutterPluginBinding) {
    // Engine is being destroyed. Null out the sink to prevent calls on
    // a dead messenger. The upload Service continues independently.
    controlChannel.setMethodCallHandler(null)
    progressChannel.setStreamHandler(null)
    eventSink = null
  }

  override fun onMethodCall(call: MethodCall, result: MethodChannel.Result) {
    when (call.method) {
      "start" -> {
        val params = UploadParams.fromMap(call.arguments as Map<*, *>)
        uploadManager = UploadManager(params) { progress ->
          // SDK callback arrives on a background thread.
          // Post to main thread before touching the event sink.
          mainHandler.post {
            eventSink?.success(progress.toMap())
          }
        }
        uploadManager?.start()
        result.success(null)
      }
      "pause" -> { uploadManager?.pause(); result.success(null) }
      "resume" -> { uploadManager?.resume(); result.success(null) }
      "cancel" -> { uploadManager?.cancel(); result.success(null) }
      else -> result.notImplemented()
    }
  }
}

UploadManager (Android Service wrapper)

// UploadManager.kt

class UploadManager(
  private val params: UploadParams,
  private val onProgress: (UploadProgress) -> Unit,
) {
  // Tracks the most recent progress for reattach scenarios.
  var latestProgress: UploadProgress? = null
    private set

  private val sdk = NativeUploadSdk()

  fun start() {
    sdk.startUpload(
      filePath = params.filePath,
      url = params.destinationUrl,
      progressCallback = { sent, total, speed, eta ->
        val progress = UploadProgress(
          bytesSent = sent,
          totalBytes = total,
          speedBytesPerSecond = speed,
          etaSeconds = eta,
        )
        latestProgress = progress
        onProgress(progress)
      },
      errorCallback = { e ->
        // Post errors to the event sink as well using error() method.
        Handler(Looper.getMainLooper()).post {
          eventSink?.error("UPLOAD_ERROR", e.message, null)
        }
      }
    )
  }

  fun pause() = sdk.pause()
  fun resume() = sdk.resume()
  fun cancel() { sdk.cancel(); latestProgress = null }
}
Critical: Thread Safety
The native SDK's progress callback fires on a background thread. The EventChannel.EventSink must only be called from the main thread — calling it from a background thread causes a crash or undefined behavior. Always post through Handler(Looper.getMainLooper()) on Android or DispatchQueue.main.async on iOS.

Swift: iOS EventChannel StreamHandler

// FileUploaderPlugin.swift

public class FileUploaderPlugin: NSObject, FlutterPlugin, FlutterStreamHandler {
  private var eventSink: FlutterEventSink?
  private var uploadManager: UploadManager?

  public static func register(with registrar: FlutterPluginRegistrar) {
    let controlChannel = FlutterMethodChannel(
      name: "file_uploader/control",
      binaryMessenger: registrar.messenger()
    )
    let progressChannel = FlutterEventChannel(
      name: "file_uploader/progress",
      binaryMessenger: registrar.messenger()
    )
    let instance = FileUploaderPlugin()
    registrar.addMethodCallDelegate(instance, channel: controlChannel)
    progressChannel.setStreamHandler(instance)
  }

  // MARK: - FlutterStreamHandler

  public func onListen(
    withArguments arguments: Any?,
    eventSink events: @escaping FlutterEventSink
  ) -> FlutterError? {
    self.eventSink = events
    // Replay latest progress if upload already running.
    if let latest = uploadManager?.latestProgress {
      events(latest.toMap())
    }
    return nil
  }

  public func onCancel(withArguments arguments: Any?) -> FlutterError? {
    self.eventSink = nil
    return nil
  }

  // MARK: - MethodCall handler

  public func handle(_ call: FlutterMethodCall, result: @escaping FlutterResult) {
    switch call.method {
    case "start":
      let args = call.arguments as! [String: Any]
      uploadManager = UploadManager(params: args) { [weak self] progress in
        // SDK callback may be on any thread; dispatch to main.
        DispatchQueue.main.async {
          self?.eventSink?(progress.toMap())
        }
      }
      uploadManager?.start()
      result(nil)
    case "pause":  uploadManager?.pause();  result(nil)
    case "resume": uploadManager?.resume(); result(nil)
    case "cancel": uploadManager?.cancel(); result(nil)
    default:       result(FlutterMethodNotImplemented)
    }
  }
}

Android Activity Recreation Handling

Screen rotation destroys and recreates the Activity. The Flutter engine may or may not be recreated depending on the app's android:configChanges setting. The key insight is:

  • The upload must live in a Service, not in the plugin or Activity. Services survive Activity recreation.
  • When the engine reattaches (onAttachedToEngine fires again), the plugin re-registers the event channel and immediately emits the latest progress from uploadManager?.latestProgress.
  • The UploadManager is stored as a singleton or in the Application class — not as a field on the plugin instance, which is recreated with the engine.
// Store UploadManager in Application to survive engine recreation.
class MyApplication : Application() {
  var uploadManager: UploadManager? = null
}

// In plugin:
override fun onAttachedToEngine(binding: FlutterPlugin.FlutterPluginBinding) {
  val app = binding.applicationContext as MyApplication
  uploadManager = app.uploadManager  // Reconnect to existing upload if any
  // ... channel setup
}

Testing

// Dart unit test using a mock platform
class MockFileUploaderPlatform extends FileUploaderPlatform {
  final _controller = StreamController<UploadProgress>.broadcast();

  @override
  Stream<UploadProgress> get progressStream => _controller.stream;

  @override
  Future<void> startUpload(UploadParams params) async {
    // Emit synthetic progress events.
    for (int i = 1; i <= 5; i++) {
      await Future.delayed(const Duration(milliseconds: 10));
      _controller.add(UploadProgress(
        bytesSent: i * 1024,
        totalBytes: 5120,
        speedBytesPerSecond: 1024,
        eta: Duration(seconds: 5 - i),
      ));
    }
  }

  @override Future<void> pauseUpload() async {}
  @override Future<void> resumeUpload() async {}
  @override Future<void> cancelUpload() async {}
}

void main() {
  test('progress stream emits 5 events and reaches 100%', () async {
    FileUploaderPlatform.instance = MockFileUploaderPlatform();
    final uploader = FileUploader();

    final events = <UploadProgress>[];
    final sub = uploader.progressStream.listen(events.add);

    await uploader.startUpload(UploadParams(
      filePath: '/tmp/test.mp4',
      destinationUrl: 'https://upload.example.com',
    ));

    await Future.delayed(const Duration(milliseconds: 100));
    await sub.cancel();

    expect(events.length, 5);
    expect(events.last.fraction, closeTo(1.0, 0.01));
  });
}
