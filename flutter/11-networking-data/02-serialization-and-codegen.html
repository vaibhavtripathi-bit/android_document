Part 1 — The Serialization Landscape
From Dynamic Maps to Type-Safe Models
dart:convert gives you raw dynamic maps. The ecosystem adds layers of type safety, code generation, and immutability on top — each with different trade-offs.

1. dart:convert — The Foundation

dart:convert's jsonDecode() returns dynamic — effectively an untyped Map<String, dynamic>. This is fast but completely unsafe at compile time.

import 'dart:convert';

// Raw decoding — no type safety
final String jsonString = '{"id": 1, "name": "Alice", "role": "admin"}';
final dynamic raw = jsonDecode(jsonString);

// Cast errors are runtime, not compile-time
final id   = raw['id']   as int;    // works
final name = raw['name'] as String; // works
final age  = raw['age']  as int;    // null — key missing, silent bug
// raw['id'] as String — throws CastError at runtime, not caught by analyzer

// Manual model class
class User {
  final int id;
  final String name;
  final String role;

  const User({required this.id, required this.name, required this.role});

  factory User.fromJson(Map<String, dynamic> json) {
    return User(
      id:   json['id']   as int,
      name: json['name'] as String,
      role: json['role'] as String,
    );
  }

  Map<String, dynamic> toJson() => {
    'id':   id,
    'name': name,
    'role': role,
  };
}

// Decode:
final user = User.fromJson(jsonDecode(jsonString) as Map<String, dynamic>);

// List decoding:
final jsonList = '[{"id":1,"name":"Alice"},{"id":2,"name":"Bob"}]';
final users = (jsonDecode(jsonList) as List)
    .map((e) => User.fromJson(e as Map<String, dynamic>))
    .toList();
Warning
Manual fromJson is error-prone at scale. A renamed JSON key, a missing null check, or a wrong type cast causes runtime crashes. In production apps with 50+ model classes, use code generation.

2. json_serializable — Generated fromJson/toJson

// pubspec.yaml
// dependencies:
//   json_annotation: ^4.8.0
// dev_dependencies:
//   json_serializable: ^6.7.0
//   build_runner: ^2.4.0

import 'package:json_annotation/json_annotation.dart';
part 'user.g.dart'; // generated file

@JsonSerializable(
  explicitToJson: true, // ensures nested objects use toJson()
  includeIfNull: false, // omit null fields from toJson output
)
class User {
  final int id;
  final String name;

  // Custom key name (API uses snake_case, Dart uses camelCase)
  @JsonKey(name: 'display_name')
  final String displayName;

  // Default value when key is missing
  @JsonKey(defaultValue: 'user')
  final String role;

  // Custom serialization for enum
  @JsonKey(unknownEnumValue: UserStatus.unknown)
  final UserStatus status;

  // Ignore field in serialization
  @JsonKey(includeFromJson: false, includeToJson: false)
  final bool isDirty;

  // Custom converter for complex types (e.g., DateTime from Unix timestamp)
  @JsonKey(fromJson: _dateFromTimestamp, toJson: _dateToTimestamp)
  final DateTime createdAt;

  const User({
    required this.id,
    required this.name,
    required this.displayName,
    required this.role,
    required this.status,
    required this.createdAt,
    this.isDirty = false,
  });

  factory User.fromJson(Map<String, dynamic> json) => _$UserFromJson(json);
  Map<String, dynamic> toJson() => _$UserToJson(this);

  static DateTime _dateFromTimestamp(int ts) =>
    DateTime.fromMillisecondsSinceEpoch(ts * 1000);
  static int _dateToTimestamp(DateTime dt) =>
    dt.millisecondsSinceEpoch ~/ 1000;
}

Nested Objects and Lists

@JsonSerializable(explicitToJson: true)
class Order {
  final String id;
  final User customer;           // nested object — needs explicitToJson: true
  final List<OrderItem> items;   // list of nested objects
  final Address? shippingAddress; // nullable nested object

  const Order({
    required this.id,
    required this.customer,
    required this.items,
    this.shippingAddress,
  });

  factory Order.fromJson(Map<String, dynamic> json) => _$OrderFromJson(json);
  Map<String, dynamic> toJson() => _$OrderToJson(this);
}

Enum Serialization

// Using @JsonEnum for value-based enum serialization
@JsonEnum(valueField: 'apiValue')
enum UserRole {
  admin('ADMIN'),
  editor('EDITOR'),
  viewer('VIEWER'),
  unknown('UNKNOWN');

  const UserRole(this.apiValue);
  final String apiValue;
}

// Or use @JsonValue annotation on each value:
enum OrderStatus {
  @JsonValue('pending')  pending,
  @JsonValue('paid')     paid,
  @JsonValue('shipped')  shipped,
  @JsonValue('cancelled') cancelled,
}
Part 2 — Freezed
Immutable Data Classes + Union Types
Freezed generates immutable data classes with copyWith, deep equality, pattern matching, and optional JSON serialization. The most powerful serialization solution for complex domain models.

3. Freezed — Data Classes and Sealed Types

// pubspec.yaml
// dependencies:
//   freezed_annotation: ^2.4.0
//   json_annotation: ^4.8.0
// dev_dependencies:
//   freezed: ^2.4.0
//   json_serializable: ^6.7.0
//   build_runner: ^2.4.0

import 'package:freezed_annotation/freezed_annotation.dart';
part 'product.freezed.dart';
part 'product.g.dart'; // needed for JSON serialization

@freezed
class Product with _$Product {
  const factory Product({
    required String id,
    required String name,
    required double price,
    @Default(0) int stockCount,
    String? imageUrl,
    @Default([]) List<String> tags,
  }) = _Product;

  factory Product.fromJson(Map<String, dynamic> json) =>
      _$ProductFromJson(json);
}

// What Freezed generates:
// - Immutable class (all fields are final)
// - copyWith() method
// - == and hashCode based on all fields
// - toString() for debugging
// - fromJson/toJson (via json_serializable integration)

Using copyWith

final product = Product(id: '1', name: 'Widget', price: 9.99);

// Create a modified copy — original is unchanged
final discounted = product.copyWith(price: 7.99);
final renamed    = product.copyWith(name: 'Super Widget', tags: ['new', 'sale']);

// Deep copy for nested objects:
@freezed
class Cart with _$Cart {
  const factory Cart({
    required List<CartItem> items,
    required double total,
  }) = _Cart;
}

// Add item — must create new Cart with new list
final newCart = cart.copyWith(
  items: [...cart.items, newItem],
  total: cart.total + newItem.price,
);

Freezed Union Types (Sealed Classes)

// Result type pattern — replaces exceptions in functional style
@freezed
sealed class Result<T> with _$Result<T> {
  const factory Result.success(T data) = Success<T>;
  const factory Result.failure(String message, {Object? error}) = Failure<T>;
  const factory Result.loading() = Loading<T>;
}

// Usage with pattern matching (Dart 3):
final result = await fetchUser(userId);

final widget = switch (result) {
  Success(:final data)          => UserCard(user: data),
  Failure(:final message)       => ErrorWidget(message: message),
  Loading()                     => const CircularProgressIndicator(),
};

// Or with when():
result.when(
  success: (data)           => UserCard(user: data),
  failure: (msg, {error})   => ErrorWidget(message: msg),
  loading: ()               => const CircularProgressIndicator(),
);

Freezed for API Response Union Types

// Model a polymorphic API response
@freezed
sealed class Notification with _$Notification {
  const factory Notification.message({
    required String id,
    required String senderId,
    required String text,
    required DateTime sentAt,
  }) = MessageNotification;

  const factory Notification.friendRequest({
    required String id,
    required String fromUserId,
    required String fromUserName,
  }) = FriendRequestNotification;

  const factory Notification.system({
    required String id,
    required String title,
    required String body,
  }) = SystemNotification;

  factory Notification.fromJson(Map<String, dynamic> json) =>
      _$NotificationFromJson(json);
}

// Pattern match in widget:
switch (notification) {
  case MessageNotification(:final text, :final senderId):
    return MessageTile(text: text, senderId: senderId);
  case FriendRequestNotification(:final fromUserName):
    return FriendRequestTile(name: fromUserName);
  case SystemNotification(:final title, :final body):
    return SystemTile(title: title, body: body);
}
Staff-Level Insight
Freezed's sealed union types are the Dart equivalent of Kotlin sealed classes or Swift enums with associated values. They are the correct way to model state machines, API responses with multiple shapes, and error types. Using a plain class hierarchy requires casting and instanceof checks (fragile). Sealed types with pattern matching are exhaustive — the compiler tells you when you've missed a case.

4. Dart 3 Records for Simple DTOs

// Records — anonymous, structural tuple types
// Perfect for simple multi-value returns without defining a class

// Function returning multiple values
(String name, int age) getUserInfo(String id) {
  return ('Alice', 30);
}

final (name, age) = getUserInfo('user-1');

// Named record fields
({String userId, String displayName, bool isAdmin}) parseAuthToken(String token) {
  // parse JWT...
  return (userId: 'u1', displayName: 'Alice', isAdmin: false);
}

final info = parseAuthToken(token);
print(info.userId);       // 'u1'
print(info.displayName);  // 'Alice'

// Records as map keys (structural equality built-in):
final cache = <(String, int), List<Product>>{};
cache[('electronics', 1)] = products; // (category, page) as key

// Parallel async with records (Dart 3):
final (user, posts, followers) = await (
  fetchUser(id),
  fetchPosts(id),
  fetchFollowerCount(id),
).wait;

5. Dart 3 Sealed Classes for Sum Types

// Native sealed classes (Dart 3) — without Freezed
sealed class NetworkResult<T> {}

final class NetworkSuccess<T> extends NetworkResult<T> {
  final T data;
  const NetworkSuccess(this.data);
}

final class NetworkError<T> extends NetworkResult<T> {
  final String message;
  final int? statusCode;
  const NetworkError(this.message, {this.statusCode});
}

final class NetworkLoading<T> extends NetworkResult<T> {
  const NetworkLoading();
}

// Exhaustive switch — compiler warns if case is missing:
NetworkResult<User> result = await fetchUser();

final display = switch (result) {
  NetworkSuccess(:final data)    => UserView(user: data),
  NetworkError(:final message)   => ErrorView(message: message),
  NetworkLoading()               => const LoadingView(),
}; // compiler error if any case is missing

6. Manual Serialization — Performance-Critical Code

// For high-throughput paths (e.g., parsing 10,000 messages from WebSocket),
// hand-written parsing avoids reflection overhead.

class Message {
  final String id;
  final String type;
  final String content;
  final int timestamp;

  const Message({
    required this.id,
    required this.type,
    required this.content,
    required this.timestamp,
  });

  // Optimized fromJson — direct field access, no reflection
  factory Message.fromJson(Map<String, dynamic> json) {
    return Message(
      id:        json['id'] as String,
      type:      json['type'] as String? ?? 'text',
      content:   json['content'] as String,
      timestamp: json['ts'] as int,
    );
  }

  Map<String, dynamic> toJson() => {
    'id':      id,
    'type':    type,
    'content': content,
    'ts':      timestamp,
  };
}

// Batch parsing on isolate to avoid janking the UI:
Future<List<Message>> parseMessages(String jsonString) async {
  return compute(_parseOnIsolate, jsonString);
}

List<Message> _parseOnIsolate(String jsonString) {
  final list = jsonDecode(jsonString) as List;
  return list.map((e) => Message.fromJson(e as Map<String, dynamic>)).toList();
}

7. Handling Nulls, Defaults, and Schema Evolution

// Robust fromJson for evolving APIs:
@JsonSerializable()
class UserProfile {
  final String id;
  final String name;

  // Nullable — field added in API v2, absent in v1 responses
  final String? bio;

  // Default for missing fields (backward compat)
  @JsonKey(defaultValue: false)
  final bool isPremium;

  // Custom null handling
  @JsonKey(
    fromJson: _parseNullableDate,
    toJson: _serializeDate,
  )
  final DateTime? lastSeen;

  static DateTime? _parseNullableDate(dynamic value) {
    if (value == null) return null;
    if (value is String) return DateTime.tryParse(value);
    if (value is int) return DateTime.fromMillisecondsSinceEpoch(value);
    return null;
  }

  static String? _serializeDate(DateTime? dt) => dt?.toIso8601String();
}

9. build_runner Workflow

# One-time build (CI, before production build):
dart run build_runner build --delete-conflicting-outputs

# Watch mode (development):
dart run build_runner watch --delete-conflicting-outputs

# Clean and rebuild (fixes stale cache issues):
dart run build_runner clean
dart run build_runner build --delete-conflicting-outputs

# Common issues:
# - "Already exists" error: use --delete-conflicting-outputs
# - Part file not found: ensure part directive matches filename exactly
# - Build hangs: check for circular imports in generated files
