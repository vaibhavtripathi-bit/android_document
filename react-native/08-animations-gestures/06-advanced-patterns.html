1. Physics-Based Animations

Physics-based animations feel natural because they model real-world behavior — springs, gravity, friction, and momentum. Reanimated's withSpring and withDecay are the primary tools.

1.1 Spring Physics

import { withSpring, withDecay } from 'react-native-reanimated';

// Spring configurations for different feel
const SPRING_CONFIGS = {
  gentle: { damping: 20, stiffness: 100, mass: 1 },
  bouncy: { damping: 8, stiffness: 150, mass: 0.5 },
  stiff: { damping: 30, stiffness: 300, mass: 1 },
  wobbly: { damping: 4, stiffness: 80, mass: 1 },
  snappy: { damping: 15, stiffness: 200, mass: 0.8 },
};

// Gravity simulation
function useGravity(initialY: number) {
  const y = useSharedValue(initialY);
  const velocity = useSharedValue(0);
  const GRAVITY = 9.8;
  const GROUND = 500;
  const BOUNCE_FACTOR = 0.6;

  useFrameCallback((frameInfo) => {
    const dt = (frameInfo.timeSincePreviousFrame ?? 16) / 1000;
    velocity.value += GRAVITY * dt * 60;
    y.value += velocity.value * dt;

    if (y.value >= GROUND) {
      y.value = GROUND;
      velocity.value = -velocity.value * BOUNCE_FACTOR;
      if (Math.abs(velocity.value) < 1) {
        velocity.value = 0;
      }
    }
  });

  return y;
}

1.2 Decay with Boundaries

// Momentum scrolling with snap points
const pan = Gesture.Pan()
  .onEnd((event) => {
    const targetX = translateX.value + event.velocityX * 0.2;
    const snapPoints = [0, -CARD_WIDTH, -CARD_WIDTH * 2];
    const closest = snapPoints.reduce((prev, curr) =>
      Math.abs(curr - targetX) < Math.abs(prev - targetX) ? curr : prev
    );
    translateX.value = withSpring(closest, SPRING_CONFIGS.snappy);
  });

2. Bottom Sheet

function BottomSheet({ snapPoints }: { snapPoints: number[] }) {
  const translateY = useSharedValue(snapPoints[0]);
  const context = useSharedValue(0);

  const pan = Gesture.Pan()
    .onStart(() => {
      context.value = translateY.value;
    })
    .onUpdate((event) => {
      const newY = context.value + event.translationY;
      const maxY = snapPoints[snapPoints.length - 1];
      const minY = snapPoints[0];
      translateY.value = clamp(newY, minY, maxY);
    })
    .onEnd((event) => {
      const projected = translateY.value + event.velocityY * 0.1;
      const closest = snapPoints.reduce((prev, curr) =>
        Math.abs(curr - projected) < Math.abs(prev - projected) ? curr : prev
      );
      translateY.value = withSpring(closest, {
        damping: 20,
        stiffness: 150,
        velocity: event.velocityY,
      });
    });

  const animatedStyle = useAnimatedStyle(() => ({
    transform: [{ translateY: translateY.value }],
  }));

  const backdropStyle = useAnimatedStyle(() => ({
    opacity: interpolate(
      translateY.value,
      [snapPoints[0], snapPoints[snapPoints.length - 1]],
      [0.5, 0]
    ),
  }));

  return (
    <>
      <Animated.View style={[styles.backdrop, backdropStyle]} />
      <GestureDetector gesture={pan}>
        <Animated.View style={[styles.sheet, animatedStyle]}>
          <View style={styles.handle} />
          {/* Sheet content */}
        </Animated.View>
      </GestureDetector>
    </>
  );
}

3. Swipeable Cards

// Tinder-style swipeable card stack
function CardStack({ cards, onSwipe }) {
  return (
    <View style={styles.stack}>
      {cards.slice(0, 3).map((card, index) => (
        <SwipeCard
          key={card.id}
          card={card}
          index={index}
          onSwipe={onSwipe}
        />
      )).reverse()}
    </View>
  );
}

function SwipeCard({ card, index, onSwipe }) {
  const translateX = useSharedValue(0);
  const translateY = useSharedValue(0);
  const THRESHOLD = SCREEN_WIDTH * 0.3;

  const pan = Gesture.Pan()
    .enabled(index === 0)
    .onUpdate((e) => {
      translateX.value = e.translationX;
      translateY.value = e.translationY;
    })
    .onEnd((e) => {
      if (Math.abs(translateX.value) > THRESHOLD) {
        const direction = translateX.value > 0 ? 'right' : 'left';
        translateX.value = withTiming(
          Math.sign(translateX.value) * SCREEN_WIDTH * 1.5,
          { duration: 300 },
          () => runOnJS(onSwipe)(card.id, direction)
        );
      } else {
        translateX.value = withSpring(0);
        translateY.value = withSpring(0);
      }
    });

  const animatedStyle = useAnimatedStyle(() => {
    const rotate = interpolate(
      translateX.value,
      [-SCREEN_WIDTH, 0, SCREEN_WIDTH],
      [-15, 0, 15]
    );
    const stackScale = interpolate(index, [0, 1, 2], [1, 0.95, 0.9]);

    return {
      transform: [
        { translateX: translateX.value },
        { translateY: translateY.value + index * 8 },
        { rotate: `${rotate}deg` },
        { scale: stackScale },
      ],
      zIndex: 3 - index,
    };
  });

  return (
    <GestureDetector gesture={pan}>
      <Animated.View style={[styles.card, animatedStyle]}>
        <Image source={{ uri: card.imageUrl }} style={styles.cardImage} />
      </Animated.View>
    </GestureDetector>
  );
}

4. Pull-to-Refresh

// Custom pull-to-refresh with Reanimated
function CustomPullToRefresh({ onRefresh, children }) {
  const translateY = useSharedValue(0);
  const isRefreshing = useSharedValue(false);
  const TRIGGER_THRESHOLD = 80;

  const pan = Gesture.Pan()
    .onUpdate((e) => {
      if (e.translationY > 0 && !isRefreshing.value) {
        // Rubber-band effect: diminishing returns past threshold
        translateY.value = e.translationY * 0.5;
      }
    })
    .onEnd(() => {
      if (translateY.value > TRIGGER_THRESHOLD) {
        isRefreshing.value = true;
        translateY.value = withSpring(TRIGGER_THRESHOLD);
        runOnJS(onRefresh)();
      } else {
        translateY.value = withSpring(0);
      }
    });

  const spinnerStyle = useAnimatedStyle(() => ({
    transform: [
      { translateY: translateY.value - 40 },
      { rotate: `${interpolate(translateY.value, [0, TRIGGER_THRESHOLD], [0, 360])}deg` },
    ],
    opacity: interpolate(translateY.value, [0, TRIGGER_THRESHOLD * 0.5], [0, 1]),
  }));

  return (
    <GestureDetector gesture={pan}>
      <Animated.View>
        <Animated.View style={spinnerStyle}>
          <ActivityIndicator />
        </Animated.View>
        {children}
      </Animated.View>
    </GestureDetector>
  );
}

5. Skeleton Loading

function SkeletonLoader({ width, height }: { width: number; height: number }) {
  const shimmer = useSharedValue(0);

  useEffect(() => {
    shimmer.value = withRepeat(
      withTiming(1, { duration: 1200, easing: Easing.linear }),
      -1,
      false
    );
  }, []);

  const animatedStyle = useAnimatedStyle(() => ({
    opacity: interpolate(
      shimmer.value,
      [0, 0.5, 1],
      [0.3, 0.7, 0.3]
    ),
  }));

  return (
    <Animated.View
      style={[
        { width, height, backgroundColor: '#E0E0E0', borderRadius: 4 },
        animatedStyle,
      ]}
    />
  );
}

6. Lottie Integration

import LottieView from 'lottie-react-native';

// Basic Lottie animation
<LottieView
  source={require('./animations/loading.json')}
  autoPlay
  loop
  style={{ width: 200, height: 200 }}
/>

// Controlled Lottie with Reanimated
function ControlledLottie() {
  const progress = useSharedValue(0);
  const animatedProps = useAnimatedProps(() => ({
    progress: progress.value,
  }));

  const pan = Gesture.Pan()
    .onUpdate((e) => {
      progress.value = clamp(e.translationX / 300, 0, 1);
    });

  return (
    <GestureDetector gesture={pan}>
      <AnimatedLottieView
        source={require('./animations/progress.json')}
        animatedProps={animatedProps}
        style={{ width: 300, height: 300 }}
      />
    </GestureDetector>
  );
}
Key Insight

Lottie renders After Effects animations exported as JSON. It's ideal for complex illustrations, loading states, and micro-interactions that would be impractical to code manually. However, Lottie animations are rendered on the main thread by default — complex animations can impact performance. For performance-critical paths, consider using Skia-based rendering or simpler Reanimated animations instead.

7. SVG & Canvas Animations

import Animated, { useAnimatedProps } from 'react-native-reanimated';
import Svg, { Path, Circle } from 'react-native-svg';

const AnimatedPath = Animated.createAnimatedComponent(Path);
const AnimatedCircle = Animated.createAnimatedComponent(Circle);

// Animated progress ring
function ProgressRing({ progress }: { progress: SharedValue<number> }) {
  const RADIUS = 40;
  const CIRCUMFERENCE = 2 * Math.PI * RADIUS;

  const animatedProps = useAnimatedProps(() => ({
    strokeDashoffset: CIRCUMFERENCE * (1 - progress.value),
  }));

  return (
    <Svg width={100} height={100}>
      <Circle cx={50} cy={50} r={RADIUS} stroke="#E0E0E0" strokeWidth={4} fill="none" />
      <AnimatedCircle
        cx={50} cy={50} r={RADIUS}
        stroke="#4A90D9" strokeWidth={4} fill="none"
        strokeDasharray={CIRCUMFERENCE}
        animatedProps={animatedProps}
        strokeLinecap="round"
        rotation={-90} origin={"50, 50"}
      />
    </Svg>
  );
}

// Measure animation frame drops
import { useFrameCallback } from 'react-native-reanimated';

function useFrameDropMonitor() {
  const frameCount = useSharedValue(0);
  const droppedFrames = useSharedValue(0);

  useFrameCallback((frameInfo) => {
    frameCount.value++;
    const elapsed = frameInfo.timeSincePreviousFrame ?? 16;
    if (elapsed > 20) { // More than 20ms = likely dropped frame
      droppedFrames.value++;
    }
  });

  return { frameCount, droppedFrames };
}

9. Architect Considerations

Staff-Level Mental Model

Advanced animations in React Native follow a layered architecture: (1) Reanimated + Gesture Handler for gesture-driven interactions (bottom sheets, swipeable cards, pull-to-refresh) — these run entirely on the UI thread; (2) Lottie for complex illustrative animations designed by motion designers — these are fire-and-forget; (3) SVG + Reanimated for data-driven animations (progress rings, charts) — these combine declarative SVG with imperative animation control; (4) Skia for custom rendering that exceeds what the view system can do (shaders, blur, custom drawing). A Staff engineer selects the right layer based on the animation's complexity, interactivity requirements, and performance budget — not by defaulting to the most powerful tool for every case.
