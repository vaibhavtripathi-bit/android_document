Jest Testing for React Native

Jest is the default test runner for React Native, but configuring it correctly for the RN environment — with native module mocks, timer handling, async patterns, and component testing via @testing-library/react-native — requires deep understanding of both Jest internals and RN's module system.

// Mock Reanimated

  require('react-native-reanimated/mock')

// Mock gesture handler
jest.mock('react-native-gesture-handler', () => ({
  GestureHandlerRootView: ({ children }) => children,
  Swipeable: 'Swipeable',
  DrawerLayout: 'DrawerLayout',
  State: {},
  PanGestureHandler: 'PanGestureHandler',
  TapGestureHandler: 'TapGestureHandler',
}));

// Mock safe area
jest.mock('react-native-safe-area-context', () => ({
  SafeAreaProvider: ({ children }) => children,
  SafeAreaView: ({ children }) => children,
  useSafeAreaInsets: () => ({ top: 0, bottom: 0, left: 0, right: 0 }),
}));
Warning

transformIgnorePatterns is critical — React Native packages ship untranspiled ES modules that Jest can't parse. You must exclude RN packages from the ignore pattern so Jest transforms them. Missing a package here causes SyntaxError: Unexpected token 'export'. Maintain this list as you add dependencies.

2. Module Mocking

// Manual mock: __mocks__/react-native-camera.js
export const RNCamera = 'RNCamera';
export default { RNCamera };

// Inline mock with factory
jest.mock('../services/api', () => ({
  fetchUser: jest.fn(),
  updateUser: jest.fn(),

// Partial mock — keep real implementation, override specific exports
jest.mock('../utils/analytics', () => ({
  ...jest.requireActual('../utils/analytics'),
  trackEvent: jest.fn(),  // only mock tracking
}));

// Spy on module method
import * as api from '../services/api';
jest.spyOn(api, 'fetchUser').mockResolvedValue(mockUser);

// Mock with different implementations per test
const mockFetch = jest.fn();
jest.mock('../services/api', () => ({ fetchUser: mockFetch }));

it('handles success', async () => {
  mockFetch.mockResolvedValueOnce(mockUser);
  // ... test
});

it('handles error', async () => {
  mockFetch.mockRejectedValueOnce(new Error('Network error'));
  // ... test
});

3. Timer Mocking

// Modern fake timers (recommended)
beforeEach(() => {
  jest.useFakeTimers();
});

afterEach(() => {
  jest.useRealTimers();
});

it('debounces search input', () => {
  const onSearch = jest.fn();
  const { getByPlaceholderText } = render(
    <SearchInput onSearch={onSearch} debounceMs={300} />
  );

  fireEvent.changeText(getByPlaceholderText('Search'), 'hel');
  fireEvent.changeText(getByPlaceholderText('Search'), 'hello');

  expect(onSearch).not.toHaveBeenCalled();

  jest.advanceTimersByTime(300);

  expect(onSearch).toHaveBeenCalledTimes(1);
  expect(onSearch).toHaveBeenCalledWith('hello');
});

// Testing setInterval
it('polls for updates every 5 seconds', () => {
  const onPoll = jest.fn();
  render(<Poller onPoll={onPoll} intervalMs={5000} />);

  jest.advanceTimersByTime(15000);

  expect(onPoll).toHaveBeenCalledTimes(3);
});

4. Async Testing

// waitFor — polls until assertion passes
it('loads and displays user data', async () => {
  mockFetchUser.mockResolvedValue({ name: 'John' });

  render(<UserProfile userId="123" />);

  // Loading state
  expect(screen.getByText('Loading...')).toBeTruthy();

  // Wait for data
  await waitFor(() => {
    expect(screen.getByText('John')).toBeTruthy();
  });

  // Loading gone
  expect(screen.queryByText('Loading...')).toBeNull();
});

// findBy — combines getBy + waitFor
it('shows error message', async () => {
  mockFetchUser.mockRejectedValue(new Error('Not found'));
  render(<UserProfile userId="999" />);

  const error = await screen.findByText('Not found');
  expect(error).toBeTruthy();
});

// Testing with MSW (Mock Service Worker)
import { setupServer } from 'msw/node';
import { http, HttpResponse } from 'msw';

const server = setupServer(
  http.get('/api/user/:id', ({ params }) => {
    return HttpResponse.json({ id: params.id, name: 'John' });
  })
);

beforeAll(() => server.listen());
afterEach(() => server.resetHandlers());
afterAll(() => server.close());

5. @testing-library/react-native

import { render, screen, fireEvent, within }
  from '@testing-library/react-native';

// Query priority (prefer accessible queries)
screen.getByRole('button', { name: 'Submit' });  // best
screen.getByLabelText('Email input');             // good
screen.getByPlaceholderText('Enter email');        // acceptable
screen.getByText('Welcome');                      // acceptable
screen.getByTestId('submit-btn');                 // last resort

// Query variants
screen.getByText('X');      // throws if not found
screen.queryByText('X');    // returns null if not found
screen.findByText('X');     // async, waits for element
screen.getAllByText('X');   // returns array, throws if empty

// Scoped queries with within()
const header = screen.getByTestId('header');
const title = within(header).getByText('Settings');

// Custom matchers
expect(button).toBeDisabled();
expect(input).toHaveProp('value', 'hello');
expect(view).toHaveStyle({ backgroundColor: 'red' });
expect(element).toBeVisible();
expect(element).toHaveAccessibilityState({ checked: true });

6. Component Testing Patterns

// Testing form submission
it('submits form with valid data', async () => {
  const onSubmit = jest.fn();
  render(<ContactForm onSubmit={onSubmit} />);

  fireEvent.changeText(screen.getByLabelText('Name'), 'John');
  fireEvent.changeText(screen.getByLabelText('Email'), 'john@test.com');
  fireEvent.press(screen.getByRole('button', { name: 'Submit' }));

  await waitFor(() => {
    expect(onSubmit).toHaveBeenCalledWith({
      name: 'John',
      email: 'john@test.com',
    });
  });
});

// Testing FlatList rendering
it('renders list items', () => {
  const items = [
    { id: '1', title: 'Item 1' },
    { id: '2', title: 'Item 2' },
  ];
  render(<ItemList items={items} />);

  expect(screen.getByText('Item 1')).toBeTruthy();
  expect(screen.getByText('Item 2')).toBeTruthy();
});

// Testing conditional rendering
it('shows empty state when no items', () => {
  render(<ItemList items={[]} />);
  expect(screen.getByText('No items found')).toBeTruthy();
  expect(screen.queryByTestId('item-list')).toBeNull();
});

const Stack = createNativeStackNavigator();

const renderWithNavigation = (component, { initialParams } = {}) => {
  const Screen = () => component;
  return render(
    <NavigationContainer>

        <Stack.Screen
          name="Test"
          component={Screen}
          initialParams={initialParams}
        />
      </Stack.Navigator>
    </NavigationContainer>
  );
};

it('navigates to detail on item press', async () => {
  const navigate = jest.fn();
  jest.spyOn(require('@react-navigation/native'), 'useNavigation')
    .mockReturnValue({ navigate });

  render(<ItemList />);
  fireEvent.press(screen.getByText('Item 1'));

});

10. Architect Considerations

Architect Insight

Jest testing best practices for Staff engineers: (1) Prefer @testing-library/react-native queries by role/label over testID — this tests accessibility for free. (2) Use MSW over manual mocks for API testing — it tests the actual fetch/axios code path. (3) Create test utilities: renderWithProviders() that wraps components in all necessary providers (navigation, theme, store). (4) Set per-module coverage thresholds — higher for business logic, lower for UI. (5) Run tests in CI with --ci --forceExit --detectOpenHandles to catch hanging tests.
