// Optimized fetch configuration
async function optimizedFetch(url: string, options = {}) {
  const controller = new AbortController();
  const timeoutId = setTimeout(() => controller.abort(), 10000);

  try {
    const response = await fetch(url, {
      ...options,
      signal: controller.signal,
      headers: {
        'Accept-Encoding': 'gzip',
        'Accept': 'application/json',
        ...options.headers,
      },
    });

    if (!response.ok) {
      throw new Error(`HTTP ${response.status}`);
    }

    return response;
  } finally {
    clearTimeout(timeoutId);
  }
}

function OptimizedImage({ url, width, height }) {
  // Request exact size from CDN (avoid downloading 4K for a thumbnail)
  const optimizedUrl = `${url}?w=${width * 2}&h=${height * 2}&fit=crop&auto=format,compress`;

  return (
    <FastImage
      source={{
        uri: optimizedUrl,
        priority: FastImage.priority.normal,
        cache: FastImage.cacheControl.immutable,
      }}
      style={{ width, height }}
      resizeMode={FastImage.resizeMode.cover}
    />
  );
}

// Progressive image loading pattern
function ProgressiveImage({ thumbnailUrl, fullUrl, style }) {
  const [loaded, setLoaded] = useState(false);

  return (
    <View style={style}>
      {!loaded && (
        <FastImage
          source={{ uri: thumbnailUrl }}
          style={[StyleSheet.absoluteFill, { filter: 'blur(10)' }]}
          resizeMode="cover"
        />
      )}
      <FastImage
        source={{ uri: fullUrl }}
        style={StyleSheet.absoluteFill}
        resizeMode="cover"
        onLoad={() => setLoaded(true)}
      />
    </View>
  );
}

3. Request Batching

// Batch multiple API calls into a single request
class RequestBatcher {
  private queue: Array<{ endpoint: string; resolve: Function; reject: Function }> = [];
  private timer: NodeJS.Timeout | null = null;
  private batchDelay = 50; // ms to wait for more requests

  async request(endpoint: string): Promise<any> {
    return new Promise((resolve, reject) => {
      this.queue.push({ endpoint, resolve, reject });

      if (!this.timer) {
        this.timer = setTimeout(() => this.flush(), this.batchDelay);
      }
    });
  }

  private async flush() {
    const batch = [...this.queue];
    this.queue = [];
    this.timer = null;

    try {
      const response = await fetch('/api/batch', {
        method: 'POST',
        body: JSON.stringify({
          requests: batch.map(b => ({ endpoint: b.endpoint })),
        }),
      });

      const results = await response.json();
      batch.forEach((item, i) => item.resolve(results[i]));
    } catch (error) {
      batch.forEach(item => item.reject(error));
    }
  }
}

4. Caching Strategies

Caching Layers in React Native ═══════════════════════════════ Layer 1: In-Memory Cache (fastest, volatile) ┌──────────────────────────────────────────┐ │ React Query / SWR / Apollo cache │ │ TTL: seconds to minutes │ │ Size: limited by JS heap │ └──────────────────────────────────────────┘ │ miss ▼ Layer 2: Persistent Cache (fast, survives restart) ┌──────────────────────────────────────────┐ │ MMKV / AsyncStorage / SQLite │ │ TTL: hours to days │ │ Size: limited by disk │ └──────────────────────────────────────────┘ │ miss ▼ Layer 3: HTTP Cache (native, automatic) ┌──────────────────────────────────────────┐ │ NSURLCache (iOS) / OkHttp Cache (Android) │ │ TTL: based on Cache-Control headers │ │ Size: configurable (default 10-50MB) │ └──────────────────────────────────────────┘ │ miss ▼ Layer 4: Network Request ┌──────────────────────────────────────────┐ │ Actual HTTP request to server │ └──────────────────────────────────────────┘ 
// MMKV — fastest persistent storage for RN
import { MMKV } from 'react-native-mmkv';

const storage = new MMKV();

class PersistentCache {
  get<T>(key: string): T | null {
    const raw = storage.getString(key);
    if (!raw) return null;

    const { data, expiry } = JSON.parse(raw);
    if (Date.now() > expiry) {
      storage.delete(key);
      return null;
    }
    return data;
  }

  set<T>(key: string, data: T, ttlMs: number) {
    storage.set(key, JSON.stringify({
      data,
      expiry: Date.now() + ttlMs,
    }));
  }
}

5. Offline Support

// Offline-first architecture
function useOfflineQuery(key, fetcher) {
  const [data, setData] = useState(() => cache.get(key));
  const [isOnline] = useNetInfo();

  useEffect(() => {
    if (!isOnline) return;

    fetcher()
      .then((freshData) => {
        setData(freshData);
        cache.set(key, freshData, 60 * 60 * 1000);
      })
      .catch(() => {
        // Network failed — stale cache is better than nothing
      });
  }, [key, isOnline]);

  return { data, isStale: !isOnline };
}

// Offline mutation queue
class OfflineMutationQueue {
  private queue: Array<{ mutation: Function; params: any }> = [];

  enqueue(mutation: Function, params: any) {
    this.queue.push({ mutation, params });
    storage.set('mutation_queue', JSON.stringify(this.queue));
  }

  async flush() {
    while (this.queue.length > 0) {
      const { mutation, params } = this.queue[0];
      try {
        await mutation(params);
        this.queue.shift();
        storage.set('mutation_queue', JSON.stringify(this.queue));
      } catch {
        break; // retry later
      }
    }
  }
}

7. Network State Monitoring

// @react-native-community/netinfo
import NetInfo from '@react-native-community/netinfo';

function useNetworkAwareRequests() {
  const [netState, setNetState] = useState(null);

  useEffect(() => {
    const unsubscribe = NetInfo.addEventListener(state => {
      setNetState(state);
    });
    return unsubscribe;
  }, []);

  const shouldReduceData = useMemo(() => {
    if (!netState) return false;
    return (
      netState.type === 'cellular' &&
      (netState.details?.cellularGeneration === '2g' ||
       netState.details?.cellularGeneration === '3g')
    );
  }, [netState]);

  return {
    isConnected: netState?.isConnected ?? true,
    isWifi: netState?.type === 'wifi',
    shouldReduceData,
    connectionType: netState?.type,
  };
}

9. Request Prioritization

// Request priority queue
class PriorityRequestQueue {
  private queues = {
    critical: [],  // auth, payment
    high: [],      // visible content
    normal: [],    // prefetch
    low: [],       // analytics, logging
  };

  private concurrency = { critical: 4, high: 3, normal: 2, low: 1 };
  private active = { critical: 0, high: 0, normal: 0, low: 0 };

  async enqueue(priority, requestFn) {
    return new Promise((resolve, reject) => {
      this.queues[priority].push({ requestFn, resolve, reject });
      this.processQueue(priority);
    });
  }

  private async processQueue(priority) {
    while (
      this.queues[priority].length > 0 &&
      this.active[priority] < this.concurrency[priority]
    ) {
      const { requestFn, resolve, reject } = this.queues[priority].shift();
      this.active[priority]++;

      requestFn()
        .then(resolve)
        .catch(reject)
        .finally(() => {
          this.active[priority]--;
          this.processQueue(priority);
        });
    }
  }
}
Staff-Level Mental Model

Network optimization on mobile is about resilience, not just speed. Users switch between WiFi and cellular, enter tunnels, and use the app in areas with poor coverage. The Staff-level approach: (1) cache aggressively at multiple layers (memory → disk → HTTP), (2) design for offline-first (the app should work without network), (3) adapt to connection quality (reduce image quality on slow connections), (4) prioritize requests (auth and visible content before analytics), (5) batch requests to reduce connection overhead, and (6) use optimistic updates for instant perceived performance.
