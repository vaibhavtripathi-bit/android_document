1. Permission Model

Permission Flow (iOS & Android) ════════════════════════════════ ┌─────────────┐ check() ┌──────────────────┐ │ App Needs │────────────────►│ Permission Status │ │ Capability │ │ │ └─────────────┘ │ UNAVAILABLE ──► Feature disabled │ │ DENIED ──────► Show rationale │ │ BLOCKED ─────► Open Settings │ │ GRANTED ─────► Use feature │ │ LIMITED ─────► Partial access │ │ NOT_DETERMINED ──┐ │ └──────────────────┘ │ │ │ request() │ │ │ ▼ │ System Dialog │ (one-time on iOS) │ │ │ ┌──────┴──────┐ │ │ │ │ GRANTED DENIED │ │ │ │ ▼ ▼ │ Use feature Show rationale│ 
// react-native-permissions — unified permission API
import {
  check, request, PERMISSIONS, RESULTS, openSettings
} from 'react-native-permissions';
import { Platform, Alert } from 'react-native';

async function requestCameraPermission(): Promise<boolean> {
  const permission = Platform.select({
    ios: PERMISSIONS.IOS.CAMERA,
    android: PERMISSIONS.ANDROID.CAMERA,
  });

  if (!permission) return false;

  const status = await check(permission);

  switch (status) {
    case RESULTS.GRANTED:
    case RESULTS.LIMITED:
      return true;

    case RESULTS.DENIED: {
      const result = await request(permission);
      return result === RESULTS.GRANTED;
    }

    case RESULTS.BLOCKED:
      Alert.alert(
        'Camera Permission Required',
        'Please enable camera access in Settings.',
        [
          { text: 'Cancel', style: 'cancel' },
          { text: 'Open Settings', onPress: openSettings },
        ]
      );
      return false;

    case RESULTS.UNAVAILABLE:
      return false;
  }
}
iOS Permission Gotcha

On iOS, request() only shows the system dialog once. After the user denies, subsequent request() calls return BLOCKED without showing any dialog. You must direct users to Settings. On Android 11+, if the user denies twice, the permission is permanently blocked. Always check before requesting, and show a rationale UI before the system dialog.

2. Camera API

// react-native-camera / react-native-vision-camera
import { Camera, useCameraDevice, useCodeScanner }
  from 'react-native-vision-camera';

function CameraScreen() {
  const device = useCameraDevice('back');
  const { hasPermission, requestPermission } = useCameraPermission();

  const codeScanner = useCodeScanner({
    codeTypes: ['qr', 'ean-13'],
    onCodeScanned: (codes) => {
      console.log(`Scanned ${codes.length} codes`);
    },
  });

  if (!hasPermission) {
    return <PermissionRequest onRequest={requestPermission} />;
  }
  if (!device) {
    return <Text>No camera available</Text>;
  }

  return (
    <Camera
      device={device}
      isActive={true}
      photo={true}
      video={true}
      codeScanner={codeScanner}
      style={StyleSheet.absoluteFill}
      frameProcessor={frameProcessor}  // Worklet-based processing
    />
  );
}

// Frame processor — runs on native camera thread via Worklets
const frameProcessor = useFrameProcessor((frame) => {
  'worklet';
  const objects = detectObjects(frame);
  console.log(`Detected ${objects.length} objects`);
}, []);

3. Geolocation

// react-native-geolocation-service — high-accuracy location
import Geolocation from 'react-native-geolocation-service';

async function getCurrentLocation(): Promise<Position> {
  const hasPermission = await requestLocationPermission();
  if (!hasPermission) throw new Error('Location permission denied');

  return new Promise((resolve, reject) => {
    Geolocation.getCurrentPosition(
      (position) => resolve(position),
      (error) => reject(error),
      {
        enableHighAccuracy: true,
        timeout: 15000,
        maximumAge: 10000,
        forceRequestLocation: true,
        showLocationDialog: true,  // Android: show enable GPS dialog
      }
    );
  });
}

// Continuous tracking with cleanup
function useLocationTracking() {
  const [location, setLocation] = useState(null);

  useEffect(() => {
    const watchId = Geolocation.watchPosition(
      (pos) => setLocation(pos.coords),
      (err) => console.error(err),
      {
        enableHighAccuracy: true,
        distanceFilter: 10,       // meters between updates
        interval: 5000,           // Android: ms between updates
        fastestInterval: 2000,    // Android: fastest update rate
        forceLocationManager: false, // use FusedLocationProvider
      }
    );

    return () => Geolocation.clearWatch(watchId);
  }, []);

  return location;
}

4. Biometrics

// react-native-biometrics — Face ID / Touch ID / Fingerprint
import ReactNativeBiometrics, { BiometryTypes }
  from 'react-native-biometrics';

const biometrics = new ReactNativeBiometrics({
  allowDeviceCredentials: true,  // fallback to PIN/pattern
});

async function authenticateUser(): Promise<boolean> {
  const { available, biometryType } =
    await biometrics.isSensorAvailable();

  if (!available) {
    return fallbackToPasswordAuth();
  }

  const promptMessage = biometryType === BiometryTypes.FaceID
    ? 'Confirm your identity with Face ID'
    : 'Confirm your identity with fingerprint';

  const { success } = await biometrics.simplePrompt({
    promptMessage,
    cancelButtonText: 'Use Password',
  });

  return success;
}

// Cryptographic biometric authentication (more secure)
async function signWithBiometrics(payload: string) {
  const { keysExist } = await biometrics.biometricKeysExist();
  if (!keysExist) {
    await biometrics.createKeys();
  }

  const { signature } = await biometrics.createSignature({
    promptMessage: 'Sign transaction',
    payload,
  });

  return signature; // RSA signature, verify on server
}

5. Push Notifications

// @react-native-firebase/messaging — FCM integration
import messaging from '@react-native-firebase/messaging';
import notifee from '@notifee/react-native';

async function setupNotifications() {
  const authStatus = await messaging().requestPermission();
  const enabled =
    authStatus === messaging.AuthorizationStatus.AUTHORIZED ||
    authStatus === messaging.AuthorizationStatus.PROVISIONAL;

  if (!enabled) return;

  const token = await messaging().getToken();
  await sendTokenToServer(token);

  messaging().onTokenRefresh(async (newToken) => {
    await sendTokenToServer(newToken);
  });
}

// Foreground notification display via Notifee
messaging().onMessage(async (remoteMessage) => {
  await notifee.displayNotification({
    title: remoteMessage.notification?.title,
    body: remoteMessage.notification?.body,
    android: {
      channelId: 'default',
      pressAction: { id: 'default' },
      importance: AndroidImportance.HIGH,
    },
    ios: {
      foregroundPresentationOptions: {
        badge: true, sound: true, banner: true,
      },
    },
  });
});

// Background handler — registered at top level (index.js)
messaging().setBackgroundMessageHandler(async (remoteMessage) => {
  // Process silently — no UI access
  await syncDataFromNotification(remoteMessage.data);
});

// react-native-fs — file system access
import RNFS from 'react-native-fs';

const paths = {
  documents: RNFS.DocumentDirectoryPath,
  cache: RNFS.CachesDirectoryPath,
  temp: RNFS.TemporaryDirectoryPath,
  external: RNFS.ExternalDirectoryPath,  // Android only
};

async function fileOperations() {
  const filePath = `${paths.documents}/data.json`;

  await RNFS.writeFile(filePath, JSON.stringify(data), 'utf8');
  const content = await RNFS.readFile(filePath, 'utf8');
  const exists = await RNFS.exists(filePath);
  const stat = await RNFS.stat(filePath);

  // Download with progress
  const download = RNFS.downloadFile({
    fromUrl: 'https://example.com/large-file.zip',
    toFile: `${paths.cache}/download.zip`,
    progress: (res) => {
      const pct = (res.bytesWritten / res.contentLength) * 100;
      updateProgress(pct);
    },
    progressDivider: 10, // report every 10%
  });

  const result = await download.promise;
}

7. Contacts & Calendar

// react-native-contacts
import Contacts from 'react-native-contacts';

async function loadContacts() {
  const permission = await requestContactsPermission();
  if (!permission) return [];

  const contacts = await Contacts.getAll();
  return contacts
    .filter(c => c.phoneNumbers.length > 0)
    .map(c => ({
      id: c.recordID,
      name: `${c.givenName} ${c.familyName}`.trim(),
      phone: c.phoneNumbers[0]?.number,
      thumbnail: c.thumbnailPath,
    }))
    .sort((a, b) => a.name.localeCompare(b.name));
}

8. Platform API Abstraction Patterns

// Abstraction layer for platform capabilities
interface PlatformCapabilities {
  hasBiometrics: boolean;
  biometryType: 'face' | 'fingerprint' | 'iris' | 'none';
  hasCamera: boolean;
  hasNFC: boolean;
  hasARKit: boolean;
  platformVersion: number;
}

async function detectCapabilities(): Promise<PlatformCapabilities> {
  const [biometrics, camera] = await Promise.allSettled([
    ReactNativeBiometrics.isSensorAvailable(),
    check(Platform.select({
      ios: PERMISSIONS.IOS.CAMERA,
      android: PERMISSIONS.ANDROID.CAMERA,
    })),
  ]);

  return {
    hasBiometrics: biometrics.status === 'fulfilled'
      && biometrics.value.available,
    biometryType: mapBiometryType(biometrics),
    hasCamera: camera.status === 'fulfilled'
      && camera.value !== RESULTS.UNAVAILABLE,
    hasNFC: Platform.OS === 'android' && Platform.Version >= 19,
    hasARKit: Platform.OS === 'ios' && parseInt(Platform.Version) >= 11,
    platformVersion: Platform.Version,
  };
}

// Feature flag + capability gating
function useFeatureGate(feature: string) {
  const capabilities = useCapabilities();
  const featureFlags = useFeatureFlags();

  return useMemo(() => ({
    available: capabilities[feature] && featureFlags[feature],
    reason: !capabilities[feature]
      ? 'device_unsupported'
      : !featureFlags[feature]
        ? 'feature_disabled'
        : null,
  }), [capabilities, featureFlags, feature]);
}

11. Graceful Degradation

// Graceful degradation pattern
function AuthenticationScreen() {
  const { available, biometryType } = useBiometrics();
  const [authMethod, setAuthMethod] = useState('loading');

  useEffect(() => {
    if (available && biometryType === 'face') {
      setAuthMethod('faceId');
    } else if (available) {
      setAuthMethod('fingerprint');
    } else if (DeviceInfo.hasNotch()) {
      setAuthMethod('pin');
    } else {
      setAuthMethod('password');
    }
  }, [available, biometryType]);

  switch (authMethod) {
    case 'faceId':
      return <FaceIdAuth fallback={() => setAuthMethod('pin')} />;
    case 'fingerprint':
      return <FingerprintAuth fallback={() => setAuthMethod('pin')} />;
    case 'pin':
      return <PinAuth />;
    case 'password':
      return <PasswordAuth />;
    default:
      return <LoadingSpinner />;
  }
}
Staff-Level Mental Model

Platform API access at Staff level means designing for the matrix of: (platform) × (OS version) × (device capability) × (permission state) × (feature flag). A robust architecture pre-computes a capability map at startup, gates features declaratively, provides graceful fallbacks at every level, and never crashes on a missing capability. The permission flow should be a first-class UX concern, not an afterthought — users who deny permissions should still have a functional (if reduced) app experience.
