Native Integration (Brownfield)

Integrating React Native into existing native apps — "brownfield" — is the most common enterprise adoption pattern. This doc covers embedding RN surfaces in native navigation stacks, sharing state, managing multiple bundles, and the architectural decisions that determine success or failure at scale.

1. Brownfield Architecture

Brownfield Integration Architecture ════════════════════════════════════ Existing Native App ┌─────────────────────────────────────────────────────┐ │ │ │ Native Navigation Stack │ │ ┌──────────┐ ┌──────────┐ ┌──────────────────┐ │ │ │ Native │ │ Native │ │ RN Surface │ │ │ │ Screen A │→│ Screen B │→│ (RCTRootView) │ │ │ │ (UIKit/ │ │ │ │ │ │ │ │ Compose)│ │ │ │ ┌──────────────┐│ │ │ └──────────┘ └──────────┘ │ │ React Tree ││ │ │ │ │ (JS Bundle) ││ │ │ ┌──────────────────────┐ │ └──────────────┘│ │ │ │ Shared Services │ │ │ │ │ │ • Auth │◄───│ Native Modules │ │ │ │ • Analytics │ │ (bridge to app) │ │ │ │ • Networking │ └──────────────────┘ │ │ │ • Storage │ │ │ └──────────────────────┘ │ └─────────────────────────────────────────────────────┘
Architect Insight

Brownfield is harder than greenfield because you must solve: (1) bridge lifecycle management within an existing app lifecycle, (2) navigation interop between native and RN stacks, (3) shared state synchronization, (4) bundle loading without blocking the native UI, and (5) incremental migration without breaking existing features. The key architectural decision is whether to use a single shared bridge instance or multiple isolated instances.

2. iOS Embedding (RCTRootView)

// AppDelegate.swift — initialize bridge once
class AppDelegate: UIResponder, UIApplicationDelegate {
  var bridge: RCTBridge?

  func application(
    _ application: UIApplication,
    didFinishLaunchingWithOptions launchOptions: [UIApplication.LaunchOptionsKey: Any]?
  ) -> Bool {
    // Initialize bridge early — can be done lazily too
    bridge = RCTBridge(
      delegate: self,
      launchOptions: launchOptions
    )
    return true
  }
}

extension AppDelegate: RCTBridgeDelegate {
  func sourceURL(for bridge: RCTBridge!) -> URL! {
    #if DEBUG
    return RCTBundleURLProvider.sharedSettings()
      .jsBundleURL(forBundleRoot: "index")
    #else
    return Bundle.main.url(
      forResource: "main", withExtension: "jsbundle"
    )
    #endif
  }
}

// Embedding RN in a UIViewController
class ReactNativeViewController: UIViewController {
  private var reactView: RCTRootView?

  override func viewDidLoad() {
    super.viewDidLoad()

    guard let bridge = (UIApplication.shared.delegate as?
      AppDelegate)?.bridge else { return }

    reactView = RCTRootView(
      bridge: bridge,
      moduleName: "SettingsScreen",
      initialProperties: [
        "userId": currentUser.id,
        "theme": "dark",
      ]
    )

    reactView?.frame = view.bounds
    reactView?.autoresizingMask = [.flexibleWidth, .flexibleHeight]
    view.addSubview(reactView!)
  }

  override func viewWillDisappear(_ animated: Bool) {
    super.viewWillDisappear(animated)
    // Don't invalidate bridge — it's shared
  }
}

3. Android Embedding (ReactActivity)

// Application class — initialize ReactNativeHost
class MyApplication : Application(), ReactApplication {

  private val reactNativeHost = object : ReactNativeHost(this) {
    override fun getUseDeveloperSupport() = BuildConfig.DEBUG

    override fun getPackages(): List<ReactPackage> {
      val packages = PackageList(this@MyApplication).packages
      packages.add(SharedServicesPackage())
      return packages
    }

    override fun getJSMainModuleName() = "index"
  }

  override fun getReactNativeHost() = reactNativeHost
}

// Embedding RN in a Fragment
class ReactNativeFragment : Fragment() {
  private var reactRootView: ReactRootView? = null
  private var reactInstanceManager: ReactInstanceManager? = null

  override fun onCreateView(
    inflater: LayoutInflater, container: ViewGroup?,
    savedInstanceState: Bundle?
  ): View {
    reactRootView = ReactRootView(requireContext())
    reactInstanceManager = (requireActivity().application
      as ReactApplication).reactNativeHost.reactInstanceManager

    val initialProps = Bundle().apply {
      putString("userId", currentUser.id)
      putString("theme", "dark")
    }

    reactRootView?.startReactApplication(
      reactInstanceManager,
      "SettingsScreen",
      initialProps
    )

    return reactRootView!!
  }

  override fun onDestroyView() {
    super.onDestroyView()
    reactRootView?.unmountReactApplication()
    reactRootView = null
  }
}

5. Shared Native Modules

// Shared native module — exposes existing app services to RN
class SharedAuthModule(
  reactContext: ReactApplicationContext,
  private val authManager: AuthManager  // existing app service
) : ReactContextBaseJavaModule(reactContext) {

  override fun getName() = "SharedAuth"

  @ReactMethod
  fun getAccessToken(promise: Promise) {
    val token = authManager.currentToken
    if (token != null) {
      promise.resolve(token.accessToken)
    } else {
      promise.reject("E_NO_TOKEN", "Not authenticated")
    }
  }

  @ReactMethod
  fun logout(promise: Promise) {
    authManager.logout {
      promise.resolve(null)
    }
  }
}

// Package that injects existing services
class SharedServicesPackage(
  private val authManager: AuthManager,
  private val analyticsManager: AnalyticsManager
) : ReactPackage {

  override fun createNativeModules(
    reactContext: ReactApplicationContext
  ): List<NativeModule> = listOf(
    SharedAuthModule(reactContext, authManager),
    SharedAnalyticsModule(reactContext, analyticsManager),
  )

  override fun createViewManagers(
    reactContext: ReactApplicationContext
  ): List<ViewManager<*, *>> = emptyList()
}

6. Navigation Between Native and RN

// Native module for navigation interop
class NavigationBridge(
  reactContext: ReactApplicationContext
) : ReactContextBaseJavaModule(reactContext) {

  override fun getName() = "NavigationBridge"

  @ReactMethod
  fun pushNativeScreen(screenName: String, params: ReadableMap) {
    currentActivity?.let { activity ->
      when (screenName) {
        "NativeProfile" -> {
          val intent = Intent(activity, ProfileActivity::class.java)
          intent.putExtra("userId", params.getString("userId"))
          activity.startActivity(intent)
        }
        "NativeSettings" -> {
          val intent = Intent(activity, SettingsActivity::class.java)
          activity.startActivity(intent)
        }
      }
    }
  }

  @ReactMethod
  fun popToNative() {
    currentActivity?.finish()
  }

  @ReactMethod
  fun presentRNModal(moduleName: String, props: ReadableMap) {
    currentActivity?.let { activity ->
      val intent = Intent(activity, ReactModalActivity::class.java)
      intent.putExtra("moduleName", moduleName)
      intent.putExtra("props", props.toHashMap().toBundle())
      activity.startActivity(intent)
    }
  }
}

// JS side — using navigation bridge
import { NativeModules } from 'react-native';
const { NavigationBridge } = NativeModules;

function SettingsScreen() {
  return (
    <View>
      <Button
        title="Open Native Profile"
        onPress={() => NavigationBridge.pushNativeScreen(
          'NativeProfile', { userId: '123' }
        )}
      />
      <Button
        title="Go Back to Native"
        onPress={() => NavigationBridge.popToNative()}
      />
    </View>
  );
}

7. Bundle Loading Strategies

Strategy Startup Impact Memory Use Case
Eager (app launch) +200-500ms cold start Always loaded RN is primary UI
Lazy (on first RN screen) Delay on first RN screen Loaded on demand RN is secondary feature
Pre-warm (background) No visible delay Loaded after native init Best UX, higher memory
Split bundles Per-feature loading Minimal per feature Large apps, many RN features 
// Pre-warming strategy — iOS
class ReactNativePrewarmer {
  static let shared = ReactNativePrewarmer()
  private var bridge: RCTBridge?

  func prewarm() {
    DispatchQueue.global(qos: .utility).async {
      self.bridge = RCTBridge(delegate: self, launchOptions: nil)
    }
  }

  func createRootView(
    moduleName: String,
    props: [String: Any]?
  ) -> RCTRootView? {
    guard let bridge = bridge else { return nil }
    return RCTRootView(
      bridge: bridge,
      moduleName: moduleName,
      initialProperties: props
    )
  }
}

8. Native-to-JS Communication

// Sending data from native to JS

// Method 1: Initial properties (one-time, at mount)
let rootView = RCTRootView(
  bridge: bridge,
  moduleName: "MyScreen",
  initialProperties: ["data": serializedData]
)

// Method 2: appProperties update (re-renders React tree)
rootView.appProperties = ["data": updatedData]

// Method 3: Event emitter (push events anytime)
bridge.eventDispatcher().sendAppEvent(
  withName: "onNativeDataUpdate",
  body: ["key": "value"]
)

// Method 4: Direct JS function call
bridge.enqueueJSCall(
  "RCTDeviceEventEmitter",
  method: "emit",
  args: ["customEvent", ["payload": data]],
  completion: nil
)

// JS side — listening for native events
import { DeviceEventEmitter } from 'react-native';

useEffect(() => {
  const sub = DeviceEventEmitter.addListener(
    'onNativeDataUpdate',
    (data) => dispatch(updateFromNative(data))
  );
  return () => sub.remove();
}, []);

10. Migration Strategy

Phase Action Risk
1. Infrastructure Add RN dependency, bridge setup, CI/CD integration Build time increase, binary size
2. Leaf screens Migrate settings, about, help — low-risk screens Low — isolated, easy rollback
3. Shared services Bridge auth, analytics, networking to RN Medium — integration complexity
4. Feature screens Migrate core features one at a time High — navigation interop, state sharing
5. Navigation Move navigation stack to RN (react-navigation) Highest — complete paradigm shift
Staff-Level Mental Model

Brownfield integration is fundamentally an organizational problem, not just a technical one. The biggest challenges are: (1) native and RN teams sharing ownership of the bridge layer, (2) maintaining two build systems (Gradle + Metro, Xcode + Metro), (3) versioning native modules alongside the JS bundle, and (4) deciding when to stop migrating. The most successful brownfield integrations treat the native-RN boundary as an API contract with versioning, testing, and documentation — not as an implementation detail.
