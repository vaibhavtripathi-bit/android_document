Expo Modules API

The Expo Modules API is a modern, declarative approach to building native modules using Swift and Kotlin DSLs. It eliminates boilerplate, provides type-safe bindings, and integrates with Expo's config plugin and prebuild system for a streamlined native development experience.

1. Expo Modules Overview

Expo Modules API vs Traditional Native Modules ═══════════════════════════════════════════════ Traditional (RCT/ReactMethod): ────────────────────────────── iOS: RCTBridgeModule + ObjC macros + bridging headers Android: ReactContextBaseJavaModule + @ReactMethod + ReactPackage Lines: ~150-300 per platform for a simple module Expo Modules API: ───────────────── iOS: Swift DSL (ExpoModulesCore) Android: Kotlin DSL (expo-modules-core) Lines: ~30-60 per platform for the same module ┌────────────────────────────────────────────────────┐ │ Module("MyModule") { │ │ Function("doWork") { input: String -> String in │ │ return process(input) │ │ } │ │ } │ │ │ │ That's it. No macros, no registration boilerplate, │ │ no bridging headers, no ReactPackage. │ └────────────────────────────────────────────────────┘
Architect Insight

Expo Modules API is built on top of JSI (same as TurboModules) but provides a dramatically simpler developer experience. Under the hood, expo-modules-core generates the JSI bindings, type conversions, and module registration automatically from the DSL declarations. It's essentially a code-generation framework that targets JSI, similar to how React Native's codegen works but with a more ergonomic API surface.

// Creating a new Expo module
// npx create-expo-module my-native-module

// Generated project structure:
// my-native-module/
// ├── src/
// │   └── MyNativeModule.ts          ← JS API
// ├── ios/
// │   └── MyNativeModule.swift        ← Swift implementation
// ├── android/
// │   └── src/main/java/expo/modules/
// │       └── mynativemodule/
// │           └── MyNativeModule.kt   ← Kotlin implementation
// ├── expo-module.config.json         ← Module config
// └── package.json

// expo-module.config.json
{
  "platforms": ["ios", "android"],
  "ios": {
    "modules": ["MyNativeModule"]
  },
  "android": {
    "modules": ["expo.modules.mynativemodule.MyNativeModule"]
  }
}

// src/MyNativeModule.ts — JS API
import { requireNativeModule } from 'expo-modules-core';

const MyNativeModule = requireNativeModule('MyNativeModule');

export function doWork(input: string): string {
  return MyNativeModule.doWork(input);
}

export async function doWorkAsync(input: string): Promise<string> {
  return await MyNativeModule.doWorkAsync(input);
}

3. Swift DSL

// ios/MyNativeModule.swift
import ExpoModulesCore

public class MyNativeModule: Module {
  public func definition() -> ModuleDefinition {
    Name("MyNativeModule")

    // Constants — available synchronously on JS side
    Constants([
      "PI": Double.pi,
      "PLATFORM": "ios",
      "VERSION": Bundle.main.infoDictionary?["CFBundleShortVersionString"]
    ])

    // Synchronous function
    Function("doWork") { (input: String) -> String in
      return input.uppercased()
    }

    // Async function — runs on background thread
    AsyncFunction("doWorkAsync") { (input: String) -> String in
      let result = await heavyComputation(input)
      return result
    }

    // Function with complex types
    Function("processData") { (data: [String: Any]) -> [String: Any] in
      let values = data["values"] as? [Double] ?? []
      let sum = values.reduce(0, +)
      return ["sum": sum, "count": values.count]
    }

    // Events
    Events("onStatusChange", "onError")

    // Lifecycle
    OnCreate {
      print("Module created")
    }

    OnDestroy {
      print("Module destroyed")
    }

    OnAppEntersBackground {
      // Save state
    }

    OnAppBecomesActive {
      // Resume operations
    }
  }

  private func emitStatusChange(_ status: String) {
    sendEvent("onStatusChange", [
      "status": status,
      "timestamp": Date().timeIntervalSince1970
    ])
  }
}

4. Kotlin DSL

// android/.../MyNativeModule.kt
package expo.modules.mynativemodule

import expo.modules.kotlin.modules.Module
import expo.modules.kotlin.modules.ModuleDefinition

class MyNativeModule : Module() {
  override fun definition() = ModuleDefinition {
    Name("MyNativeModule")

    Constants(
      "PI" to Math.PI,
      "PLATFORM" to "android",
      "VERSION" to BuildConfig.VERSION_NAME
    )

    Function("doWork") { input: String ->
      input.uppercase()
    }

    AsyncFunction("doWorkAsync") { input: String ->
      heavyComputation(input)
    }

    Function("processData") { data: Map<String, Any> ->
      val values = (data["values"] as? List<Double>) ?: emptyList()
      val sum = values.sum()
      mapOf("sum" to sum, "count" to values.size)
    }

    Events("onStatusChange", "onError")

    OnCreate { }
    OnDestroy { }
    OnActivityEntersForeground { }
    OnActivityEntersBackground { }
  }
}

5. View Definition

// Swift — native view with Expo Modules
public class MyChartModule: Module {
  public func definition() -> ModuleDefinition {
    Name("MyChart")

    View(MyChartView.self) {
      Prop("data") { (view: MyChartView, data: [[String: Any]]) in
        view.updateData(data)
      }

      Prop("color") { (view: MyChartView, color: UIColor) in
        view.chartColor = color
      }

      Prop("animated") { (view: MyChartView, animated: Bool) in
        view.isAnimated = animated
      }

      Events("onDataPointTap")

      OnViewDidUpdateProps { view in
        view.refresh()
      }
    }
  }
}

class MyChartView: ExpoView {
  var chartColor: UIColor = .blue
  var isAnimated: Bool = true
  let onDataPointTap = EventDispatcher()

  func updateData(_ data: [[String: Any]]) {
    // Update chart rendering
  }

  func refresh() {
    // Re-render with current props
  }
}

// JS side
import { requireNativeViewManager } from 'expo-modules-core';

const NativeChartView = requireNativeViewManager('MyChart');

function Chart({ data, color, onTap }) {
  return (
    <NativeChartView
      data={data}
      color={color}
      animated={true}
      onDataPointTap={onTap}
      style={{ height: 200 }}
    />

}

6. Event Handling

// Typed events with Expo Modules

// Swift
Events("onProgress", "onComplete", "onError")

func startDownload() {
  sendEvent("onProgress", [
    "bytesWritten": 1024,
    "totalBytes": 10240,
    "percentage": 0.1
  ])

  sendEvent("onComplete", [
    "filePath": "/path/to/file",
    "size": 10240
  ])
}

// JS side — typed event listener
import { EventEmitter } from 'expo-modules-core';

const emitter = new EventEmitter(MyNativeModule);

function useDownloadProgress() {
  const [progress, setProgress] = useState(0);

  useEffect(() => {
    const sub = emitter.addListener('onProgress', (event) => {
      setProgress(event.percentage);
    });
    return () => sub.remove();
  }, []);

  return progress;
}

7. Async Functions

// Swift — async/await support
AsyncFunction("fetchData") { (url: String) -> [String: Any] in
  let (data, response) = try await URLSession.shared.data(
    from: URL(string: url)!
  )
  let json = try JSONSerialization.jsonObject(with: data)
  return json as! [String: Any]
}

// Kotlin — coroutine support
AsyncFunction("fetchData") Coroutine { url: String ->
  withContext(Dispatchers.IO) {
    val response = httpClient.get(url)
    val json = JSONObject(response.body)
    json.toMap()
  }
}

8. Expo Config Plugins

// Config plugin — modifies native projects at prebuild time
// plugin/src/index.ts
import {
  ConfigPlugin, withInfoPlist, withAndroidManifest,
  withEntitlementsPlist
} from 'expo/config-plugins';

const withMyModule: ConfigPlugin<{ apiKey: string }> = (
  config, { apiKey }
) => {
  // iOS: Add to Info.plist
  config = withInfoPlist(config, (config) => {
    config.modResults.NSCameraUsageDescription =
      'Required for scanning';
    config.modResults.MY_API_KEY = apiKey;
    return config;
  });

  // iOS: Add entitlements
  config = withEntitlementsPlist(config, (config) => {
    config.modResults['com.apple.developer.nfc.readersession.formats']
      = ['TAG'];
    return config;
  });

  // Android: Modify AndroidManifest.xml
  config = withAndroidManifest(config, (config) => {
    const mainApp = config.modResults.manifest.application?.[0];
    mainApp?.['meta-data']?.push({
      $: {
        'android:name': 'MY_API_KEY',
        'android:value': apiKey,
      },
    });
    return config;
  });

  return config;
};

export default withMyModule;

// app.json — using the config plugin
{
  "expo": {
    "plugins": [
      ["my-native-module", { "apiKey": "abc123" }]
    ]
  }
}

12. Expo Go Limitations

Expo Go Cannot Run Custom Native Code

Expo Go is a pre-built app with a fixed set of native modules. Custom native modules (whether Expo Modules API, TurboModules, or legacy NativeModules) cannot run in Expo Go. You must use a development build (npx expo run:ios or EAS Build with developmentClient: true) to test custom native code. This is the most common confusion for teams adopting Expo.

Staff-Level Mental Model

Expo Modules API represents the future direction of native module development in React Native. It solves the two biggest pain points: (1) boilerplate — a module that takes 300 lines with RCT macros takes 30 lines with Expo DSL, and (2) native project management — config plugins + prebuild mean native directories are generated artifacts, not hand-maintained code. The trade-off is vendor lock-in to the Expo ecosystem and a learning curve for the plugin system. For greenfield projects, Expo Modules is the recommended approach. For existing brownfield apps, the migration cost must be weighed against the DX benefits.
