C++ TurboModules

C++ TurboModules enable truly cross-platform native code — write once in C++, share across iOS and Android. They leverage JSI HostObjects for zero-overhead JS interop, making them ideal for performance-critical modules like crypto, image processing, and database engines.

1. C++ TurboModule Architecture

C++ TurboModule — Cross-Platform Architecture ══════════════════════════════════════════════ ┌──────────────────────┐ │ TypeScript Spec │ │ NativeCrypto.ts │ └──────────┬───────────┘ │ codegen ┌──────────▼───────────┐ │ C++ JSI Interface │ │ NativeCryptoSpec.h │ └──────────┬───────────┘ │ ┌──────────▼───────────┐ │ C++ Implementation │ ◄── Shared across platforms │ CryptoModule.cpp │ │ CryptoModule.h │ └──────────┬───────────┘ │ ┌────────────────┴────────────────┐ │ │ ┌─────────▼──────────┐ ┌──────────▼─────────┐ │ iOS Binding │ │ Android Binding │ │ CryptoModule.mm │ │ CryptoModule.cpp │ │ (ObjC++ bridge) │ │ (JNI bridge) │ │ Xcode project │ │ CMakeLists.txt │ └────────────────────┘ └────────────────────┘ JS ──► JSI ──► C++ (shared) ──► Platform APIs (if needed) │ └── Zero serialization, direct function calls
Architect Insight

C++ TurboModules are the highest-performance integration point in React Native. They bypass both the bridge AND platform-specific module layers. A JS call to a C++ TurboModule goes: JS → JSI (C++ virtual dispatch) → your C++ code. No JNI, no ObjC message dispatch, no serialization. This makes them ~100x faster than bridge calls and ~10x faster than platform-specific TurboModules for compute-heavy operations.

2. Shared C++ Implementation

// cpp/CryptoEngine.h — shared C++ header
#pragma once
#include <string>
#include <vector>
#include <memory>
#include <optional>

namespace crypto {

struct HashResult {
  std::string hex;
  std::vector<uint8_t> bytes;
  size_t length;
};

class CryptoEngine {
public:
  static std::shared_ptr<CryptoEngine> create();

  HashResult sha256(const std::string& input) const;
  HashResult hmacSha256(
    const std::string& key,
    const std::string& data
  ) const;

  std::string encrypt(
    const std::string& plaintext,
    const std::string& key
  ) const;

  std::optional<std::string> decrypt(
    const std::string& ciphertext,
    const std::string& key
  ) const;

  std::string generateRandomBytes(size_t count) const;

private:
  CryptoEngine() = default;
};

} // namespace crypto

// cpp/CryptoEngine.cpp — shared implementation
#include "CryptoEngine.h"
#include <openssl/sha.h>
#include <openssl/hmac.h>
#include <openssl/rand.h>

namespace crypto {

std::shared_ptr<CryptoEngine> CryptoEngine::create() {
  return std::shared_ptr<CryptoEngine>(
    new CryptoEngine()
  );
}

HashResult CryptoEngine::sha256(const std::string& input) const {
  unsigned char hash[SHA256_DIGEST_LENGTH];
  SHA256(
    reinterpret_cast<const unsigned char*>(input.data()),
    input.size(),
    hash
  );

  std::string hex;
  hex.reserve(SHA256_DIGEST_LENGTH * 2);
  for (int i = 0; i < SHA256_DIGEST_LENGTH; i++) {
    char buf[3];
    snprintf(buf, sizeof(buf), "%02x", hash[i]);
    hex += buf;
  }

  return HashResult{
    .hex = std::move(hex),
    .bytes = std::vector<uint8_t>(hash, hash + SHA256_DIGEST_LENGTH),
    .length = SHA256_DIGEST_LENGTH
  };
}

} // namespace crypto

// CryptoTurboModule.h — JSI binding
#pragma once
#include <ReactCommon/TurboModule.h>
#include <ReactCommon/CallInvoker.h>
#include "CryptoEngine.h"

namespace facebook::react {

class CryptoTurboModule : public TurboModule {
public:
  CryptoTurboModule(
    std::shared_ptr<CallInvoker> jsInvoker
  );

  jsi::Value sha256(
    jsi::Runtime& rt,
    const jsi::Value& input
  );

  jsi::Value encrypt(
    jsi::Runtime& rt,
    const jsi::Value& plaintext,
    const jsi::Value& key
  );

private:
  std::shared_ptr<crypto::CryptoEngine> engine_;
};

} // namespace facebook::react

// CryptoTurboModule.cpp — implementation
#include "CryptoTurboModule.h"

namespace facebook::react {

CryptoTurboModule::CryptoTurboModule(
  std::shared_ptr<CallInvoker> jsInvoker
) : TurboModule("NativeCrypto", std::move(jsInvoker)),
    engine_(crypto::CryptoEngine::create()) {

  methodMap_["sha256"] = MethodMetadata{
    1, // arg count
    [this](jsi::Runtime& rt, const jsi::Value* args, size_t count) {
      return this->sha256(rt, args[0]);
    }
  };
}

jsi::Value CryptoTurboModule::sha256(
  jsi::Runtime& rt,
  const jsi::Value& input
) {
  auto str = input.asString(rt).utf8(rt);
  auto result = engine_->sha256(str);

  auto obj = jsi::Object(rt);
  obj.setProperty(rt, "hex",
    jsi::String::createFromUtf8(rt, result.hex));
  obj.setProperty(rt, "length",
    jsi::Value((double)result.length));
  return obj;
}

} // namespace facebook::react

4. JSI HostObject for C++ Classes

// Exposing a C++ class as a JS object via HostObject
class DatabaseHostObject : public jsi::HostObject {
public:
  DatabaseHostObject(std::shared_ptr<Database> db)
    : db_(std::move(db)) {}

  jsi::Value get(jsi::Runtime& rt, const jsi::PropNameID& name) override {
    auto propName = name.utf8(rt);

    if (propName == "execute") {
      return jsi::Function::createFromHostFunction(
        rt, name, 2,
        [this](jsi::Runtime& rt, const jsi::Value& thisVal,
               const jsi::Value* args, size_t count) -> jsi::Value {
          auto sql = args[0].asString(rt).utf8(rt);
          auto result = db_->execute(sql);
          return resultToJSI(rt, result);
        }
      );
    }

    if (propName == "close") {
      return jsi::Function::createFromHostFunction(
        rt, name, 0,
        [this](jsi::Runtime& rt, const jsi::Value&,
               const jsi::Value*, size_t) -> jsi::Value {
          db_->close();
          return jsi::Value::undefined();
        }
      );
    }

    return jsi::Value::undefined();
  }

private:
  std::shared_ptr<Database> db_;
};

// Installing the HostObject on the JS global
void install(jsi::Runtime& rt) {
  auto db = std::make_shared<Database>("/path/to/db");
  auto hostObj = std::make_shared<DatabaseHostObject>(db);

  rt.global().setProperty(
    rt,
    "__database",
    jsi::Object::createFromHostObject(rt, hostObj)
  );
}

// JS side — using the HostObject
const db = global.__database;
const result = db.execute('SELECT * FROM users WHERE id = ?', [1]);
db.close();

// Safe memory patterns
class SafeHostObject : public jsi::HostObject {
public:
  ~SafeHostObject() {
    // Called when JS garbage collects the proxy object
    // Release native resources here
    engine_.reset();
  }

private:
  std::shared_ptr<Engine> engine_;
  // weak_ptr breaks cycles if Engine references back to HostObject
  std::weak_ptr<Callback> weakCallback_;

6. Error Handling Across C++/JS

// Throwing JS errors from C++
jsi::Value safeExecute(
  jsi::Runtime& rt,
  const jsi::Value* args,
  size_t count
) {
  try {
    auto result = engine_->compute(
      args[0].asString(rt).utf8(rt)
    );
    return jsi::String::createFromUtf8(rt, result);
  } catch (const std::invalid_argument& e) {
    throw jsi::JSError(rt,
      std::string("Invalid argument: ") + e.what()
    );
  } catch (const std::runtime_error& e) {
    throw jsi::JSError(rt,
      std::string("Runtime error: ") + e.what()
    );
  } catch (...) {
    throw jsi::JSError(rt, "Unknown native error");
  }
}

// JS side — catching native errors
try {
  const result = NativeCrypto.encrypt(data, key);
} catch (error) {
  // error.message contains the C++ error string
  console.error('Native error:', error.message);
}

9. Third-Party C++ Library Integration

// Integrating SQLite as a C++ TurboModule
// Uses sqlite3.c amalgamation (single-file build)

#include "sqlite3.h"

class SQLiteEngine {
public:
  SQLiteEngine(const std::string& path) {
    int rc = sqlite3_open(path.c_str(), &db_);
    if (rc != SQLITE_OK) {
      throw std::runtime_error(sqlite3_errmsg(db_));
    }
    sqlite3_exec(db_, "PRAGMA journal_mode=WAL", nullptr,
                 nullptr, nullptr);
  }

  ~SQLiteEngine() {
    if (db_) sqlite3_close(db_);
  }

  std::vector<Row> execute(
    const std::string& sql,
    const std::vector<Binding>& bindings = {}
  ) {
    sqlite3_stmt* stmt;
    sqlite3_prepare_v2(db_, sql.c_str(), -1, &stmt, nullptr);

    for (size_t i = 0; i < bindings.size(); i++) {
      bindValue(stmt, i + 1, bindings[i]);
    }

    std::vector<Row> results;
    while (sqlite3_step(stmt) == SQLITE_ROW) {
      results.push_back(extractRow(stmt));
    }

    sqlite3_finalize(stmt);
    return results;
  }

private:
  sqlite3* db_ = nullptr;
};

10. Testing C++ Modules

// Google Test for C++ module logic
#include <gtest/gtest.h>
#include "CryptoEngine.h"

TEST(CryptoEngineTest, SHA256ProducesCorrectHash) {
  auto engine = crypto::CryptoEngine::create();
  auto result = engine->sha256("hello");

  EXPECT_EQ(result.hex,
    "2cf24dba5fb0a30e26e83b2ac5b9e29e"
    "1b161e5c1fa7425e73043362938b9824");
  EXPECT_EQ(result.length, 32);
}

TEST(CryptoEngineTest, EncryptDecryptRoundTrip) {
  auto engine = crypto::CryptoEngine::create();
  const std::string key = "secret-key-256bit-padded-here!!";
  const std::string plaintext = "sensitive data";

  auto encrypted = engine->encrypt(plaintext, key);
  auto decrypted = engine->decrypt(encrypted, key);

  ASSERT_TRUE(decrypted.has_value());
  EXPECT_EQ(decrypted.value(), plaintext);
}
Staff-Level Mental Model

C++ TurboModules represent the convergence of React Native's architecture: JS (Hermes) → JSI (C++ interface) → your C++ code → platform APIs. The entire chain is C++ except the JS source code. This is why React Native's New Architecture is fundamentally a C++ rewrite — Fabric renderer, TurboModules, and the bridge replacement are all C++. Understanding C++ TurboModules means understanding the actual runtime architecture, not just the JS API surface.
