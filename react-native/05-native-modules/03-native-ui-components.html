Native UI Components

Custom native UI components let you wrap platform views (MapView, VideoPlayer, WebView) and expose them as React components. This doc covers the full lifecycle from ViewManager registration through Fabric component creation with ShadowNodes and ComponentDescriptors.

Architect Insight

Native UI components differ from native modules in a fundamental way: modules expose functions, while UI components expose views with props, events, and layout participation. The ViewManager acts as a factory — it creates view instances, sets props, and handles commands. In Fabric (New Architecture), this is replaced by ComponentDescriptor + ShadowNode for C++-level layout computation via Yoga.

2. Android ViewManager

// CircularProgressViewManager.kt
package com.myapp.views

import com.facebook.react.uimanager.SimpleViewManager
import com.facebook.react.uimanager.ThemedReactContext
import com.facebook.react.uimanager.annotations.ReactProp
import com.facebook.react.uimanager.annotations.ReactPropGroup
import com.facebook.react.common.MapBuilder

class CircularProgressViewManager : SimpleViewManager<CircularProgressView>() {

  override fun getName(): String = "RCTCircularProgress"

  override fun createViewInstance(
    context: ThemedReactContext
  ): CircularProgressView {
    return CircularProgressView(context)
  }

  @ReactProp(name = "progress", defaultFloat = 0f)
  fun setProgress(view: CircularProgressView, progress: Float) {
    view.setProgress(progress.coerceIn(0f, 1f))
  }

  @ReactProp(name = "strokeWidth", defaultFloat = 4f)
  fun setStrokeWidth(view: CircularProgressView, width: Float) {
    view.setStrokeWidth(width)
  }

  @ReactProp(name = "color", customType = "Color")
  fun setColor(view: CircularProgressView, color: Int?) {
    color?.let { view.setProgressColor(it) }
  }

  @ReactProp(name = "animated", defaultBoolean = true)
  fun setAnimated(view: CircularProgressView, animated: Boolean) {
    view.setAnimated(animated)
  }

  // Export event mappings: native event name → JS prop name
  override fun getExportedCustomDirectEventTypeConstants():
      MutableMap<String, Any>? {
    return MapBuilder.builder<String, Any>()
      .put("onProgressComplete",
        MapBuilder.of("registrationName", "onProgressComplete"))
      .put("onProgressChange",
        MapBuilder.of("registrationName", "onProgressChange"))
      .build()
  }

  // Direct commands — imperative API from JS
  override fun getCommandsMap(): MutableMap<String, Int> {
    return MapBuilder.of(
      "reset", COMMAND_RESET,
      "animateTo", COMMAND_ANIMATE_TO
    )
  }

  override fun receiveCommand(
    view: CircularProgressView,
    commandId: Int,
    args: ReadableArray?
  ) {
    when (commandId) {
      COMMAND_RESET -> view.reset()
      COMMAND_ANIMATE_TO -> {
        val target = args?.getDouble(0)?.toFloat() ?: 0f
        val duration = args?.getInt(1) ?: 300
        view.animateTo(target, duration)
      }
    }
  }

  companion object {
    const val COMMAND_RESET = 1
    const val COMMAND_ANIMATE_TO = 2
  }
}

5. Event Handling (Native → JS)

Native views communicate back to JS through events. There are two event types: bubbling (propagates up the view tree) and direct (sent only to the target component).

// Android — dispatching events from native view
class CircularProgressView(context: Context) : View(context) {

  private fun onProgressChanged(newProgress: Float) {
    val event = Arguments.createMap().apply {
      putDouble("progress", newProgress.toDouble())
      putDouble("timestamp", System.currentTimeMillis().toDouble())
    }
    val reactContext = context as ReactContext
    reactContext.getJSModule(RCTEventEmitter::class.java)
      .receiveEvent(id, "onProgressChange", event)
  }

  private fun onComplete() {
    val event = Arguments.createMap()
    val reactContext = context as ReactContext
    reactContext.getJSModule(RCTEventEmitter::class.java)
      .receiveEvent(id, "onProgressComplete", event)
  }
}

// iOS — dispatching events from native view
- (void)progressDidChange:(float)newProgress {
  if (self.onProgressChange) {
    self.onProgressChange(@{
      @"progress": @(newProgress),
      @"timestamp": @(CACurrentMediaTime())
    });
  }
}

- (void)progressDidComplete {
  if (self.onProgressComplete) {
    self.onProgressComplete(@{});
  }
}

// JS side — receiving events
const CircularProgress = requireNativeComponent('RCTCircularProgress');

function MyProgress() {
  return (
    <CircularProgress
      progress={0.75}
      strokeWidth={6}
      color="#7c3aed"
      onProgressChange={(e) => {
        console.log('Progress:', e.nativeEvent.progress);
      }}
      onProgressComplete={() => {
        console.log('Done!');
      }}
    />
  );
}
Event Type Distinction

RCTBubblingEventBlock (iOS) / getExportedCustomBubblingEventTypeConstants (Android): events bubble up the React tree (like DOM events). RCTDirectEventBlock / getExportedCustomDirectEventTypeConstants: events go directly to the component. Use direct events for component-specific notifications (onLoad, onError). Use bubbling events for interaction events that parent components might want to intercept.

6. Direct Commands

Commands provide an imperative API for controlling native views from JS — bypassing the declarative prop flow for operations like "scroll to position" or "play video".

// JS side — sending commands via UIManager
import { UIManager, findNodeHandle } from 'react-native';

function MyProgress() {
  const ref = useRef(null);

  const reset = () => {
    UIManager.dispatchViewManagerCommand(
      findNodeHandle(ref.current),
      UIManager.getViewManagerConfig('RCTCircularProgress')
        .Commands.reset,
      []  // arguments array
    );
  };

  const animateTo = (target, duration) => {
    UIManager.dispatchViewManagerCommand(
      findNodeHandle(ref.current),
      UIManager.getViewManagerConfig('RCTCircularProgress')
        .Commands.animateTo,
      [target, duration]
    );
  };

  return (
    <>
      <CircularProgress ref={ref} progress={0.5} />
      <Button title="Reset" onPress={reset} />
      <Button title="Animate to 100%"
              onPress={() => animateTo(1.0, 500)} />
    </>
  );
}

override fun onAfterUpdateTransaction(view: CircularProgressView) {
  super.onAfterUpdateTransaction(view)
  // Called after ALL props are set in a single update batch
  view.applyPendingChanges()
}

override fun onDropViewInstance(view: CircularProgressView) {
  super.onDropViewInstance(view)
  view.cleanup() // stop animations, remove listeners
}

// iOS — cleanup in view dealloc

  [self.displayLink invalidate];

8. Fabric Component Creation

9. Measuring and Layout

// Android — custom measurement in ViewManager
override fun getShadowNodeClass(): Class<out LayoutShadowNode> {
  return CircularProgressShadowNode::class.java
}

override fun createShadowNodeInstance(): LayoutShadowNode {
  return CircularProgressShadowNode()
}

class CircularProgressShadowNode : LayoutShadowNode(),
    YogaMeasureFunction {

  init { setMeasureFunction(this) }

  override fun measure(
    node: YogaNode, width: Float, widthMode: YogaMeasureMode,
    height: Float, heightMode: YogaMeasureMode
  ): Long {
    val size = 48f // default intrinsic size
    val measuredWidth = when (widthMode) {
      YogaMeasureMode.EXACTLY -> width
      YogaMeasureMode.AT_MOST -> minOf(size, width)
      else -> size
    }
    val measuredHeight = when (heightMode) {
      YogaMeasureMode.EXACTLY -> height
      YogaMeasureMode.AT_MOST -> minOf(size, height)
      else -> size
    }
    return YogaMeasureOutput.make(measuredWidth, measuredHeight)
  }
}

// iOS — intrinsic content size
- (CGSize)intrinsicContentSize {
  return CGSizeMake(48, 48);
}

// Fabric — C++ measurement function
Size CircularProgressMeasure(
    SurfaceId surfaceId,
    const LayoutConstraints &layoutConstraints,
    const CircularProgressProps &props) {
  float size = 48.0f;
  return {
    std::min(layoutConstraints.maximumSize.width, size),
    std::min(layoutConstraints.maximumSize.height, size)
  };
}

10. Accessibility for Custom Components

// Android — accessibility in custom view
class CircularProgressView(context: Context) : View(context) {

  init {
    importantForAccessibility = IMPORTANT_FOR_ACCESSIBILITY_YES
  }

  override fun onInitializeAccessibilityNodeInfo(
    info: AccessibilityNodeInfo
  ) {
    super.onInitializeAccessibilityNodeInfo(info)
    info.className = "android.widget.ProgressBar"
    info.roleDescription = "Progress indicator"
    info.rangeInfo = AccessibilityNodeInfo.RangeInfo.obtain(
      AccessibilityNodeInfo.RangeInfo.RANGE_TYPE_PERCENT,
      0f, 100f, progress * 100f
    )
    info.contentDescription =
      "Progress: ${(progress * 100).toInt()} percent"
  }
}

// iOS — accessibility in custom view
- (instancetype)init {
  self = [super init];
  self.isAccessibilityElement = YES;
  self.accessibilityTraits = UIAccessibilityTraitUpdatesFrequently;
  return self;
}

- (NSString *)accessibilityValue {
  return [NSString stringWithFormat:@"%d percent",
    (int)(self.progress * 100)];
}

- (NSString *)accessibilityLabel {
  return @"Circular progress indicator";
}

// JS side — accessibility props

Staff-Level Mental Model

Native UI components are the most complex integration point in React Native. They participate in three systems simultaneously: React's declarative prop/event model, Yoga's layout engine, and the platform's native view hierarchy. In Fabric, the C++ ShadowNode layer adds a fourth dimension — layout computation happens in C++ before platform views are even created. Mastering this stack means understanding how a single prop change flows from JSX through the reconciler, across JSI, into C++ layout, and finally to a platform view mutation.
