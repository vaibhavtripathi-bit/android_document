1. AsyncSequence Protocol

AsyncSequence is the async counterpart of Sequence. Where Sequence provides elements synchronously, AsyncSequence provides elements that may arrive over time, with each element potentially requiring an await.

// The AsyncSequence protocol
protocol AsyncSequence {
    associatedtype Element
    associatedtype AsyncIterator: AsyncIteratorProtocol
    func makeAsyncIterator() -> AsyncIterator
}

protocol AsyncIteratorProtocol {
    associatedtype Element
    mutating func next() async throws -> Element?
    // Returns nil when the sequence is finished
}
Sequence vs AsyncSequence: Sequence (synchronous): ┌───┐ ┌───┐ ┌───┐ ┌───┐ ┌─────┐ │ 1 │→│ 2 │→│ 3 │→│ 4 │→│ nil │ All available immediately └───┘ └───┘ └───┘ └───┘ └─────┘ AsyncSequence (asynchronous): ┌───┐ ┌───┐ ┌───┐ ┌───┐ ┌─────┐ │ 1 │→···│ 2 │→·········│ 3 │→·│ 4 │→······│ nil │ └───┘ └───┘ └───┘ └───┘ └─────┘ t=0 t=100ms t=500ms t=510ms t=2s Elements arrive over time. Consumer awaits each one.

2. for-await Pattern

// Basic for-await iteration
func processNotifications() async {
    let notifications = NotificationCenter.default.notifications(
        named: .didReceiveData
    )

    for await notification in notifications {
        guard let data = notification.userInfo?["data"] as? Data else { continue }
        await processData(data)
    }
    // This line is reached when the sequence finishes (iterator returns nil)
    // or when the task is cancelled
}

// Streaming bytes from a URL
func downloadWithProgress(url: URL) async throws {
    let (bytes, response) = try await URLSession.shared.bytes(from: url)
    let totalBytes = response.expectedContentLength

    var receivedBytes: Int64 = 0
    var buffer = Data()

    for try await byte in bytes {
        buffer.append(byte)
        receivedBytes += 1

        if receivedBytes % 1024 == 0 {
            let progress = Double(receivedBytes) / Double(totalBytes)
            await updateProgress(progress)
        }
    }
}

// Reading lines from a file
func processLogFile(url: URL) async throws {
    for try await line in url.lines {
        if line.contains("ERROR") {
            await reportError(line)
        }
    }
}
Deep Dive: How for-await Compiles

The compiler transforms for await x in sequence into a while loop that calls next() on the iterator. Each next() call is an await point — the task suspends until the next element is available. When next() returns nil, the loop exits. If the task is cancelled, the next await point will check cancellation (if the sequence supports it).

// for await x in sequence { body(x) }
// compiles to approximately:
var iterator = sequence.makeAsyncIterator()
while let x = try await iterator.next() {
    body(x)
}

3. AsyncStream & AsyncThrowingStream

AsyncStream is the primary way to create custom async sequences. It bridges imperative code (callbacks, delegates, timers) into the async sequence world.

3.1 AsyncStream with Continuation

// Bridging a delegate-based API to AsyncStream
class LocationTracker: NSObject, CLLocationManagerDelegate {
    private let manager = CLLocationManager()
    private var continuation: AsyncStream<CLLocation>.Continuation?

    var locations: AsyncStream<CLLocation> {
        AsyncStream { continuation in
            self.continuation = continuation
            manager.delegate = self
            manager.startUpdatingLocation()

            // Called when the stream is cancelled or finished
            continuation.onTermination = { @Sendable _ in
                self.manager.stopUpdatingLocation()
            }
        }
    }

    func locationManager(
        _ manager: CLLocationManager,
        didUpdateLocations locations: [CLLocation]
    ) {
        for location in locations {
            continuation?.yield(location)
        }
    }

    func locationManager(
        _ manager: CLLocationManager,
        didFailWithError error: Error
    ) {
        continuation?.finish()
    }
}

// Usage
let tracker = LocationTracker()
for await location in tracker.locations {
    print("Lat: \(location.coordinate.latitude)")
}

3.2 AsyncStream with Unfolding

// Simpler pattern for pull-based sequences
let countdown = AsyncStream<Int>.init(unfolding: {
    try? await Task.sleep(for: .seconds(1))
    return Int.random(in: 1...100)
})

// Timer-based stream
func timerStream(interval: Duration) -> AsyncStream<Date> {
    AsyncStream { continuation in
        let task = Task {
            while !Task.isCancelled {
                continuation.yield(Date())
                try? await Task.sleep(for: interval)
            }
            continuation.finish()
        }

        continuation.onTermination = { _ in
            task.cancel()
        }
    }
}

3.3 AsyncThrowingStream

// Stream that can produce errors
func eventStream() -> AsyncThrowingStream<ServerEvent, Error> {
    AsyncThrowingStream { continuation in
        let connection = WebSocketConnection(url: serverURL)

        connection.onMessage = { message in
            do {
                let event = try JSONDecoder().decode(ServerEvent.self, from: message)
                continuation.yield(event)
            } catch {
                continuation.finish(throwing: error)
            }
        }

        connection.onDisconnect = { error in
            if let error = error {
                continuation.finish(throwing: error)
            } else {
                continuation.finish()
            }
        }

        continuation.onTermination = { _ in
            connection.disconnect()
        }

        connection.connect()
    }
}

// Usage with error handling
do {
    for try await event in eventStream() {
        await handleEvent(event)
    }
} catch {
    print("Stream failed: \(error)")
}
Common Pitfall: AsyncStream Continuation Lifetime

The continuation must not be used after calling finish(). Yielding after finish is silently ignored, but it indicates a logic error. Also, if you store the continuation as a property, ensure it's properly cleaned up — a retained continuation that never finishes will keep the consuming task suspended forever.

4. Backpressure Handling

Backpressure occurs when a producer generates data faster than the consumer can process it. AsyncSequence has a natural backpressure model: the producer is implicitly paused because the consumer must await each element.

4.1 AsyncStream Buffering Policies

// AsyncStream.Continuation.BufferingPolicy
let stream1 = AsyncStream<Int>(bufferingPolicy: .unbounded) { continuation in
    // No limit — can cause unbounded memory growth!
}

let stream2 = AsyncStream<Int>(bufferingPolicy: .bufferingNewest(10)) { continuation in
    // Keeps newest 10 elements, drops oldest when full
}

let stream3 = AsyncStream<Int>(bufferingPolicy: .bufferingOldest(10)) { continuation in
    // Keeps oldest 10 elements, drops newest when full
}
Backpressure Models: Pull-based (AsyncSequence default): Producer ──yield──▶ Buffer ──await next()──▶ Consumer │ │ │ Producer blocked │ │ until consumer calls │ │ next() │ └────────────────────────┘ Buffered (AsyncStream): Producer ──yield──▶ [Buffer: max N] ──await──▶ Consumer │ │ │ If buffer full: │ │ .bufferingNewest → drop oldest │ .bufferingOldest → drop newest │ .unbounded → grow forever ⚠️ └───────────────────────────┘
Common Pitfall: Unbounded Buffering

The default buffering policy for AsyncStream is .unbounded. If your producer is faster than your consumer (e.g., sensor data at 100Hz, consumer processing at 10Hz), the buffer grows without limit, eventually causing an out-of-memory crash. Always specify a bounded buffering policy for high-frequency streams.

5. Bridging Combine → AsyncSequence

// Built-in bridge: Publisher.values
let publisher = NotificationCenter.default.publisher(for: .someNotification)

for await notification in publisher.values {
    print("Received: \(notification)")
}

// @Published property as AsyncSequence
class SettingsStore: ObservableObject {
    @Published var theme: Theme = .light
}

let store = SettingsStore()
for await theme in store.$theme.values {
    await applyTheme(theme)
}

// Custom bridge for complex Combine pipelines
extension Publisher where Failure == Never {
    func toAsyncStream() -> AsyncStream<Output> {
        AsyncStream { continuation in
            let cancellable = self.sink { completion in
                continuation.finish()
            } receiveValue: { value in
                continuation.yield(value)
            }

            continuation.onTermination = { _ in
                cancellable.cancel()
            }
        }
    }
}

// Bridge with Combine operators applied first
let debouncedSearch = searchPublisher
    .debounce(for: .milliseconds(300), scheduler: RunLoop.main)
    .removeDuplicates()
    .values  // AsyncSequence

for await query in debouncedSearch {
    await performSearch(query)
}

6. withCheckedContinuation Patterns

// Pattern 1: Single-shot callback → async
func requestPermission() async -> Bool {
    await withCheckedContinuation { continuation in
        UNUserNotificationCenter.current().requestAuthorization(
            options: [.alert, .sound]
        ) { granted, _ in
            continuation.resume(returning: granted)
        }
    }
}

// Pattern 2: Result-based callback → async throws
func authenticate() async throws -> Token {
    try await withCheckedThrowingContinuation { continuation in
        authService.authenticate { result in
            continuation.resume(with: result)
        }
    }
}

// Pattern 3: Delegate → async (one-shot)
func pickImage() async -> UIImage? {
    await withCheckedContinuation { continuation in
        let picker = UIImagePickerController()
        let delegate = PickerDelegate { image in
            continuation.resume(returning: image)
        }
        picker.delegate = delegate
        present(picker)
    }
}

// Pattern 4: Wrapping DispatchQueue.async
func runOnQueue<T: Sendable>(
    _ queue: DispatchQueue,
    work: @Sendable @escaping () -> T
) async -> T {
    await withCheckedContinuation { continuation in
        queue.async {
            let result = work()
            continuation.resume(returning: result)
        }
    }
}

7. Building Custom Async Sequences

// Custom AsyncSequence: Paginated API fetcher
struct PaginatedFetcher<Item: Decodable & Sendable>: AsyncSequence {
    typealias Element = [Item]

    let baseURL: URL
    let pageSize: Int

    struct AsyncIterator: AsyncIteratorProtocol {
        let baseURL: URL
        let pageSize: Int
        var currentPage = 0
        var isExhausted = false

        mutating func next() async throws -> [Item]? {
            guard !isExhausted else { return nil }
            try Task.checkCancellation()

            var components = URLComponents(url: baseURL, resolvingAgainstBaseURL: false)!
            components.queryItems = [
                URLQueryItem(name: "page", value: "\(currentPage)"),
                URLQueryItem(name: "size", value: "\(pageSize)")
            ]

            let (data, _) = try await URLSession.shared.data(from: components.url!)
            let items = try JSONDecoder().decode([Item].self, from: data)

            if items.count < pageSize {
                isExhausted = true
            }
            currentPage += 1

            return items.isEmpty ? nil : items
        }
    }

    func makeAsyncIterator() -> AsyncIterator {
        AsyncIterator(baseURL: baseURL, pageSize: pageSize)
    }
}

// Usage: automatically paginates through all results
let fetcher = PaginatedFetcher<User>(
    baseURL: URL(string: "https://api.example.com/users")!,
    pageSize: 50
)

for try await page in fetcher {
    for user in page {
        await processUser(user)
    }
}

8. AsyncSequence Operators

// Built-in operators (mirror Sequence operators)
let numbers = AsyncStream<Int> { /* ... */ }

// map
for await doubled in numbers.map({ $0 * 2 }) { }

// filter
for await even in numbers.filter({ $0.isMultiple(of: 2) }) { }

// compactMap
for await value in strings.compactMap({ Int($0) }) { }

// prefix (take first N)
for await value in numbers.prefix(10) { }

// drop (skip first N)
for await value in numbers.dropFirst(5) { }

// first (single value)
let first = await numbers.first(where: { $0 > 100 })

// reduce
let sum = await numbers.reduce(0, +)

// contains
let hasNegative = await numbers.contains(where: { $0 < 0 })

// Chaining operators
for await result in events
    .filter { $0.type == .userAction }
    .map { $0.payload }
    .prefix(100) {
    await process(result)
}

9. Architect: Choosing Between Combine, AsyncSequence, and Observation

Architect Must-Know

In 2024+, iOS developers have three reactive/streaming frameworks: Combine, AsyncSequence, and the Observation framework. Choosing the right one for each layer of your architecture is a critical Staff-level decision.

Aspect Combine AsyncSequence Observation
Introduced iOS 13 (2019) iOS 15 (2021) iOS 17 (2023)
Paradigm Reactive (push) Async iteration (pull) Observation (automatic)
Backpressure Demand-based (explicit) Natural (await-based) N/A (synchronous)
Operators Rich (100+ operators) Basic (map, filter, etc.) None (not a stream)
Error handling Typed errors in pipeline throws / try await N/A
Cancellation AnyCancellable Task cancellation withObservationTracking scope
SwiftUI integration @Published + ObservableObject .task { for await } @Observable (native)
Best for Complex event processing Streaming data, bridging UI state management
Best Practice: Recommended Architecture (2024+)
  • UI Layer: Observation framework (@Observable) for SwiftUI, @MainActor for UIKit
  • ViewModel: @Observable + async/await for data loading
  • Service Layer: async/await + AsyncSequence for streaming APIs
  • Legacy Integration: Combine where existing code uses it; bridge to AsyncSequence via .values
  • Complex Event Processing: Combine for debounce/throttle/combineLatest chains (richer operators)
  • New streaming APIs: AsyncStream for wrapping delegates and callbacks
