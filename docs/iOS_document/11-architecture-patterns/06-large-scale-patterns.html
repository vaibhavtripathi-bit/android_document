1. Feature Flags

Feature flags (feature toggles) decouple deployment from release. Code ships to production behind a flag, and the flag controls whether users see the feature. This enables trunk-based development, gradual rollouts, and instant kill switches.

1.1 Feature Flag Types

TypeLifetimePurposeExample
Release flagDays-weeksHide incomplete featuresNew checkout flow
Experiment flagWeeks-monthsA/B test variantsButton color test
Ops flagPermanentRuntime controlKill switch for video
Permission flagPermanentUser entitlementsPremium features

1.2 Feature Flag Implementation

protocol FeatureFlagProvider: Sendable {
    func isEnabled(_ flag: FeatureFlag) -> Bool
    func value<T: Decodable>(for flag: FeatureFlag) -> T?
    func observe(_ flag: FeatureFlag) -> AsyncStream<Bool>
}

enum FeatureFlag: String, CaseIterable, Sendable {
    case newCheckoutFlow = "new_checkout_flow"
    case aiRecommendations = "ai_recommendations"
    case videoPlayerV2 = "video_player_v2"
    case darkModeAutoSwitch = "dark_mode_auto_switch"
    case killSwitchPayments = "kill_switch_payments"

    var defaultValue: Bool {
        switch self {
        case .killSwitchPayments: return true
        default: return false
        }
    }
}

final class FeatureFlagService: FeatureFlagProvider, @unchecked Sendable {
    private let remoteConfig: RemoteConfigProvider
    private let localStorage: UserDefaults
    private let overrides: [FeatureFlag: Bool]
    private let lock = NSLock()
    private var cache: [FeatureFlag: Bool] = [:]

    func isEnabled(_ flag: FeatureFlag) -> Bool {
        // Priority: debug override > remote config > local cache > default
        if let override = overrides[flag] {
            return override
        }

        lock.lock()
        defer { lock.unlock() }

        if let cached = cache[flag] {
            return cached
        }

        let value = remoteConfig.bool(forKey: flag.rawValue) ?? flag.defaultValue
        cache[flag] = value
        return value
    }

    func observe(_ flag: FeatureFlag) -> AsyncStream<Bool> {
        AsyncStream { continuation in
            continuation.yield(isEnabled(flag))
            let observer = remoteConfig.addObserver(forKey: flag.rawValue) { newValue in
                continuation.yield(newValue as? Bool ?? flag.defaultValue)
            }
            continuation.onTermination = { _ in
                observer.remove()
            }
        }
    }
}

1.3 Gradual Rollout

struct RolloutConfig: Codable {
    let flag: String
    let percentage: Double
    let targetGroups: [String]
    let excludedUserIds: [String]
    let minAppVersion: String?
}

final class GradualRollout {
    func isUserInRollout(userId: String, config: RolloutConfig) -> Bool {
        guard !config.excludedUserIds.contains(userId) else {
            return false
        }

        if let minVersion = config.minAppVersion,
           Bundle.main.appVersion < minVersion {
            return false
        }

        // Deterministic bucketing: same user always gets same result
        let hash = "\(userId):\(config.flag)".hashValue
        let bucket = abs(hash) % 100
        return Double(bucket) < config.percentage
    }
}

1.4 Kill Switch Pattern

final class KillSwitch {
    private let flags: FeatureFlagProvider

    func isServiceAvailable(_ service: ServiceType) -> Bool {
        switch service {
        case .payments:
            return flags.isEnabled(.killSwitchPayments)
        case .videoStreaming:
            return flags.isEnabled(.killSwitchVideo)
        case .search:
            return flags.isEnabled(.killSwitchSearch)
        }
    }
}

// Usage in feature code
func processPayment() async throws {
    guard killSwitch.isServiceAvailable(.payments) else {
        throw ServiceError.temporarilyUnavailable(
            message: "Payments are temporarily unavailable. Please try again later."
        )
    }
    // ... process payment
}

struct Experiment: Codable, Sendable {

    let startDate: Date
    let endDate: Date?
}

    let id: String
    let name: String
    let weight: Double
    let parameters: [String: AnyCodable]
}

protocol ExperimentService: Sendable {
    func variant(for experimentId: String) -> Variant?
    func logExposure(experimentId: String, variantId: String)
}

final class ABTestingService: ExperimentService, @unchecked Sendable {
    private var assignments: [String: Variant] = [:]
    private let analytics: AnalyticsProtocol

    func variant(for experimentId: String) -> Variant? {
        guard let variant = assignments[experimentId] else {
            return nil
        }
        logExposure(experimentId: experimentId, variantId: variant.id)
        return variant
    }

    func logExposure(experimentId: String, variantId: String) {
        analytics.track(ExperimentExposureEvent(
            experimentId: experimentId,
            variantId: variantId
        ))
    }
}

3. Analytics Abstraction Layer

An analytics abstraction layer decouples your app code from specific analytics providers (Firebase, Amplitude, Mixpanel). This enables switching providers without touching feature code.

3.1 Protocol-Based Abstraction

protocol AnalyticsEvent: Sendable {
    var name: String { get }
    var parameters: [String: any Sendable] { get }
}

protocol AnalyticsProvider: Sendable {
    func track(_ event: AnalyticsEvent)
    func setUserProperty(_ key: String, value: String?)
    func identify(userId: String)
    func reset()
}

final class AnalyticsService: Sendable {
    private let providers: [AnalyticsProvider]
    private let middleware: [AnalyticsMiddleware]

    init(providers: [AnalyticsProvider], middleware: [AnalyticsMiddleware] = []) {
        self.providers = providers
        self.middleware = middleware
    }

    func track(_ event: AnalyticsEvent) {
        var processedEvent = event
        for mw in middleware {
            processedEvent = mw.process(processedEvent)
        }
        for provider in providers {
            provider.track(processedEvent)
        }
    }
}

// Concrete providers
final class FirebaseAnalyticsProvider: AnalyticsProvider {
    func track(_ event: AnalyticsEvent) {
        Analytics.logEvent(event.name, parameters: event.parameters)
    }
}

final class AmplitudeProvider: AnalyticsProvider {
    func track(_ event: AnalyticsEvent) {
        Amplitude.instance().logEvent(event.name, withEventProperties: event.parameters)
    }
}

3.2 Type-Safe Event Definitions

enum AppAnalyticsEvent: AnalyticsEvent {
    case screenViewed(name: String, source: String?)
    case productViewed(id: String, category: String)
    case addedToCart(productId: String, price: Decimal, quantity: Int)
    case purchaseCompleted(orderId: String, total: Decimal, itemCount: Int)
    case searchPerformed(query: String, resultCount: Int)
    case errorOccurred(domain: String, code: Int, message: String)

    var name: String {
        switch self {
        case .screenViewed: return "screen_viewed"
        case .productViewed: return "product_viewed"
        case .addedToCart: return "added_to_cart"
        case .purchaseCompleted: return "purchase_completed"
        case .searchPerformed: return "search_performed"
        case .errorOccurred: return "error_occurred"
        }
    }

    var parameters: [String: any Sendable] {
        switch self {
        case let .screenViewed(name, source):
            var params: [String: any Sendable] = ["screen_name": name]
            if let source { params["source"] = source }
            return params
        case let .productViewed(id, category):
            return ["product_id": id, "category": category]
        case let .addedToCart(productId, price, quantity):
            return ["product_id": productId, "price": NSDecimalNumber(decimal: price), "quantity": quantity]
        case let .purchaseCompleted(orderId, total, itemCount):
            return ["order_id": orderId, "total": NSDecimalNumber(decimal: total), "item_count": itemCount]
        case let .searchPerformed(query, resultCount):
            return ["query": query, "result_count": resultCount]
        case let .errorOccurred(domain, code, message):
            return ["domain": domain, "code": code, "message": message]
        }
    }
}

4.3 Structured Logging with OSLogStore

import OSLog

func exportRecentLogs() throws -> [LogEntry] {
    let store = try OSLogStore(scope: .currentProcessIdentifier)
    let position = store.position(
        date: Date().addingTimeInterval(-3600)
    )
    let entries = try store.getEntries(at: position)
        .compactMap { $0 as? OSLogEntryLog }
        .filter { $0.subsystem == Bundle.main.bundleIdentifier }
        .map { entry in
            LogEntry(
                date: entry.date,
                level: entry.level.rawValue,
                category: entry.category,
                message: entry.composedMessage
            )
        }
    return entries
}

5. Crash Reporting Integration

5.1 Crash Reporter Abstraction

protocol CrashReporter: Sendable {
    func recordError(_ error: Error, metadata: [String: String])
    func recordBreadcrumb(_ message: String, category: String)
    func setUserIdentifier(_ id: String?)
    func setCustomKey(_ key: String, value: String)
}

final class CrashReportingService: CrashReporter {
    private let reporters: [CrashReporter]

    func recordError(_ error: Error, metadata: [String: String] = [:]) {
        var enrichedMetadata = metadata
        enrichedMetadata["app_version"] = Bundle.main.appVersion
        enrichedMetadata["build_number"] = Bundle.main.buildNumber
        enrichedMetadata["os_version"] = ProcessInfo.processInfo.operatingSystemVersionString

        for reporter in reporters {
            reporter.recordError(error, metadata: enrichedMetadata)
        }
    }

    func recordBreadcrumb(_ message: String, category: String) {
        for reporter in reporters {
            reporter.recordBreadcrumb(message, category: category)
        }
    }
}

// Automatic breadcrumb recording via coordinator
extension Coordinator {
    func trackNavigation(to screen: String) {
        crashReporter.recordBreadcrumb(
            "Navigated to \(screen)",
            category: "navigation"
        )
    }
}

6.2 Implementation

actor RemoteConfigService {
    private var activeConfig: [String: AnyCodable] = [:]
    private var fetchedConfig: [String: AnyCodable] = [:]
    private let defaults: [String: AnyCodable]
    private let networkClient: APIClient
    private let storage: ConfigStorage

    func fetchAndActivate() async throws {
        let response = try await networkClient.fetch(ConfigResponse.self, from: .remoteConfig)
        fetchedConfig = response.values
        try await storage.save(fetchedConfig)
        activate()
    }

    func activate() {
        activeConfig = fetchedConfig
    }

    func value<T: Decodable>(forKey key: String) -> T? {
        if let value = activeConfig[key] {
            return value.value as? T
        }
        return defaults[key]?.value as? T
    }

    func bool(forKey key: String) -> Bool? {
        value(forKey: key)
    }
}

8. Architect: Platform Infrastructure for 20+ Teams

Architect Must-Know: Building Platform Infrastructure

At 20+ feature teams, platform infrastructure becomes a product itself. The platform team builds SDKs that feature teams consume. The quality of these SDKs directly impacts every team's velocity, reliability, and ability to experiment safely.

8.1 Platform Team Responsibilities

AnalyticsEvent tracking, user propertiesProduct insights, funnel analysis

8.2 SDK Design Principles

// Platform SDK design principles for internal teams:
//
// 1. ZERO CONFIGURATION: Works out of the box
//    PlatformSDK.initialize() вҖ” one line in AppDelegate
//
// 2. PROTOCOL-BASED: Feature teams depend on protocols, not implementations
//    import PlatformInterfaces (protocols only, no implementation)
//    import PlatformSDK (implementation, only in app target)
//
// 3. FAIL-SAFE: SDK failures never crash the app
//    Feature flags return defaults if service is down
//    Analytics silently drops events if queue is full
//
// 4. TESTABLE: Mock implementations for unit tests
//    import PlatformMocks (in test targets only)
//
// 5. OBSERVABLE: SDK health metrics exposed
//    Config fetch latency, event queue depth, error rates

Best Practice: Feature Flag Hygiene

Feature flags are technical debt. Every flag adds a code path that must be tested. Establish a policy: release flags are removed within 2 sprints of full rollout. Experiment flags are removed when the experiment concludes. Ops flags are the only permanent flags. Track flag age in your dashboard and alert when flags exceed their expected lifetime.
