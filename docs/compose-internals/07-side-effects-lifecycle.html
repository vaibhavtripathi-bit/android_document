3. LaunchedEffect — Internals

// What LaunchedEffect actually does (simplified from source):
@Composable
fun LaunchedEffect(key1: Any?, block: suspend CoroutineScope.() -> Unit) {
    val applyContext = currentComposer.applyCoroutineContext
    remember(key1) {
        LaunchedEffectImpl(applyContext, block)
    }
}

// LaunchedEffectImpl is a RememberObserver:
internal class LaunchedEffectImpl(
    parentContext: CoroutineContext,
    private val task: suspend CoroutineScope.() -> Unit
) : RememberObserver {
    private val scope = CoroutineScope(parentContext + Job())

    override fun onRemembered() {
        // Called when the effect enters the composition
        scope.launch(block = task)
    }

    override fun onForgotten() {
        // Called when the effect leaves the composition
        scope.cancel() // Cancels the coroutine
    }

    override fun onAbandoned() {
        scope.cancel()
    }
}

// Key insight: LaunchedEffect is just remember() + RememberObserver
// When the key changes, the old remembered value is "forgotten" (cancelled)
// and a new one is "remembered" (started)
Staff-Level Insight — RememberObserver Is the Foundation

All effects are built on RememberObserver — an interface that gets callbacks when a value enters (onRemembered) or leaves (onForgotten) the slot table. LaunchedEffect starts a coroutine on onRemembered and cancels it on onForgotten. DisposableEffect calls the effect block on onRemembered and the onDispose block on onForgotten. Understanding this pattern lets you build custom effects for any lifecycle-aware resource management.

4. DisposableEffect — Register/Unregister Pattern

@Composable
fun LifecycleObserver(
    lifecycleOwner: LifecycleOwner = LocalLifecycleOwner.current,
    onEvent: (Lifecycle.Event) -> Unit
) {
    val currentOnEvent by rememberUpdatedState(onEvent)

    DisposableEffect(lifecycleOwner) {
        val observer = LifecycleEventObserver { _, event ->
            currentOnEvent(event)
        }
        lifecycleOwner.lifecycle.addObserver(observer)

        onDispose {
            lifecycleOwner.lifecycle.removeObserver(observer)
        }
    }
}

rememberUpdatedState — Why It Exists

// Problem: LaunchedEffect captures the lambda at creation time
@Composable
fun Timer(onTick: () -> Unit) {
    // BUG: if onTick changes, the old lambda is still captured
    LaunchedEffect(Unit) {
        while (true) {
            delay(1000)
            onTick() // This is the ORIGINAL onTick, not the latest one
        }
    }
}

// Fix: rememberUpdatedState keeps a reference to the latest value
@Composable
fun Timer(onTick: () -> Unit) {
    val currentOnTick by rememberUpdatedState(onTick)

    LaunchedEffect(Unit) {
        while (true) {
            delay(1000)
            currentOnTick() // Always calls the LATEST onTick
        }
    }
}

// How it works: rememberUpdatedState returns a State object
// The State is updated on every recomposition with the latest value
// The LaunchedEffect reads the State (not the lambda directly)
// So it always gets the current value without restarting the effect

5. SideEffect — Synchronizing with External State

// SideEffect runs AFTER every successful composition
// It's NOT a coroutine — it runs synchronously on the main thread

@Composable
fun AnalyticsScreen(screenName: String) {
    SideEffect {
        // Runs after every recomposition
        // Safe to access Compose state here — composition is complete
        analytics.setCurrentScreen(screenName)
    }
}

// Key difference from LaunchedEffect:
// - SideEffect has NO key — it runs every time
// - SideEffect has NO cleanup — it's fire-and-forget
// - SideEffect runs SYNCHRONOUSLY after composition
// - LaunchedEffect runs ASYNCHRONOUSLY in a coroutine

6. CompositionLocal — Implicit Data Passing

// CompositionLocal provides values down the tree without explicit parameters

// Define a CompositionLocal:
val LocalUserTheme = compositionLocalOf<Theme> {
    error("No theme provided") // Default if not provided
}

// Provide a value:
@Composable
fun App() {
    CompositionLocalProvider(LocalUserTheme provides DarkTheme) {
        // Everything inside sees DarkTheme
        MainScreen()
    }
}

// Read the value:
@Composable
fun SomeDeepChild() {
    val theme = LocalUserTheme.current // Reads from nearest provider
    Box(Modifier.background(theme.backgroundColor))
}

// staticCompositionLocalOf — for values that rarely change
val LocalDensity = staticCompositionLocalOf<Density> { /* ... */ }
// When a static local changes, ALL readers recompose (broad invalidation)
// When a regular local changes, only the reading scopes recompose (precise invalidation)
// Use static for things like Density, LayoutDirection that change only on config change
Staff-Level Insight — compositionLocalOf vs staticCompositionLocalOf

compositionLocalOf uses the snapshot system to track reads — only composables that actually call .current recompose when the value changes. This is precise but has overhead (snapshot read tracking per access). staticCompositionLocalOf doesn't track reads — when the value changes, it invalidates the entire subtree below the provider. This sounds worse, but for values that almost never change (Density, LayoutDirection, Context), the lack of per-read tracking overhead makes it faster overall. Rule: use static for stable infrastructure values, regular for frequently-changing values.

7. movableContentOf — Preserving State Across Moves

// movableContentOf lets you move composable content between different
// positions in the tree WITHOUT losing state

@Composable
fun AdaptiveLayout(isWide: Boolean) {
    val sharedContent = remember {
        movableContentOf {
            // This content's state survives being moved
            var count by remember { mutableStateOf(0) }
            Button(onClick = { count++ }) {
                Text("Count: $count")
            }
        }
    }

    if (isWide) {
        Row {
            SidePanel()
            sharedContent() // Content is HERE when wide
        }
    } else {
        Column {
            sharedContent() // Content MOVES here when narrow
            BottomBar()
        }
    }
    // When isWide changes, the button's count is PRESERVED
    // Without movableContentOf, switching branches would reset count to 0
}

// How it works internally:
// movableContentOf wraps the content in a MovableGroup
// When the content moves to a new position, the runtime:
// 1. Removes the group from the old position in the slot table
// 2. Inserts it at the new position
// 3. All remembered values, state, and effects are preserved
// 4. The Applier moves the nodes in the tree (no recreation)
Staff-Level Insight — movableContentOf Use Cases

This is an advanced API for adaptive layouts where the same content appears in different tree positions based on screen size or orientation. Without it, switching between layouts would destroy and recreate all state. The key constraint: the content can only exist in one place at a time. If you call the movable content lambda twice in the same composition, it throws. For shared elements across navigation transitions, this is the foundation of Compose's shared element transitions API.

8. Effect Ordering and Timing

@Composable
fun EffectTimingDemo() {
    // 1. Composition phase (synchronous, main thread)
    val state = remember { mutableStateOf(0) }
    Text("Value: ${state.value}")

    // 2. SideEffect — runs AFTER composition succeeds
    SideEffect {
        println("SideEffect: composition complete")
    }

    // 3. DisposableEffect — onRemembered called after composition
    DisposableEffect(Unit) {
        println("DisposableEffect: entered composition")
        onDispose {
            println("DisposableEffect: leaving composition")
        }
    }

    // 4. LaunchedEffect — coroutine launched after composition
    LaunchedEffect(Unit) {
        println("LaunchedEffect: coroutine started")
    }

    // Execution order on first composition:
    // 1. Composition runs (remember, Text, etc.)
    // 2. Changes applied to node tree (Applier)
    // 3. SideEffect runs
    // 4. DisposableEffect.onRemembered runs
    // 5. LaunchedEffect coroutine launches (may not execute immediately)
    // 6. Layout phase (measure + place)
    // 7. Draw phase
}
Common Effect Mistakes
  • Using LaunchedEffect for one-shot events: LaunchedEffect(Unit) { navigateTo(screen) } — this runs on every recomposition if the composable leaves and re-enters composition. Use a ViewModel event channel instead.
  • Missing key in LaunchedEffect: LaunchedEffect(Unit) { loadData(id) } — if id changes, the effect doesn't restart. Fix: LaunchedEffect(id) { loadData(id) }
  • Doing work in SideEffect that needs cleanup: SideEffect has no dispose mechanism. If you register a listener in SideEffect, you can never unregister it. Use DisposableEffect instead.
  • Reading state in DisposableEffect's onDispose: The composition may have already changed by the time onDispose runs. Capture values before the effect, not inside onDispose.

9. Compose Testing Internals

Compose testing is built on the semantics tree, not the LayoutNode tree. Understanding the internals helps write reliable, non-flaky tests:

// Basic test structure:
class MyScreenTest {
    @get:Rule
    val composeTestRule = createComposeRule()

    @Test
    fun counterIncrements() {
        composeTestRule.setContent {
            CounterScreen()
        }

        // Find nodes via semantics tree:
        composeTestRule.onNodeWithText("Count: 0").assertIsDisplayed()
        composeTestRule.onNodeWithText("Increment").performClick()
        composeTestRule.onNodeWithText("Count: 1").assertIsDisplayed()
    }
}

How Compose Tests Work Internally

// ComposeTestRule does several things:
// 1. Creates a test Recomposer with a CONTROLLABLE frame clock
//    (not tied to Choreographer — tests control when frames advance)
// 2. Installs a test CoroutineDispatcher
// 3. Provides synchronization: waits for all recompositions,
//    animations, and pending effects to complete before assertions

// The synchronization mechanism:
// Before every assertion or action, ComposeTestRule:
// a) Advances the frame clock until no more recompositions are pending
// b) Drains the main thread dispatcher
// c) Waits for all LaunchedEffect coroutines to suspend
// d) Only THEN executes the assertion

// This is why Compose tests are less flaky than View tests —
// the framework guarantees the UI is in a stable state before assertions

// Semantic matchers search the semantics tree:
onNodeWithText("Hello")        // Matches Text("Hello") or contentDescription
onNodeWithTag("my_tag")        // Matches Modifier.testTag("my_tag")
onNodeWithContentDescription() // Matches semantics { contentDescription = "..." }
onAllNodesWithText("Item")     // Returns all matching nodes

// Actions dispatch through the semantics tree:
performClick()        // Finds the onClick semantic action and invokes it
performTextInput()    // Finds the text input semantic action
performScrollTo()     // Scrolls until the node is visible

Testing with Async and Animations

// Waiting for async operations:
composeTestRule.waitUntil(timeoutMillis = 5000) {
    composeTestRule.onAllNodesWithText("Loaded").fetchSemanticsNodes().isNotEmpty()
}

// Controlling time for animations:
composeTestRule.mainClock.autoAdvance = false // Pause the clock
composeTestRule.mainClock.advanceTimeBy(500)  // Advance by 500ms
// Now assert the animation state at exactly 500ms

// Testing LaunchedEffect / Flow collection:
@Test
fun loadDataOnAppear() {
    val fakeRepo = FakeRepository()
    composeTestRule.setContent {
        DataScreen(repository = fakeRepo)
    }
    // ComposeTestRule automatically waits for LaunchedEffect to complete
    // (if it suspends on a state write that triggers recomposition)
    composeTestRule.onNodeWithText("Data loaded").assertIsDisplayed()
}
Staff-Level Insight — Testing Strategy at Scale

For L6 interviews, the testing strategy matters as much as the API knowledge. The recommended approach: (1) Unit test ViewModels with runTest (no Compose dependency). (2) Component tests with createComposeRule() for individual composables — test behavior via semantics, not implementation. (3) Screenshot tests with Roborazzi or Paparazzi for visual regression. (4) Integration tests with createAndroidComposeRule<Activity>() for navigation flows. Key principle: test through the semantics tree, never through internal state. If you can't test something, it's missing a semantic property — add testTag or contentDescription.
