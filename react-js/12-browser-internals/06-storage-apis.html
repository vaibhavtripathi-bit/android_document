Web Storage (localStorage / sessionStorage)

// localStorage: persists across sessions, shared across tabs
localStorage.setItem('theme', 'dark');
const theme = localStorage.getItem('theme'); // 'dark'
localStorage.removeItem('theme');
localStorage.clear();

// LIMITATIONS:
// 1. Synchronous — blocks main thread
// 2. Strings only — must JSON.stringify/parse
// 3. ~5-10 MB limit (varies by browser)
// 4. No indexing or querying
// 5. No transactions

// Type-safe wrapper
class TypedStorage<T> {
  constructor(private key: string, private defaultValue: T) {}

  get(): T {
    try {
      const raw = localStorage.getItem(this.key);
      return raw ? JSON.parse(raw) : this.defaultValue;
    } catch {
      return this.defaultValue;
    }
  }

  set(value: T): void {
    localStorage.setItem(this.key, JSON.stringify(value));
  }

  remove(): void {
    localStorage.removeItem(this.key);
  }
}

// React hook for localStorage
function useLocalStorage<T>(key: string, initialValue: T) {
  const [value, setValue] = useState<T>(() => {
    try {
      const item = localStorage.getItem(key);
      return item ? JSON.parse(item) : initialValue;
    } catch {
      return initialValue;
    }
  });

  useEffect(() => {
    localStorage.setItem(key, JSON.stringify(value));
  }, [key, value]);

  // Listen for changes from other tabs
  useEffect(() => {
    const handler = (e: StorageEvent) => {
      if (e.key === key && e.newValue) {
        setValue(JSON.parse(e.newValue));
      }
    };
    window.addEventListener('storage', handler);
    return () => window.removeEventListener('storage', handler);
  }, [key]);

  return [value, setValue] as const;
}

// sessionStorage: per-tab, cleared when tab closes
// Use for: form drafts, wizard state, tab-specific UI state
sessionStorage.setItem('formDraft', JSON.stringify(formData));

IndexedDB

// IndexedDB: full database in the browser
// Async, transactional, supports indexes and cursors

// Modern wrapper using idb library (by Jake Archibald)
import { openDB } from 'idb';

const dbPromise = openDB('my-app', 1, {
  upgrade(db) {
    const store = db.createObjectStore('users', { keyPath: 'id' });
    store.createIndex('email', 'email', { unique: true });
    store.createIndex('role', 'role');

    db.createObjectStore('cache', { keyPath: 'key' });
  },
});

// CRUD operations
async function addUser(user) {
  const db = await dbPromise;
  await db.put('users', user);
}

async function getUser(id) {
  const db = await dbPromise;
  return db.get('users', id);
}

async function getUsersByRole(role) {
  const db = await dbPromise;
  return db.getAllFromIndex('users', 'role', role);
}

async function deleteUser(id) {
  const db = await dbPromise;
  await db.delete('users', id);
}

// Transaction for atomic operations
async function transferFunds(fromId, toId, amount) {
  const db = await dbPromise;
  const tx = db.transaction('accounts', 'readwrite');
  const store = tx.objectStore('accounts');

  const from = await store.get(fromId);
  const to = await store.get(toId);

  from.balance -= amount;
  to.balance += amount;

  await store.put(from);
  await store.put(to);
  await tx.done;
}

// React hook for IndexedDB
function useIndexedDB<T>(storeName: string, key: string) {
  const [data, setData] = useState<T | null>(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    let cancelled = false;
    dbPromise.then(db => db.get(storeName, key)).then(result => {
      if (!cancelled) {
        setData(result || null);
        setLoading(false);
      }
    });
    return () => { cancelled = true; };
  }, [storeName, key]);

  const update = useCallback(async (value: T) => {
    const db = await dbPromise;
    await db.put(storeName, value);
    setData(value);
  }, [storeName]);

  return { data, loading, update };
}

Cache API

// Cache API: stores Request/Response pairs
// Primarily used with Service Workers but available on main thread

// Cache static assets
async function precacheAssets(urls) {
  const cache = await caches.open('static-v1');
  await cache.addAll(urls);
}

// Cache API responses
async function cachedFetch(url, options = {}) {
  const cache = await caches.open('api-cache');
  const cached = await cache.match(url);

  if (cached && !options.forceRefresh) {
    const age = Date.now() - new Date(cached.headers.get('date')).getTime();
    if (age < (options.maxAge || 300000)) {
      return cached.json();
    }
  }

  const response = await fetch(url);
  cache.put(url, response.clone());
  return response.json();
}

// Cache management
async function cleanOldCaches(currentVersion) {
  const keys = await caches.keys();
  await Promise.all(
    keys
      .filter(key => key !== currentVersion)
      .map(key => caches.delete(key))
  );
}

Cookies

// Cookies: small data sent with every HTTP request to the origin
// Use for: authentication tokens, session IDs, preferences

// Setting cookies (server-side is preferred for security)
// Set-Cookie: session=abc123; HttpOnly; Secure; SameSite=Strict; Path=/; Max-Age=86400

// Client-side cookie access (limited)
document.cookie = 'theme=dark; path=/; max-age=31536000; SameSite=Lax';

// Reading cookies (painful API)
function getCookie(name) {
  const match = document.cookie.match(new RegExp(`(^| )${name}=([^;]+)`));
  return match ? decodeURIComponent(match[2]) : null;
}

// Cookie Store API (modern, async, observable)
// Available in Chrome/Edge (not Firefox/Safari yet)
if ('cookieStore' in window) {
  // Read
  const cookie = await cookieStore.get('session');

  // Write
  await cookieStore.set({
    name: 'theme',
    value: 'dark',
    expires: Date.now() + 365 * 24 * 60 * 60 * 1000,
    sameSite: 'lax',
  });

  // Observe changes
  cookieStore.addEventListener('change', (event) => {
    for (const cookie of event.changed) {
      console.log('Cookie changed:', cookie.name, cookie.value);
    }
  });
}

// Cookie security attributes:
// HttpOnly  — not accessible via document.cookie (XSS protection)
// Secure    — only sent over HTTPS
// SameSite  — controls cross-site sending:
//   Strict: never sent cross-site
//   Lax: sent on top-level navigations (default)
//   None: always sent (requires Secure)
// Path      — restricts cookie to URL path
// Domain    — restricts cookie to domain
// Max-Age   — expiration in seconds

Origin Private File System (OPFS)

// OPFS: high-performance file system access
// Much faster than IndexedDB for large binary data

// Get the root directory
const root = await navigator.storage.getDirectory();

// Create and write a file
const fileHandle = await root.getFileHandle('data.json', { create: true });
const writable = await fileHandle.createWritable();
await writable.write(JSON.stringify(largeData));
await writable.close();

// Read a file
const file = await fileHandle.getFile();
const text = await file.text();
const data = JSON.parse(text);

// Synchronous access in Web Workers (much faster)
// Only available in dedicated workers
const accessHandle = await fileHandle.createSyncAccessHandle();
const buffer = new ArrayBuffer(accessHandle.getSize());
accessHandle.read(buffer);
accessHandle.write(newBuffer);
accessHandle.flush();
accessHandle.close();

// Use case: SQLite in the browser (via sql.js or wa-sqlite)
// OPFS provides the fast file I/O that SQLite needs

Storage Quotas and Eviction

// Check storage quota
const estimate = await navigator.storage.estimate();
console.log(`Used: ${estimate.usage} bytes`);
console.log(`Quota: ${estimate.quota} bytes`);
console.log(`Available: ${estimate.quota - estimate.usage} bytes`);
// Typical quota: ~50% of available disk space

// Request persistent storage (prevents eviction)
const persisted = await navigator.storage.persist();
if (persisted) {
  console.log('Storage will not be evicted');
} else {
  console.log('Storage may be evicted under pressure');
}

// Check if already persisted
const isPersisted = await navigator.storage.persisted();

// Eviction order (when storage pressure):
// 1. Non-persisted origins, least recently used first
// 2. All data for an origin is evicted together
//    (localStorage, IndexedDB, Cache API, OPFS)
// 3. Persisted origins are NOT evicted (user must clear manually)

Cross-Tab Communication

// Method 1: BroadcastChannel (recommended)
const channel = new BroadcastChannel('app-state');

// Send
channel.postMessage({ type: 'THEME_CHANGED', theme: 'dark' });

// Receive (in other tabs)
channel.onmessage = (event) => {
  if (event.data.type === 'THEME_CHANGED') {
    setTheme(event.data.theme);
  }
};

// Cleanup
channel.close();

// React hook
function useBroadcastChannel(channelName) {
  const [lastMessage, setLastMessage] = useState(null);
  const channelRef = useRef(null);

  useEffect(() => {
    const channel = new BroadcastChannel(channelName);
    channelRef.current = channel;
    channel.onmessage = (e) => setLastMessage(e.data);
    return () => channel.close();
  }, [channelName]);

  const postMessage = useCallback((data) => {
    channelRef.current?.postMessage(data);
  }, []);

  return { lastMessage, postMessage };
}

// Method 2: localStorage events (wider support)
// Fires in OTHER tabs when localStorage changes
window.addEventListener('storage', (event) => {
  if (event.key === 'auth-state') {
    const newState = JSON.parse(event.newValue);
    if (!newState.isLoggedIn) {
      redirectToLogin();
    }
  }
});

// Method 3: SharedWorker (shared across tabs)
// shared-worker.js
const connections = new Set();

self.onconnect = (event) => {
  const port = event.ports[0];
  connections.add(port);

  port.onmessage = (e) => {
    // Broadcast to all connected tabs
    connections.forEach(p => {
      if (p !== port) p.postMessage(e.data);
    });
  };

  port.start();
};

// In each tab:
const worker = new SharedWorker('shared-worker.js');
worker.port.onmessage = (e) => handleMessage(e.data);
worker.port.postMessage({ type: 'SYNC', data: stateUpdate });

Structured Clone Algorithm

// Structured clone: deep copy algorithm used by:
// - postMessage (workers, BroadcastChannel)
// - IndexedDB (storing/retrieving)
// - structuredClone() (explicit API)
// - history.pushState()

// What CAN be cloned:
// ✓ Primitives (string, number, boolean, null, undefined)
// ✓ Date, RegExp, Blob, File, FileList
// ✓ ArrayBuffer, TypedArrays, DataView
// ✓ Map, Set
// ✓ Plain objects, Arrays
// ✓ Circular references (handled correctly!)

// What CANNOT be cloned:
// ✗ Functions
// ✗ DOM nodes
// ✗ Symbols
// ✗ Property descriptors (getters/setters)
// ✗ Prototype chain (only own properties)
// ✗ Error objects (partially — message and name only)

// structuredClone() — explicit deep clone
const original = {
  date: new Date(),
  nested: { map: new Map([['key', 'value']]) },
  buffer: new ArrayBuffer(8),
};

const clone = structuredClone(original);
// clone.date instanceof Date → true
// clone.nested.map instanceof Map → true
// clone.buffer instanceof ArrayBuffer → true
// clone !== original → true (deep copy)

// Transfer instead of clone (zero-copy for ArrayBuffers)
const buffer = new ArrayBuffer(1024 * 1024); // 1MB
const transferred = structuredClone(buffer, { transfer: [buffer] });
// buffer.byteLength → 0 (transferred, not copied)
// transferred.byteLength → 1048576
Interview Tip

When discussing storage, match the API to the use case: localStorage for small, synchronous key-value data (theme, preferences). IndexedDB for structured data that needs indexing (offline cache, large datasets). Cache API for HTTP response caching (Service Worker). OPFS for high-performance binary data (SQLite, file processing). Cookies for data that must be sent to the server (auth tokens). Knowing the tradeoffs shows platform expertise.
