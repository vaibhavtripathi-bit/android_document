Fetch API Internals

// Fetch API: modern replacement for XMLHttpRequest
// Returns a Promise that resolves to a Response object

// Basic fetch with error handling
async function fetchJSON(url, options = {}) {
  const response = await fetch(url, {
    headers: { 'Content-Type': 'application/json', ...options.headers },
    signal: options.signal,
    ...options,
  });

  if (!response.ok) {
    throw new HttpError(response.status, await response.text());
  }

  return response.json();
}

// Streaming response body
async function fetchWithProgress(url, onProgress) {
  const response = await fetch(url);
  const contentLength = response.headers.get('Content-Length');
  const total = parseInt(contentLength, 10);
  let received = 0;

  const reader = response.body.getReader();
  const chunks = [];

  while (true) {
    const { done, value } = await reader.read();
    if (done) break;

    chunks.push(value);
    received += value.length;
    onProgress({ received, total, percentage: (received / total) * 100 });
  }

  const blob = new Blob(chunks);
  return blob;
}

// Fetch with timeout and retry
async function resilientFetch(url, options = {}) {
  const { timeout = 10000, retries = 3, ...fetchOptions } = options;

  for (let attempt = 0; attempt <= retries; attempt++) {
    const controller = new AbortController();
    const timeoutId = setTimeout(() => controller.abort(), timeout);

    try {
      const response = await fetch(url, {
        ...fetchOptions,
        signal: controller.signal,
      });
      clearTimeout(timeoutId);

      if (response.ok) return response;
      if (response.status < 500) throw new HttpError(response.status);
    } catch (err) {
      clearTimeout(timeoutId);
      if (attempt === retries) throw err;
      await new Promise(r => setTimeout(r, 2 ** attempt * 1000));
    }
  }
}

HTTP/3 (QUIC)

// HTTP/3 uses QUIC (UDP-based) instead of TCP
// Solves HTTP/2's TCP-level head-of-line blocking

// HTTP/2 problem: TCP packet loss blocks ALL streams
// If one packet is lost, TCP retransmission blocks everything

// HTTP/3 solution: each QUIC stream is independent
// Packet loss in one stream doesn't affect others

// Key benefits:
// 1. No TCP head-of-line blocking
// 2. 0-RTT connection establishment (vs TCP's 1-3 RTT)
// 3. Connection migration (survives IP changes, e.g., WiFi → cellular)
// 4. Built-in TLS 1.3 (always encrypted)

// Check if HTTP/3 is being used:
const response = await fetch('https://example.com');
// Chrome DevTools → Network → Protocol column shows "h3"

// HTTP/3 is negotiated via Alt-Svc header:
// Alt-Svc: h3=":443"; ma=86400
// Browser tries QUIC, falls back to HTTP/2 if unavailable

Service Workers for Network Interception

// Service Worker: programmable network proxy
// Intercepts ALL network requests from the page

// Registration
if ('serviceWorker' in navigator) {
  navigator.serviceWorker.register('/sw.js', {
    scope: '/',
  });
}

// sw.js — Service Worker
const CACHE_NAME = 'app-v1';
const STATIC_ASSETS = ['/index.html', '/app.js', '/style.css'];

// Install: pre-cache static assets
self.addEventListener('install', (event) => {
  event.waitUntil(
    caches.open(CACHE_NAME).then(cache => cache.addAll(STATIC_ASSETS))
  );
  self.skipWaiting();
});

// Activate: clean up old caches
self.addEventListener('activate', (event) => {
  event.waitUntil(
    caches.keys().then(keys =>
      Promise.all(
        keys.filter(k => k !== CACHE_NAME).map(k => caches.delete(k))
      )
    )
  );
  self.clients.claim();
});

// Fetch: intercept and respond
self.addEventListener('fetch', (event) => {
  const { request } = event;

  if (request.destination === 'document') {
    // Network-first for HTML (always get latest)
    event.respondWith(networkFirst(request));
  } else if (STATIC_ASSETS.includes(new URL(request.url).pathname)) {
    // Cache-first for static assets
    event.respondWith(cacheFirst(request));
  } else if (request.url.includes('/api/')) {
    // Stale-while-revalidate for API calls
    event.respondWith(staleWhileRevalidate(request));
  }
});

async function cacheFirst(request) {
  const cached = await caches.match(request);
  return cached || fetch(request);
}

async function networkFirst(request) {
  try {
    const response = await fetch(request);
    const cache = await caches.open(CACHE_NAME);
    cache.put(request, response.clone());
    return response;
  } catch {
    return caches.match(request);
  }
}

async function staleWhileRevalidate(request) {
  const cache = await caches.open(CACHE_NAME);
  const cached = await cache.match(request);

  const fetchPromise = fetch(request).then(response => {
    cache.put(request, response.clone());
    return response;
  });

  return cached || fetchPromise;
}

WebSocket Protocol

// WebSocket: full-duplex communication over a single TCP connection
// Starts as HTTP upgrade, then switches to WS protocol

class ReconnectingWebSocket {
  constructor(url, options = {}) {
    this.url = url;
    this.maxRetries = options.maxRetries || 10;
    this.retryDelay = options.retryDelay || 1000;
    this.retryCount = 0;
    this.handlers = new Map();
    this.connect();
  }

  connect() {
    this.ws = new WebSocket(this.url);

    this.ws.onopen = () => {
      this.retryCount = 0;
      this.handlers.get('open')?.forEach(h => h());
    };

    this.ws.onmessage = (event) => {
      const data = JSON.parse(event.data);
      this.handlers.get(data.type)?.forEach(h => h(data.payload));
      this.handlers.get('message')?.forEach(h => h(data));
    };

    this.ws.onclose = (event) => {
      if (!event.wasClean && this.retryCount < this.maxRetries) {
        const delay = this.retryDelay * Math.pow(2, this.retryCount);
        setTimeout(() => {
          this.retryCount++;
          this.connect();
        }, delay);
      }
    };

    this.ws.onerror = () => {};
  }

  on(event, handler) {
    if (!this.handlers.has(event)) this.handlers.set(event, new Set());
    this.handlers.get(event).add(handler);
    return () => this.handlers.get(event).delete(handler);
  }

  send(type, payload) {
    if (this.ws.readyState === WebSocket.OPEN) {
      this.ws.send(JSON.stringify({ type, payload }));
    }
  }

  close() {
    this.maxRetries = 0;
    this.ws.close();
  }
}

Server-Sent Events

// SSE: server-to-client streaming over HTTP
// Simpler than WebSocket for one-way data

function useServerEvents(url) {
  const [data, setData] = useState(null);
  const [connected, setConnected] = useState(false);

  useEffect(() => {
    const source = new EventSource(url);

    source.onopen = () => setConnected(true);

    source.onmessage = (event) => {
      setData(JSON.parse(event.data));
    };

    source.addEventListener('notification', (event) => {
      const notification = JSON.parse(event.data);
      showNotification(notification);
    });

    source.onerror = () => {
      setConnected(false);
      // EventSource auto-reconnects!
    };

    return () => source.close();
  }, [url]);

  return { data, connected };
}

Resource Hints

// Preload: fetch resource needed for current page (high priority)
<link rel="preload" href="/fonts/inter.woff2" as="font" type="font/woff2" crossorigin>
<link rel="preload" href="/hero.webp" as="image">
<link rel="preload" href="/critical.css" as="style">

// Prefetch: fetch resource needed for NEXT navigation (low priority)
<link rel="prefetch" href="/next-page.js">
<link rel="prefetch" href="/api/next-page-data" as="fetch">

// Preconnect: establish connection to origin (DNS + TCP + TLS)
<link rel="preconnect" href="https://api.example.com">
<link rel="preconnect" href="https://fonts.googleapis.com">

// DNS-prefetch: resolve DNS only (cheaper than preconnect)
<link rel="dns-prefetch" href="https://analytics.example.com">

// Modulepreload: preload ES modules (includes parsing)
<link rel="modulepreload" href="/components/Header.js">

// Priority Hints: fine-tune resource priority
<img src="hero.jpg" fetchpriority="high">
<img src="below-fold.jpg" fetchpriority="low">
<script src="analytics.js" fetchpriority="low"></script>
<link rel="stylesheet" href="critical.css" fetchpriority="high">

// React: programmatic prefetching
function usePrefetch(url) {
  useEffect(() => {
    const link = document.createElement('link');
    link.rel = 'prefetch';
    link.href = url;
    document.head.appendChild(link);
    return () => document.head.removeChild(link);
  }, [url]);
}

Beacon API and Background Sync

// Beacon API: guaranteed delivery of small data
// Survives page unload (unlike fetch)
window.addEventListener('visibilitychange', () => {
  if (document.visibilityState === 'hidden') {
    navigator.sendBeacon('/api/analytics', JSON.stringify({
      sessionDuration: Date.now() - sessionStart,
      events: pendingEvents,
    }));
  }
});

// Background Sync (Service Worker API)
// Retry failed requests when connectivity returns
// Registration:
async function saveData(data) {
  try {
    await fetch('/api/save', { method: 'POST', body: JSON.stringify(data) });
  } catch {
    await saveToIndexedDB('pending-sync', data);
    const reg = await navigator.serviceWorker.ready;
    await reg.sync.register('sync-pending-data');
  }
}

// In Service Worker:
self.addEventListener('sync', (event) => {
  if (event.tag === 'sync-pending-data') {
    event.waitUntil(syncPendingData());
  }
});

async function syncPendingData() {
  const pending = await getFromIndexedDB('pending-sync');
  for (const item of pending) {
    await fetch('/api/save', { method: 'POST', body: JSON.stringify(item) });
    await removeFromIndexedDB('pending-sync', item.id);
  }
}
Interview Tip

When discussing networking, show depth by connecting APIs to user experience: "I'd use preconnect for the API origin to save 100-300ms on first request. For images below the fold, I'd use fetchpriority='low' and lazy loading. For analytics, I'd use sendBeacon to ensure delivery without blocking navigation. For real-time features, I'd choose SSE over WebSocket for one-way data because it auto-reconnects and multiplexes over HTTP/2." This practical application of platform knowledge is what L6 interviewers want.
