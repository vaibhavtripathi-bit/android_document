2. Environment Variables: Client-Side Exposure

// CRITICAL: Client-side env vars are embedded in the JavaScript bundle
// They are NOT secret — anyone can read them

// .env file
VITE_API_URL=https://api.example.com        // Public — OK
VITE_STRIPE_PUBLIC_KEY=pk_live_abc123       // Public key — OK
VITE_ANALYTICS_ID=UA-12345                  // Public — OK
DATABASE_URL=postgres://user:pass@host/db   // Server-only — SAFE (no VITE_ prefix)
STRIPE_SECRET_KEY=sk_live_xyz789            // Server-only — SAFE (no VITE_ prefix)

// WRONG: exposing secrets to client
VITE_DATABASE_URL=postgres://...            // NEVER! Visible in bundle
VITE_STRIPE_SECRET=sk_live_...              // NEVER! Visible in bundle
VITE_AWS_SECRET_KEY=...                     // NEVER! Visible in bundle

// How to verify what's in your bundle:
// 1. Build the app: npm run build
// 2. Search the output: grep -r "sk_live" dist/
// 3. If found, you have a secret leak

Build-Time Embedding

Environment variables with the client prefix (VITE_, NEXT_PUBLIC_, REACT_APP_) are replaced with their literal values at build time by the bundler's DefinePlugin or define configuration. This means: (1) they appear as plain strings in the JavaScript bundle, (2) they cannot be changed without rebuilding, (3) they are visible to anyone who views the page source or inspects the bundle. Treat them as public information.

3. API Key Protection

// Pattern: Backend for Frontend (BFF) proxy
// Client never sees the actual API key

// WRONG: API key in client code
async function fetchWeather(city) {
  const response = await fetch(
    `https://api.weather.com/data?key=SECRET_API_KEY&city=${city}`
  );
  return response.json();
}

// CORRECT: Proxy through your server
// Client code:
async function fetchWeather(city) {
  const response = await fetch(`/api/weather?city=${city}`);
  return response.json();
}

// Server code (Next.js API route):
// app/api/weather/route.ts
export async function GET(request: NextRequest) {
  const city = request.nextUrl.searchParams.get('city');

  // Rate limiting
  const clientIp = request.headers.get('x-forwarded-for');
  if (await isRateLimited(clientIp)) {
    return NextResponse.json({ error: 'Rate limited' }, { status: 429 });
  }

  // Server adds the secret API key
  const response = await fetch(
    `https://api.weather.com/data?key=${process.env.WEATHER_API_KEY}&city=${city}`
  );
  const data = await response.json();

  return NextResponse.json(data);
}

// For third-party APIs that MUST be called from the client
// (e.g., Stripe, Google Maps), use restricted API keys:
// - Domain-restricted (only works from your domain)
// - Scope-restricted (read-only, specific endpoints)
// - Rate-limited

4. Secrets Management

// Development: .env files (gitignored)
// .gitignore
.env
.env.local
.env.*.local

// Production: secrets management services
// AWS Secrets Manager, HashiCorp Vault, Vercel Environment Variables

// Next.js: server-only secrets
// lib/secrets.ts
import 'server-only'; // Ensures this module can't be imported in client code

export const secrets = {
  databaseUrl: process.env.DATABASE_URL!,
  stripeSecret: process.env.STRIPE_SECRET_KEY!,
  jwtSecret: process.env.JWT_SECRET!,
};

// Validation at startup
function validateSecrets() {
  const required = ['DATABASE_URL', 'STRIPE_SECRET_KEY', 'JWT_SECRET'];
  const missing = required.filter(key => !process.env[key]);

  if (missing.length > 0) {
    throw new Error(`Missing required environment variables: ${missing.join(', ')}`);
  }
}

// Call at server startup
validateSecrets();

5. Data Encryption

// HTTPS: encryption in transit (mandatory)
// All modern React apps MUST be served over HTTPS
// HSTS header forces HTTPS:
Strict-Transport-Security: max-age=31536000; includeSubDomains; preload

// Client-side encryption with Web Crypto API
async function encryptData(plaintext: string, key: CryptoKey): Promise<string> {
  const encoder = new TextEncoder();
  const iv = crypto.getRandomValues(new Uint8Array(12)); // 96-bit IV for AES-GCM

  const encrypted = await crypto.subtle.encrypt(
    { name: 'AES-GCM', iv },
    key,
    encoder.encode(plaintext)
  );

  // Combine IV + ciphertext for storage
  const combined = new Uint8Array(iv.length + encrypted.byteLength);
  combined.set(iv);
  combined.set(new Uint8Array(encrypted), iv.length);

  return btoa(String.fromCharCode(...combined));
}

async function decryptData(ciphertext: string, key: CryptoKey): Promise<string> {
  const combined = Uint8Array.from(atob(ciphertext), c => c.charCodeAt(0));
  const iv = combined.slice(0, 12);
  const data = combined.slice(12);

  const decrypted = await crypto.subtle.decrypt(
    { name: 'AES-GCM', iv },
    key,
    data
  );

  return new TextDecoder().decode(decrypted);
}

// Generate encryption key
async function generateKey(): Promise<CryptoKey> {
  return crypto.subtle.generateKey(
    { name: 'AES-GCM', length: 256 },
    true,  // extractable
    ['encrypt', 'decrypt']
  );
}

6. PII Handling

// Personally Identifiable Information (PII) requires special handling

// Data classification
const DATA_CLASSIFICATION = {
  PUBLIC: 'public',         // Company name, product info
  INTERNAL: 'internal',     // Employee names, internal docs
  CONFIDENTIAL: 'confidential', // Customer emails, addresses
  RESTRICTED: 'restricted', // SSN, credit cards, health data
};

// PII masking for display
function maskEmail(email: string): string {
  const [local, domain] = email.split('@');
  const masked = local[0] + '***' + local[local.length - 1];
  return `${masked}@${domain}`;
}

function maskPhone(phone: string): string {
  return phone.replace(/(\d{3})\d{4}(\d{4})/, '$1****$2');
}

function maskSSN(ssn: string): string {
  return '***-**-' + ssn.slice(-4);
}

function maskCreditCard(number: string): string {
  return '**** **** **** ' + number.slice(-4);
}

// React component with PII masking
function UserProfile({ user }: { user: User }) {
  const [showEmail, setShowEmail] = useState(false);

  return (
    <div>
      <p>Name: {user.name}</p>
      <p>
        Email: {showEmail ? user.email : maskEmail(user.email)}
        <button onClick={() => setShowEmail(!showEmail)}>
          {showEmail ? 'Hide' : 'Show'}
        </button>
      </p>
      <p>Phone: {maskPhone(user.phone)}</p>
    </div>
  );
}

// Prevent PII in logs
function sanitizeForLogging(data: Record<string, unknown>) {
  const sensitiveFields = ['password', 'ssn', 'creditCard', 'token', 'secret'];
  const sanitized = { ...data };

  for (const field of sensitiveFields) {
    if (field in sanitized) {
      sanitized[field] = '[REDACTED]';
    }
  }

  return sanitized;
}

7. GDPR Compliance in Frontend

// GDPR requirements affecting React applications:

// 1. Consent management
function CookieConsent() {
  const [consent, setConsent] = useState<ConsentState | null>(null);

  useEffect(() => {
    const stored = localStorage.getItem('cookie-consent');
    if (stored) setConsent(JSON.parse(stored));
  }, []);

  const handleConsent = (preferences: ConsentState) => {
    setConsent(preferences);
    localStorage.setItem('cookie-consent', JSON.stringify(preferences));

    // Enable/disable tracking based on consent
    if (preferences.analytics) {
      initAnalytics();
    }
    if (preferences.marketing) {
      initMarketingPixels();
    }
  };

  if (consent) return null;

  return (
    <div className="cookie-banner">
      <p>We use cookies to improve your experience.</p>
      <button onClick={() => handleConsent({ essential: true, analytics: true, marketing: true })}>
        Accept All
      </button>
      <button onClick={() => handleConsent({ essential: true, analytics: false, marketing: false })}>
        Essential Only
      </button>
      <button onClick={() => openPreferences()}>Customize</button>
    </div>
  );
}

// 2. Right to data export
async function exportUserData(userId: string) {
  const response = await fetch(`/api/user/${userId}/export`, {
    credentials: 'include',
  });
  const blob = await response.blob();
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = 'my-data.json';
  a.click();
  URL.revokeObjectURL(url);
}

// 3. Right to deletion
async function requestAccountDeletion() {
  if (!confirm('This will permanently delete your account and all data.')) return;

  await fetch('/api/user/delete', {
    method: 'DELETE',
    credentials: 'include',
  });

  // Clear all local data
  localStorage.clear();
  sessionStorage.clear();

  // Clear cookies
  document.cookie.split(';').forEach(cookie => {
    const name = cookie.split('=')[0].trim();
    document.cookie = `${name}=; expires=Thu, 01 Jan 1970 00:00:00 GMT; path=/`;
  });

  window.location.href = '/goodbye';
}

8. Secure Forms

// Password field best practices
function LoginForm() {
  return (
    <form method="post" action="/api/login">
      <input
        type="email"
        name="email"
        autoComplete="email"           // Help password managers
        required
      />
      <input
        type="password"
        name="password"
        autoComplete="current-password" // Existing password
        minLength={8}
        required
      />
      <button type="submit">Login</button>
    </form>
  );
}

function RegistrationForm() {
  return (
    <form>
      <input
        type="password"
        name="new-password"
        autoComplete="new-password"     // Trigger password generator
        minLength={12}
        pattern="(?=.*\d)(?=.*[a-z])(?=.*[A-Z]).{12,}"
        title="At least 12 characters with uppercase, lowercase, and number"
        required
      />
    </form>
  );
}

// Credit card form security
function PaymentForm() {
  return (
    <form>
      <input
        type="text"
        name="cc-number"
        autoComplete="cc-number"
        inputMode="numeric"
        pattern="[0-9\s]{13,19}"
      />
      <input
        type="text"
        name="cc-exp"
        autoComplete="cc-exp"
        placeholder="MM/YY"
      />
      <input
        type="text"
        name="cc-csc"
        autoComplete="cc-csc"
        inputMode="numeric"
        maxLength={4}
      />
    </form>
  );
}

// Prevent form data in browser history
// Use POST instead of GET for sensitive forms
// Never put sensitive data in URL parameters

autoComplete Attribute Values

Proper autoComplete values help password managers and browsers: email, current-password (login), new-password (registration/change), one-time-code (2FA), cc-number, cc-exp, cc-csc. Using autoComplete="off" is generally discouraged — it breaks password managers and doesn't reliably prevent autofill. Instead, use the correct semantic values.

9. Clipboard API Security

// Clipboard API requires user gesture and secure context (HTTPS)

// Reading clipboard (requires permission)
async function pasteFromClipboard() {
  try {
    const text = await navigator.clipboard.readText();
    // Validate/sanitize before using
    return sanitizeInput(text);
  } catch (err) {
    // Permission denied or not in secure context
    console.error('Clipboard read failed:', err);
  }
}

// Writing to clipboard
async function copyToClipboard(text: string) {
  try {
    await navigator.clipboard.writeText(text);
  } catch (err) {
    // Fallback for older browsers
    const textarea = document.createElement('textarea');
    textarea.value = text;
    textarea.style.position = 'fixed';
    textarea.style.opacity = '0';
    document.body.appendChild(textarea);
    textarea.select();
    document.execCommand('copy');
    document.body.removeChild(textarea);
  }
}

// Security considerations:
// 1. Never auto-read clipboard without user action
// 2. Sanitize clipboard content before inserting into DOM
// 3. Be cautious with rich text paste (can contain HTML/scripts)
// 4. Clear sensitive data from clipboard after use
async function copyAndClearSensitive(text: string, clearAfterMs = 30000) {
  await navigator.clipboard.writeText(text);
  setTimeout(async () => {
    const current = await navigator.clipboard.readText();
    if (current === text) {
      await navigator.clipboard.writeText('');
    }
  }, clearAfterMs);
}

Architect Insight: Data Minimization

The most secure data is data you don't collect. At Staff level, advocate for data minimization: (1) only collect data you need for the feature, (2) delete data when it's no longer needed, (3) use tokenization for payment data (Stripe handles PCI compliance), (4) anonymize analytics data, (5) implement data retention policies. For credit card processing, NEVER handle raw card numbers — use Stripe Elements or similar that keep card data in an iframe on the payment provider's domain, never touching your servers.
