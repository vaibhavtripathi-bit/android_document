1. JWT (JSON Web Tokens)

1.1 JWT Structure

JWT Structure (three Base64URL-encoded parts):

  Header           Payload              Signature
  ┌──────────┐    ┌──────────────┐    ┌──────────────────────┐
  │ {        │    │ {            │    │ HMACSHA256(          │
  │  "alg":  │    │  "sub": "1", │    │   base64(header) +   │
  │   "HS256"│    │  "name":     │    │   "." +              │
  │  "typ":  │    │   "Alice",   │    │   base64(payload),   │
  │   "JWT"  │    │  "iat":      │    │   secret             │
  │ }        │    │   1700000000,│    │ )                    │
  └──────────┘    │  "exp":      │    └──────────────────────┘
       │          │   1700003600 │           │
       │          │ }            │           │
       │          └──────────────┘           │
       ▼                ▼                    ▼
  eyJhbGciOiJI.eyJzdWIiOiIxIiw.SflKxwRJSMeKKF2QT4fwp

  Sent as: Authorization: Bearer eyJhbGciOiJI...

// JWT creation (server-side)
import jwt from 'jsonwebtoken';

const accessToken = jwt.sign(
  {
    sub: user.id,           // Subject (user ID)
    email: user.email,
    role: user.role,
    iat: Math.floor(Date.now() / 1000),  // Issued at
  },
  process.env.JWT_SECRET,
  {
    expiresIn: '15m',       // Short-lived access token
    issuer: 'myapp.com',
    audience: 'myapp.com',
  }
);

// JWT verification (server-side)
function verifyToken(token) {
  try {
    const payload = jwt.verify(token, process.env.JWT_SECRET, {
      issuer: 'myapp.com',
      audience: 'myapp.com',
      algorithms: ['HS256'],  // Prevent algorithm confusion attacks
    });
    return payload;
  } catch (err) {
    if (err instanceof jwt.TokenExpiredError) {
      throw new Error('Token expired');
    }
    if (err instanceof jwt.JsonWebTokenError) {
      throw new Error('Invalid token');
    }
    throw err;
  }
}

JWT Security Pitfalls

(1) Algorithm confusion: always specify allowed algorithms in verification to prevent alg: "none" attacks. (2) Secret strength: use at least 256-bit secrets for HMAC; prefer RSA/ECDSA for distributed systems. (3) Payload is NOT encrypted: anyone can decode the payload — never put secrets in it. (4) No revocation: JWTs can't be invalidated before expiry without a blocklist. (5) Token size: JWTs grow with claims, adding overhead to every request.

1.2 Refresh Token Pattern

// Access token: short-lived (15 min), sent with every request
// Refresh token: long-lived (7-30 days), used only to get new access tokens

// Server: token pair generation
function generateTokenPair(user) {
  const accessToken = jwt.sign(
    { sub: user.id, role: user.role },
    process.env.JWT_SECRET,
    { expiresIn: '15m' }
  );

  const refreshToken = crypto.randomBytes(64).toString('hex');

  // Store refresh token in database (for revocation)
  db.refreshTokens.create({
    token: hashToken(refreshToken),
    userId: user.id,
    expiresAt: new Date(Date.now() + 30 * 24 * 60 * 60 * 1000),
  });

  return { accessToken, refreshToken };
}

// Server: refresh endpoint
app.post('/api/auth/refresh', async (req, res) => {
  const { refreshToken } = req.cookies;

  const stored = await db.refreshTokens.findOne({
    token: hashToken(refreshToken),
    expiresAt: { $gt: new Date() },
  });

  if (!stored) {
    return res.status(401).json({ error: 'Invalid refresh token' });
  }

  // Rotate refresh token (invalidate old, issue new)
  await db.refreshTokens.delete({ id: stored.id });

  const user = await db.users.findById(stored.userId);
  const tokens = generateTokenPair(user);

  res.cookie('refreshToken', tokens.refreshToken, {
    httpOnly: true, secure: true, sameSite: 'strict',
    path: '/api/auth/refresh',  // Only sent to refresh endpoint
    maxAge: 30 * 24 * 60 * 60 * 1000,
  });

  res.json({ accessToken: tokens.accessToken });
});

// Login

  const user = await authenticateUser(req.body);
  req.session.userId = user.id;
  req.session.role = user.role;

});

// Logout
app.post('/api/logout', (req, res) => {
  req.session.destroy((err) => {
    res.clearCookie('connect.sid');
    res.json({ success: true });
  });
});

// Auth middleware
function requireAuth(req, res, next) {
  if (!req.session.userId) {
    return res.status(401).json({ error: 'Not authenticated' });
  }
  next();
}

   │              │                  │                   │
   │              │──(8) API call────────────────────────▶│

// PKCE (Proof Key for Code Exchange) — required for SPAs
function generatePKCE() {
  const verifier = crypto.randomBytes(32)
    .toString('base64url');

  const challenge = crypto.createHash('sha256')
    .update(verifier)
    .digest('base64url');

  return { verifier, challenge };
}

// Initiate OAuth flow
function startOAuthLogin() {
  const { verifier, challenge } = generatePKCE();
  sessionStorage.setItem('pkce_verifier', verifier);

  const params = new URLSearchParams({
    client_id: 'your-client-id',
    redirect_uri: 'https://myapp.com/callback',
    response_type: 'code',
    scope: 'openid profile email',
    state: crypto.randomUUID(),  // CSRF protection
    code_challenge: challenge,
    code_challenge_method: 'S256',
  });

  window.location.href = `https://auth.example.com/authorize?${params}`;
}

// Handle callback
async function handleOAuthCallback(code) {
  const verifier = sessionStorage.getItem('pkce_verifier');
  sessionStorage.removeItem('pkce_verifier');

  const response = await fetch('https://auth.example.com/token', {
    method: 'POST',
    headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
    body: new URLSearchParams({
      grant_type: 'authorization_code',
      client_id: 'your-client-id',
      code,
      redirect_uri: 'https://myapp.com/callback',
      code_verifier: verifier,
    }),
  });

  const tokens = await response.json();
  return tokens; // { access_token, id_token, refresh_token }
}

// Recommended pattern: BFF (Backend for Frontend)
// Refresh token in httpOnly cookie → server
// Access token in memory → client

// Server (BFF) handles token exchange
app.post('/api/bff/login', async (req, res) => {
  const tokens = await authServer.login(req.body);

  // Store refresh token in httpOnly cookie
  res.cookie('refresh_token', tokens.refresh_token, {
    httpOnly: true,
    secure: true,
    sameSite: 'strict',
    path: '/api/bff',
    maxAge: 30 * 24 * 60 * 60 * 1000,
  });

  // Return access token to client (stored in memory)
  res.json({ accessToken: tokens.access_token });
});

// Client stores access token in memory
let accessToken = null;

async function login(credentials) {
  const response = await fetch('/api/bff/login', {
    method: 'POST',
    credentials: 'include',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(credentials),
  });
  const data = await response.json();
  accessToken = data.accessToken;  // In-memory only
}

// On page refresh, get new access token via refresh token cookie
async function refreshAccessToken() {
  const response = await fetch('/api/bff/refresh', {
    method: 'POST',
    credentials: 'include',
  });
  const data = await response.json();
  accessToken = data.accessToken;
}

Architect Insight: Token Storage Decision

The safest pattern for SPAs: (1) Store refresh tokens in httpOnly, Secure, SameSite=Strict cookies — JavaScript can't access them, so XSS can't steal them. (2) Store access tokens in memory (JavaScript variable) — they're lost on page refresh but can be silently renewed via the refresh token cookie. (3) Use a BFF (Backend for Frontend) to proxy auth requests, keeping client secrets server-side. This eliminates both XSS token theft and CSRF risks. Never store tokens in localStorage — any XSS vulnerability gives full access to the token.

5. Auth State Management in React

// Auth context with proper typing and loading states
interface AuthState {
  user: User | null;
  isAuthenticated: boolean;
  isLoading: boolean;
}

type AuthAction =
  | { type: 'LOGIN_START' }
  | { type: 'LOGIN_SUCCESS'; user: User }
  | { type: 'LOGIN_ERROR' }
  | { type: 'LOGOUT' }
  | { type: 'TOKEN_REFRESHED' };

function authReducer(state: AuthState, action: AuthAction): AuthState {
  switch (action.type) {
    case 'LOGIN_START':
      return { ...state, isLoading: true };
    case 'LOGIN_SUCCESS':
      return { user: action.user, isAuthenticated: true, isLoading: false };
    case 'LOGIN_ERROR':
      return { user: null, isAuthenticated: false, isLoading: false };
    case 'LOGOUT':
      return { user: null, isAuthenticated: false, isLoading: false };
    default:
      return state;
  }
}

function AuthProvider({ children }: { children: React.ReactNode }) {
  const [state, dispatch] = useReducer(authReducer, {
    user: null, isAuthenticated: false, isLoading: true,
  });

  useEffect(() => {
    // Check auth status on mount (silent token refresh)
    checkAuthStatus()
      .then(user => dispatch({ type: 'LOGIN_SUCCESS', user }))
      .catch(() => dispatch({ type: 'LOGIN_ERROR' }));
  }, []);

  const login = async (credentials: Credentials) => {
    dispatch({ type: 'LOGIN_START' });
    try {
      const user = await authApi.login(credentials);
      dispatch({ type: 'LOGIN_SUCCESS', user });
    } catch {
      dispatch({ type: 'LOGIN_ERROR' });
      throw new Error('Login failed');
    }
  };

  const logout = async () => {
    await authApi.logout();
    dispatch({ type: 'LOGOUT' });
  };

  return (
    <AuthContext.Provider value={{ ...state, login, logout }}>
      {children}
    </AuthContext.Provider>
  );
}

6. Protected Routes

// Protected route component
function ProtectedRoute({ children }: { children: React.ReactNode }) {
  const { isAuthenticated, isLoading } = useAuth();
  const location = useLocation();

  if (isLoading) {
    return <LoadingSpinner />;
  }

  if (!isAuthenticated) {
    return <Navigate to="/login" state={{ from: location }} replace />;
  }

  return <>{children}</>;
}

// Role-based protected route
function RequireRole({ role, children }: { role: string; children: React.ReactNode }) {
  const { user, isAuthenticated } = useAuth();

  if (!isAuthenticated) {
    return <Navigate to="/login" replace />;
  }

  if (user?.role !== role) {
    return <Navigate to="/unauthorized" replace />;
  }

  return <>{children}</>;
}

// Route configuration
function AppRoutes() {
  return (
    <Routes>
      <Route path="/login" element={<LoginPage />} />
      <Route path="/dashboard" element={
        <ProtectedRoute><Dashboard /></ProtectedRoute>
      } />
      <Route path="/admin" element={
        <RequireRole role="admin"><AdminPanel /></RequireRole>
      } />
    </Routes>
  );
}

7. SSR Authentication

// Next.js App Router: server-side auth check
// app/dashboard/page.tsx
import { cookies } from 'next/headers';
import { redirect } from 'next/navigation';

async function getUser() {
  const cookieStore = await cookies();
  const sessionToken = cookieStore.get('session');

  if (!sessionToken) return null;

  const user = await db.sessions.findUser(sessionToken.value);
  return user;
}

export default async function DashboardPage() {
  const user = await getUser();

  if (!user) {
    redirect('/login');
  }

  return <Dashboard user={user} />;
}

// Middleware-based auth (applies to multiple routes)
// middleware.ts
import { NextResponse } from 'next/server';
import type { NextRequest } from 'next/server';

export function middleware(request: NextRequest) {
  const session = request.cookies.get('session');

  if (!session && request.nextUrl.pathname.startsWith('/dashboard')) {
    return NextResponse.redirect(new URL('/login', request.url));
  }

  return NextResponse.next();
}

export const config = {
  matcher: ['/dashboard/:path*', '/admin/:path*', '/settings/:path*'],

8. NextAuth.js / Auth.js Architecture

// Auth.js (NextAuth v5) configuration
// auth.ts
import NextAuth from 'next-auth';
import Google from 'next-auth/providers/google';
import Credentials from 'next-auth/providers/credentials';
import { PrismaAdapter } from '@auth/prisma-adapter';

export const { handlers, auth, signIn, signOut } = NextAuth({
  adapter: PrismaAdapter(prisma),
  providers: [
    Google({
      clientId: process.env.GOOGLE_CLIENT_ID,
      clientSecret: process.env.GOOGLE_CLIENT_SECRET,
    }),
    Credentials({
      credentials: {
        email: { label: 'Email', type: 'email' },
        password: { label: 'Password', type: 'password' },
      },
      async authorize(credentials) {
        const user = await db.users.findByEmail(credentials.email);
        if (!user || !await verifyPassword(credentials.password, user.hash)) {
          return null;
        }
        return { id: user.id, email: user.email, name: user.name };
      },
    }),
  ],
  callbacks: {
    async jwt({ token, user }) {
      if (user) {
        token.role = user.role;
      }
      return token;
    },
    async session({ session, token }) {
      session.user.role = token.role;
      return session;
    },
    async authorized({ auth, request }) {
      return !!auth?.user; // Middleware-level auth check
    },
  },
  pages: {
    signIn: '/login',
    error: '/auth/error',
  },
});

// Using auth in Server Components
import { auth } from '@/auth';

export default async function Page() {
  const session = await auth();
  if (!session) redirect('/login');
  return <div>Welcome, {session.user.name}</div>;
}

Auth.js Architecture

Auth.js uses a layered architecture: (1) Providers handle identity verification (OAuth, credentials, email), (2) Adapters persist user/session data to databases (Prisma, Drizzle, MongoDB), (3) Callbacks customize token/session data, (4) Middleware provides route-level protection. Sessions can be JWT-based (stateless, stored in cookie) or database-based (session ID in cookie, data in DB). For most applications, JWT sessions are simpler; database sessions are needed when you require server-side session revocation.
