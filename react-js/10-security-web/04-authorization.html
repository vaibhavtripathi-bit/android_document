2. Role-Based Access Control (RBAC)

// RBAC: permissions assigned to roles, roles assigned to users
// Simple but effective for most applications

// Role definitions
const ROLES = {
  viewer: {
    permissions: ['read:articles', 'read:comments'],
  },
  editor: {
    permissions: [
      'read:articles', 'write:articles', 'read:comments', 'write:comments',
    ],
  },
  admin: {
    permissions: [
      'read:articles', 'write:articles', 'delete:articles',
      'read:comments', 'write:comments', 'delete:comments',
      'manage:users', 'manage:settings',
    ],
  },
} as const;

type Role = keyof typeof ROLES;
type Permission = typeof ROLES[Role]['permissions'][number];

// Permission checking utility
function hasPermission(userRole: Role, permission: Permission): boolean {
  return ROLES[userRole]?.permissions.includes(permission) ?? false;
}

// React hook
function usePermission(permission: Permission): boolean {
  const { user } = useAuth();
  if (!user) return false;
  return hasPermission(user.role, permission);
}

// Component-level authorization
function ArticleActions({ articleId }: { articleId: string }) {
  const canEdit = usePermission('write:articles');
  const canDelete = usePermission('delete:articles');

  return (
    <div>
      {canEdit && <button onClick={() => editArticle(articleId)}>Edit</button>}
      {canDelete && <button onClick={() => deleteArticle(articleId)}>Delete</button>}
    </div>
  );
}

3. Attribute-Based Access Control (ABAC)

// ABAC: decisions based on attributes of user, resource, action, and environment
// More flexible than RBAC for complex scenarios

interface PolicyContext {
  user: { id: string; role: string; department: string; clearanceLevel: number };
  resource: { type: string; ownerId: string; classification: string; department: string };
  action: string;
  environment: { time: Date; ipAddress: string; location: string };
}

type PolicyRule = (ctx: PolicyContext) => boolean;

const policies: Record<string, PolicyRule[]> = {
  'read:document': [
    // Users can read documents in their department
    (ctx) => ctx.user.department === ctx.resource.department,
    // Or if they have sufficient clearance
    (ctx) => ctx.user.clearanceLevel >= getClearanceRequired(ctx.resource.classification),
  ],
  'edit:document': [
    // Only the owner can edit
    (ctx) => ctx.user.id === ctx.resource.ownerId,
    // Or admins in the same department
    (ctx) => ctx.user.role === 'admin' && ctx.user.department === ctx.resource.department,
  ],
  'delete:document': [
    // Only admins can delete, and only during business hours
    (ctx) => {
      const hour = ctx.environment.time.getHours();
      return ctx.user.role === 'admin' && hour >= 9 && hour <= 17;
    },
  ],
};

function evaluatePolicy(action: string, ctx: PolicyContext): boolean {
  const rules = policies[action];
  if (!rules) return false;
  return rules.some(rule => rule(ctx)); // OR logic: any rule can grant access
}

// React hook for ABAC
function useAbacPermission(action: string, resource: Resource): boolean {
  const { user } = useAuth();
  if (!user) return false;

  return evaluatePolicy(action, {
    user,
    resource,
    action,
    environment: {
      time: new Date(),
      ipAddress: '', // Server-side only
      location: '',
    },
  });
}

4. Permission Checking in React Components

// Permission gate component
interface PermissionGateProps {
  permission: Permission;
  fallback?: React.ReactNode;
  children: React.ReactNode;
}

function PermissionGate({ permission, fallback = null, children }: PermissionGateProps) {
  const hasAccess = usePermission(permission);
  return hasAccess ? <>{children}</> : <>{fallback}</>;
}

// Usage
function Dashboard() {
  return (
    <div>
      <h1>Dashboard</h1>
      <PermissionGate permission="read:analytics">
        <AnalyticsPanel />
      </PermissionGate>

      <PermissionGate
        permission="manage:users"
        fallback={<p>Contact admin for user management.</p>}
      >
        <UserManagement />
      </PermissionGate>
    </div>
  );
}

// Higher-order component pattern
function withPermission<P extends object>(
  Component: React.ComponentType<P>,
  permission: Permission,
  Fallback: React.ComponentType = () => <div>Access Denied</div>
) {
  return function PermissionWrapper(props: P) {
    const hasAccess = usePermission(permission);
    return hasAccess ? <Component {...props} /> : <Fallback />;
  };
}

const ProtectedAdminPanel = withPermission(AdminPanel, 'manage:settings');

5. Route-Level Authorization

// Route configuration with permissions
interface RouteConfig {
  path: string;
  element: React.ReactNode;
  requiredPermission?: Permission;
  requiredRole?: Role;
}

const routes: RouteConfig[] = [
  { path: '/', element: <Home /> },
  { path: '/articles', element: <Articles />, requiredPermission: 'read:articles' },
  { path: '/admin', element: <Admin />, requiredRole: 'admin' },
  { path: '/settings', element: <Settings />, requiredPermission: 'manage:settings' },
];

function AuthorizedRoute({ config }: { config: RouteConfig }) {
  const { user, isAuthenticated, isLoading } = useAuth();

  if (isLoading) return <LoadingSpinner />;
  if (!isAuthenticated) return <Navigate to="/login" />;

  if (config.requiredPermission && !hasPermission(user.role, config.requiredPermission)) {
    return <Navigate to="/unauthorized" />;
  }

  if (config.requiredRole && user.role !== config.requiredRole) {
    return <Navigate to="/unauthorized" />;
  }

  return <>{config.element}</>;
}

// Dynamic navigation based on permissions
function Navigation() {
  const { user } = useAuth();

  const visibleRoutes = routes.filter(route => {
    if (!route.requiredPermission) return true;
    return hasPermission(user.role, route.requiredPermission);
  });

  return (
    <nav>
      {visibleRoutes.map(route => (
        <Link key={route.path} to={route.path}>{route.path}</Link>
      ))}
    </nav>
  );
}

6. API-Level Authorization

// Server-side authorization middleware (Express)
function authorize(...requiredPermissions: Permission[]) {
  return (req, res, next) => {
    const user = req.user; // Set by auth middleware

    if (!user) {
      return res.status(401).json({ error: 'Not authenticated' });
    }

    const hasAllPermissions = requiredPermissions.every(
      perm => hasPermission(user.role, perm)
    );

    if (!hasAllPermissions) {
      return res.status(403).json({
        error: 'Insufficient permissions',
        required: requiredPermissions,
      });
    }

    next();
  };
}

// Usage
app.get('/api/articles', authorize('read:articles'), getArticles);
app.post('/api/articles', authorize('write:articles'), createArticle);
app.delete('/api/articles/:id', authorize('delete:articles'), deleteArticle);

// Resource-level authorization (ownership check)
app.put('/api/articles/:id', authorize('write:articles'), async (req, res) => {
  const article = await db.articles.findById(req.params.id);

  if (!article) {
    return res.status(404).json({ error: 'Not found' });
  }

  // Admins can edit any article; others only their own
  if (req.user.role !== 'admin' && article.authorId !== req.user.id) {
    return res.status(403).json({ error: 'Cannot edit others\' articles' });
  }

  const updated = await db.articles.update(req.params.id, req.body);
  res.json(updated);
});

Architect Insight: Defense in Depth

Authorization must be enforced at EVERY layer: (1) UI layer — hide/disable unauthorized elements (UX, not security), (2) Route layer — redirect unauthorized users (UX, not security), (3) API layer — reject unauthorized requests (actual security), (4) Database layer — row-level security policies (defense in depth). The UI and route layers are for user experience only — they can be bypassed by anyone with browser DevTools. The API layer is the real security boundary. Never trust the client to enforce authorization.

7. CASL Library for React

// CASL — isomorphic authorization library
import { AbilityBuilder, createMongoAbility } from '@casl/ability';
import { Can, AbilityContext } from '@casl/react';

// Define abilities based on user role
function defineAbilitiesFor(user) {
  const { can, cannot, build } = new AbilityBuilder(createMongoAbility);

  if (user.role === 'admin') {
    can('manage', 'all'); // Can do anything
    cannot('delete', 'User', { id: user.id }); // But can't delete self
  }

  if (user.role === 'editor') {
    can('read', 'Article');
    can('create', 'Article');
    can('update', 'Article', { authorId: user.id }); // Own articles only
    can('read', 'Comment');
    can('create', 'Comment');
    can('delete', 'Comment', { authorId: user.id }); // Own comments only
  }

  if (user.role === 'viewer') {
    can('read', 'Article');
    can('read', 'Comment');
  }

  return build();
}

// React integration
function App() {
  const { user } = useAuth();
  const ability = useMemo(() => defineAbilitiesFor(user), [user]);

  return (
    <AbilityContext.Provider value={ability}>
      <Router>
        <AppRoutes />
      </Router>
    </AbilityContext.Provider>
  );
}

// Using Can component
function ArticleItem({ article }) {
  return (
    <div>
      <h2>{article.title}</h2>
      <Can I="update" this={article}>
        <button>Edit</button>
      </Can>
      <Can I="delete" this={article}>
        <button>Delete</button>
      </Can>
    </div>
  );
}

// Using useAbility hook
function ArticleActions({ article }) {
  const ability = useAbility(AbilityContext);

  return (
    <div>
      {ability.can('update', article) && <EditButton />}
      {ability.can('delete', article) && <DeleteButton />}
    </div>
  );
}

8. Server-Side Authorization in RSC

// Next.js App Router: authorization in Server Components
// No client-side JavaScript needed for auth checks

// lib/auth.ts
import { auth } from '@/auth';
import { cache } from 'react';

export const getCurrentUser = cache(async () => {
  const session = await auth();
  if (!session?.user) return null;
  return session.user;
});

export async function requirePermission(permission: string) {
  const user = await getCurrentUser();
  if (!user) throw new Error('Not authenticated');
  if (!hasPermission(user.role, permission)) {
    throw new Error('Insufficient permissions');
  }
  return user;
}

// Server Component with authorization
async function AdminDashboard() {
  const user = await requirePermission('manage:settings');

  const stats = await db.getAdminStats();

  return (
    <div>
      <h1>Admin Dashboard</h1>
      <p>Welcome, {user.name}</p>
      <StatsGrid stats={stats} />
    </div>
  );
}

// Server Action with authorization
'use server';

export async function deleteUser(userId: string) {
  const admin = await requirePermission('manage:users');

  if (admin.id === userId) {
    throw new Error('Cannot delete yourself');
  }

  await db.users.delete(userId);
  revalidatePath('/admin/users');
}

9. Feature Flags and Authorization

// Feature flags control feature visibility — distinct from authorization
// Authorization: "Is the user ALLOWED to access this?"
// Feature flags: "Is this feature ENABLED for this user?"

interface FeatureFlag {
  name: string;
  enabled: boolean;
  rules: FeatureRule[];
}

type FeatureRule =
  | { type: 'percentage'; value: number }
  | { type: 'userIds'; value: string[] }
  | { type: 'roles'; value: string[] }
  | { type: 'environment'; value: string[] };

function isFeatureEnabled(flag: FeatureFlag, user: User): boolean {
  if (!flag.enabled) return false;

  return flag.rules.some(rule => {
    switch (rule.type) {
      case 'percentage':
        return hashUserId(user.id) % 100 < rule.value;
      case 'userIds':
        return rule.value.includes(user.id);
      case 'roles':
        return rule.value.includes(user.role);
      case 'environment':
        return rule.value.includes(process.env.NODE_ENV);
      default:
        return false;
    }
  });
}

// React hook
function useFeatureFlag(flagName: string): boolean {
  const { user } = useAuth();
  const flags = useFeatureFlags(); // From context or API
  const flag = flags.find(f => f.name === flagName);
  if (!flag || !user) return false;
  return isFeatureEnabled(flag, user);
}

// Usage
function Dashboard() {
  const showNewAnalytics = useFeatureFlag('new-analytics-dashboard');

  return (
    <div>
      {showNewAnalytics ? <NewAnalytics /> : <LegacyAnalytics />}
    </div>
  );
}

10. RBAC vs ABAC Comparison

AspectRBACABAC
ComplexitySimpleComplex
FlexibilityLimited (predefined roles)High (any attribute combination)
ScalabilityRole explosion with fine-grained needsScales with attributes
AuditabilityEasy (check role assignments)Hard (complex policy evaluation)
PerformanceFast (simple lookup)Slower (policy evaluation)
Best forMost web appsEnterprise, multi-tenant, healthcare
ImplementationRole → permissions mapPolicy engine (OPA, Cedar)

When to Use ABAC

RBAC suffices for most React applications. Consider ABAC when: (1) you need resource-level permissions (user can edit THEIR articles but not others'), (2) permissions depend on context (time of day, location, device), (3) you have multi-tenant applications where tenant-specific rules vary, (4) regulatory requirements demand fine-grained audit trails. Many applications use a hybrid: RBAC for coarse-grained access (admin/editor/viewer) with ABAC-style ownership checks for resource-level decisions.
