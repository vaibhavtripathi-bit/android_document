Jest Configuration for React

// jest.config.ts — comprehensive React configuration
import type { Config } from 'jest';

const config: Config = {
  // Test environment
  testEnvironment: 'jsdom',

  // File discovery
  roots: ['<rootDir>/src'],
  testMatch: [
    '**/__tests__/**/*.{ts,tsx}',
    '**/*.{test,spec}.{ts,tsx}',
  ],
  testPathIgnorePatterns: ['/node_modules/', '/dist/', '/e2e/'],

  // Module resolution
  moduleNameMapper: {
    // Path aliases (match tsconfig paths)
    '^@/(.*)$': '<rootDir>/src/$1',
    '^@components/(.*)$': '<rootDir>/src/components/$1',

    // Static file mocks
    '\\.(css|less|scss|sass)$': 'identity-obj-proxy',
    '\\.(jpg|jpeg|png|gif|svg)$': '<rootDir>/__mocks__/fileMock.js',
  },

  // Transforms
  transform: {
    '^.+\\.tsx?$': ['@swc/jest', {
      jsc: {
        transform: {
          react: { runtime: 'automatic' },
        },
      },
    }],
  },

  // Setup files
  setupFilesAfterSetup: ['<rootDir>/jest.setup.ts'],

  // Coverage
  collectCoverageFrom: [
    'src/**/*.{ts,tsx}',
    '!src/**/*.d.ts',
    '!src/**/*.stories.{ts,tsx}',
    '!src/**/index.ts', // barrel exports
    '!src/test-utils/**',
  ],
  coverageThreshold: {
    global: {
      branches: 70,
      functions: 75,
      lines: 80,
      statements: 80,
    },
  },

  // Performance
  maxWorkers: '50%', // Use half of available CPUs
  cacheDirectory: '<rootDir>/.jest-cache',
};

export default config;
// jest.setup.ts — global test setup
import '@testing-library/jest-dom';
import { server } from './mocks/server';

// MSW setup
beforeAll(() => server.listen({ onUnhandledRequest: 'error' }));
afterEach(() => server.resetHandlers());
afterAll(() => server.close());

// Clean up after each test
afterEach(() => {
  jest.restoreAllMocks();
});

// Suppress specific console warnings in tests
const originalError = console.error;
console.error = (...args) => {
  if (
    typeof args[0] === 'string' &&
    args[0].includes('Not wrapped in act')
  ) {
    return; // Suppress known React testing warnings
  }
  originalError.call(console, ...args);
};

Vitest as Jest Alternative

Vitest is a Vite-native test runner that provides Jest-compatible APIs with significantly faster execution. It leverages Vite's transform pipeline, eliminating the need for separate Babel/SWC configuration.

FeatureJestVitest
TransformBabel/SWC (separate config)Vite (shared config)
SpeedModerate (cold start ~2-5s)Fast (instant HMR-like)
ESM supportExperimentalNative
Watch modeFile-basedHMR-based (instant re-run)
API compatibilityN/A (the standard)Jest-compatible (drop-in)
TypeScriptRequires ts-jest or @swc/jestBuilt-in via Vite
CSS handlingRequires moduleNameMapperBuilt-in via Vite plugins
SnapshotBuilt-inBuilt-in (compatible format)
CoverageIstanbul or V8Istanbul or V8 (c8)
 
// vitest.config.ts
import { defineConfig } from 'vitest/config';
import react from '@vitejs/plugin-react';
import tsconfigPaths from 'vite-tsconfig-paths';

export default defineConfig({
  plugins: [react(), tsconfigPaths()],
  test: {
    environment: 'jsdom',
    globals: true, // Use global describe, test, expect
    setupFiles: ['./vitest.setup.ts'],

    // CSS handling (built-in)
    css: true,

    // Coverage
    coverage: {
      provider: 'v8', // or 'istanbul'
      reporter: ['text', 'json', 'html', 'lcov'],
      include: ['src/**/*.{ts,tsx}'],
      exclude: [
        'src/**/*.d.ts',
        'src/**/*.stories.tsx',
        'src/test-utils/**',
      ],
      thresholds: {
        branches: 70,
        functions: 75,
        lines: 80,
        statements: 80,
      },
    },

    // Performance
    pool: 'threads', // or 'forks' for isolation
    poolOptions: {
      threads: { maxThreads: 4, minThreads: 1 },
    },
  },
});

Babel/SWC Transforms

// SWC transform (fastest) — @swc/jest
// jest.config.ts
{
  transform: {
    '^.+\\.tsx?$': ['@swc/jest', {
      jsc: {
        parser: {
          syntax: 'typescript',
          tsx: true,
          decorators: true,
        },
        transform: {
          react: {
            runtime: 'automatic',
          },
        },
      },
    }],
  },
}

// Babel transform (most compatible) — babel-jest
// jest.config.ts
{
  transform: {
    '^.+\\.tsx?$': 'babel-jest',
  },
}

// babel.config.js
module.exports = {
  presets: [
    ['@babel/preset-env', { targets: { node: 'current' } }],
    ['@babel/preset-react', { runtime: 'automatic' }],
    '@babel/preset-typescript',
  ],
};

// ts-jest (TypeScript-native, slower but type-checked)
{
  transform: {
    '^.+\\.tsx?$': ['ts-jest', {
      tsconfig: 'tsconfig.test.json',
      useESM: true,
    }],
  },
}
Transform Speed Comparison

For a typical React project with 500 test files: SWC transforms in ~2s, Babel in ~8s, ts-jest in ~15s. SWC is the recommended transform for Jest. Vitest avoids this entirely by reusing Vite's transform pipeline.

Code Coverage

// Istanbul vs V8 coverage providers

// Istanbul: Source-level instrumentation
// - Instruments code at transform time
// - More accurate for branch coverage
// - Slower (adds instrumentation overhead)
// - Works with all transforms

// V8: Engine-level coverage
// - Uses V8's built-in coverage
// - Faster (no instrumentation overhead)
// - Less accurate for some edge cases
// - Requires Node.js 10.12+

// Coverage configuration
{
  coverageProvider: 'v8', // or 'istanbul'
  coverageReporters: [
    'text',           // Terminal output
    'text-summary',   // Summary in terminal
    'json',           // Machine-readable
    'html',           // Browser-viewable report
    'lcov',           // For CI tools (Codecov, Coveralls)
    'cobertura',      // For Azure DevOps
  ],
  collectCoverageFrom: [
    'src/**/*.{ts,tsx}',
    '!src/**/*.d.ts',
    '!src/**/*.test.{ts,tsx}',
    '!src/**/*.stories.{ts,tsx}',
    '!src/test-utils/**',
    '!src/mocks/**',
    '!src/**/index.ts',
  ],
  coverageThreshold: {
    global: {
      branches: 70,
      functions: 75,
      lines: 80,
      statements: 80,
    },
    // Per-directory thresholds
    './src/utils/': {
      branches: 90,
      functions: 95,
      lines: 95,
    },
    // Per-file thresholds
    './src/lib/auth.ts': {
      branches: 100,
      functions: 100,
    },
  },
}

CI Pipeline Setup

# .github/workflows/test.yml — comprehensive CI pipeline
name: Test Suite
on:
  push:
    branches: [main]
  pull_request:
    branches: [main]

jobs:
  # Type checking (fastest feedback)
  typecheck:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-node@v4
        with: { node-version: 20, cache: 'npm' }
      - run: npm ci
      - run: npx tsc --noEmit

  # Linting
  lint:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-node@v4
        with: { node-version: 20, cache: 'npm' }
      - run: npm ci
      - run: npx eslint src/ --max-warnings 0

  # Unit + Integration tests (parallel shards)
  test:
    runs-on: ubuntu-latest
    strategy:
      matrix:
        shard: [1, 2, 3, 4]
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-node@v4
        with: { node-version: 20, cache: 'npm' }
      - run: npm ci

      - name: Run tests (shard ${{ matrix.shard }}/4)
        run: npx jest --shard=${{ matrix.shard }}/4 --coverage

      - name: Upload coverage
        uses: actions/upload-artifact@v4
        with:
          name: coverage-${{ matrix.shard }}
          path: coverage/

  # Merge coverage from all shards
  coverage:
    needs: test
    runs-on: ubuntu-latest
    steps:
      - uses: actions/download-artifact@v4
      - name: Merge and upload coverage
        run: npx istanbul-merge --out merged/coverage.json coverage-*/coverage-final.json
      - uses: codecov/codecov-action@v3
        with:
          file: merged/coverage.json

  # E2E tests (separate job, heavier)
  e2e:
    needs: [typecheck, lint]
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-node@v4
        with: { node-version: 20, cache: 'npm' }
      - run: npm ci
      - run: npx playwright install --with-deps
      - run: npm run build
      - run: npx playwright test
      - uses: actions/upload-artifact@v4
        if: failure()
        with:
          name: playwright-report
          path: playwright-report/

Parallel Test Execution

// Jest parallel execution
// jest.config.ts
{
  // Use 50% of CPUs (good for CI with shared resources)
  maxWorkers: '50%',

  // Or specify exact number
  maxWorkers: 4,

  // Run tests in band (sequentially) for debugging
  // jest --runInBand
}

// Playwright parallel execution
// playwright.config.ts
{
  // Number of parallel workers
  workers: process.env.CI ? 4 : undefined, // undefined = half of CPUs

  // Fully parallel (tests within a file run in parallel)
  fullyParallel: true,

  // Retry flaky tests
  retries: process.env.CI ? 2 : 0,
}

// Test splitting strategies for CI

Monorepo Testing

// Monorepo test configuration with workspaces
// Root jest.config.ts
{
  projects: [
    '<rootDir>/packages/ui',
    '<rootDir>/packages/utils',
    '<rootDir>/apps/web',
    '<rootDir>/apps/admin',
  ],
}

// Each package has its own jest.config.ts
// packages/ui/jest.config.ts
{
  displayName: 'ui',
  testEnvironment: 'jsdom',
  roots: ['<rootDir>/src'],
}

// Run specific project
// jest --selectProjects ui

// Run only affected packages (with Turborepo)
// turbo run test --filter=...[HEAD^1]

// Turborepo configuration
// turbo.json
{
  "pipeline": {
    "test": {
      "dependsOn": ["^build"],
      "outputs": ["coverage/**"],
      "cache": true
    },
    "test:ci": {
      "dependsOn": ["^build"],
      "outputs": ["coverage/**"],
      "cache": false
    }
  }
}

// Nx affected tests
// nx affected:test --base=main --head=HEAD

Pre-Commit Hooks

// package.json — lint-staged + husky setup
{
  "scripts": {
    "prepare": "husky install"
  },
  "lint-staged": {
    "*.{ts,tsx}": [
      "eslint --fix --max-warnings 0",
      "prettier --write"
    ],
    "*.{ts,tsx}": [
      "jest --bail --findRelatedTests"
    ],
    "*.css": [
      "stylelint --fix",
      "prettier --write"
    ]
  }
}

// .husky/pre-commit
#!/usr/bin/env sh
. "$(dirname -- "$0")/_/husky.sh"
npx lint-staged

// .husky/pre-push
#!/usr/bin/env sh
. "$(dirname -- "$0")/_/husky.sh"
npm run typecheck
npm run test:ci
// jest --findRelatedTests: runs only tests affected by staged files
// This keeps pre-commit hooks fast (seconds, not minutes)

// Example: you changed src/components/Button.tsx
// jest --findRelatedTests src/components/Button.tsx
// Runs: Button.test.tsx, Form.test.tsx (imports Button), etc.

// For larger changes, run full suite on pre-push instead
// Pre-commit: lint + format + related tests (~5-15s)
// Pre-push: typecheck + full test suite (~1-3min)
Architect Insight

Testing infrastructure is a force multiplier. A well-configured CI pipeline with parallel execution, intelligent test splitting, and cached builds can run 5000 tests in under 5 minutes. Without it, the same suite takes 30+ minutes, developers skip running tests locally, and CI becomes a bottleneck. At L6, investing in testing infrastructure (fast CI, reliable test environments, good developer experience) has higher ROI than writing more tests.
