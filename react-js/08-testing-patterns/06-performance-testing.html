React Profiler API

The React Profiler API measures rendering performance programmatically. It can be used in tests to assert that components render within acceptable time bounds and don't re-render unnecessarily.

import { Profiler } from 'react';
import { render, screen } from '@testing-library/react';

// Profiler callback captures render metrics
function onRender(
  id: string,
  phase: 'mount' | 'update',
  actualDuration: number,
  baseDuration: number,
  startTime: number,
  commitTime: number
) {
  // actualDuration: time spent rendering this update
  // baseDuration: estimated time for full re-render without memoization
  // phase: 'mount' for initial render, 'update' for re-renders
}

test('ProductList renders within performance budget', async () => {
  const renderMetrics: Array<{
    phase: string;
    actualDuration: number;
  }> = [];

  function captureMetrics(
    id: string,
    phase: string,
    actualDuration: number
  ) {
    renderMetrics.push({ phase, actualDuration });
  }

  render(
    <Profiler id="ProductList" onRender={captureMetrics}>
      <ProductList products={generateProducts(100)} />
    </Profiler>
  );

  await screen.findByText('Product 1');

  // Assert initial render is fast enough
  const mountMetric = renderMetrics.find(m => m.phase === 'mount');
  expect(mountMetric?.actualDuration).toBeLessThan(50); // 50ms budget
});

// Testing that memoization prevents unnecessary re-renders
test('memoized child does not re-render on parent state change', async () => {
  const childRenders: number[] = [];

  function ChildProfiler({ children }) {
    return (
      <Profiler
        id="child"
        onRender={(id, phase, duration) => {
          childRenders.push(duration);
        }}
      >
        {children}
      </Profiler>
    );
  }

  render(
    <ParentWithState>
      <ChildProfiler>
        <MemoizedExpensiveChild data={staticData} />
      </ChildProfiler>
    </ParentWithState>
  );

  // Initial render
  expect(childRenders).toHaveLength(1);

  // Trigger parent state change
  await userEvent.click(screen.getByRole('button', { name: /toggle/i }));

  // Child should NOT have re-rendered
  expect(childRenders).toHaveLength(1);
});

// Import cost testing — ensure specific imports are tree-shaken
// tests/bundle/tree-shaking.test.ts
import { execSync } from 'child_process';

test('lodash is tree-shaken (only used functions bundled)', () => {
  const output = execSync('npx source-map-explorer dist/static/js/main*.js --json')
    .toString();
  const analysis = JSON.parse(output);

  const lodashSize = Object.entries(analysis.results[0].files)
    .filter(([path]) => path.includes('lodash'))
    .reduce((sum, [, data]) => sum + data.size, 0);

  // Should be small (individual functions) not full lodash
  expect(lodashSize).toBeLessThan(10000); // 10KB max
});

Memory Leak Detection

// Detecting memory leaks from event listeners
test('cleanup removes all event listeners', () => {
  const addSpy = jest.spyOn(window, 'addEventListener');
  const removeSpy = jest.spyOn(window, 'removeEventListener');

  const { unmount } = render(<WindowEventComponent />);

  const addedEvents = addSpy.mock.calls.map(([event]) => event);
  unmount();
  const removedEvents = removeSpy.mock.calls.map(([event]) => event);

  // Every added listener should be removed
  for (const event of addedEvents) {
    expect(removedEvents).toContain(event);
  }

  addSpy.mockRestore();
  removeSpy.mockRestore();
});

// Detecting memory leaks from subscriptions
test('unsubscribes from store on unmount', () => {
  const unsubscribe = jest.fn();
  const subscribe = jest.fn().mockReturnValue(unsubscribe);
  const store = { subscribe, getState: () => ({}) };

  const { unmount } = render(
    <StoreProvider store={store}>
      <ConnectedComponent />
    </StoreProvider>
  );

  expect(subscribe).toHaveBeenCalled();

  unmount();
  expect(unsubscribe).toHaveBeenCalled();
});

// Detecting state updates after unmount
test('no state updates after unmount', async () => {
  const consoleSpy = jest.spyOn(console, 'error').mockImplementation();

  server.use(
    http.get('/api/data', async () => {
      await delay(100);
      return HttpResponse.json({ data: 'test' });
    })
  );

  const { unmount } = render(<AsyncComponent />);

  // Unmount before fetch completes
  unmount();

  // Wait for the fetch to complete
  await new Promise(resolve => setTimeout(resolve, 200));

  // Should not see "Can't perform a React state update on unmounted component"
  const stateUpdateWarnings = consoleSpy.mock.calls.filter(
    call => call[0]?.includes?.('unmounted')
  );
  expect(stateUpdateWarnings).toHaveLength(0);

  consoleSpy.mockRestore();
});

Core Web Vitals Testing

// Automated CWV testing with web-vitals library
import { onCLS, onFCP, onLCP, onTTFB } from 'web-vitals';

// In E2E tests (Playwright)
test('Core Web Vitals meet thresholds', async ({ page }) => {
  // Inject web-vitals measurement
  await page.addInitScript(() => {
    window.__webVitals = {};
    // These will be populated by the web-vitals library
  });

  await page.goto('/');
  await page.waitForLoadState('networkidle');

  // Collect metrics
  const metrics = await page.evaluate(() => {
    return new Promise(resolve => {
      const results = {};
      new PerformanceObserver(list => {
        for (const entry of list.getEntries()) {
          if (entry.entryType === 'largest-contentful-paint') {
            results.lcp = entry.startTime;
          }
        }
      }).observe({ type: 'largest-contentful-paint', buffered: true });

      new PerformanceObserver(list => {
        let cls = 0;
        for (const entry of list.getEntries()) {
          if (!entry.hadRecentInput) cls += entry.value;
        }
        results.cls = cls;
      }).observe({ type: 'layout-shift', buffered: true });

      const fcp = performance.getEntriesByName('first-contentful-paint')[0];
      results.fcp = fcp?.startTime ?? 0;

      const nav = performance.getEntriesByType('navigation')[0];
      results.ttfb = nav?.responseStart ?? 0;

      setTimeout(() => resolve(results), 3000);
    });
  });

  // Assert thresholds (Good ratings per Google)
  expect(metrics.lcp).toBeLessThan(2500);   // LCP < 2.5s
  expect(metrics.fcp).toBeLessThan(1800);   // FCP < 1.8s
  expect(metrics.cls).toBeLessThan(0.1);    // CLS < 0.1
  expect(metrics.ttfb).toBeLessThan(800);   // TTFB < 800ms
});

Regression Detection

// Track performance metrics over time
// Compare against baseline (previous release)

// performance-regression.test.ts
import baseline from './performance-baseline.json';

test('no performance regression from baseline', async () => {
  const current = await measurePerformance('/');

  // Allow 10% regression tolerance
  const tolerance = 1.1;

  expect(current.lcp).toBeLessThan(baseline.lcp * tolerance);
  expect(current.fcp).toBeLessThan(baseline.fcp * tolerance);
  expect(current.bundleSize).toBeLessThan(baseline.bundleSize * tolerance);
  expect(current.renderTime).toBeLessThan(baseline.renderTime * tolerance);

  // CLS should not regress at all
  expect(current.cls).toBeLessThanOrEqual(baseline.cls);
});

Architect Insight

Performance testing is most valuable as a regression gate, not an absolute measure. A single Lighthouse score varies by 5-10% between runs. Instead, track trends: is the bundle getting bigger? Is LCP increasing? Are render times growing? Set budgets with enough headroom for variance, and investigate when metrics trend upward. At Google's scale, automated performance regression detection in CI prevents the "death by a thousand cuts" that slowly degrades user experience.
