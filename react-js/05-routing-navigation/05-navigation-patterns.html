1. Programmatic Navigation

Programmatic navigation is triggered by code rather than user link clicks. Common use cases include post-login redirects, form submission redirects, and conditional navigation.

// useNavigate — the primary hook for programmatic navigation
import { useNavigate } from 'react-router-dom';

function LoginForm() {
  const navigate = useNavigate();
  const location = useLocation();

  async function handleSubmit(e) {
    e.preventDefault();
    const formData = new FormData(e.target);
    await login(formData);

    // Redirect to where user came from, or default to dashboard
    const from = location.state?.from ?? '/dashboard';
    navigate(from, { replace: true });  // replace login page in history
  }

  return <form onSubmit={handleSubmit}>...</form>;
}

// Navigation in loaders/actions (outside React components)
import { redirect } from 'react-router-dom';

async function loginAction({ request }) {
  const formData = await request.formData();
  const result = await authenticate(formData);

  if (!result.success) {
    return { error: result.message };
  }

  const url = new URL(request.url);
  const redirectTo = url.searchParams.get('redirectTo') ?? '/dashboard';
  return redirect(redirectTo);  // returns a Response with 302 status
}

// useBlocker — prevent navigation (unsaved changes)
function EditForm() {
  const [isDirty, setIsDirty] = useState(false);

  const blocker = useBlocker(
    ({ currentLocation, nextLocation }) =>
      isDirty && currentLocation.pathname !== nextLocation.pathname
  );

  return (
    <div>
      <form onChange={() => setIsDirty(true)}>...</form>
      {blocker.state === 'blocked' && (
        <dialog open>
          <p>You have unsaved changes. Leave anyway?</p>
          <button onClick={() => blocker.proceed()}>Leave</button>
          <button onClick={() => blocker.reset()}>Stay</button>
        </dialog>
      )}
    </div>
  );
}

2. Protected Routes & Auth Guards

// Pattern 1: Wrapper component (simple)
function ProtectedRoute({ children }) {
  const { user, isLoading } = useAuth();
  const location = useLocation();

  if (isLoading) return <LoadingScreen />;

  if (!user) {
    return <Navigate to="/login" state={{ from: location.pathname }} replace />;
  }

  return children;
}

// Usage in routes
{
  path: 'dashboard',
  element: <ProtectedRoute><Dashboard /></ProtectedRoute>,
}

// Pattern 2: Loader-based auth (data router — preferred)
async function protectedLoader({ request }) {
  const user = await getAuthenticatedUser();
  if (!user) {
    const url = new URL(request.url);
    return redirect(`/login?redirectTo=${url.pathname}`);
  }
  return { user };
}

const routes = [
  {
    path: 'dashboard',
    loader: protectedLoader,
    element: <Dashboard />,
  },
];

// Pattern 3: Layout route guard (protects all children)
const routes = [
  {
    path: '/',
    element: <PublicLayout />,
    children: [
      { path: 'login', element: <Login /> },
      { path: 'signup', element: <Signup /> },
    ],
  },
  {
    path: '/',
    element: <AuthenticatedLayout />,  // checks auth, renders Outlet or redirects
    loader: protectedLoader,
    children: [
      { path: 'dashboard', element: <Dashboard /> },
      { path: 'settings', element: <Settings /> },
      { path: 'profile', element: <Profile /> },
    ],
  },
];
Architect Insight

Loader-based auth guards are superior to component-based guards because: (1) auth check happens before any component renders, avoiding flash of protected content, (2) the redirect happens at the routing level, not the rendering level, (3) it works with SSR since loaders run on the server, and (4) it integrates with React Router's data loading pipeline, allowing auth + data fetch in a single round trip.

3. Role-Based Route Access

// Role-based access with route metadata
const routes = [
  {
    path: 'admin',
    element: <AdminLayout />,
    handle: { requiredRole: 'admin' },  // route metadata
    loader: async ({ request }) => {
      const user = await getUser();
      if (!user) return redirect('/login');
      if (user.role !== 'admin') {
        throw new Response('Forbidden', { status: 403 });
      }
      return { user };
    },
    children: [
      { path: 'users', element: <ManageUsers /> },
      {
        path: 'billing',
        element: <Billing />,
        handle: { requiredRole: 'super_admin' },
        loader: async () => {
          const user = await getUser();
          if (user.role !== 'super_admin') {
            throw new Response('Forbidden', { status: 403 });
          }
          return { billing: await fetchBilling() };
        },
      },
    ],
  },
];

// Generic role-checking loader factory
function requireRole(...roles) {
  return async function loader({ request }) {
    const user = await getUser();
    if (!user) return redirect(`/login?redirectTo=${new URL(request.url).pathname}`);
    if (!roles.includes(user.role)) {
      throw new Response('Forbidden', { status: 403 });
    }
    return { user };
  };
}

// Usage
{ path: 'admin', loader: requireRole('admin', 'super_admin'), element: <Admin /> }

4. Breadcrumb Generation

// Using route handle metadata + useMatches
const routes = [
  {
    path: '/',
    handle: { breadcrumb: 'Home' },
    element: <RootLayout />,
    children: [
      {
        path: 'products',
        handle: { breadcrumb: 'Products' },
        element: <ProductsLayout />,
        children: [
          {
            path: ':productId',
            handle: {
              breadcrumb: (match) => match.data?.product?.name ?? 'Product',
            },
            loader: productLoader,
            element: <ProductDetail />,
          },
        ],
      },
    ],
  },
];

// Breadcrumb component
function Breadcrumbs() {
  const matches = useMatches();

  const crumbs = matches
    .filter(match => match.handle?.breadcrumb)
    .map(match => {
      const breadcrumb = typeof match.handle.breadcrumb === 'function'
        ? match.handle.breadcrumb(match)
        : match.handle.breadcrumb;

      return { label: breadcrumb, path: match.pathname };
    });

  return (
    <nav aria-label="breadcrumb">
      <ol>
        {crumbs.map((crumb, i) => (
          <li key={crumb.path}>
            {i < crumbs.length - 1 ? (
              <Link to={crumb.path}>{crumb.label}</Link>
            ) : (
              <span aria-current="page">{crumb.label}</span>
            )}
          </li>
        ))}
      </ol>
    </nav>
  );
}

5. Tab-Based Navigation State

// Pattern 1: Tabs as search params (shareable, bookmarkable)
function UserProfile() {
  const [searchParams, setSearchParams] = useSearchParams();
  const activeTab = searchParams.get('tab') ?? 'overview';

  const tabs = [
    { id: 'overview', label: 'Overview', content: <Overview /> },
    { id: 'posts', label: 'Posts', content: <Posts /> },
    { id: 'settings', label: 'Settings', content: <Settings /> },
  ];

  return (
    <div>
      <div role="tablist">
        {tabs.map(tab => (
          <button
            key={tab.id}
            role="tab"
            aria-selected={activeTab === tab.id}
            onClick={() => setSearchParams({ tab: tab.id })}
          >
            {tab.label}
          </button>
        ))}
      </div>
      <div role="tabpanel">
        {tabs.find(t => t.id === activeTab)?.content}
      </div>
    </div>
  );
}

// Pattern 2: Tabs as nested routes (each tab has its own URL)
const routes = [
  {
    path: 'users/:id',
    element: <UserProfile />,
    children: [
      { index: true, element: <Navigate to="overview" replace /> },
      { path: 'overview', element: <Overview /> },
      { path: 'posts', element: <Posts /> },
      { path: 'settings', element: <Settings /> },
    ],
  },
];

6. Modal Routes (Parallel Routes)

Modal routes allow a modal to have its own URL while the background page remains visible. This enables deep linking to modals and proper back button behavior.

// Pattern: location state for modal background
function PhotoGrid() {
  const location = useLocation();

  return (
    <div>
      {photos.map(photo => (
        <Link
          key={photo.id}
          to={`/photos/${photo.id}`}
          state={{ backgroundLocation: location }}
        >
          <img src={photo.thumbnail} alt={photo.title} />
        </Link>
      ))}
    </div>
  );
}

// Root: render modal on top of background location
function App() {
  const location = useLocation();
  const backgroundLocation = location.state?.backgroundLocation;

  return (
    <div>
      {/* Always render routes for background (or current) location */}
      <Routes location={backgroundLocation ?? location}>
        <Route path="/" element={<Layout />}>
          <Route index element={<PhotoGrid />} />
          <Route path="photos/:id" element={<PhotoPage />} />
        </Route>
      </Routes>

      {/* If navigated from grid, show modal on top */}
      {backgroundLocation && (
        <Routes>
          <Route path="photos/:id" element={<PhotoModal />} />
        </Routes>
      )}
    </div>
  );
}

// Direct URL access (/photos/42) → full page
// Click from grid → modal over grid (background preserved)
// Back button → closes modal, returns to grid
Modal Route Benefits

Modal routes provide: (1) deep linkable modals — sharing /photos/42 shows the full page, (2) proper back button behavior — back closes the modal, (3) SEO — the URL is crawlable, (4) progressive enhancement — works without JavaScript as a full page. This is the pattern used by Instagram, Twitter, and most modern web apps.

7. Search Param State Management

// Type-safe search params hook
function useTypedSearchParams<T>(schema) {
  const [searchParams, setSearchParams] = useSearchParams();

  const parsed = useMemo(() => {
    const result = {};
    for (const [key, config] of Object.entries(schema)) {
      const raw = searchParams.get(key);
      result[key] = raw !== null ? config.parse(raw) : config.defaultValue;
    }
    return result;
  }, [searchParams, schema]);

  const update = useCallback((updates) => {
    setSearchParams(prev => {
      const next = new URLSearchParams(prev);
      for (const [key, value] of Object.entries(updates)) {
        if (value === undefined || value === schema[key]?.defaultValue) {
          next.delete(key);  // remove defaults from URL
        } else {
          next.set(key, schema[key].serialize(value));
        }
      }
      return next;
    });
  }, [setSearchParams, schema]);

  return [parsed, update];
}

// Usage
const schema = {
  page: { parse: Number, serialize: String, defaultValue: 1 },
  sort: { parse: String, serialize: String, defaultValue: 'date' },
  filter: { parse: (v) => v.split(','), serialize: (v) => v.join(','), defaultValue: [] },
};

function ProductList() {
  const [params, setParams] = useTypedSearchParams(schema);
  // params.page: number, params.sort: string, params.filter: string[]

  setParams({ page: 2, filter: ['electronics', 'books'] });
  // URL: ?page=2&filter=electronics,books (sort omitted — it's the default)
}

8. Deep Linking Strategies

// Deep linking: every meaningful UI state has a URL

// Level 1: Route-based (page level)
// /products/42

// Level 2: Search params (filters, pagination)
// /products?category=electronics&page=2&sort=price-asc

// Level 3: Hash fragments (scroll to section)
// /products/42#reviews

// Level 4: Complex state in URL
// /dashboard?view=grid&columns=name,price,stock&selected=42,43,44

// Scroll to hash on navigation
function ScrollToHash() {
  const location = useLocation();

  useEffect(() => {
    if (location.hash) {
      const element = document.querySelector(location.hash);
      element?.scrollIntoView({ behavior: 'smooth' });
    }
  }, [location.hash]);

  return null;
}

// Generate shareable deep links
function ShareButton({ filters }) {
  function generateLink() {
    const params = new URLSearchParams();
    Object.entries(filters).forEach(([k, v]) => {
      if (v != null) params.set(k, String(v));
    });
    return `${window.location.origin}${window.location.pathname}?${params}`;
  }

  return (
    <button onClick={() => navigator.clipboard.writeText(generateLink())}>
      Copy Link
    </button>
  );
}

9. URL as State: Encoding Complex State

The URL is the most shareable, bookmarkable, and resilient state container in web applications. Encoding UI state in the URL enables deep linking, browser history, and link sharing.

// Encoding complex filter state in URL
const filterSchema = {
  category: { type: 'string', multi: false },
  tags: { type: 'array', separator: ',' },
  priceRange: { type: 'range', separator: '-' },
  inStock: { type: 'boolean' },
  sort: { type: 'string', default: 'relevance' },
};

// Serialize state → URL
function stateToSearchParams(state) {
  const params = new URLSearchParams();

  if (state.category) params.set('category', state.category);
  if (state.tags?.length) params.set('tags', state.tags.join(','));
  if (state.priceRange) params.set('price', `${state.priceRange[0]}-${state.priceRange[1]}`);
  if (state.inStock) params.set('inStock', '1');
  if (state.sort !== 'relevance') params.set('sort', state.sort);

  return params;
}

// Parse URL → state
function searchParamsToState(params) {
  return {
    category: params.get('category') ?? null,
    tags: params.get('tags')?.split(',') ?? [],
    priceRange: params.get('price')?.split('-').map(Number) ?? null,
    inStock: params.get('inStock') === '1',
    sort: params.get('sort') ?? 'relevance',
  };
}

// Compressed state for very complex UIs (e.g., dashboard layouts)
function encodeState(state) {
  const json = JSON.stringify(state);
  const compressed = pako.deflate(json);
  return btoa(String.fromCharCode(...compressed));
}

function decodeState(encoded) {
  const binary = atob(encoded);
  const bytes = Uint8Array.from(binary, c => c.charCodeAt(0));
  const json = pako.inflate(bytes, { to: 'string' });
  return JSON.parse(json);
}
Architect Insight: What Belongs in the URL

Not all state belongs in the URL. Use this decision framework: URL state — anything that should survive a page refresh, be shareable via link, or be bookmarkable (filters, pagination, sort, selected tab, view mode). Component state — ephemeral UI state (dropdown open/closed, hover state, animation state). Server state — data from APIs (use React Query/SWR). Global state — cross-cutting concerns (auth, theme, locale). The URL is the most durable state container — if a user shares a link, the recipient should see the same view.

State Type Storage Shareable Survives Refresh Example
URL params URL path Yes Yes /users/42
Search params URL query Yes Yes ?page=2&sort=name
Hash URL fragment Yes Yes #section-3
Location state History API No Yes (session) Modal background
Component state React memory No No Dropdown open
