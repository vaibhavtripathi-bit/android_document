useState Is useReducer Internally

In React's source code, useState is literally implemented as useReducer with a built-in reducer called basicStateReducer:

// ReactFiberHooks.js — simplified
function basicStateReducer(state, action) {
  return typeof action === 'function' ? action(state) : action;
}

function mountState(initialState) {
  // Delegates to mountReducer with basicStateReducer
  const hook = mountWorkInProgressHook();
  if (typeof initialState === 'function') {
    initialState = initialState();
  }
  hook.memoizedState = hook.baseState = initialState;

  const queue = {
    pending: null,
    lanes: NoLanes,
    dispatch: null,
    lastRenderedReducer: basicStateReducer,
    lastRenderedState: initialState,
  };
  hook.queue = queue;

  const dispatch = queue.dispatch = dispatchSetState.bind(
    null, currentlyRenderingFiber, queue
  );
  return [hook.memoizedState, dispatch];
}

function updateState(initialState) {
  // Literally calls updateReducer
  return updateReducer(basicStateReducer, initialState);
}
Architect Insight

basicStateReducer handles both direct values (setState(5)) and functional updates (setState(prev => prev + 1)). If the action is a function, it's called with the current state. Otherwise, the action itself becomes the new state. This is why useState and useReducer share identical update queue machinery.

The Update Queue: Circular Linked List

When you call setState, React doesn't immediately update state. Instead, it creates an update object and appends it to a circular linked list on the hook's queue:

type Update<S, A> = {
  lane: Lane,              // priority lane
  revertLane: Lane,        // for transitions
  action: A,               // the value or function passed to setState
  hasEagerState: boolean,  // was eager computation attempted?
  eagerState: S | null,    // pre-computed state (if eager)
  next: Update<S, A>,     // next update (circular)
};

type UpdateQueue<S, A> = {
  pending: Update<S, A> | null,  // last update (circular list)
  lanes: Lanes,
  dispatch: ((A) => void) | null,
  lastRenderedReducer: ((S, A) => S) | null,
  lastRenderedState: S | null,
};

Why Circular?

The pending pointer points to the last update, and pending.next points to the first. This allows O(1) appending without traversing the list:

Circular Update Queue (pending points to last): After setState(1): pending → [Update A: action=1] ──┐ ▲ │ └───────────────────────┘ After setState(2): pending → [Update B: action=2] → [Update A: action=1] ──┐ ▲ │ └───────────────────────────────────────────────┘ After setState(prev => prev + 1): pending → [Update C: action=fn] → [Update A] → [Update B] ──┐ ▲ │ └────────────────────────────────────────────────────┘ Processing order: A → B → C (pending.next is first) 
function dispatchSetState(fiber, queue, action) {
  const lane = requestUpdateLane(fiber);

  const update = {
    lane,
    revertLane: NoLane,
    action,
    hasEagerState: false,
    eagerState: null,
    next: null,
  };

  // Append to circular list
  const pending = queue.pending;
  if (pending === null) {
    update.next = update; // circular: points to itself
  } else {
    update.next = pending.next; // new.next = first
    pending.next = update;       // old_last.next = new
  }
  queue.pending = update; // pending = new (last)

  // ... eager state check, then schedule re-render
}

Eager State Bail-Out

Before scheduling a re-render, React attempts an eager state computation. If the new state is identical to the current state (via Object.is), React can skip the re-render entirely:

function dispatchSetState(fiber, queue, action) {
  // ... create update, append to queue ...

  const alternate = fiber.alternate;
  if (
    fiber.lanes === NoLanes &&
    (alternate === null || alternate.lanes === NoLanes)
  ) {
    // The fiber has no pending work — safe to eagerly compute
    const lastRenderedReducer = queue.lastRenderedReducer;
    if (lastRenderedReducer !== null) {
      const currentState = queue.lastRenderedState;
      const eagerState = lastRenderedReducer(currentState, action);
      update.hasEagerState = true;
      update.eagerState = eagerState;

      if (Object.is(eagerState, currentState)) {
        // Bail out! Don't schedule a re-render.
        enqueueConcurrentHookUpdateAndEagerlyBailout(
          fiber, queue, update
        );
        return;
      }
    }
  }

  // State changed — schedule re-render
  scheduleUpdateOnFiber(root, fiber, lane);
}
Bail-Out Conditions

Eager bail-out only works when the fiber has no pending updates (fiber.lanes === NoLanes). If there are already queued updates, React can't know the final state without processing the entire queue, so it must schedule a re-render. This is why sometimes setState with the same value still causes a render — there were pending updates from a previous setState.

Object.is Comparison — Not Deep Equality

// Object.is behavior:
Object.is(1, 1)           // true — bail out
Object.is('a', 'a')       // true — bail out
Object.is(NaN, NaN)       // true — bail out (unlike ===)
Object.is(0, -0)          // false — no bail out (unlike ===)

// Reference types — NEVER bail out with new objects:
Object.is({a: 1}, {a: 1}) // false — different references
Object.is([1,2], [1,2])   // false — different references

// Common mistake:
const [user, setUser] = useState({ name: 'Alice', age: 30 });
setUser({ ...user, age: 30 }); // WILL re-render (new object ref)
setUser(user);                  // WON'T re-render (same ref)

Update Queue Processing

During render, React processes the entire update queue to compute the final state:

function updateReducer(reducer, initialArg, init) {
  const hook = updateWorkInProgressHook();
  const queue = hook.queue;
  queue.lastRenderedReducer = reducer;

  const current = currentHook;
  let baseQueue = current.baseQueue;
  const pendingQueue = queue.pending;

  // Merge pending updates into base queue
  if (pendingQueue !== null) {
    if (baseQueue !== null) {
      // Concatenate circular lists
      const baseFirst = baseQueue.next;
      const pendingFirst = pendingQueue.next;
      baseQueue.next = pendingFirst;
      pendingQueue.next = baseFirst;
    }
    current.baseQueue = baseQueue = pendingQueue;
    queue.pending = null;
  }

  if (baseQueue !== null) {
    const first = baseQueue.next;
    let newState = current.baseState;
    let newBaseState = null;
    let newBaseQueueFirst = null;
    let newBaseQueueLast = null;
    let update = first;

    do {
      const updateLane = update.lane;
      if (!isSubsetOfLanes(renderLanes, updateLane)) {
        // Insufficient priority — skip and keep in base queue
        // (for concurrent rendering priority handling)
        const clone = { /* clone update */ };
        if (newBaseQueueLast === null) {
          newBaseQueueFirst = newBaseQueueLast = clone;
          newBaseState = newState;
        } else {
          newBaseQueueLast = newBaseQueueLast.next = clone;
        }
      } else {
        // Process this update
        if (update.hasEagerState) {
          newState = update.eagerState; // use pre-computed
        } else {
          newState = reducer(newState, update.action);
        }
      }
      update = update.next;
    } while (update !== null && update !== first);

    if (!Object.is(newState, hook.memoizedState)) {
      markWorkInProgressReceivedUpdate(); // signal that state changed
    }

    hook.memoizedState = newState;
    hook.baseState = newBaseState ?? newState;
    hook.baseQueue = newBaseQueueLast;
    queue.lastRenderedState = newState;
  }

  return [hook.memoizedState, queue.dispatch];
}

Batched Updates

Since React 18, all state updates are automatically batched — even inside setTimeout, promises, and native event handlers:

function handleClick() {
  setCount(c => c + 1);  // queued
  setFlag(f => !f);       // queued
  setName('Alice');        // queued
  // React processes all three in a single re-render
}

// Pre-React 18: only batched inside React event handlers
// React 18+: batched everywhere via automatic batching

// To opt OUT of batching (rare):
import { flushSync } from 'react-dom';

function handleClick() {
  flushSync(() => {
    setCount(c => c + 1); // renders immediately
  });
  // DOM is updated here
  flushSync(() => {
    setFlag(f => !f);     // renders immediately
  });
  // DOM is updated here
}
Automatic Batching (React 18+): setState(A) ──┐ setState(B) ──┤ All queued in same microtask setState(C) ──┘ │ ▼ scheduleUpdateOnFiber() │ ▼ Single re-render processes A → B → C │ ▼ Single DOM commit flushSync breaks batching: flushSync(() => setState(A)) → render → commit → DOM updated flushSync(() => setState(B)) → render → commit → DOM updated

Lazy Initialization Pattern

When useState receives a function, it calls it only on mount — not on every render:

// BAD: expensive computation runs every render
const [data, setData] = useState(parseExpensiveJSON(rawData));

// GOOD: lazy initializer — runs only on mount
const [data, setData] = useState(() => parseExpensiveJSON(rawData));

// Internal implementation:
function mountState(initialState) {
  const hook = mountWorkInProgressHook();
  if (typeof initialState === 'function') {
    initialState = initialState(); // called once during mount
  }
  hook.memoizedState = hook.baseState = initialState;
  // ...
}

// On update, initialState is completely ignored:
function updateState(initialState) {
  return updateReducer(basicStateReducer, initialState);
  // initialState param is never used in updateReducer
}

Functional Updates: prev => next

Functional updates are essential when the next state depends on the previous state, especially with batching:

// Problem: stale state with direct values
function Counter() {
  const [count, setCount] = useState(0);

  function incrementThree() {
    setCount(count + 1); // all three read count=0
    setCount(count + 1); // from the closure
    setCount(count + 1); // result: count=1, not 3
  }

  // Solution: functional updates
  function incrementThree() {
    setCount(prev => prev + 1); // 0 → 1
    setCount(prev => prev + 1); // 1 → 2
    setCount(prev => prev + 1); // 2 → 3
  }
}

// How it works internally (basicStateReducer):
function basicStateReducer(state, action) {
  return typeof action === 'function'
    ? action(state)  // functional update: call with current state
    : action;         // direct value: replace state
}

// Queue processing applies each update sequentially:
// Update 1: basicStateReducer(0, prev => prev + 1) → 1
// Update 2: basicStateReducer(1, prev => prev + 1) → 2
// Update 3: basicStateReducer(2, prev => prev + 1) → 3

useReducer Deep Dive

// useReducer provides the same queue machinery with a custom reducer
function mountReducer(reducer, initialArg, init) {
  const hook = mountWorkInProgressHook();
  const initialState = init !== undefined ? init(initialArg) : initialArg;
  hook.memoizedState = hook.baseState = initialState;

  const queue = {
    pending: null,
    lanes: NoLanes,
    dispatch: null,
    lastRenderedReducer: reducer,
    lastRenderedState: initialState,
  };
  hook.queue = queue;

  const dispatch = queue.dispatch = dispatchReducerAction.bind(
    null, currentlyRenderingFiber, queue
  );
  return [hook.memoizedState, dispatch];
}

// Key difference: dispatchReducerAction does NOT attempt eager bail-out
// because the reducer might have side effects or depend on external state.
function dispatchReducerAction(fiber, queue, action) {
  const lane = requestUpdateLane(fiber);
  const update = {
    lane,
    revertLane: NoLane,
    action,
    hasEagerState: false,
    eagerState: null,
    next: null,
  };
  // Always schedules — no eager bail-out
  enqueueConcurrentHookUpdate(fiber, queue, update, lane);
  scheduleUpdateOnFiber(root, fiber, lane);
}
useState vs useReducer Bail-Out

useState (via dispatchSetState) attempts eager bail-out with Object.is. useReducer (via dispatchReducerAction) does not attempt eager bail-out — it always schedules a re-render. However, during the render phase, if the reducer produces the same state, React may still bail out of the child subtree via Object.is comparison.

Architect Insight — When to Choose useReducer

Use useReducer when: (1) next state depends on previous state in complex ways, (2) multiple sub-values are updated together, (3) you want to pass dispatch down instead of multiple callbacks, (4) you want to extract state logic for testing. The dispatch function is stable (like useState's setter), so it won't cause child re-renders when passed as a prop.
