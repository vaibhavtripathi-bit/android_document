useRef Internal Implementation

useRef is the simplest hook internally. It creates a plain object with a .current property and stores it on the hook's memoizedState. The same object persists across all renders.

// ReactFiberHooks.js — simplified
function mountRef(initialValue) {
  const hook = mountWorkInProgressHook();
  const ref = { current: initialValue };
  hook.memoizedState = ref;
  return ref;
}

function updateRef(initialValue) {
  const hook = updateWorkInProgressHook();
  return hook.memoizedState; // return the SAME object
  // initialValue is completely ignored on updates
}

// That's it. No queue, no reducer, no effect list.
// The ref object is created once and returned on every render.
Architect Insight

A ref is just a mutable container that survives re-renders because it's stored on the fiber's hook linked list. Unlike state, mutating .current does NOT trigger a re-render. This makes refs the escape hatch for values that need to persist but shouldn't drive the UI.

// Use ref for:
// 1. DOM element references
const inputRef = useRef(null);
<input ref={inputRef} />
inputRef.current.focus();

// 2. Mutable values that don't affect rendering
const intervalRef = useRef(null);
const renderCountRef = useRef(0);
renderCountRef.current++; // track renders without causing re-render

// 3. Previous value tracking
function usePrevious(value) {
  const ref = useRef();
  useEffect(() => {
    ref.current = value;
  });
  return ref.current; // returns value from previous render
}

// 4. Stable callback reference (avoiding stale closures)
function useEventCallback(fn) {
  const ref = useRef(fn);
  ref.current = fn; // always points to latest closure
  return useCallback((...args) => ref.current(...args), []);
}

DOM Refs: How React Attaches Them

// When you write: <div ref={myRef} />
// React handles ref attachment during the commit phase:

// During commit (commitAttachRef):
function commitAttachRef(finishedWork) {
  const ref = finishedWork.ref;
  if (ref !== null) {
    const instance = finishedWork.stateNode; // DOM node

    if (typeof ref === 'function') {
      // Callback ref
      ref(instance);
    } else {
      // Object ref (useRef)
      ref.current = instance;
    }
  }
}

// During unmount or ref change (commitDetachRef):
function commitDetachRef(current) {
  const ref = current.ref;
  if (ref !== null) {
    if (typeof ref === 'function') {
      ref(null);       // callback ref receives null
    } else {
      ref.current = null; // object ref cleared
    }
  }
}
Ref Lifecycle During Commit: Mount: DOM node created → commitAttachRef → ref.current = DOM node Update (ref unchanged): No ref work needed Update (ref changed, e.g., conditional ref): commitDetachRef(old) → old.current = null commitAttachRef(new) → new.current = DOM node Unmount: commitDetachRef → ref.current = null → DOM node removed

Callback Refs

Callback refs are functions that React calls with the DOM node (on mount) or null (on unmount). They offer more control than object refs.

// Basic callback ref
function MeasuredComponent() {
  const [height, setHeight] = useState(0);

  const measuredRef = useCallback((node) => {
    if (node !== null) {
      setHeight(node.getBoundingClientRect().height);
    }
  }, []);

  return <div ref={measuredRef}>Content</div>;
}

// Callback ref advantages:
// 1. Notified when ref attaches/detaches (unlike useRef)
// 2. Can perform side effects on attachment
// 3. Works with conditional rendering
// 4. Can be used for ref merging

// Ref merging pattern (multiple consumers):
function mergeRefs(...refs) {
  return (node) => {
    refs.forEach(ref => {
      if (typeof ref === 'function') {
        ref(node);
      } else if (ref != null) {
        ref.current = node;
      }
    });
  };
}

function MyComponent({ externalRef }) {
  const internalRef = useRef(null);
  const mergedRef = useMemo(
    () => mergeRefs(internalRef, externalRef),
    [externalRef]
  );
  return <div ref={mergedRef} />;
}

Ref Forwarding with forwardRef

// Problem: refs don't pass through function components
function FancyInput(props) {
  return <input className="fancy" {...props} />;
}
// <FancyInput ref={myRef} /> — ref is NOT forwarded to <input>

// Solution: forwardRef
const FancyInput = forwardRef(function FancyInput(props, ref) {
  return <input className="fancy" ref={ref} {...props} />;
});
// <FancyInput ref={myRef} /> — myRef.current = <input> DOM node

// Internal: forwardRef creates a special element type
function forwardRef(render) {
  return {
    $$typeof: REACT_FORWARD_REF_TYPE,
    render, // the component function that receives (props, ref)
  };
}

// During reconciliation, React detects REACT_FORWARD_REF_TYPE
// and passes the ref as the second argument to render().
React 19: forwardRef No Longer Needed

In React 19, ref is passed as a regular prop. Function components can accept ref directly in their props without forwardRef. The forwardRef wrapper still works but is no longer necessary.

// React 19 — ref as a prop (no forwardRef needed)
function FancyInput({ ref, ...props }) {
  return <input className="fancy" ref={ref} {...props} />;
}

// Works directly:
<FancyInput ref={myRef} />

useImperativeHandle

useImperativeHandle customizes the value exposed via a ref. Instead of exposing the DOM node, you expose a custom API.

const FancyInput = forwardRef(function FancyInput(props, ref) {
  const inputRef = useRef(null);

  useImperativeHandle(ref, () => ({
    focus() {
      inputRef.current.focus();
    },
    scrollIntoView() {
      inputRef.current.scrollIntoView({ behavior: 'smooth' });
    },
    // DOM node is NOT exposed — only these methods
  }), []); // deps array — recreate handle when deps change

  return <input ref={inputRef} {...props} />;
});

// Usage:
function Parent() {
  const inputRef = useRef(null);
  return (
    <>
      <FancyInput ref={inputRef} />
      <button onClick={() => inputRef.current.focus()}>Focus</button>
      {/* inputRef.current.value would be undefined — not exposed */}
    </>
  );
}

Internal Implementation

// useImperativeHandle is implemented as a layout effect:
function mountImperativeHandle(ref, create, deps) {
  const effectDeps = deps != null ? deps.concat([ref]) : null;

  mountEffectImpl(
    UpdateEffect | LayoutStaticEffect,
    HookLayout,
    imperativeHandleEffect.bind(null, create, ref),
    effectDeps
  );
}

function imperativeHandleEffect(create, ref) {
  if (typeof ref === 'function') {
    const refCallback = ref;
    const inst = create();
    refCallback(inst);
    return () => refCallback(null);
  } else if (ref !== null && ref !== undefined) {
    const refObject = ref;
    const inst = create();
    refObject.current = inst;
    return () => { refObject.current = null; };
  }
}

// Key: it runs as a LAYOUT effect (synchronous, after DOM mutations)
// This ensures the imperative handle is available before useEffect runs

Ref Cleanup in React 19

React 19 introduces cleanup functions for callback refs, similar to how useEffect returns a cleanup function:

// React 19: callback ref with cleanup
function MyComponent() {
  const ref = useCallback((node) => {
    if (node) {
      const observer = new ResizeObserver(entries => {
        console.log('Resized:', entries[0].contentRect);
      });
      observer.observe(node);

      // NEW in React 19: return cleanup function
      return () => {
        observer.disconnect();
      };
    }
  }, []);

  return <div ref={ref}>Observed</div>;
}

// Before React 19, you had to track cleanup manually:
function MyComponent() {
  const observerRef = useRef(null);

  const ref = useCallback((node) => {
    if (observerRef.current) {
      observerRef.current.disconnect();
    }
    if (node) {
      observerRef.current = new ResizeObserver(/* ... */);
      observerRef.current.observe(node);
    }
  }, []);

  return <div ref={ref}>Observed</div>;
}

Why Refs Survive Re-Renders

// Refs survive because the ref OBJECT is stored on the fiber's
// hook linked list, and the same object is returned every render.

// Render 1: mountRef(0) → creates { current: 0 }, stores on hook
// Render 2: updateRef(0) → returns hook.memoizedState (same object)
// Render 3: updateRef(0) → returns hook.memoizedState (same object)

// The object identity never changes:
const ref = useRef(0);
// ref === ref from previous render (Object.is returns true)

// But .current can be mutated freely:
ref.current = 42; // no re-render, immediate mutation

// This is fundamentally different from state:
const [val, setVal] = useState(0);
// setVal(42) → schedules re-render, new value available next render

DOM Measurement Patterns

// Pattern 1: Measure after render with useLayoutEffect
function Tooltip({ children, targetRef }) {
  const tooltipRef = useRef(null);
  const [position, setPosition] = useState({ top: 0, left: 0 });

  useLayoutEffect(() => {
    const targetRect = targetRef.current.getBoundingClientRect();
    const tooltipRect = tooltipRef.current.getBoundingClientRect();
    setPosition({
      top: targetRect.bottom + 8,
      left: targetRect.left + (targetRect.width - tooltipRect.width) / 2,
    });
  }, [targetRef]);

  return (
    <div ref={tooltipRef} style={{ position: 'fixed', ...position }}>
      {children}
    </div>
  );
}

// Pattern 2: ResizeObserver with callback ref
function useResizeObserver() {
  const [size, setSize] = useState({ width: 0, height: 0 });

  const ref = useCallback((node) => {
    if (!node) return;
    const observer = new ResizeObserver(([entry]) => {
      setSize({
        width: entry.contentRect.width,
        height: entry.contentRect.height,
      });
    });
    observer.observe(node);
    return () => observer.disconnect(); // React 19 cleanup
  }, []);

  return [ref, size];
}

// Pattern 3: IntersectionObserver
function useLazyLoad() {
  const [isVisible, setIsVisible] = useState(false);

  const ref = useCallback((node) => {
    if (!node) return;
    const observer = new IntersectionObserver(
      ([entry]) => {
        if (entry.isIntersecting) {
          setIsVisible(true);
          observer.disconnect();
        }
      },
      { threshold: 0.1 }
    );
    observer.observe(node);
    return () => observer.disconnect();
  }, []);

  return [ref, isVisible];
}
Architect Insight — Ref Anti-Patterns

Don't read or write ref.current during render for values that change — this breaks concurrent rendering because React may render a component multiple times before committing. Refs should be read/written in effects, event handlers, or callbacks. The exception is lazy initialization: if (ref.current === null) ref.current = new Thing().
