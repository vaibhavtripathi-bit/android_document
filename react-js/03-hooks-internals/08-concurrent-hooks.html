useTransition

useTransition marks a state update as a transition — a low-priority update that can be interrupted by urgent updates (user input). It returns [isPending, startTransition].

function SearchPage() {
  const [query, setQuery] = useState('');
  const [results, setResults] = useState([]);
  const [isPending, startTransition] = useTransition();

  function handleChange(e) {
    const value = e.target.value;
    setQuery(value);              // urgent: update input immediately

    startTransition(() => {
      setResults(search(value));  // transition: can be interrupted
    });
  }

  return (
    <div>
      <input value={query} onChange={handleChange} />
      {isPending && <Spinner />}
      <ResultsList results={results} />
    </div>
  );
}

Internal Mechanism

// Simplified useTransition implementation:
function mountTransition() {
  const [isPending, setPending] = mountState(false);
  const start = startTransition.bind(null, setPending);
  const hook = mountWorkInProgressHook();
  hook.memoizedState = start;
  return [isPending, start];
}

function startTransition(setPending, callback) {
  setPending(true); // urgent update: show pending state

  const prevTransition = ReactCurrentBatchConfig.transition;
  ReactCurrentBatchConfig.transition = {};

  try {
    setPending(false); // transition update: will be deferred
    callback();        // state updates inside get transition lane
  } finally {
    ReactCurrentBatchConfig.transition = prevTransition;
  }
}

// When ReactCurrentBatchConfig.transition is set,
// requestUpdateLane() returns TransitionLane instead of SyncLane.
// This means updates inside startTransition get lower priority.
useTransition Priority Flow: User types "abc" rapidly: Input: "a" ─── SyncLane (urgent) ──────────── renders immediately Search: "a" ── TransitionLane (deferred) ──── starts rendering... │ Input: "ab" ── SyncLane (urgent) ──────────── interrupts transition! Search: "ab" ─ TransitionLane (deferred) ──── starts rendering... │ Input: "abc" ─ SyncLane (urgent) ──────────── interrupts transition! Search: "abc"─ TransitionLane (deferred) ──── renders to completion ✓ Result: input always responsive, search renders when idle
Architect Insight — Lanes and Priority

React's lane system assigns bit-flags to updates: SyncLane (0b0001) for urgent, TransitionLane (0b0001000000...) for transitions. During work loop, React checks if higher-priority lanes have pending work. If so, it abandons the current transition render and starts the urgent one. The abandoned work is discarded (no wasted DOM mutations).

useDeferredValue

useDeferredValue returns a deferred copy of a value. During urgent updates, it returns the previous value, then re-renders with the new value at transition priority.

function SearchResults({ query }) {
  const deferredQuery = useDeferredValue(query);
  const isStale = query !== deferredQuery;

  return (
    <div style={{ opacity: isStale ? 0.5 : 1 }}>
      <ExpensiveList query={deferredQuery} />
    </div>
  );
}

// Internal implementation (simplified):
function mountDeferredValue(value) {
  const hook = mountWorkInProgressHook();
  hook.memoizedState = value;
  return value;
}

function updateDeferredValue(value) {
  const hook = updateWorkInProgressHook();
  const prevValue = hook.memoizedState;

  if (Object.is(value, prevValue)) {
    return value; // no change
  }

  // Check if this is an urgent render
  if (isCurrentTreeHiddenOrDetached()) {
    hook.memoizedState = value;
    return value;
  }

  // Return old value for this render, schedule transition for new value
  const shouldDeferValue = !includesOnlyNonUrgentLanes(renderLanes);
  if (shouldDeferValue) {
    // Keep old value, schedule re-render at transition priority
    markWorkInProgressReceivedUpdate();
    return prevValue;
  } else {
    hook.memoizedState = value;
    return value;
  }
}

useId

useId generates a unique ID that is stable across server and client rendering. It solves the hydration mismatch problem with auto-generated IDs.

function FormField({ label }) {
  const id = useId();
  return (
    <div>
      <label htmlFor={id}>{label}</label>
      <input id={id} />
    </div>
  );
}

// Multiple IDs from one hook:
function PasswordField() {
  const id = useId();
  return (
    <div>
      <label htmlFor={`${id}-password`}>Password</label>
      <input id={`${id}-password`} type="password" />
      <p id={`${id}-hint`}>Must be 8+ characters</p>
    </div>
  );
}

How useId Generates Stable IDs

// useId generates IDs based on the component's position in the tree.
// The algorithm uses the fiber tree path to create deterministic IDs:

// Internal (simplified):
function mountId() {
  const hook = mountWorkInProgressHook();

  const root = getWorkInProgressRoot();
  let id = ':' + root.identifierPrefix;

  // Build ID from tree path (position of this fiber in the tree)
  const treeId = getTreeId();
  id += 'r' + treeId + ':';

  hook.memoizedState = id;
  return id;
}

// The tree ID encodes the path through the fiber tree:
// - Each level contributes bits based on sibling index
// - This makes the ID deterministic regardless of render order
// - Server and client traverse the tree in the same order
//   → same IDs generated on both sides

// Example IDs: ":r0:", ":r1:", ":r2:", ":r1a:", ":r1b:"
Don't Use useId for List Keys

useId generates IDs based on tree position, not data identity. Using it for list keys would produce unstable keys when items are reordered. Use data-derived keys (item.id) for lists.

useSyncExternalStore

useSyncExternalStore safely reads from external stores (Redux, Zustand, browser APIs) without tearing — a bug where different parts of the UI show inconsistent state during concurrent rendering.

// Signature:
const snapshot = useSyncExternalStore(
  subscribe,    // function to subscribe to store changes
  getSnapshot,  // function to read current value
  getServerSnapshot // optional: for SSR
);

// Example: subscribing to browser online status
function useOnlineStatus() {
  return useSyncExternalStore(
    (callback) => {
      window.addEventListener('online', callback);
      window.addEventListener('offline', callback);
      return () => {
        window.removeEventListener('online', callback);
        window.removeEventListener('offline', callback);
      };
    },
    () => navigator.onLine,       // client snapshot
    () => true                     // server snapshot (assume online)
  );
}

// Example: subscribing to a Redux-like store
function useStoreSelector(store, selector) {
  return useSyncExternalStore(
    store.subscribe,
    () => selector(store.getState()),
    () => selector(store.getInitialState())
  );
}

The Tearing Problem

Tearing in Concurrent Rendering (without useSyncExternalStore): External Store: value = "A" React starts rendering tree: Component1 reads store → "A" Component2 reads store → "A" │ ├── React yields to browser (concurrent mode) │ └── Store updates: value = "B" │ Component3 reads store → "B" ← TEARING! Inconsistent with 1 and 2 Component4 reads store → "B" With useSyncExternalStore: React takes a SNAPSHOT before rendering All components read from the same snapshot If store updates mid-render, React re-renders from scratch 
// Internal implementation (simplified):
function mountSyncExternalStore(subscribe, getSnapshot, getServerSnapshot) {
  const fiber = currentlyRenderingFiber;
  const hook = mountWorkInProgressHook();

  const nextSnapshot = getSnapshot();
  hook.memoizedState = nextSnapshot;

  const inst = { hasValue: true, value: nextSnapshot };
  hook.queue = inst;

  // Subscribe via effect
  mountEffect(
    subscribeToStore.bind(null, fiber, inst, subscribe),
    [subscribe]
  );

  // Check for changes via layout effect (synchronous)
  mountLayoutEffect(
    updateStoreInstance.bind(null, fiber, inst, getSnapshot),
    [getSnapshot]
  );

  return nextSnapshot;
}

// The key: it uses a LAYOUT effect to synchronously check
// if the store changed during render. If it did, it forces
// a synchronous re-render to prevent tearing.

useInsertionEffect

// useInsertionEffect fires BEFORE DOM mutations.
// Designed for CSS-in-JS libraries (styled-components, emotion).

function useCSS(rule) {
  useInsertionEffect(() => {
    if (!isInserted(rule)) {
      const sheet = getGlobalStyleSheet();
      sheet.insertRule(rule);
    }
  }, [rule]);
}

// Timing comparison:
// 1. useInsertionEffect → fires (inject styles)
// 2. DOM mutations       → React updates DOM
// 3. useLayoutEffect     → fires (read DOM measurements)
// 4. Browser paint       → screen updates
// 5. useEffect           → fires (side effects)

// Restrictions:
// - Cannot access refs (DOM not yet mutated)
// - Cannot schedule state updates
// - Intended ONLY for CSS-in-JS style injection

How Concurrent Hooks Interact with Lanes

// React's lane system (simplified):
const SyncLane           = 0b0000000000000000000000000000001;
const InputContinuousLane = 0b0000000000000000000000000000100;
const DefaultLane        = 0b0000000000000000000000000010000;
const TransitionLane1    = 0b0000000000000000000001000000000;
const TransitionLane2    = 0b0000000000000000000010000000000;
// ... more transition lanes

// How hooks map to lanes:
// useState/useReducer:  SyncLane (urgent by default)
// startTransition:      TransitionLane (deferred)
// useDeferredValue:     creates TransitionLane re-render
// useSyncExternalStore: SyncLane (forces synchronous)

// The work loop processes lanes by priority:
function performConcurrentWorkOnRoot(root) {
  const lanes = getNextLanes(root);

  if (includesSyncLane(lanes)) {
    // Synchronous: cannot be interrupted
    renderRootSync(root, lanes);
  } else {
    // Concurrent: can be interrupted by higher-priority work
    renderRootConcurrent(root, lanes);
  }
}

// shouldYield() checks if higher-priority work is pending:
function workLoopConcurrent() {
  while (workInProgress !== null && !shouldYield()) {
    performUnitOfWork(workInProgress);
  }
  // If shouldYield() returns true, we pause and check for urgent work
}
Architect Insight — Choosing the Right Concurrent Hook

useTransition: You control the state update and want to mark it as non-urgent. useDeferredValue: You receive a value (prop) and want to defer expensive re-renders. useSyncExternalStore: You subscribe to a store outside React's state management. useId: You need SSR-safe unique IDs. useInsertionEffect: You're building a CSS-in-JS library (not for app code).

Concurrent Patterns

// Pattern: Transition + Suspense
function TabContainer() {
  const [tab, setTab] = useState('home');
  const [isPending, startTransition] = useTransition();

  function selectTab(nextTab) {
    startTransition(() => {
      setTab(nextTab); // if new tab suspends, show old tab with isPending
    });
  }

  return (
    <div>
      <TabBar
        selectedTab={tab}
        onSelect={selectTab}
        isPending={isPending}
      />
      <Suspense fallback={<Spinner />}>
        <TabContent tab={tab} />
      </Suspense>
    </div>
  );
}

// Pattern: Deferred value for expensive rendering
function TypeaheadSearch() {
  const [text, setText] = useState('');
  const deferredText = useDeferredValue(text);

  return (
    <div>
      <input value={text} onChange={e => setText(e.target.value)} />
      <Suspense fallback={<Spinner />}>
        <SearchResults query={deferredText} />
      </Suspense>
    </div>
  );
}

// Pattern: External store with selector
function useSelector(selector) {
  const store = useContext(StoreContext);
  return useSyncExternalStore(
    store.subscribe,
    () => selector(store.getState()),
    () => selector(store.getInitialState())
  );
}

function CartCount() {
  const count = useSelector(state => state.cart.items.length);
  return <span>{count}</span>;
}
