What is use()?

use() is a new React 19 API that can read the value of a resource — either a Promise or a Context. Unlike other hooks, use() can be called conditionally and inside loops.

import { use } from 'react';

// Reading a promise (integrates with Suspense):
function UserProfile({ userPromise }) {
  const user = use(userPromise); // suspends until resolved
  return <h1>{user.name}</h1>;
}

// Reading context (replaces useContext):
function ThemeButton() {
  const theme = use(ThemeContext);
  return <button className={theme}>Click</button>;
}

// Conditional calling (unique to use()):
function StatusMessage({ shouldLoad, messagePromise }) {
  if (shouldLoad) {
    const message = use(messagePromise); // OK inside if!
    return <p>{message}</p>;
  }
  return <p>Not loaded</p>;
}
Architect Insight — Why use() Breaks the Rules

use() is technically not a hook — it doesn't create a hook node on the fiber's linked list. It's a special API that reads from external resources. Because it doesn't rely on positional identity in a linked list, it can be called conditionally. The use prefix is a naming convention for consistency, not a structural requirement.

Reading Promises with Suspense

// use() with promises integrates with React's Suspense mechanism:

async function fetchUser(id) {
  const res = await fetch(`/api/users/${id}`);
  return res.json();
}

function UserPage({ userId }) {
  // Create the promise OUTSIDE the component that reads it
  const userPromise = fetchUser(userId);

  return (
    <Suspense fallback={<Spinner />}>
      <UserProfile userPromise={userPromise} />
    </Suspense>
  );
}

function UserProfile({ userPromise }) {
  const user = use(userPromise);
  // This line only executes AFTER the promise resolves.
  // While pending, React suspends and shows the Suspense fallback.
  return <div>{user.name}</div>;
}

How use() Suspends

use(promise) execution flow: Component renders → use(promise) called │ ├─ Promise status: PENDING │ └─ React throws the promise (Suspense protocol) │ └─ Nearest <Suspense> catches it │ └─ Shows fallback UI │ └─ When promise resolves: │ └─ React re-renders the suspended component │ └─ use(promise) returns the resolved value │ ├─ Promise status: FULFILLED │ └─ Returns resolved value immediately (no suspension) │ └─ Promise status: REJECTED └─ Throws the rejection reason └─ Nearest ErrorBoundary catches it 
// Simplified internal implementation:
function use(usable) {
  if (usable !== null && typeof usable === 'object') {
    if (typeof usable.then === 'function') {
      // It's a thenable/promise
      return useThenable(usable);
    }
    if (usable.$$typeof === REACT_CONTEXT_TYPE) {
      // It's a context
      return readContext(usable);
    }
  }
  throw new Error('use() requires a promise or context');
}

function useThenable(thenable) {
  const status = thenable.status;

  switch (status) {
    case 'fulfilled':
      return thenable.value;
    case 'rejected':
      throw thenable.reason;
    default:
      // Pending — suspend
      if (typeof status !== 'string') {
        // Track the promise
        thenable.status = 'pending';
        thenable.then(
          value => {
            thenable.status = 'fulfilled';
            thenable.value = value;
          },
          reason => {
            thenable.status = 'rejected';
            thenable.reason = reason;
          }
        );
      }
      // Throw the promise to trigger Suspense
      throw thenable;
  }
}

Promise Caching Requirements

Critical: Promise Identity Matters

use() tracks promises by reference. If you create a new promise on every render, React will suspend infinitely because each render sees a new, pending promise. The promise must be created outside the rendering component or cached.

// BUG: infinite suspension loop
function UserProfile({ userId }) {
  // Creates a NEW promise every render → suspends → re-renders → new promise → ...
  const user = use(fetch(`/api/users/${userId}`).then(r => r.json()));
  return <div>{user.name}</div>;
}

// FIX 1: Create promise in parent, pass as prop
function UserPage({ userId }) {
  const [userPromise] = useState(() => fetchUser(userId));
  return (
    <Suspense fallback={<Spinner />}>
      <UserProfile userPromise={userPromise} />
    </Suspense>
  );
}

// FIX 2: Use a caching layer
const cache = new Map();
function fetchUserCached(userId) {
  if (!cache.has(userId)) {
    cache.set(userId, fetchUser(userId));
  }
  return cache.get(userId);
}

function UserProfile({ userId }) {
  const user = use(fetchUserCached(userId)); // same promise ref
  return <div>{user.name}</div>;
}

// FIX 3: Use React.cache (for Server Components)
const fetchUser = cache(async (id) => {
  const res = await fetch(`/api/users/${id}`);
  return res.json();
});

Reading Context (Replacing useContext)

// use() can read context, replacing useContext:

// Before (useContext):
function ThemedButton() {
  const theme = useContext(ThemeContext);
  return <button style={{ background: theme.primary }}>Click</button>;
}

// After (use):
function ThemedButton() {
  const theme = use(ThemeContext);
  return <button style={{ background: theme.primary }}>Click</button>;
}

// Key advantage: use() can be called conditionally
function StatusDisplay({ showTheme }) {
  if (showTheme) {
    const theme = use(ThemeContext); // OK — use() allows this
    return <div style={{ color: theme.text }}>Themed</div>;
  }
  return <div>Default</div>;
}

// With useContext, this would violate rules of hooks:
function StatusDisplay({ showTheme }) {
  if (showTheme) {
    const theme = useContext(ThemeContext); // ERROR: conditional hook
  }
  // ...
}

Conditional Calling: Breaking Rules of Hooks

// use() can be called:
// - Inside if/else blocks
// - Inside loops
// - Inside try/catch
// - After early returns

// This is valid:
function DataDisplay({ condition, promise1, promise2 }) {
  if (condition) {
    const data1 = use(promise1);
    return <div>{data1.value}</div>;
  } else {
    const data2 = use(promise2);
    return <div>{data2.value}</div>;
  }
}

// Inside a loop:
function MultiDataDisplay({ promises }) {
  const results = [];
  for (const promise of promises) {
    results.push(use(promise)); // each may suspend independently
  }
  return <ul>{results.map((r, i) => <li key={i}>{r.name}</li>)}</ul>;
}

// After early return:
function ConditionalFetch({ shouldFetch, dataPromise }) {
  if (!shouldFetch) {
    return <div>Not fetching</div>;
  }
  const data = use(dataPromise); // only called when shouldFetch is true
  return <div>{data.value}</div>;
}

// STILL NOT ALLOWED: use() inside try/catch in the same component
// (because thrown promises would be caught by your try/catch
//  instead of propagating to Suspense)
function Broken({ promise }) {
  try {
    const data = use(promise); // DON'T: catches the thrown promise
    return <div>{data}</div>;
  } catch (e) {
    return <div>Error</div>; // catches Suspense throw, not errors
  }
}

// BEFORE: useEffect waterfall pattern
function UserPosts({ userId }) {
  const [user, setUser] = useState(null);
  const [posts, setPosts] = useState(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    let cancelled = false;
    setLoading(true);

    fetchUser(userId).then(user => {
      if (cancelled) return;
      setUser(user);
      return fetchPosts(user.id); // waterfall: waits for user
    }).then(posts => {
      if (cancelled) return;
      setPosts(posts);
      setLoading(false);
    });

    return () => { cancelled = true; };
  }, [userId]);

  if (loading) return <Spinner />;
  return <div>{user.name}: {posts.length} posts</div>;
}

// AFTER: use() with parallel fetching
function UserPosts({ userId }) {
  const userPromise = fetchUserCached(userId);
  const postsPromise = fetchPostsCached(userId);

  return (
    <Suspense fallback={<Spinner />}>
      <UserPostsContent
        userPromise={userPromise}
        postsPromise={postsPromise}
      />
    </Suspense>
  );
}

function UserPostsContent({ userPromise, postsPromise }) {
  const user = use(userPromise);   // both fetched in parallel
  const posts = use(postsPromise); // no waterfall
  return <div>{user.name}: {posts.length} posts</div>;
}

Error Handling with use()

// When a promise rejects, use() throws the error.
// Catch it with an ErrorBoundary:

function App() {
  return (
    <ErrorBoundary fallback={<ErrorMessage />}>
      <Suspense fallback={<Spinner />}>
        <UserProfile userPromise={fetchUser(1)} />
      </Suspense>
    </ErrorBoundary>
  );
}

// ErrorBoundary implementation:
class ErrorBoundary extends React.Component {
  state = { hasError: false, error: null };

  static getDerivedStateFromError(error) {
    return { hasError: true, error };
  }

  render() {
    if (this.state.hasError) {
      return this.props.fallback;
    }
    return this.props.children;
  }
}

// Nested boundaries for granular error handling:
function Dashboard() {
  return (
    <div>
      <ErrorBoundary fallback={<p>User failed</p>}>
        <Suspense fallback={<Spinner />}>
          <UserWidget />
        </Suspense>
      </ErrorBoundary>

      <ErrorBoundary fallback={<p>Stats failed</p>}>
        <Suspense fallback={<Spinner />}>
          <StatsWidget />
        </Suspense>
      </ErrorBoundary>
    </div>
  );
}

use() with Server Components

// In Server Components, you can use async/await directly:
async function UserProfile({ userId }) {
  const user = await fetchUser(userId); // server-side fetch
  return <div>{user.name}</div>;
}

// In Client Components, use use() with promises from the server:
'use client';
function UserProfile({ userPromise }) {
  const user = use(userPromise);
  return <div>{user.name}</div>;
}

// Server Component passes promise to Client Component:
async function UserPage({ userId }) {
  const userPromise = fetchUser(userId); // starts fetch on server
  return (
    <Suspense fallback={<Spinner />}>
      <UserProfile userPromise={userPromise} />
    </Suspense>
  );
}
use() Adoption Strategy

For existing apps: (1) Keep useEffect-based fetching for now if it works. (2) Adopt use() for new features that benefit from Suspense integration. (3) Use a data fetching library (TanStack Query) that handles caching — it works with both patterns. (4) For Server Components, prefer async/await over use().
