TypeScript with React

FC type controversy, generic components, discriminated unions, event handler types, ref typing, context typing, template literal types, strict mode, and type-safe routing patterns.

1. React.FC — The Controversy

The React.FC (FunctionComponent) type was once the standard way to type React components. It's now widely discouraged by the React and TypeScript communities.

// React.FC definition (simplified)
type FC<P = {}> = FunctionComponent<P>;

interface FunctionComponent<P = {}> {
  (props: P): ReactElement | null;
  displayName?: string;
  defaultProps?: Partial<P>;  // Deprecated in React 18.2+
}

// Using React.FC (discouraged)
const Button: React.FC<ButtonProps> = ({ label, onClick }) => {
  return <button onClick={onClick}>{label}</button>;
};

// Problems with React.FC:
// 1. Implicitly accepts children (fixed in React 18 types)
// 2. Doesn't support generics easily
// 3. Breaks defaultProps type inference
// 4. Adds unnecessary complexity

// Preferred: plain function with typed props
interface ButtonProps {
  label: string;
  onClick: () => void;
  variant?: 'primary' | 'secondary';
}

function Button({ label, onClick, variant = 'primary' }: ButtonProps) {
  return <button className={variant} onClick={onClick}>{label}</button>;
}

// Or arrow function with explicit return type
const Button = ({ label, onClick }: ButtonProps): JSX.Element => {
  return <button onClick={onClick}>{label}</button>;
};

Architect Insight: FC in React 18+ Types

In @types/react@18, the implicit children prop was removed from FC. This fixed the biggest complaint, but the community consensus remains: use plain functions with typed props. The FC type adds no value that a return type annotation doesn't provide, and it complicates generic components.

3. Generic Components

// Generic list component
interface ListProps<T> {
  items: T[];
  renderItem: (item: T, index: number) => React.ReactNode;
  keyExtractor: (item: T) => string;
  onSelect?: (item: T) => void;
}

function List<T>({ items, renderItem, keyExtractor, onSelect }: ListProps<T>) {
  return (
    <ul>
      {items.map((item, i) => (
        <li key={keyExtractor(item)} onClick={() => onSelect?.(item)}>
          {renderItem(item, i)}
        </li>
      ))}
    </ul>
  );
}

// TypeScript infers T from usage:
<List
  items={users}                          // T = User
  renderItem={(user) => user.name}       // user: User (inferred)
  keyExtractor={(user) => user.id}       // user: User (inferred)
  onSelect={(user) => console.log(user)} // user: User (inferred)
/>

// Generic select component
interface SelectProps<T extends string | number> {
  options: Array<{ value: T; label: string }>;
  value: T;
  onChange: (value: T) => void;
}

function Select<T extends string | number>({
  options, value, onChange
}: SelectProps<T>) {
  return (
    <select
      value={value}
      onChange={(e) => onChange(e.target.value as T)}
    >
      {options.map((opt) => (
        <option key={opt.value} value={opt.value}>{opt.label}</option>
      ))}
    </select>
  );
}

// Generic table with column definitions
interface Column<T> {
  key: keyof T & string;
  header: string;
  render?: (value: T[keyof T], row: T) => React.ReactNode;
  sortable?: boolean;
}

interface TableProps<T> {
  data: T[];
  columns: Column<T>[];
  onRowClick?: (row: T) => void;
}

function Table<T extends Record<string, unknown>>({
  data, columns, onRowClick
}: TableProps<T>) {
  return (
    <table>
      <thead>
        <tr>
          {columns.map((col) => <th key={col.key}>{col.header}</th>)}
        </tr>
      </thead>
      <tbody>
        {data.map((row, i) => (
          <tr key={i} onClick={() => onRowClick?.(row)}>
            {columns.map((col) => (
              <td key={col.key}>
                {col.render ? col.render(row[col.key], row) : String(row[col.key])}
              </td>
            ))}
          </tr>
        ))}
      </tbody>
    </table>
  );
}

4. Discriminated Unions for Props

// Pattern: mutually exclusive prop combinations
type ButtonProps =
  | { variant: 'link'; href: string; onClick?: never }
  | { variant: 'button'; onClick: () => void; href?: never }
  | { variant: 'submit'; form: string; onClick?: never; href?: never };

function Button(props: ButtonProps) {
  switch (props.variant) {
    case 'link':
      return <a href={props.href}>Link</a>;
    case 'button':
      return <button onClick={props.onClick}>Button</button>;
    case 'submit':
      return <button type="submit" form={props.form}>Submit</button>;
  }
}

// Notification component with discriminated union
type NotificationProps =
  | { type: 'success'; message: string; duration?: number }
  | { type: 'error'; message: string; retryAction: () => void }
  | { type: 'loading'; progress: number };

function Notification(props: NotificationProps) {
  switch (props.type) {
    case 'success':
      return <div className="success">{props.message}</div>;
    case 'error':
      return (
        <div className="error">
          {props.message}
          <button onClick={props.retryAction}>Retry</button>
        </div>
      );
    case 'loading':
      return <div className="loading">{props.progress}%</div>;
  }
}

// Exhaustive checking with never
function assertNever(x: never): never {
  throw new Error(`Unexpected value: ${x}`);
}

function getIcon(props: NotificationProps) {
  switch (props.type) {
    case 'success': return '✓';
    case 'error': return '✗';
    case 'loading': return '⟳';
    default: return assertNever(props); // Compile error if case missed
  }
}

5. Event Handler Types

// React event types mirror DOM events with generics
function Form() {
  const handleChange = (e: React.ChangeEvent<HTMLInputElement>) => {
    console.log(e.target.value);  // string
  };

  const handleSubmit = (e: React.FormEvent<HTMLFormElement>) => {
    e.preventDefault();
  };

  const handleClick = (e: React.MouseEvent<HTMLButtonElement>) => {
    console.log(e.clientX, e.clientY);
  };

  const handleKeyDown = (e: React.KeyboardEvent<HTMLInputElement>) => {
    if (e.key === 'Enter') { /* submit */ }
  };

  const handleFocus = (e: React.FocusEvent<HTMLInputElement>) => {
    e.target.select();
  };

  const handleDrag = (e: React.DragEvent<HTMLDivElement>) => {
    e.dataTransfer.setData('text/plain', 'data');
  };

  return (
    <form onSubmit={handleSubmit}>
      <input onChange={handleChange} onKeyDown={handleKeyDown} />
      <button onClick={handleClick}>Submit</button>
    </form>
  );
}

// Event handler prop types (for component interfaces)
interface FormFieldProps {

  onBlur?: React.FocusEventHandler<HTMLInputElement>;
  onKeyDown?: React.KeyboardEventHandler<HTMLInputElement>;

6. Ref Types

// useRef with DOM element
function TextInput() {
  const inputRef = useRef<HTMLInputElement>(null);

  const focus = () => {
    inputRef.current?.focus();  // HTMLInputElement | null
  };

  return <input ref={inputRef} />;
}

// useRef as mutable container (no null)
function Timer() {
  const intervalRef = useRef<ReturnType<typeof setInterval>>(undefined);

  useEffect(() => {
    intervalRef.current = setInterval(() => tick(), 1000);
    return () => clearInterval(intervalRef.current);
  }, []);
}

// forwardRef with TypeScript
interface InputProps {
  label: string;
  error?: string;
}

const Input = forwardRef<HTMLInputElement, InputProps>(
  function Input({ label, error }, ref) {
    return (
      <div>
        <label>{label}</label>
        <input ref={ref} />
        {error && <span>{error}</span>}
      </div>
    );
  }
);

// Callback refs
function MeasuredBox() {
  const [height, setHeight] = useState(0);

  const measuredRef = useCallback((node: HTMLDivElement | null) => {
    if (node !== null) {
      setHeight(node.getBoundingClientRect().height);
    }
  }, []);

  return <div ref={measuredRef}>Content</div>;
}

// useImperativeHandle
interface EditorHandle {
  focus: () => void;
  getValue: () => string;
  setValue: (value: string) => void;
}

const Editor = forwardRef<EditorHandle, EditorProps>(
  function Editor(props, ref) {
    const inputRef = useRef<HTMLTextAreaElement>(null);

    useImperativeHandle(ref, () => ({
      focus: () => inputRef.current?.focus(),
      getValue: () => inputRef.current?.value ?? '',
      setValue: (v) => { if (inputRef.current) inputRef.current.value = v; },
    }));

    return <textarea ref={inputRef} />;
  }
);

7. Context Typing

// Typed context with proper null handling
interface AuthContextType {
  user: User | null;
  login: (credentials: Credentials) => Promise<void>;
  logout: () => void;
  isLoading: boolean;
}

const AuthContext = createContext<AuthContextType | null>(null);

function useAuth(): AuthContextType {
  const context = useContext(AuthContext);
  if (context === null) {
    throw new Error('useAuth must be used within AuthProvider');
  }
  return context;
}

function AuthProvider({ children }: { children: React.ReactNode }) {
  const [user, setUser] = useState<User | null>(null);
  const [isLoading, setIsLoading] = useState(true);

  const login = async (credentials: Credentials) => {
    setIsLoading(true);
    const user = await authApi.login(credentials);
    setUser(user);
    setIsLoading(false);
  };

  const logout = () => { setUser(null); };

  return (
    <AuthContext.Provider value={{ user, login, logout, isLoading }}>
      {children}
    </AuthContext.Provider>
  );
}

// Generic context factory
function createSafeContext<T>(displayName: string) {
  const Context = createContext<T | undefined>(undefined);
  Context.displayName = displayName;

  function useContext_() {
    const ctx = useContext(Context);
    if (ctx === undefined) {
      throw new Error(`use${displayName} must be used within ${displayName}Provider`);
    }
    return ctx;
  }

  return [Context.Provider, useContext_] as const;
}

const [ThemeProvider, useTheme] = createSafeContext<ThemeContextType>('Theme');

// Type-safe CSS property values
type CSSUnit = 'px' | 'rem' | 'em' | '%' | 'vh' | 'vw';

// Type-safe color tokens
type ColorShade = 50 | 100 | 200 | 300 | 400 | 500 | 600 | 700 | 800 | 900;
type ColorName = 'red' | 'blue' | 'green' | 'gray';
type ColorToken = `${ColorName}-${ColorShade}`;

function getColor(token: ColorToken): string {
  const colors: Record<ColorToken, string> = {
    'red-500': '#ef4444',
    'blue-500': '#3b82f6',
    // ...
  } as Record<ColorToken, string>;
  return colors[token];
}

9. Strict Mode Configuration

// tsconfig.json strict mode flags
{
  "compilerOptions": {
    "strict": true,
    // Equivalent to enabling ALL of these:
    "noImplicitAny": true,           // Error on implicit any
    "strictNullChecks": true,        // null/undefined are distinct types
    "strictFunctionTypes": true,     // Contravariant function params
    "strictBindCallApply": true,     // Type-check bind/call/apply
    "strictPropertyInitialization": true, // Class props must be initialized
    "noImplicitThis": true,          // Error on implicit this
    "alwaysStrict": true,            // Emit "use strict"
    "useUnknownInCatchVariables": true, // catch(e) → e: unknown

    // Additional strictness (not in "strict"):
    "noUncheckedIndexedAccess": true,  // T[] access returns T | undefined
    "exactOptionalPropertyTypes": true, // undefined !== missing
    "noImplicitReturns": true,
    "noFallthroughCasesInSwitch": true,
    "noImplicitOverride": true,
  }
}

noUncheckedIndexedAccess Impact

Enabling noUncheckedIndexedAccess means arr[0] returns T | undefined instead of T. This catches real bugs but requires more null checks. For React, this affects Object.keys() iterations, array destructuring, and record lookups. It's worth enabling but expect a migration effort on existing codebases.

10. Type-Safe Routing

// Type-safe route definitions (React Router v6 pattern)
const routes = {
  home: '/',
  users: '/users',
  userDetail: '/users/:userId',
  userPosts: '/users/:userId/posts/:postId',
  settings: '/settings',
} as const;

type Routes = typeof routes;
type RoutePath = Routes[keyof Routes];

// Extract params from route path
type ExtractParams<T extends string> =
  T extends `${string}:${infer Param}/${infer Rest}`
    ? { [K in Param | keyof ExtractParams<Rest>]: string }
    : T extends `${string}:${infer Param}`
      ? { [K in Param]: string }
      : {};

// Usage
type UserDetailParams = ExtractParams<typeof routes.userDetail>;
// { userId: string }

type UserPostsParams = ExtractParams<typeof routes.userPosts>;
// { userId: string; postId: string }

// Type-safe navigation helper
function buildPath<T extends RoutePath>(
  route: T,
  params: ExtractParams<T>
): string {
  let path: string = route;
  for (const [key, value] of Object.entries(params)) {
    path = path.replace(`:${key}`, value as string);
  }
  return path;
}

buildPath(routes.userDetail, { userId: '123' }); // '/users/123'
