useState for Simple State

useState is the right choice for independent, simple values: toggles, counters, form inputs, selection state. Each useState call manages one concern.

// Simple independent state values
function UserForm() {
  const [name, setName] = useState('');
  const [email, setEmail] = useState('');
  const [agreed, setAgreed] = useState(false);

  return (
    <form>
      <input value={name} onChange={e => setName(e.target.value)} />
      <input value={email} onChange={e => setEmail(e.target.value)} />
      <input
        type="checkbox"
        checked={agreed}
        onChange={e => setAgreed(e.target.checked)}
      />
    </form>
  );
}

// When multiple states always change together, consider grouping:
function DragHandler() {
  // BAD: two states that always update together
  const [x, setX] = useState(0);
  const [y, setY] = useState(0);

  // BETTER: single state object
  const [position, setPosition] = useState({ x: 0, y: 0 });

  function handleDrag(e) {
    setPosition({ x: e.clientX, y: e.clientY });
  }
}

useReducer for Complex Transitions

Use useReducer when state transitions are complex, interdependent, or need to be modeled explicitly.

// Complex state with interdependent transitions
type FetchState<T> = 
  | { status: 'idle' }
  | { status: 'loading' }
  | { status: 'success'; data: T }
  | { status: 'error'; error: Error };

type FetchAction<T> =
  | { type: 'FETCH_START' }
  | { type: 'FETCH_SUCCESS'; data: T }
  | { type: 'FETCH_ERROR'; error: Error }
  | { type: 'RESET' };

function fetchReducer<T>(state: FetchState<T>, action: FetchAction<T>): FetchState<T> {
  switch (action.type) {
    case 'FETCH_START':
      return { status: 'loading' };
    case 'FETCH_SUCCESS':
      return { status: 'success', data: action.data };
    case 'FETCH_ERROR':
      return { status: 'error', error: action.error };
    case 'RESET':
      return { status: 'idle' };
  }
}

function UserProfile({ userId }) {
  const [state, dispatch] = useReducer(fetchReducer, { status: 'idle' });

  useEffect(() => {
    dispatch({ type: 'FETCH_START' });
    fetchUser(userId)
      .then(data => dispatch({ type: 'FETCH_SUCCESS', data }))
      .catch(error => dispatch({ type: 'FETCH_ERROR', error }));
  }, [userId]);

  switch (state.status) {
    case 'idle':    return null;
    case 'loading': return <Spinner />;
    case 'success': return <div>{state.data.name}</div>;
    case 'error':   return <div>Error: {state.error.message}</div>;
  }
}
Architect Insight — useState vs useReducer Decision

Choose useReducer when: (1) next state depends on previous state in complex ways, (2) multiple sub-values update together (impossible states should be unrepresentable), (3) you want to pass dispatch down instead of multiple callbacks, (4) state transitions need to be testable independently. Choose useState for everything else — it's simpler and sufficient for most cases.

State Colocation Principle

State should live as close as possible to where it's used. This minimizes re-renders, improves code locality, and makes components more self-contained.

State Colocation Spectrum: Local State Lifted State Context/Store ────────── ──────────── ────────────── Component-level Shared ancestor App-wide useState/useReducer Props drilling Provider/Store ◄─── Prefer this end ───────────────── Use only when needed ──► Rule: Start local. Lift only when a sibling needs it. Don't put everything in global state "just in case." 
// ANTI-PATTERN: everything in global state
const useGlobalStore = create((set) => ({
  searchQuery: '',           // only used in SearchBar
  isDropdownOpen: false,     // only used in Dropdown
  selectedTab: 'home',       // only used in TabBar
  formData: {},              // only used in Form
  // ... 50 more local concerns in global state
}));

// CORRECT: colocate state with its consumer
function SearchBar() {
  const [query, setQuery] = useState(''); // local — only SearchBar needs it
  return <input value={query} onChange={e => setQuery(e.target.value)} />;
}

function Dropdown() {
  const [isOpen, setIsOpen] = useState(false); // local
  return (
    <div>
      <button onClick={() => setIsOpen(!isOpen)}>Toggle</button>
      {isOpen && <DropdownMenu />}
    </div>
  );
}

Lifting State Up vs Pushing State Down

Lifting State Up

// When siblings need to share state, lift to their common ancestor:

// BEFORE: each panel tracks its own active state
function Accordion() {
  return (
    <div>
      <Panel title="Section 1">Content 1</Panel>
      <Panel title="Section 2">Content 2</Panel>
    </div>
  );
}
function Panel({ title, children }) {
  const [isOpen, setIsOpen] = useState(false); // can't enforce "only one open"
  return /* ... */;
}

// AFTER: lift state to parent to enforce single-open constraint
function Accordion() {
  const [activeIndex, setActiveIndex] = useState(null);
  return (
    <div>
      <Panel
        title="Section 1"
        isOpen={activeIndex === 0}
        onToggle={() => setActiveIndex(activeIndex === 0 ? null : 0)}
      >Content 1</Panel>
      <Panel
        title="Section 2"
        isOpen={activeIndex === 1}
        onToggle={() => setActiveIndex(activeIndex === 1 ? null : 1)}
      >Content 2</Panel>
    </div>
  );
}

Pushing State Down

// When only part of a component needs state, push it into a child:

// BEFORE: entire App re-renders when hover state changes
function App() {
  const [isHovered, setIsHovered] = useState(false);
  return (
    <div>
      <ExpensiveTree /> {/* re-renders unnecessarily */}
      <div
        onMouseEnter={() => setIsHovered(true)}
        onMouseLeave={() => setIsHovered(false)}
        style={{ color: isHovered ? 'red' : 'black' }}
      >
        Hover me
      </div>
    </div>
  );
}

// AFTER: push hover state into its own component
function App() {
  return (
    <div>
      <ExpensiveTree /> {/* no longer re-renders */}
      <HoverableText />
    </div>
  );
}

function HoverableText() {
  const [isHovered, setIsHovered] = useState(false);
  return (
    <div
      onMouseEnter={() => setIsHovered(true)}
      onMouseLeave={() => setIsHovered(false)}
      style={{ color: isHovered ? 'red' : 'black' }}
    >
      Hover me
    </div>
  );
}

Component Composition to Avoid Prop Drilling

// PROBLEM: prop drilling through intermediate components
function App() {
  const [user, setUser] = useState(null);
  return <Layout user={user} />;
}
function Layout({ user }) {
  return <Sidebar user={user} />;  // Layout doesn't use user
}
function Sidebar({ user }) {
  return <UserInfo user={user} />; // Sidebar doesn't use user
}
function UserInfo({ user }) {
  return <span>{user?.name}</span>; // only consumer
}

// SOLUTION: component composition (children pattern)
function App() {
  const [user, setUser] = useState(null);
  return (
    <Layout
      sidebar={<Sidebar userInfo={<UserInfo user={user} />} />}
    />
  );
}
function Layout({ sidebar }) {
  return <div className="layout">{sidebar}</div>;
}
function Sidebar({ userInfo }) {
  return <div className="sidebar">{userInfo}</div>;
}

// Layout and Sidebar don't know about user at all.
// The JSX element is created at the top and passed down as a prop.

State Initialization Patterns

// Pattern 1: Lazy initialization (expensive computation)
const [data, setData] = useState(() => {
  return JSON.parse(localStorage.getItem('data') || '{}');
});

// Pattern 2: Derived initial state from props
function EditForm({ initialValues }) {
  const [values, setValues] = useState(initialValues);
  // WARNING: this only uses initialValues on MOUNT.
  // If initialValues prop changes, state won't update.
  // This is intentional for "uncontrolled with initial value" pattern.
}

// Pattern 3: Reset state when key changes
function EditForm({ userId, initialValues }) {
  const [values, setValues] = useState(initialValues);
  // To reset when userId changes, use key prop on parent:
  // <EditForm key={userId} initialValues={...} />
  // key change forces remount → fresh state
}

// Pattern 4: useReducer with init function
const [state, dispatch] = useReducer(
  reducer,
  initialArg,
  (arg) => ({ ...arg, computed: derive(arg) }) // init function
);

// Pattern 5: Controlled vs uncontrolled pattern
function Input({ value: controlledValue, defaultValue, onChange }) {
  const [internalValue, setInternalValue] = useState(defaultValue ?? '');
  const isControlled = controlledValue !== undefined;
  const value = isControlled ? controlledValue : internalValue;

  function handleChange(e) {
    if (!isControlled) setInternalValue(e.target.value);
    onChange?.(e.target.value);
  }

  return <input value={value} onChange={handleChange} />;
}

Derived State: Compute vs Store

Anti-Pattern: Storing Derived State

If a value can be computed from existing state or props, compute it during render instead of storing it in separate state. Storing derived values creates synchronization bugs and unnecessary re-render cycles.

// ANTI-PATTERN: storing derived state
function TodoList({ todos }) {
  const [filteredTodos, setFilteredTodos] = useState([]);
  const [filter, setFilter] = useState('all');

  useEffect(() => {
    setFilteredTodos(
      filter === 'all' ? todos : todos.filter(t => t.status === filter)
    );
  }, [todos, filter]);
  // Problems:
  // 1. Extra render cycle (render → effect → setState → re-render)
  // 2. Stale data between renders (filteredTodos lags by one render)
  // 3. More state to manage and synchronize

  // CORRECT: compute during render
  const filteredTodos = filter === 'all'
    ? todos
    : todos.filter(t => t.status === filter);
  // Computed synchronously, always in sync, no extra render

  // For expensive computations, use useMemo:
  const filteredTodos = useMemo(
    () => filter === 'all'
      ? todos
      : todos.filter(t => t.status === filter),
    [todos, filter]
  );
}

// Another common anti-pattern:
function UserList({ users }) {
  const [sortedUsers, setSortedUsers] = useState([]);
  const [sortKey, setSortKey] = useState('name');

  // DON'T: sync derived state
  useEffect(() => {
    setSortedUsers([...users].sort((a, b) => a[sortKey].localeCompare(b[sortKey])));
  }, [users, sortKey]);

  // DO: compute inline
  const sortedUsers = useMemo(
    () => [...users].sort((a, b) => a[sortKey].localeCompare(b[sortKey])),
    [users, sortKey]
  );
}

State Update Batching Implications

// React 18 batches ALL state updates automatically.
// This affects how you think about state structure:

// With batching, multiple setStates = one render:
function handleSubmit() {
  setLoading(true);     // queued
  setError(null);       // queued
  setData(null);        // queued
  // → single re-render with all three changes
}

// This means useReducer's advantage of "atomic multi-value updates"
// is less important in React 18 — useState batching achieves the same.

// However, useReducer still wins for:
// 1. Impossible state prevention (type safety)
type State =
  | { status: 'idle' }
  | { status: 'loading' }
  | { status: 'success'; data: Data }
  | { status: 'error'; error: Error };
// Can't have status='success' without data

// 2. Complex transition logic
function reducer(state, action) {
  switch (action.type) {
    case 'ADD_ITEM':
      if (state.items.length >= state.maxItems) return state; // guard
      return {
        ...state,
        items: [...state.items, action.item],
        total: state.total + action.item.price,
        lastModified: Date.now(),
      };
  }
}

// 3. Testability
test('ADD_ITEM respects maxItems', () => {
  const state = { items: [a, b, c], maxItems: 3, total: 30 };
  const result = reducer(state, { type: 'ADD_ITEM', item: d });
  expect(result).toBe(state); // unchanged — guard prevented add
});
State Management Decision Tree

Is it used by one component? → useState/useReducer locally. Is it shared by siblings? → Lift to common ancestor. Is it deeply nested? → Component composition or Context. Is it app-wide? → Context (if rarely changes) or external store (if frequent updates). Is it server data? → TanStack Query/SWR. Is it complex with many transitions? → useReducer or XState.
