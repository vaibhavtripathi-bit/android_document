Controlled vs Uncontrolled Forms

AspectControlledUncontrolled
State ownerReact (useState)DOM (ref)
Value accessstate variableref.current.value
Re-rendersEvery keystrokeNone (until submit)
Validation timingReal-time (on change)On submit
PerformanceCan be slow for large formsFast (no re-renders)
ComplexityMore code per fieldLess code, less control
 
// Controlled form — React owns the state
function ControlledForm() {
  const [name, setName] = useState('');
  const [email, setEmail] = useState('');

  function handleSubmit(e) {
    e.preventDefault();
    console.log({ name, email }); // values from state
  }

  return (
    <form onSubmit={handleSubmit}>
      <input value={name} onChange={e => setName(e.target.value)} />
      <input value={email} onChange={e => setEmail(e.target.value)} />
      <button type="submit">Submit</button>
    </form>
  );
}

// Uncontrolled form — DOM owns the state
function UncontrolledForm() {
  const nameRef = useRef(null);
  const emailRef = useRef(null);

  function handleSubmit(e) {
    e.preventDefault();
    console.log({
      name: nameRef.current.value,
      email: emailRef.current.value,
    });
  }

  return (
    <form onSubmit={handleSubmit}>
      <input ref={nameRef} defaultValue="" />
      <input ref={emailRef} defaultValue="" />
      <button type="submit">Submit</button>
    </form>
  );
}

// Hybrid: FormData API (modern approach)
function FormDataForm() {
  function handleSubmit(e) {
    e.preventDefault();
    const formData = new FormData(e.target);
    const data = Object.fromEntries(formData);
    console.log(data); // { name: '...', email: '...' }
  }

  return (
    <form onSubmit={handleSubmit}>
      <input name="name" />
      <input name="email" />
      <button type="submit">Submit</button>
    </form>
  );
}

React Hook Form

React Hook Form (RHF) uses uncontrolled inputs with refs for performance, while providing a controlled-like API. It minimizes re-renders by isolating field state.

import { useForm, useWatch, useFormContext, Controller } from 'react-hook-form';

function RegistrationForm() {
  const {
    register,      // connects input to form state
    handleSubmit,  // wraps submit handler with validation
    formState: { errors, isSubmitting, isDirty, isValid },
    watch,         // subscribe to field changes
    setValue,      // programmatically set values
    reset,         // reset form to defaults
    trigger,       // manually trigger validation
  } = useForm({
    defaultValues: {
      name: '',
      email: '',
      role: 'user',
    },
    mode: 'onBlur', // validate on blur (onChange, onSubmit, onBlur, all)
  });

  const onSubmit = async (data) => {
    await submitToServer(data);
  };

  return (
    <form onSubmit={handleSubmit(onSubmit)}>
      <input
        {...register('name', {
          required: 'Name is required',
          minLength: { value: 2, message: 'Min 2 characters' },
        })}
      />
      {errors.name && <span>{errors.name.message}</span>}

      <input
        {...register('email', {
          required: 'Email is required',
          pattern: {
            value: /^[A-Z0-9._%+-]+@[A-Z0-9.-]+\.[A-Z]{2,}$/i,
            message: 'Invalid email',
          },
        })}
      />
      {errors.email && <span>{errors.email.message}</span>}

      <select {...register('role')}>
        <option value="user">User</option>
        <option value="admin">Admin</option>
      </select>

      <button type="submit" disabled={isSubmitting}>
        {isSubmitting ? 'Submitting...' : 'Submit'}
      </button>
    </form>
  );
}

How register() Works

// register('name') returns:
{
  name: 'name',
  ref: (element) => { /* stores ref to DOM element */ },
  onChange: (event) => { /* updates internal state, triggers validation */ },
  onBlur: (event) => { /* triggers validation if mode includes blur */ },
}

// Internally, RHF stores form state in a ref (not useState),
// so field changes don't cause re-renders.
// Only specific subscriptions (errors, isDirty, etc.) trigger re-renders.

function FormWithCustomInput() {

Schema Validation with Zod/Yup

import { useForm } from 'react-hook-form';
import { zodResolver } from '@hookform/resolvers/zod';
import { z } from 'zod';

// Define schema:
const registrationSchema = z.object({
  name: z.string().min(2, 'Min 2 characters').max(50),
  email: z.string().email('Invalid email'),
  password: z.string()
    .min(8, 'Min 8 characters')
    .regex(/[A-Z]/, 'Must contain uppercase')
    .regex(/[0-9]/, 'Must contain number'),
  confirmPassword: z.string(),
  age: z.number().min(18, 'Must be 18+').optional(),
  role: z.enum(['user', 'admin', 'moderator']),
}).refine(data => data.password === data.confirmPassword, {
  message: 'Passwords must match',
  path: ['confirmPassword'],
});

type RegistrationData = z.infer<typeof registrationSchema>;

function RegistrationForm() {
  const { register, handleSubmit, formState: { errors } } = useForm<RegistrationData>({
    resolver: zodResolver(registrationSchema),
    defaultValues: { role: 'user' },
  });

  return (
    <form onSubmit={handleSubmit(onSubmit)}>
      <input {...register('name')} />
      {errors.name && <span>{errors.name.message}</span>}
      {/* ... other fields */}
    </form>
  );
}

Performance: Minimizing Re-Renders

// RHF's key performance optimization: isolate re-renders per field

// useWatch — subscribe to specific fields without re-rendering entire form:
function PriceDisplay() {
  const price = useWatch({ name: 'price' });
  const quantity = useWatch({ name: 'quantity' });
  // Only this component re-renders when price or quantity changes
  return <p>Total: ${price * quantity}</p>;
}

// useFormContext — access form in nested components without prop drilling:
function NestedField() {
  const { register, formState: { errors } } = useFormContext();
  return (
    <div>
      <input {...register('address.street')} />
      {errors.address?.street && <span>{errors.address.street.message}</span>}
    </div>
  );
}

function ParentForm() {
  const methods = useForm();
  return (
    <FormProvider {...methods}>
      <form onSubmit={methods.handleSubmit(onSubmit)}>
        <NestedField />
      </form>
    </FormProvider>
  );
}
Re-render Comparison (form with 20 fields): Controlled (useState per field): User types in field #5 → setState → entire form re-renders (20 fields) → 20 input re-renders per keystroke React Hook Form: User types in field #5 → ref update (no re-render) → only error display re-renders (if validation changes) → 0-1 re-renders per keystroke

Field Arrays

import { useFieldArray, useForm } from 'react-hook-form';

function DynamicForm() {
  const { control, register, handleSubmit } = useForm({
    defaultValues: {
      items: [{ name: '', quantity: 1 }],
    },
  });

  const { fields, append, remove, move, swap } = useFieldArray({
    control,
    name: 'items',
  });

  return (
    <form onSubmit={handleSubmit(onSubmit)}>
      {fields.map((field, index) => (
        <div key={field.id}>
          <input {...register(`items.${index}.name`)} />
          <input
            type="number"
            {...register(`items.${index}.quantity`, { valueAsNumber: true })}
          />
          <button type="button" onClick={() => remove(index)}>Remove</button>
        </div>
      ))}
      <button type="button" onClick={() => append({ name: '', quantity: 1 })}>
        Add Item
      </button>
      <button type="submit">Submit</button>
    </form>
  );
}

Server Actions and Form Handling in React 19

// React 19 introduces server actions for form handling:

// Server action (runs on the server):
'use server';
async function createUser(formData: FormData) {
  const name = formData.get('name');
  const email = formData.get('email');

  // Validate on server
  if (!name || !email) {
    return { error: 'Name and email required' };
  }

  await db.users.create({ name, email });
  revalidatePath('/users');
  return { success: true };
}

// Client component using server action:
function CreateUserForm() {
  return (
    <form action={createUser}>
      <input name="name" required />
      <input name="email" type="email" required />
      <button type="submit">Create</button>
    </form>
  );
}

useFormStatus & useFormState (React 19)

import { useFormStatus, useFormState } from 'react-dom';

// useFormStatus — read pending state of parent form
function SubmitButton() {
  const { pending, data, method, action } = useFormStatus();
  // Must be a child of a <form> element
  return (
    <button type="submit" disabled={pending}>
      {pending ? 'Submitting...' : 'Submit'}
    </button>
  );
}

// useFormState — read form action return value
function CreateUserForm() {
  const [state, formAction] = useFormState(createUser, null);

  return (
    <form action={formAction}>
      {state?.error && <p className="error">{state.error}</p>}
      {state?.success && <p className="success">User created!</p>}
      <input name="name" required />
      <input name="email" type="email" required />
      <SubmitButton />
    </form>
  );
}

// useOptimistic — show optimistic state during form submission
import { useOptimistic } from 'react';

function TodoList({ todos }) {
  const [optimisticTodos, addOptimisticTodo] = useOptimistic(
    todos,
    (state, newTodo) => [...state, { ...newTodo, pending: true }]
  );

  async function addTodo(formData) {
    const text = formData.get('text');
    addOptimisticTodo({ id: Date.now(), text }); // show immediately
    await createTodo(text); // server action
  }

  return (
    <div>
      {optimisticTodos.map(todo => (
        <div key={todo.id} style={{ opacity: todo.pending ? 0.5 : 1 }}>
          {todo.text}
        </div>
      ))}
      <form action={addTodo}>
        <input name="text" />
        <SubmitButton />
      </form>
    </div>
  );
}
Architect Insight — Form Strategy Decision

For L6 interviews: (1) Simple forms (3-5 fields): controlled with useState. (2) Complex forms (10+ fields, validation): React Hook Form + Zod. (3) Server-rendered forms: React 19 server actions + useFormStatus. (4) Never put form state in global stores (Redux/Zustand) — forms are inherently local. (5) Schema validation (Zod) provides both runtime validation and TypeScript types from a single source of truth.
