API Abstraction Layers

In large applications, direct fetch calls scattered throughout components create tight coupling to API implementation details. A layered abstraction provides a clean separation between data fetching mechanics and business logic.

Data Layer Architecture ═══════════════════════════════════════════════════════════ ┌─────────────────────────────────────────────────────┐ │ COMPONENTS │ │ useUsers() useProducts() useOrders() │ └──────────────────────┬──────────────────────────────┘ │ Custom hooks ┌──────────────────────┴──────────────────────────────┐ │ QUERY LAYER │ │ React Query / SWR / Apollo │ │ (caching, deduplication, background refresh) │ └──────────────────────┬──────────────────────────────┘ │ Repository methods ┌──────────────────────┴──────────────────────────────┐ │ REPOSITORY LAYER │ │ UserRepository ProductRepository OrderRepository │ │ (business logic, data transformation) │ └──────────────────────┬──────────────────────────────┘ │ API client methods ┌──────────────────────┴──────────────────────────────┐ │ API CLIENT │ │ Axios / Fetch wrapper │ │ (auth headers, base URL, interceptors, retry) │ └──────────────────────┬──────────────────────────────┘ │ HTTP ┌────┴────┐ │ Backend │ └─────────┘ 
// Layer 1: API Client — handles transport concerns
class ApiClient {
  private baseUrl: string;
  private getToken: () => string | null;

  constructor(config: { baseUrl: string; getToken: () => string | null }) {
    this.baseUrl = config.baseUrl;
    this.getToken = config.getToken;
  }

  private async request<T>(endpoint: string, options: RequestInit = {}): Promise<T> {
    const token = this.getToken();
    const response = await fetch(`${this.baseUrl}${endpoint}`, {
      ...options,
      headers: {
        'Content-Type': 'application/json',
        ...(token ? { Authorization: `Bearer ${token}` } : {}),
        ...options.headers,
      },
    });

    if (!response.ok) {
      const error = await response.json().catch(() => ({}));
      throw new ApiError(response.status, error.message, error.code);
    }

    return response.json();
  }

  get<T>(endpoint: string, params?: Record<string, string>) {
    const url = params ? `${endpoint}?${new URLSearchParams(params)}` : endpoint;
    return this.request<T>(url);
  }

  post<T>(endpoint: string, data: unknown) {
    return this.request<T>(endpoint, {
      method: 'POST',
      body: JSON.stringify(data),
    });
  }

  put<T>(endpoint: string, data: unknown) {
    return this.request<T>(endpoint, {
      method: 'PUT',
      body: JSON.stringify(data),
    });
  }

  delete<T>(endpoint: string) {
    return this.request<T>(endpoint, { method: 'DELETE' });
  }
}

// Layer 2: Repository — handles business logic
class UserRepository {
  constructor(private api: ApiClient) {}

  async getUsers(filters?: UserFilters): Promise<User[]> {
    const raw = await this.api.get<ApiUser[]>('/users', filters);
    return raw.map(this.toDomain);
  }

  async getUser(id: string): Promise<User> {
    const raw = await this.api.get<ApiUser>(`/users/${id}`);
    return this.toDomain(raw);
  }

  async createUser(data: CreateUserInput): Promise<User> {
    const raw = await this.api.post<ApiUser>('/users', this.toApi(data));
    return this.toDomain(raw);
  }

  private toDomain(raw: ApiUser): User {
    return {
      id: raw.id,
      fullName: `${raw.first_name} ${raw.last_name}`,
      email: raw.email_address,
      role: raw.role_id === 1 ? 'admin' : 'user',
      createdAt: new Date(raw.created_at),
    };
  }

  private toApi(data: CreateUserInput): Partial<ApiUser> {
    const [firstName, ...rest] = data.fullName.split(' ');
    return {
      first_name: firstName,
      last_name: rest.join(' '),
      email_address: data.email,
    };
  }
}

// Layer 3: React Query hooks — handles caching and UI state
function useUsers(filters?: UserFilters) {
  const repo = useUserRepository();
  return useQuery({
    queryKey: ['users', filters],
    queryFn: () => repo.getUsers(filters),
    staleTime: 5 * 60 * 1000,
  });
}

function useCreateUser() {
  const repo = useUserRepository();
  const queryClient = useQueryClient();

  return useMutation({
    mutationFn: (data: CreateUserInput) => repo.createUser(data),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['users'] });
    },
  });
}
Architect Insight

The repository pattern provides a critical boundary: components never see API response shapes. When the backend changes field names from email_address to email, you update one mapper in the repository — not 50 components. This is especially valuable during API migrations or when consuming multiple backends.

Data Normalization

Normalization eliminates data duplication by storing entities in flat lookup tables, similar to a relational database. This prevents inconsistencies when the same entity appears in multiple places.

// DENORMALIZED (typical API response)

      title: 'React Patterns',
      author: { id: 'a1', name: 'Alice', avatar: 'alice.jpg' },

// Problem: Alice appears twice — update one, the other is stale

// NORMALIZED (entity tables)
const normalizedState = {

Optimistic Updates Architecture

// Optimistic update with React Query
function useToggleLike(postId: string) {
  const queryClient = useQueryClient();

  return useMutation({
    mutationFn: (liked: boolean) =>
      api.post(`/posts/${postId}/like`, { liked }),

    onMutate: async (liked) => {
      await queryClient.cancelQueries({ queryKey: ['post', postId] });

      const previous = queryClient.getQueryData<Post>(['post', postId]);

      queryClient.setQueryData<Post>(['post', postId], (old) => ({
        ...old!,
        liked,
        likeCount: old!.likeCount + (liked ? 1 : -1),
      }));

      return { previous };
    },

    onError: (_err, _liked, context) => {
      queryClient.setQueryData(['post', postId], context?.previous);
      toast.error('Failed to update. Please try again.');
    },

    onSettled: () => {
      queryClient.invalidateQueries({ queryKey: ['post', postId] });
    },
  });
}

// Optimistic update with pending queue (offline-capable)
interface PendingMutation {
  id: string;
  type: 'like' | 'comment' | 'edit';
  payload: unknown;
  timestamp: number;
  retryCount: number;
  status: 'pending' | 'in-flight' | 'failed';
}

class MutationQueue {
  private queue: PendingMutation[] = [];
  private processing = false;

  add(mutation: Omit<PendingMutation, 'id' | 'timestamp' | 'retryCount' | 'status'>) {
    const entry: PendingMutation = {
      ...mutation,
      id: crypto.randomUUID(),
      timestamp: Date.now(),
      retryCount: 0,
      status: 'pending',
    };
    this.queue.push(entry);
    this.persist();
    this.process();
    return entry.id;
  }

  private async process() {
    if (this.processing || !navigator.onLine) return;
    this.processing = true;

    while (this.queue.length > 0) {
      const mutation = this.queue[0];
      mutation.status = 'in-flight';

      try {
        await this.execute(mutation);
        this.queue.shift();
      } catch (error) {
        mutation.retryCount++;
        mutation.status = 'failed';
        if (mutation.retryCount >= 3) {
          this.queue.shift();
          this.onPermanentFailure(mutation);
        } else {
          await this.delay(Math.pow(2, mutation.retryCount) * 1000);
        }
      }
    }

    this.processing = false;
    this.persist();
  }

  private persist() {
    localStorage.setItem('mutation-queue', JSON.stringify(this.queue));
  }
}

Real-Time Data Patterns

// WebSocket with React Query integration
function useRealtimeQuery<T>(queryKey: string[], fetchFn: () => Promise<T>, wsChannel: string) {
  const queryClient = useQueryClient();

  useEffect(() => {
    const ws = new WebSocket(`wss://api.example.com/ws`);

    ws.onopen = () => {
      ws.send(JSON.stringify({ type: 'subscribe', channel: wsChannel }));
    };

    ws.onmessage = (event) => {
      const update = JSON.parse(event.data);
      queryClient.setQueryData(queryKey, (old: T) => {
        return applyUpdate(old, update);
      });
    };

    return () => {
      ws.send(JSON.stringify({ type: 'unsubscribe', channel: wsChannel }));
      ws.close();
    };
  }, [queryKey, wsChannel, queryClient]);

  return useQuery({ queryKey, queryFn: fetchFn });
}

// Server-Sent Events for one-way real-time
function useSSE<T>(url: string, queryKey: string[]) {
  const queryClient = useQueryClient();

  useEffect(() => {
    const source = new EventSource(url);

    source.onmessage = (event) => {
      const data = JSON.parse(event.data) as T;
      queryClient.setQueryData(queryKey, data);
    };

    source.onerror = () => {
      source.close();
      setTimeout(() => {
        // Reconnect logic
      }, 5000);
    };

    return () => source.close();
  }, [url, queryKey, queryClient]);

  return useQuery<T>({ queryKey, enabled: false });
}

// Polling fallback (simplest, most reliable)
function usePolling<T>(queryKey: string[], fetchFn: () => Promise<T>, interval = 5000) {
  return useQuery({
    queryKey,
    queryFn: fetchFn,
    refetchInterval: interval,
    refetchIntervalInBackground: false,
  });
}

Offline-First Architecture

// Offline-first data layer
class OfflineFirstRepository {
  constructor(
    private api: ApiClient,
    private cache: IndexedDBCache,
    private syncQueue: MutationQueue
  ) {}

  async getUsers(): Promise<User[]> {
    try {
      const remote = await this.api.get<User[]>('/users');
      await this.cache.set('users', remote);
      return remote;
    } catch (error) {
      if (!navigator.onLine) {
        const cached = await this.cache.get<User[]>('users');
        if (cached) return cached;
      }
      throw error;
    }
  }

  async createUser(data: CreateUserInput): Promise<User> {
    const optimisticUser: User = {
      id: `temp-${crypto.randomUUID()}`,
      ...data,
      _pending: true,
    };

    const cached = await this.cache.get<User[]>('users') || [];
    await this.cache.set('users', [...cached, optimisticUser]);

    if (navigator.onLine) {
      const user = await this.api.post<User>('/users', data);
      await this.cache.set('users',
        cached.map(u => u.id === optimisticUser.id ? user : u)
      );
      return user;
    } else {
      this.syncQueue.add({
        type: 'createUser',
        payload: data,
        optimisticId: optimisticUser.id,
      });
      return optimisticUser;
    }
  }
}

CQRS Pattern Adaptation for React

CQRS in React ═══════════════════════════════════════════════════════════ Traditional CQRS: Commands (writes) → Write Model → Event Store Queries (reads) → Read Model → Projections React Adaptation: Mutations (useMutation) → API POST/PUT/DELETE → Invalidate Queries (useQuery) → API GET → Cache ┌─────────────────────────────────────────────────────┐ │ COMPONENTS │ ├──────────────────┬──────────────────────────────────┤ │ QUERY SIDE │ COMMAND SIDE │ │ │ │ │ useQuery() │ useMutation() │ │ useInfinite() │ optimistic updates │ │ prefetch() │ mutation queue │ │ │ │ │ Stale-while- │ Fire-and-forget │ │ revalidate │ or await confirmation │ │ │ │ │ Read from cache │ Write to API │ │ Background │ Invalidate queries │ │ refresh │ on success │ └──────────────────┴──────────────────────────────────┘ 
// CQRS-style separation in React
// queries.ts — read side
export const userQueries = {
  all: () => ({ queryKey: ['users'], queryFn: () => userRepo.getAll() }),
  detail: (id: string) => ({
    queryKey: ['users', id],
    queryFn: () => userRepo.getById(id),
  }),
  byRole: (role: string) => ({
    queryKey: ['users', { role }],
    queryFn: () => userRepo.getByRole(role),
  }),
};

// commands.ts — write side
export function useCreateUser() {
  const queryClient = useQueryClient();
  return useMutation({
    mutationFn: userRepo.create,
    onSuccess: (newUser) => {
      queryClient.setQueryData(
        userQueries.detail(newUser.id).queryKey,
        newUser
      );
      queryClient.invalidateQueries({ queryKey: ['users'] });
    },
  });
}

export function useUpdateUser() {
  const queryClient = useQueryClient();
  return useMutation({
    mutationFn: ({ id, data }: { id: string; data: UpdateUserInput }) =>
      userRepo.update(id, data),
    onMutate: async ({ id, data }) => {
      await queryClient.cancelQueries({ queryKey: ['users', id] });
      const previous = queryClient.getQueryData(['users', id]);
      queryClient.setQueryData(['users', id], (old: User) => ({ ...old, ...data }));
      return { previous };
    },
    onError: (_err, { id }, context) => {
      queryClient.setQueryData(['users', id], context?.previous);
    },
    onSettled: (_data, _err, { id }) => {
      queryClient.invalidateQueries({ queryKey: ['users', id] });
    },
  });
}
Interview Tip

When discussing data architecture, emphasize the separation of concerns: transport (API client), business logic (repository), caching (React Query), and UI state (component). Explain that this layering makes it possible to swap backends, add offline support, or change caching strategies without touching component code. This is the kind of systems thinking expected at L6.
