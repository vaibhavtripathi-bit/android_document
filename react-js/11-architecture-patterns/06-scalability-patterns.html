Feature-Based Folder Structure

As codebases grow beyond 50+ developers, the folder structure becomes a critical architectural decision. Feature-based organization scales better than type-based because changes are colocated.

Barrel Exports and Tree-Shaking

// Barrel export (index.ts re-exports everything)
// features/auth/index.ts
export * from './components/LoginPage';
export * from './components/SignupPage';
export * from './hooks/useAuth';
export * from './hooks/usePermissions';

// THE PROBLEM: importing one thing pulls in everything
import { useAuth } from '@/features/auth';
// Bundler may include LoginPage, SignupPage, usePermissions too!

// WHY: barrel files create a single module that re-exports.
// If the bundler can't prove unused exports have no side effects,
// it includes them.

// SOLUTIONS:

// 1. Mark package as side-effect-free
// package.json
{ "sideEffects": false }

// 2. Use direct imports instead of barrel
import { useAuth } from '@/features/auth/hooks/useAuth';

// 3. Use named re-exports (not wildcard)
export { useAuth } from './hooks/useAuth';
export { LoginPage } from './components/LoginPage';

// 4. Configure bundler to handle barrels
// next.config.js
module.exports = {
  experimental: {
    optimizePackageImports: ['@/features/auth'],
  },
};
Critical — Barrel Export Performance

In a large monorepo, barrel exports can add 100KB+ to bundles. Webpack's tree-shaking struggles with barrels that re-export components with side effects (CSS imports, module-level code). Measure with webpack-bundle-analyzer and consider direct imports for large feature modules. Next.js 13.5+ added optimizePackageImports specifically to address this.

Lazy Loading Strategies

// Route-level code splitting (most common)
const Dashboard = lazy(() => import('./features/dashboard/DashboardPage'));
const Settings = lazy(() => import('./features/settings/SettingsPage'));
const Analytics = lazy(() => import('./features/analytics/AnalyticsPage'));

function AppRoutes() {
  return (
    <Suspense fallback={<PageSkeleton />}>
      <Routes>
        <Route path="/dashboard" element={<Dashboard />} />
        <Route path="/settings" element={<Settings />} />
        <Route path="/analytics" element={<Analytics />} />
      </Routes>
    </Suspense>
  );
}

// Component-level lazy loading
const HeavyChart = lazy(() => import('./components/HeavyChart'));

function Dashboard() {
  const [showChart, setShowChart] = useState(false);
  return (
    <div>
      <button onClick={() => setShowChart(true)}>Show Chart</button>
      {showChart && (
        <Suspense fallback={<ChartSkeleton />}>
          <HeavyChart />
        </Suspense>
      )}
    </div>
  );
}

// Preloading on hover/focus (anticipatory loading)
function NavLink({ to, children }) {
  const preload = useCallback(() => {
    const route = routeMap[to];
    if (route?.preload) route.preload();
  }, [to]);

  return (
    <Link
      to={to}
      onMouseEnter={preload}
      onFocus={preload}
    >
      {children}
    </Link>
  );
}

const routeMap = {
  '/dashboard': {
    component: lazy(() => import('./DashboardPage')),
    preload: () => import('./DashboardPage'),
  },
  '/analytics': {
    component: lazy(() => import('./AnalyticsPage')),
    preload: () => import('./AnalyticsPage'),
  },
};

// Intersection Observer-based lazy loading
function LazySection({ importFn, fallback, ...props }) {
  const [Component, setComponent] = useState(null);
  const ref = useRef(null);

  useEffect(() => {
    const observer = new IntersectionObserver(
      ([entry]) => {
        if (entry.isIntersecting) {
          importFn().then(mod => setComponent(() => mod.default));
          observer.disconnect();
        }
      },
      { rootMargin: '200px' }
    );

    if (ref.current) observer.observe(ref.current);
    return () => observer.disconnect();
  }, [importFn]);

  return (
    <div ref={ref}>
      {Component ? <Component {...props} /> : fallback}
    </div>
  );
}

// Feature-scoped stores (Zustand)
// features/cart/store/cartStore.ts
interface CartStore {
  items: CartItem[];
  addItem: (item: CartItem) => void;
  removeItem: (id: string) => void;
  clear: () => void;
  total: () => number;
}

export const useCartStore = create<CartStore>((set, get) => ({
  items: [],
  addItem: (item) => set((state) => ({
    items: [...state.items, item],
  })),
  removeItem: (id) => set((state) => ({
    items: state.items.filter(i => i.id !== id),
  })),
  clear: () => set({ items: [] }),
  total: () => get().items.reduce((sum, i) => sum + i.price * i.quantity, 0),
}));

// Derived state with selectors (prevents unnecessary re-renders)
function CartBadge() {
  const itemCount = useCartStore((state) => state.items.length);
  return <Badge count={itemCount} />;
}

function CartTotal() {
  const total = useCartStore((state) =>
    state.items.reduce((sum, i) => sum + i.price * i.quantity, 0)
  );
  return <span>${total.toFixed(2)}</span>;
}

Feature Flags Architecture

// Feature flag provider
interface FeatureFlags {
  [key: string]: boolean | string | number;
}

const FeatureFlagContext = createContext<FeatureFlags>({});

function FeatureFlagProvider({ children }: { children: React.ReactNode }) {
  const { data: flags, isLoading } = useQuery({
    queryKey: ['feature-flags'],
    queryFn: () => fetch('/api/flags').then(r => r.json()),
    staleTime: 5 * 60 * 1000,
    refetchInterval: 60 * 1000,
  });

  if (isLoading) return <AppSkeleton />;

  return (
    <FeatureFlagContext.Provider value={flags || {}}>
      {children}
    </FeatureFlagContext.Provider>
  );
}

function useFeatureFlag(flag: string): boolean {
  const flags = useContext(FeatureFlagContext);
  return Boolean(flags[flag]);
}

function useFeatureVariant(flag: string): string {
  const flags = useContext(FeatureFlagContext);
  return String(flags[flag] || 'control');
}

// Usage
function SearchPage() {
  const useNewSearch = useFeatureFlag('new-search-algorithm');
  const searchVariant = useFeatureVariant('search-ui-experiment');

  if (useNewSearch) {
    return searchVariant === 'grid'
      ? <SearchGridView />
      : <SearchListView />;
  }

  return <LegacySearch />;
}

// Feature gate component
function FeatureGate({ flag, children, fallback = null }) {
  const enabled = useFeatureFlag(flag);
  return enabled ? children : fallback;
}

<FeatureGate flag="premium-analytics" fallback={<UpgradePrompt />}>
  <PremiumAnalytics />
</FeatureGate>

A/B Testing Infrastructure

// A/B test hook with analytics integration
interface Experiment {
  id: string;
  variant: string;
  payload?: Record<string, unknown>;
}

function useExperiment(experimentId: string): Experiment {
  const { data: experiments } = useQuery({
    queryKey: ['experiments'],
    queryFn: fetchExperiments,
  });
  const analytics = useAnalytics();

  const experiment = useMemo(() => {
    const exp = experiments?.[experimentId];
    return {
      id: experimentId,
      variant: exp?.variant || 'control',
      payload: exp?.payload,
    };
  }, [experiments, experimentId]);

  useEffect(() => {
    analytics.track('experiment_exposure', {
      experiment_id: experimentId,
      variant: experiment.variant,
    });
  }, [experimentId, experiment.variant, analytics]);

  return experiment;
}

// Usage
function CheckoutPage() {
  const { variant } = useExperiment('checkout-flow-v2');

  switch (variant) {
    case 'single-page':
      return <SinglePageCheckout />;
    case 'multi-step':
      return <MultiStepCheckout />;
    default:
      return <OriginalCheckout />;
  }
}

Analytics Abstraction Layer

// Analytics abstraction — decouple from vendor
interface AnalyticsProvider {
  track(event: string, properties?: Record<string, unknown>): void;
  identify(userId: string, traits?: Record<string, unknown>): void;
  page(name: string, properties?: Record<string, unknown>): void;
}

class AnalyticsService {
  private providers: AnalyticsProvider[] = [];

  addProvider(provider: AnalyticsProvider) {
    this.providers.push(provider);
  }

  track(event: string, properties?: Record<string, unknown>) {
    this.providers.forEach(p => {
      try { p.track(event, properties); }
      catch (e) { console.warn('Analytics error:', e); }
    });
  }

  identify(userId: string, traits?: Record<string, unknown>) {
    this.providers.forEach(p => p.identify(userId, traits));
  }

  page(name: string, properties?: Record<string, unknown>) {
    this.providers.forEach(p => p.page(name, properties));
  }
}

// Providers
class GoogleAnalyticsProvider implements AnalyticsProvider {
  track(event: string, properties?: Record<string, unknown>) {
    gtag('event', event, properties);
  }
  identify(userId: string) { gtag('set', { user_id: userId }); }
  page(name: string) { gtag('event', 'page_view', { page_title: name }); }
}

class MixpanelProvider implements AnalyticsProvider {
  track(event: string, properties?: Record<string, unknown>) {
    mixpanel.track(event, properties);
  }
  identify(userId: string, traits?: Record<string, unknown>) {
    mixpanel.identify(userId);
    if (traits) mixpanel.people.set(traits);
  }
  page(name: string) { mixpanel.track('Page View', { page: name }); }
}

// React hook
const AnalyticsContext = createContext<AnalyticsService>(new AnalyticsService());

function useAnalytics() {
  return useContext(AnalyticsContext);
}

// Auto-track page views
function usePageTracking() {
  const location = useLocation();
  const analytics = useAnalytics();

  useEffect(() => {
    analytics.page(location.pathname, {
      path: location.pathname,
      search: location.search,
    });
  }, [location, analytics]);
}
Interview Tip

When discussing scalability, focus on organizational scaling, not just technical. A Staff engineer thinks about: (1) How do 50 developers work in this codebase without stepping on each other? (2) How do we ship features independently? (3) How do we measure the impact of changes? Feature flags, module boundaries, code ownership (CODEOWNERS), and analytics abstraction are the tools that enable organizational velocity.
