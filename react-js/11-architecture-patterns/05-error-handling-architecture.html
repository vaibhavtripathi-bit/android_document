Error Categorization

Before designing error handling, categorize errors by their nature, recoverability, and user impact. Each category demands a different handling strategy.

Error Boundary Hierarchy

Error boundaries should be layered — from page-level catch-all down to widget-level isolation. The goal is to contain failures to the smallest possible blast radius.

Error Boundary Hierarchy ═══════════════════════════════════════════════════════════ ┌─────────────────────────────────────────────────────┐ │ ROOT ERROR BOUNDARY (App level) │ │ Catches: Unrecoverable errors │ │ Fallback: Full-page error with "Reload App" │ │ Reports: Always (critical) │ │ │ │ ┌───────────────────────────────────────────────┐ │ │ │ ROUTE ERROR BOUNDARY (per page) │ │ │ │ Catches: Page-level render errors │ │ │ │ Fallback: "This page had an error" + nav │ │ │ │ Reports: Always │ │ │ │ │ │ │ │ ┌─────────────────┐ ┌─────────────────────┐ │ │ │ │ │ WIDGET BOUNDARY │ │ WIDGET BOUNDARY │ │ │ │ │ │ (sidebar) │ │ (main content) │ │ │ │ │ │ Fallback: hide │ │ Fallback: skeleton │ │ │ │ │ │ or placeholder │ │ or retry button │ │ │ │ │ └─────────────────┘ └─────────────────────┘ │ │ │ └───────────────────────────────────────────────┘ │ └─────────────────────────────────────────────────────┘ 
// Reusable ErrorBoundary with reset and reporting
interface ErrorBoundaryProps {
  children: React.ReactNode;
  fallback: React.ReactNode | ((props: FallbackProps) => React.ReactNode);
  onError?: (error: Error, errorInfo: React.ErrorInfo) => void;
  resetKeys?: unknown[];
  level: 'root' | 'route' | 'widget';
}

interface FallbackProps {
  error: Error;
  resetErrorBoundary: () => void;
}

class ErrorBoundary extends React.Component<ErrorBoundaryProps, { error: Error | null }> {
  state = { error: null };

  static getDerivedStateFromError(error: Error) {
    return { error };
  }

  componentDidCatch(error: Error, errorInfo: React.ErrorInfo) {
    this.props.onError?.(error, errorInfo);

    errorReporter.report({
      error,
      componentStack: errorInfo.componentStack,
      level: this.props.level,
      timestamp: Date.now(),
      url: window.location.href,
      userAgent: navigator.userAgent,
    });
  }

  componentDidUpdate(prevProps: ErrorBoundaryProps) {
    if (this.state.error && this.props.resetKeys) {
      const changed = this.props.resetKeys.some(
        (key, i) => key !== prevProps.resetKeys?.[i]
      );
      if (changed) this.setState({ error: null });
    }
  }

  resetErrorBoundary = () => {
    this.setState({ error: null });
  };

  render() {
    if (this.state.error) {
      const { fallback } = this.props;
      if (typeof fallback === 'function') {
        return fallback({
          error: this.state.error,
          resetErrorBoundary: this.resetErrorBoundary,
        });
      }
      return fallback;
    }
    return this.props.children;
  }
}

// Usage: layered boundaries
function App() {
  return (
    <ErrorBoundary level="root" fallback={<FullPageError />}>
      <AuthProvider>
        <Router>
          <Routes>
            <Route path="/dashboard" element={
              <ErrorBoundary level="route" fallback={RouteErrorFallback}>
                <DashboardPage />
              </ErrorBoundary>
            } />
          </Routes>
        </Router>
      </AuthProvider>
    </ErrorBoundary>
  );
}

function DashboardPage() {

      </ErrorBoundary>
      <ErrorBoundary level="widget" fallback={<WidgetPlaceholder />}>

}

  });

// React 19 createRoot error callbacks
const root = createRoot(document.getElementById('root'), {
  onCaughtError(error, errorInfo) {
    // Error caught by an error boundary
    errorReporter.report({
      type: 'caught-render-error',
      error,
      componentStack: errorInfo.componentStack,
      severity: 'warning',
    });
  },
  onUncaughtError(error, errorInfo) {
    // Error NOT caught by any error boundary
    errorReporter.report({
      type: 'uncaught-render-error',
      error,
      componentStack: errorInfo.componentStack,
      severity: 'critical',
    });
  },
  onRecoverableError(error, errorInfo) {
    // Hydration mismatch or recoverable error
    errorReporter.report({
      type: 'recoverable-error',
      error,
      componentStack: errorInfo.componentStack,
      severity: 'info',
    });
  },
});

// Sentry integration with React
import * as Sentry from '@sentry/react';

Sentry.init({

  replaysOnErrorSampleRate: 1.0,

    if (error instanceof NetworkError) {
      event.level = 'warning';

      return null; // Don't report validation errors

function useErrorReporter() {
  const user = useAuth();

    Sentry.withScope((scope) => {
      scope.setUser({ id: user?.id, email: user?.email });
      scope.setTags({
        page: window.location.pathname,
        feature: context?.feature,
      });
      scope.setContext('additional', context || {});
      Sentry.captureException(error);
    });
  }, [user]);
}

Retry Strategies

// Exponential backoff with jitter
async function retryWithBackoff<T>(
  fn: () => Promise<T>,
  options: {
    maxRetries?: number;
    baseDelay?: number;
    maxDelay?: number;
    retryOn?: (error: unknown) => boolean;
    onRetry?: (attempt: number, error: unknown) => void;
  } = {}
): Promise<T> {
  const {
    maxRetries = 3,
    baseDelay = 1000,
    maxDelay = 30000,
    retryOn = isRetryable,
    onRetry,
  } = options;

  let lastError: unknown;

  for (let attempt = 0; attempt <= maxRetries; attempt++) {
    try {
      return await fn();
    } catch (error) {
      lastError = error;

      if (attempt === maxRetries || !retryOn(error)) {
        throw error;
      }

      const delay = Math.min(
        baseDelay * Math.pow(2, attempt) + Math.random() * 1000,
        maxDelay
      );

      onRetry?.(attempt + 1, error);
      await new Promise(resolve => setTimeout(resolve, delay));
    }
  }

  throw lastError;
}

function isRetryable(error: unknown): boolean {
  if (error instanceof NetworkError) return true;
  if (error instanceof HttpError) {
    return [408, 429, 500, 502, 503, 504].includes(error.status);
  }
  return false;
}

// React Query built-in retry
const { data } = useQuery({
  queryKey: ['users'],
  queryFn: fetchUsers,
  retry: 3,
  retryDelay: (attemptIndex) => Math.min(1000 * 2 ** attemptIndex, 30000),
  retryOnMount: true,
});

Circuit Breaker Pattern in Frontend

// Circuit breaker prevents cascading failures
type CircuitState = 'CLOSED' | 'OPEN' | 'HALF_OPEN';

class CircuitBreaker {
  private state: CircuitState = 'CLOSED';
  private failureCount = 0;
  private lastFailureTime = 0;
  private successCount = 0;

  constructor(
    private readonly threshold: number = 5,
    private readonly resetTimeout: number = 30000,
    private readonly halfOpenMax: number = 3
  ) {}

  async execute<T>(fn: () => Promise<T>, fallback?: () => T): Promise<T> {
    if (this.state === 'OPEN') {
      if (Date.now() - this.lastFailureTime > this.resetTimeout) {
        this.state = 'HALF_OPEN';
        this.successCount = 0;
      } else {
        if (fallback) return fallback();
        throw new CircuitOpenError('Circuit breaker is OPEN');
      }
    }

    try {
      const result = await fn();
      this.onSuccess();
      return result;
    } catch (error) {
      this.onFailure();
      if (fallback) return fallback();
      throw error;
    }
  }

  private onSuccess() {
    if (this.state === 'HALF_OPEN') {
      this.successCount++;
      if (this.successCount >= this.halfOpenMax) {
        this.state = 'CLOSED';
        this.failureCount = 0;
      }
    } else {
      this.failureCount = 0;
    }
  }

  private onFailure() {
    this.failureCount++;
    this.lastFailureTime = Date.now();
    if (this.failureCount >= this.threshold) {
      this.state = 'OPEN';
    }
  }
}

// Usage with API calls
const searchCircuit = new CircuitBreaker(5, 30000);

async function searchProducts(query: string) {
  return searchCircuit.execute(
    () => api.get(`/search?q=${query}`),
    () => cachedResults.get(query) || []  // fallback to cache
  );
}

Graceful Degradation

// Feature degradation based on error state
function Dashboard() {
  const revenue = useQuery({ queryKey: ['revenue'], queryFn: fetchRevenue });
  const activity = useQuery({ queryKey: ['activity'], queryFn: fetchActivity });
  const recommendations = useQuery({ queryKey: ['recs'], queryFn: fetchRecs });

  return (
    <div className="dashboard">
      {/* Critical: show error state */}
      {revenue.error
        ? <RevenueError onRetry={revenue.refetch} />
        : <RevenueChart data={revenue.data} />
      }

      {/* Important: show stale data with warning */}
      {activity.error
        ? <ActivityStale data={activity.data} warning="Data may be outdated" />
        : <ActivityFeed data={activity.data} />
      }

      {/* Nice-to-have: hide entirely on error */}
      {!recommendations.error && recommendations.data && (
        <Recommendations data={recommendations.data} />
      )}
    </div>
  );
}

function VideoPlayer({ src }) {
  if (!window.MediaSource) {
    return <video src={src} controls />;  // basic fallback
  }

  if (!MediaSource.isTypeSupported('video/mp4; codecs="avc1.42E01E"')) {
    return <VideoTranscoded src={src} />;  // transcoded fallback
  }

  return <AdaptivePlayer src={src} />;  // full MSE player
}
Critical — Error Boundary Limitations

Error boundaries do NOT catch: (1) Event handler errors — use try/catch. (2) Async errors (setTimeout, promises) — use global handlers. (3) Server-side rendering errors — use try/catch in SSR. (4) Errors in the error boundary itself — use a parent boundary. Design your error strategy knowing these gaps.

Interview Tip

When asked about error handling, don't just mention error boundaries. Present a complete strategy: (1) Categorize errors by type. (2) Layer boundaries by blast radius. (3) Implement retry with backoff for transient errors. (4) Use circuit breakers for failing services. (5) Degrade gracefully based on feature criticality. (6) Report with context for debugging. This holistic view is what L6 interviewers look for.
