What Are Micro-Frontends?

Micro-frontends extend the microservices concept to the frontend: independently developed, tested, and deployed frontend applications that compose into a single user experience. The primary motivation is organizational scaling — enabling multiple teams to work on the same product without blocking each other.

Module Federation (Webpack 5)

Module Federation is a Webpack 5 feature that allows JavaScript applications to dynamically load code from other applications at runtime. It's the most popular approach for React micro-frontends.

// HOST (Shell) — webpack.config.js
const { ModuleFederationPlugin } = require('webpack').container;

module.exports = {
  plugins: [
    new ModuleFederationPlugin({
      name: 'shell',
      remotes: {
        search: 'search@http://localhost:3001/remoteEntry.js',
        product: 'product@http://localhost:3002/remoteEntry.js',
        cart: 'cart@http://localhost:3003/remoteEntry.js',
      },
      shared: {
        react: { singleton: true, requiredVersion: '^18.0.0' },
        'react-dom': { singleton: true, requiredVersion: '^18.0.0' },
        'react-router-dom': { singleton: true, requiredVersion: '^6.0.0' },
      },
    }),
  ],
};

// REMOTE (Search MFE) — webpack.config.js
module.exports = {
  plugins: [
    new ModuleFederationPlugin({
      name: 'search',
      filename: 'remoteEntry.js',
      exposes: {
        './SearchPage': './src/pages/SearchPage',
        './SearchBar': './src/components/SearchBar',
      },
      shared: {
        react: { singleton: true, requiredVersion: '^18.0.0' },
        'react-dom': { singleton: true, requiredVersion: '^18.0.0' },
      },
    }),
  ],
};

// Runtime remote loading (for A/B testing, feature flags)
async function loadRemote(scope, module) {
  await __webpack_init_sharing__('default');
  const container = window[scope];
  await container.init(__webpack_share_scopes__.default);
  const factory = await container.get(module);
  return factory();
}

// Usage: dynamically decide which remote to load
const remoteUrl = featureFlags.newSearch
  ? 'http://cdn.example.com/search-v2/remoteEntry.js'
  : 'http://cdn.example.com/search-v1/remoteEntry.js';

await loadScript(remoteUrl);
const SearchModule = await loadRemote('search', './SearchPage');
Architect Insight

Module Federation's singleton: true for React is critical — having two React instances causes hooks to break (the "Invalid hook call" error). The requiredVersion field provides a safety net: if versions are incompatible, Webpack falls back to loading a separate copy, which you should detect and alert on.

iframe-Based Isolation

// iframe MFE wrapper with communication
function MicroFrontend({ url, name, onMessage }) {
  const iframeRef = useRef(null);

  useEffect(() => {
    const handler = (event) => {
      if (event.origin !== new URL(url).origin) return;
      if (event.data.source !== name) return;
      onMessage(event.data);
    };
    window.addEventListener('message', handler);
    return () => window.removeEventListener('message', handler);
  }, [url, name, onMessage]);

  const sendMessage = useCallback((data) => {
    iframeRef.current?.contentWindow?.postMessage(
      { source: 'shell', ...data },
      new URL(url).origin
    );
  }, [url]);

  return (
    <iframe
      ref={iframeRef}
      src={url}
      title={name}
      style={{ border: 'none', width: '100%', height: '100%' }}
      sandbox="allow-scripts allow-same-origin allow-forms"
    />
  );
}

// Inside iframe MFE:
window.addEventListener('message', (event) => {
  if (event.data.source !== 'shell') return;
  if (event.data.type === 'AUTH_TOKEN') {
    setAuthToken(event.data.token);
  }
});

// Notify shell of navigation
window.parent.postMessage(
  { source: 'product', type: 'NAVIGATE', path: '/product/123' },
  'https://shell.example.com'
);

Web Components as MFE Boundary

// Wrap React MFE as a Web Component
class SearchElement extends HTMLElement {
  connectedCallback() {
    const shadow = this.attachShadow({ mode: 'open' });
    const mountPoint = document.createElement('div');
    shadow.appendChild(mountPoint);

    // Import styles into shadow DOM
    const style = document.createElement('link');
    style.rel = 'stylesheet';
    style.href = '/search/styles.css';
    shadow.appendChild(style);

    this._root = ReactDOM.createRoot(mountPoint);
    this._root.render(
      <SearchApp
        authToken={this.getAttribute('auth-token')}
        onNavigate={(path) => {
          this.dispatchEvent(new CustomEvent('navigate', {
            detail: { path },
            bubbles: true,
            composed: true,
          }));
        }}
      />
    );
  }

  disconnectedCallback() {
    this._root?.unmount();
  }

  static get observedAttributes() {
    return ['auth-token'];
  }

  attributeChangedCallback(name, oldVal, newVal) {
    if (name === 'auth-token' && this._root) {
      this._root.render(<SearchApp authToken={newVal} />);
    }
  }
}

customElements.define('search-mfe', SearchElement);

// Usage in shell (any framework):
// <search-mfe auth-token="xxx"></search-mfe>

Cross-Micro-Frontend Communication

// Pattern 1: Custom Events (loosely coupled)
// Publisher (Cart MFE)
function addToCart(product) {
  cart.add(product);
  window.dispatchEvent(new CustomEvent('cart:updated', {
    detail: { count: cart.items.length, total: cart.total },
  }));
}

// Subscriber (Shell — header cart badge)
function CartBadge() {
  const [count, setCount] = useState(0);
  useEffect(() => {
    const handler = (e) => setCount(e.detail.count);
    window.addEventListener('cart:updated', handler);
    return () => window.removeEventListener('cart:updated', handler);
  }, []);
  return <Badge count={count} />;
}

// Pattern 2: Shared State Store
// shared-store.ts (loaded by shell, exposed to MFEs)
import { create } from 'zustand';

export const useSharedStore = create((set) => ({
  user: null,
  cart: { items: [], total: 0 },
  setUser: (user) => set({ user }),
  addToCart: (item) => set((state) => ({
    cart: {
      items: [...state.cart.items, item],
      total: state.cart.total + item.price,
    },
  })),
}));

// Pattern 3: Event Bus (typed)
type EventMap = {
  'cart:add': { productId: string; quantity: number };
  'cart:remove': { productId: string };
  'auth:login': { user: User };
  'auth:logout': void;
  'navigate': { path: string };
};

class TypedEventBus {
  private handlers = new Map<string, Set<Function>>();

  on<K extends keyof EventMap>(event: K, handler: (data: EventMap[K]) => void) {
    if (!this.handlers.has(event)) this.handlers.set(event, new Set());
    this.handlers.get(event)!.add(handler);
    return () => this.handlers.get(event)!.delete(handler);
  }

  emit<K extends keyof EventMap>(event: K, data: EventMap[K]) {
    this.handlers.get(event)?.forEach(h => h(data));
  }
}

export const eventBus = new TypedEventBus();
// Exposed on window for cross-MFE access
(window as any).__EVENT_BUS__ = eventBus;

Routing Across Micro-Frontends

// Shell owns top-level routing
// Each MFE owns its sub-routes

// Shell router
function ShellRouter() {
  return (
    <BrowserRouter>
      <ShellLayout>
        <Routes>
          <Route path="/search/*" element={<SearchMFE />} />
          <Route path="/product/*" element={<ProductMFE />} />
          <Route path="/cart/*" element={<CartMFE />} />
        </Routes>
      </ShellLayout>
    </BrowserRouter>
  );
}

// Search MFE internal routing (uses MemoryRouter or basename)
function SearchApp({ basename = '/search' }) {
  return (
    <MemoryRouter initialEntries={[window.location.pathname.replace(basename, '')]}>
      <Routes>
        <Route path="/" element={<SearchHome />} />
        <Route path="/results" element={<SearchResults />} />
        <Route path="/filters" element={<SearchFilters />} />
      </Routes>
    </MemoryRouter>
  );
}

// Cross-MFE navigation via shell
function navigateToProduct(productId) {
  window.dispatchEvent(new CustomEvent('navigate', {
    detail: { path: `/product/${productId}` },
  }));
}

// Shell listens and navigates
useEffect(() => {
  const handler = (e) => navigate(e.detail.path);
  window.addEventListener('navigate', handler);
  return () => window.removeEventListener('navigate', handler);
}, [navigate]);
Critical — Routing Pitfall

Never use two BrowserRouter instances — they'll fight over history.pushState. The shell owns BrowserRouter; MFEs use MemoryRouter or receive a basename prop. For URL synchronization, the shell listens to MFE navigation events and updates the browser URL.

Interview Tip

When asked about micro-frontends, always start with "why" — organizational scaling. Then discuss the tradeoff: deployment independence and team autonomy vs increased complexity in shared state, routing, and bundle size. A Staff engineer knows that micro-frontends are an organizational pattern, not a technical one. If you have one team, you don't need micro-frontends.
