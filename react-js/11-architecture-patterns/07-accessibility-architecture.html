WCAG 2.1 AA Compliance

WCAG (Web Content Accessibility Guidelines) 2.1 Level AA is the standard for most organizations. It's organized around four principles: Perceivable, Operable, Understandable, and Robust (POUR).

PrincipleKey Requirements (AA)React Implementation
PerceivableText alternatives, captions, contrast 4.5:1alt props, aria-label, CSS custom properties for theme contrast
OperableKeyboard accessible, enough time, no seizuresFocus management, keyboard handlers, prefers-reduced-motion
UnderstandableReadable, predictable, input assistanceError messages, labels, consistent navigation
RobustCompatible with assistive techSemantic HTML, ARIA roles, testing with screen readers

Focus Management in SPAs

SPAs break the browser's natural focus management because page navigations don't trigger a full page load. Staff engineers must implement focus management as a core architectural concern.

// Focus management on route change
function useFocusOnNavigate() {
  const location = useLocation();
  const mainRef = useRef<HTMLElement>(null);

  useEffect(() => {
    mainRef.current?.focus();
    document.title = getPageTitle(location.pathname);
  }, [location.pathname]);

  return mainRef;
}

function App() {
  const mainRef = useFocusOnNavigate();
  return (
    <div>
      <SkipLink />
      <Header />
      <main ref={mainRef} tabIndex={-1} id="main-content">
        <Routes>{/* ... */}</Routes>
      </main>
      <Footer />
    </div>
  );
}

// Focus trap for modals
function useFocusTrap(ref: RefObject<HTMLElement>, active: boolean) {
  useEffect(() => {
    if (!active || !ref.current) return;

    const element = ref.current;
    const focusableSelector =
      'button, [href], input, select, textarea, [tabindex]:not([tabindex="-1"])';
    const focusableElements = element.querySelectorAll(focusableSelector);
    const firstFocusable = focusableElements[0] as HTMLElement;
    const lastFocusable = focusableElements[focusableElements.length - 1] as HTMLElement;

    const previouslyFocused = document.activeElement as HTMLElement;
    firstFocusable?.focus();

    const handleKeyDown = (e: KeyboardEvent) => {
      if (e.key !== 'Tab') return;

      if (e.shiftKey) {
        if (document.activeElement === firstFocusable) {
          e.preventDefault();
          lastFocusable?.focus();
        }
      } else {
        if (document.activeElement === lastFocusable) {
          e.preventDefault();
          firstFocusable?.focus();
        }
      }
    };

    element.addEventListener('keydown', handleKeyDown);
    return () => {
      element.removeEventListener('keydown', handleKeyDown);
      previouslyFocused?.focus();
    };
  }, [ref, active]);
}

// Usage
function Dialog({ open, onClose, children }) {
  const dialogRef = useRef(null);
  useFocusTrap(dialogRef, open);

  useEffect(() => {
    const handleEscape = (e) => {
      if (e.key === 'Escape') onClose();
    };
    if (open) document.addEventListener('keydown', handleEscape);
    return () => document.removeEventListener('keydown', handleEscape);
  }, [open, onClose]);

  if (!open) return null;
  return createPortal(
    <div className="dialog-overlay" onClick={onClose}>
      <div ref={dialogRef} role="dialog" aria-modal="true" onClick={e => e.stopPropagation()}>
        {children}
      </div>
    </div>,
    document.body
  );
}

Keyboard Navigation Patterns

// Roving tabindex for composite widgets
function useRovingTabIndex(items: string[], orientation: 'horizontal' | 'vertical' = 'horizontal') {
  const [activeIndex, setActiveIndex] = useState(0);

  const handleKeyDown = useCallback((e: React.KeyboardEvent) => {
    const nextKey = orientation === 'horizontal' ? 'ArrowRight' : 'ArrowDown';
    const prevKey = orientation === 'horizontal' ? 'ArrowLeft' : 'ArrowUp';

    switch (e.key) {
      case nextKey:
        e.preventDefault();
        setActiveIndex(i => (i + 1) % items.length);
        break;
      case prevKey:
        e.preventDefault();
        setActiveIndex(i => (i - 1 + items.length) % items.length);
        break;
      case 'Home':
        e.preventDefault();
        setActiveIndex(0);
        break;
      case 'End':
        e.preventDefault();
        setActiveIndex(items.length - 1);
        break;
    }
  }, [items.length, orientation]);

  const getItemProps = useCallback((index: number) => ({
    tabIndex: index === activeIndex ? 0 : -1,
    'data-active': index === activeIndex,
    ref: (el: HTMLElement | null) => {
      if (index === activeIndex) el?.focus();
    },
  }), [activeIndex]);

  return { activeIndex, handleKeyDown, getItemProps };
}

// Usage: toolbar with roving tabindex
function Toolbar({ items }) {
  const { handleKeyDown, getItemProps } = useRovingTabIndex(
    items.map(i => i.id),
    'horizontal'
  );

  return (
    <div role="toolbar" aria-label="Formatting" onKeyDown={handleKeyDown}>
      {items.map((item, i) => (
        <button key={item.id} {...getItemProps(i)} onClick={item.action}>
          {item.icon}
          <span className="sr-only">{item.label}</span>
        </button>
      ))}
    </div>
  );
}

Reduced Motion

// Hook for respecting motion preferences
function usePrefersReducedMotion() {
  const [prefersReduced, setPrefersReduced] = useState(() =>
    window.matchMedia('(prefers-reduced-motion: reduce)').matches
  );

  useEffect(() => {
    const mq = window.matchMedia('(prefers-reduced-motion: reduce)');
    const handler = (e: MediaQueryListEvent) => setPrefersReduced(e.matches);
    mq.addEventListener('change', handler);
    return () => mq.removeEventListener('change', handler);
  }, []);

  return prefersReduced;
}

// CSS approach (preferred — no JS needed)
/* Default: with animation */
.fade-in {

}

@keyframes fadeIn {
  from { opacity: 0; transform: translateY(10px); }
  to   { opacity: 1; transform: translateY(0); }

/* Reduced motion: instant */
@media (prefers-reduced-motion: reduce) {
  .fade-in {
    animation: none;
  }

  *::before,
  *::after {
    animation-duration: 0.01ms !important;
    animation-iteration-count: 1 !important;
    transition-duration: 0.01ms !important;
    scroll-behavior: auto !important;
  }
}

Live Regions for Dynamic Content

// Announce dynamic changes to screen readers
function useAnnounce() {
  const announce = useCallback((message: string, priority: 'polite' | 'assertive' = 'polite') => {
    const el = document.getElementById(`aria-live-${priority}`);
    if (el) {
      el.textContent = '';
      requestAnimationFrame(() => {
        el.textContent = message;
      });
    }
  }, []);

  return announce;
}

// Live region container (render once at app root)
function LiveRegions() {
  return (
    <>
      <div id="aria-live-polite" aria-live="polite" aria-atomic="true" className="sr-only" />
      <div id="aria-live-assertive" aria-live="assertive" aria-atomic="true" className="sr-only" />
    </>
  );
}

// Usage
function SearchResults({ results, query }) {
  const announce = useAnnounce();

  useEffect(() => {
    announce(`${results.length} results found for "${query}"`);
  }, [results.length, query, announce]);

  return (
    <div>
      <h2>{results.length} results for "{query}"</h2>
      {results.map(r => <ResultCard key={r.id} result={r} />)}
    </div>
  );
}

// Toast notifications with live region
function Toast({ message, type }) {
  return (
    <div role="status" aria-live="polite" className={`toast toast-${type}`}>
      {type === 'error' && <AlertIcon aria-hidden="true" />}
      {message}
    </div>
  );
}

Accessible Forms

// Accessible form field component
function FormField({ label, error, hint, required, children, id }) {
  const fieldId = id || useId();
  const errorId = `${fieldId}-error`;
  const hintId = `${fieldId}-hint`;

  return (
    <div className="form-field">
      <label htmlFor={fieldId}>
        {label}
        {required && <span aria-hidden="true"> *</span>}
        {required && <span className="sr-only"> (required)</span>}
      </label>

      {hint && <p id={hintId} className="field-hint">{hint}</p>}

      {React.cloneElement(children, {
        id: fieldId,
        'aria-required': required,
        'aria-invalid': !!error,
        'aria-describedby': [
          hint ? hintId : null,
          error ? errorId : null,
        ].filter(Boolean).join(' ') || undefined,
      })}

      {error && (
        <p id={errorId} className="field-error" role="alert">
          {error}
        </p>
      )}
    </div>
  );
}

// Usage
<form onSubmit={handleSubmit} noValidate>
  <FormField label="Email" error={errors.email} required hint="We'll never share your email">
    <input type="email" value={email} onChange={e => setEmail(e.target.value)} />
  </FormField>

  <FormField label="Password" error={errors.password} required hint="Minimum 8 characters">
    <input type="password" value={password} onChange={e => setPassword(e.target.value)} />
  </FormField>

  <button type="submit" aria-busy={isSubmitting}>
    {isSubmitting ? 'Submitting...' : 'Submit'}
  </button>
</form>

Accessibility Testing Automation

// 1. jest-axe for unit tests
import { render } from '@testing-library/react';
import { axe, toHaveNoViolations } from 'jest-axe';

expect.extend(toHaveNoViolations);

test('Button has no a11y violations', async () => {
  const { container } = render(<Button>Click me</Button>);
  const results = await axe(container);
  expect(results).toHaveNoViolations();
});

// 2. Cypress + axe for integration tests
describe('Login Page', () => {
  it('has no a11y violations', () => {
    cy.visit('/login');
    cy.injectAxe();
    cy.checkA11y(null, {
      runOnly: {
        type: 'tag',
        values: ['wcag2a', 'wcag2aa'],
      },
    });
  });

  it('is keyboard navigable', () => {
    cy.visit('/login');
    cy.get('body').tab();
    cy.focused().should('have.attr', 'href', '#main-content');
    cy.focused().tab();
    cy.focused().should('have.attr', 'type', 'email');
  });
});

Architect Insight

Accessibility is an architectural concern, not a feature to bolt on later. At Staff level, embed a11y into: (1) Component API design — every interactive component gets keyboard + ARIA support by default. (2) Design system — contrast ratios and focus styles are tokens, not afterthoughts. (3) CI pipeline — axe-core runs on every PR. (4) Code review checklist — a11y is a merge requirement. This systemic approach prevents the "accessibility sprint" anti-pattern.

Interview Tip

When asked about accessibility, demonstrate that you think about it architecturally: (1) Semantic HTML first, ARIA second. (2) Focus management is a routing concern. (3) Live regions for dynamic content. (4) Automated testing at every level. (5) Design tokens enforce contrast ratios. Don't just list ARIA attributes — show how a11y is woven into the system design.
